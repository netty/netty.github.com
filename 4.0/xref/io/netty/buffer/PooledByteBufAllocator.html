48      static {
49          int defaultPageSize = SystemPropertyUtil.getInt("io.netty.allocator.pageSize", 8192);
50          Throwable pageSizeFallbackCause = null;
51          try {
52              validateAndCalculatePageShifts(defaultPageSize);
53          } catch (Throwable t) {
54              pageSizeFallbackCause = t;
55              defaultPageSize = 8192;
56          }
57          DEFAULT_PAGE_SIZE = defaultPageSize;
58  
59          int defaultMaxOrder = SystemPropertyUtil.getInt("io.netty.allocator.maxOrder", 11);
60          Throwable maxOrderFallbackCause = null;
61          try {
62              validateAndCalculateChunkSize(DEFAULT_PAGE_SIZE, defaultMaxOrder);
63          } catch (Throwable t) {
64              maxOrderFallbackCause = t;
65              defaultMaxOrder = 11;
66          }
67          DEFAULT_MAX_ORDER = defaultMaxOrder;
68  
69          // Determine reasonable default for nHeapArena and nDirectArena.
70          // Assuming each arena has 3 chunks, the pool should not consume more than 50% of max memory.
71          final Runtime runtime = Runtime.getRuntime();
72  
73          // Use 2 * cores by default to reduce condition as we use 2 * cores for the number of EventLoops
74          // in NIO and EPOLL as well. If we choose a smaller number we will run into hotspots as allocation and
75          // deallocation needs to be synchronized on the PoolArena.
76          // See https://github.com/netty/netty/issues/3888
77          final int defaultMinNumArena = runtime.availableProcessors() * 2;
78          final int defaultChunkSize = DEFAULT_PAGE_SIZE << DEFAULT_MAX_ORDER;
79          DEFAULT_NUM_HEAP_ARENA = Math.max(0,
80                  SystemPropertyUtil.getInt(
81                          "io.netty.allocator.numHeapArenas",
82                          (int) Math.min(
83                                  defaultMinNumArena,
84                                  runtime.maxMemory() / defaultChunkSize / 2 / 3)));
85          DEFAULT_NUM_DIRECT_ARENA = Math.max(0,
86                  SystemPropertyUtil.getInt(
87                          "io.netty.allocator.numDirectArenas",
88                          (int) Math.min(
89                                  defaultMinNumArena,
90                                  PlatformDependent.maxDirectMemory() / defaultChunkSize / 2 / 3)));
91  
92          // cache sizes
93          DEFAULT_TINY_CACHE_SIZE = SystemPropertyUtil.getInt("io.netty.allocator.tinyCacheSize", 512);
94          DEFAULT_SMALL_CACHE_SIZE = SystemPropertyUtil.getInt("io.netty.allocator.smallCacheSize", 256);
95          DEFAULT_NORMAL_CACHE_SIZE = SystemPropertyUtil.getInt("io.netty.allocator.normalCacheSize", 64);
96  
97          // 32 kb is the default maximum capacity of the cached buffer. Similar to what is explained in
98          // 'Scalable memory allocation using jemalloc'
99          DEFAULT_MAX_CACHED_BUFFER_CAPACITY = SystemPropertyUtil.getInt(
100                 "io.netty.allocator.maxCachedBufferCapacity", 32 * 1024);
101 
102         // the number of threshold of allocations when cached entries will be freed up if not frequently used
103         DEFAULT_CACHE_TRIM_INTERVAL = SystemPropertyUtil.getInt(
104                 "io.netty.allocator.cacheTrimInterval", 8192);
105 
106         if (logger.isDebugEnabled()) {
107             logger.debug("-Dio.netty.allocator.numHeapArenas: {}", DEFAULT_NUM_HEAP_ARENA);
108             logger.debug("-Dio.netty.allocator.numDirectArenas: {}", DEFAULT_NUM_DIRECT_ARENA);
109             if (pageSizeFallbackCause == null) {
110                 logger.debug("-Dio.netty.allocator.pageSize: {}", DEFAULT_PAGE_SIZE);
111             } else {
112                 logger.debug("-Dio.netty.allocator.pageSize: {}", DEFAULT_PAGE_SIZE, pageSizeFallbackCause);
113             }
114             if (maxOrderFallbackCause == null) {

116             } else {
117                 logger.debug("-Dio.netty.allocator.maxOrder: {}", DEFAULT_MAX_ORDER, maxOrderFallbackCause);
118             }
119             logger.debug("-Dio.netty.allocator.chunkSize: {}", DEFAULT_PAGE_SIZE << DEFAULT_MAX_ORDER);
120             logger.debug("-Dio.netty.allocator.tinyCacheSize: {}", DEFAULT_TINY_CACHE_SIZE);
121             logger.debug("-Dio.netty.allocator.smallCacheSize: {}", DEFAULT_SMALL_CACHE_SIZE);
122             logger.debug("-Dio.netty.allocator.normalCacheSize: {}", DEFAULT_NORMAL_CACHE_SIZE);
123             logger.debug("-Dio.netty.allocator.maxCachedBufferCapacity: {}", DEFAULT_MAX_CACHED_BUFFER_CAPACITY);
124             logger.debug("-Dio.netty.allocator.cacheTrimInterval: {}", DEFAULT_CACHE_TRIM_INTERVAL);
125         }

157     public PooledByteBufAllocator(boolean preferDirect, int nHeapArena, int nDirectArena, int pageSize, int maxOrder,
158                                   int tinyCacheSize, int smallCacheSize, int normalCacheSize) {
159         super(preferDirect);
160         threadCache = new PoolThreadLocalCache();
161         this.tinyCacheSize = tinyCacheSize;
162         this.smallCacheSize = smallCacheSize;
163         this.normalCacheSize = normalCacheSize;
164         final int chunkSize = validateAndCalculateChunkSize(pageSize, maxOrder);
165 
166         if (nHeapArena < 0) {
167             throw new IllegalArgumentException("nHeapArena: " + nHeapArena + " (expected: >= 0)");
168         }
169         if (nDirectArena < 0) {
170             throw new IllegalArgumentException("nDirectArea: " + nDirectArena + " (expected: >= 0)");
171         }
172 
173         int pageShifts = validateAndCalculatePageShifts(pageSize);
174 
175         if (nHeapArena > 0) {
176             heapArenas = newArenaArray(nHeapArena);
177             List<PoolArenaMetric> metrics = new ArrayList<PoolArenaMetric>(heapArenas.length);
178             for (int i = 0; i < heapArenas.length; i ++) {
179                 PoolArena.HeapArena arena = new PoolArena.HeapArena(this, pageSize, maxOrder, pageShifts, chunkSize);
180                 heapArenas[i] = arena;
181                 metrics.add(arena);
182             }
183             heapArenaMetrics = Collections.unmodifiableList(metrics);
184         } else {
185             heapArenas = null;
186             heapArenaMetrics = Collections.emptyList();
187         }
188 
189         if (nDirectArena > 0) {
190             directArenas = newArenaArray(nDirectArena);
191             List<PoolArenaMetric> metrics = new ArrayList<PoolArenaMetric>(directArenas.length);
192             for (int i = 0; i < directArenas.length; i ++) {
193                 PoolArena.DirectArena arena = new PoolArena.DirectArena(
194                         this, pageSize, maxOrder, pageShifts, chunkSize);
195                 directArenas[i] = arena;
196                 metrics.add(arena);
197             }
198             directArenaMetrics = Collections.unmodifiableList(metrics);
199         } else {
200             directArenas = null;
201             directArenaMetrics = Collections.emptyList();
202         }
203     }

219     private static int validateAndCalculatePageShifts(int pageSize) {
220         if (pageSize < MIN_PAGE_SIZE) {
221             throw new IllegalArgumentException("pageSize: " + pageSize + " (expected: " + MIN_PAGE_SIZE + "+)");
222         }
223 
224         if ((pageSize & pageSize - 1) != 0) {
225             throw new IllegalArgumentException("pageSize: " + pageSize + " (expected: power of 2)");
226         }
227 
228         // Logarithm base 2. At this point we know that pageSize is a power of two.
229         return Integer.SIZE - 1 - Integer.numberOfLeadingZeros(pageSize);
230     }
231 
232     private static int validateAndCalculateChunkSize(int pageSize, int maxOrder) {
233         if (maxOrder > 14) {
234             throw new IllegalArgumentException("maxOrder: " + maxOrder + " (expected: 0-14)");
235         }
236 
237         // Ensure the resulting chunkSize does not overflow.
238         int chunkSize = pageSize;
239         for (int i = maxOrder; i > 0; i --) {
240             if (chunkSize > MAX_CHUNK_SIZE / 2) {
241                 throw new IllegalArgumentException(String.format(
242                         "pageSize (%d) << maxOrder (%d) must not exceed %d", pageSize, maxOrder, MAX_CHUNK_SIZE));
243             }
244             chunkSize <<= 1;
245         }
246         return chunkSize;
247     }

249     @Override
250     protected ByteBuf newHeapBuffer(int initialCapacity, int maxCapacity) {
251         PoolThreadCache cache = threadCache.get();
252         PoolArena<byte[]> heapArena = cache.heapArena;
253 
254         ByteBuf buf;
255         if (heapArena != null) {
256             buf = heapArena.allocate(cache, initialCapacity, maxCapacity);
257         } else {
258             buf = new UnpooledHeapByteBuf(this, initialCapacity, maxCapacity);
259         }
260 
261         return toLeakAwareBuffer(buf);
262     }
263 
264     @Override
265     protected ByteBuf newDirectBuffer(int initialCapacity, int maxCapacity) {
266         PoolThreadCache cache = threadCache.get();
267         PoolArena<ByteBuffer> directArena = cache.directArena;
268 
269         ByteBuf buf;
270         if (directArena != null) {
271             buf = directArena.allocate(cache, initialCapacity, maxCapacity);
272         } else {
273             if (PlatformDependent.hasUnsafe()) {
274                 buf = new UnpooledUnsafeDirectByteBuf(this, initialCapacity, maxCapacity);
275             } else {
276                 buf = new UnpooledDirectByteBuf(this, initialCapacity, maxCapacity);
277             }
278         }
279 
280         return toLeakAwareBuffer(buf);
281     }

354     final class PoolThreadLocalCache extends FastThreadLocal<PoolThreadCache> {
355         private final AtomicInteger index = new AtomicInteger();
356         final AtomicInteger caches = new AtomicInteger();
357 
358         @Override
359         protected PoolThreadCache initialValue() {
360             caches.incrementAndGet();
361             final int idx = index.getAndIncrement();
362             final PoolArena<byte[]> heapArena;
363             final PoolArena<ByteBuffer> directArena;
364 
365             if (heapArenas != null) {
366                 heapArena = heapArenas[Math.abs(idx % heapArenas.length)];
367             } else {
368                 heapArena = null;
369             }
370 
371             if (directArenas != null) {
372                 directArena = directArenas[Math.abs(idx % directArenas.length)];
373             } else {
374                 directArena = null;
375             }
376             return new PoolThreadCache(
377                     heapArena, directArena, tinyCacheSize, smallCacheSize, normalCacheSize,
378                     DEFAULT_MAX_CACHED_BUFFER_CAPACITY, DEFAULT_CACHE_TRIM_INTERVAL);
379         }
380 
381         @Override
382         protected void onRemoval(PoolThreadCache value) {
383             value.free();
384             caches.decrementAndGet();
385         }
386     }
