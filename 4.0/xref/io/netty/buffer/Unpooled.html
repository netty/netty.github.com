28  /**
29   * Creates a new {@link ByteBuf} by allocating new space or by wrapping
30   * or copying existing byte arrays, byte buffers and a string.
31   *
32   * <h3>Use static import</h3>
33   * This classes is intended to be used with Java 5 static import statement:
34   *
35   * <pre>
36   * import static io.netty.buffer.{@link Unpooled}.*;
37   *
38   * {@link ByteBuf} heapBuffer    = buffer(128);
39   * {@link ByteBuf} directBuffer  = directBuffer(256);
40   * {@link ByteBuf} wrappedBuffer = wrappedBuffer(new byte[128], new byte[256]);
41   * {@link ByteBuf} copiedBuffe r = copiedBuffer({@link ByteBuffer}.allocate(128));
42   * </pre>
43   *
44   * <h3>Allocating a new buffer</h3>
45   *
46   * Three buffer types are provided out of the box.
47   *
48   * <ul>
49   * <li>{@link #buffer(int)} allocates a new fixed-capacity heap buffer.</li>
50   * <li>{@link #directBuffer(int)} allocates a new fixed-capacity direct buffer.</li>
51   * </ul>
52   *
53   * <h3>Creating a wrapped buffer</h3>
54   *
55   * Wrapped buffer is a buffer which is a view of one or more existing
56   * byte arrays and byte buffers.  Any changes in the content of the original
57   * array or buffer will be visible in the wrapped buffer.  Various wrapper
58   * methods are provided and their name is all {@code wrappedBuffer()}.
59   * You might want to take a look at the methods that accept varargs closely if
60   * you want to create a buffer which is composed of more than one array to
61   * reduce the number of memory copy.
62   *
63   * <h3>Creating a copied buffer</h3>
64   *
65   * Copied buffer is a deep copy of one or more existing byte arrays, byte
66   * buffers or a string.  Unlike a wrapped buffer, there's no shared data
67   * between the original data and the copied buffer.  Various copy methods are
68   * provided and their name is all {@code copiedBuffer()}.  It is also convenient
69   * to use this operation to merge multiple buffers into one buffer.
70   */
71  public final class Unpooled {

177     /**
178      * Creates a new buffer which wraps the specified NIO buffer's current
179      * slice.  A modification on the specified buffer's content will be
180      * visible to the returned buffer.
181      */
182     public static ByteBuf wrappedBuffer(ByteBuffer buffer) {
183         if (!buffer.hasRemaining()) {
184             return EMPTY_BUFFER;
185         }
186         if (buffer.hasArray()) {
187             return wrappedBuffer(
188                     buffer.array(),
189                     buffer.arrayOffset() + buffer.position(),
190                     buffer.remaining()).order(buffer.order());
191         } else if (PlatformDependent.hasUnsafe()) {
192             if (buffer.isReadOnly()) {
193                 if (buffer.isDirect()) {
194                     return new ReadOnlyUnsafeDirectByteBuf(ALLOC, buffer);
195                 } else {
196                     return new ReadOnlyByteBufferBuf(ALLOC, buffer);
197                 }
198             } else {
199                 return new UnpooledUnsafeDirectByteBuf(ALLOC, buffer, buffer.remaining());
200             }
201         } else {
202             if (buffer.isReadOnly()) {
203                 return new ReadOnlyByteBufferBuf(ALLOC, buffer);
204             }  else {
205                 return new UnpooledDirectByteBuf(ALLOC, buffer, buffer.remaining());
206             }
207         }
208     }

256     /**
257      * Creates a new big-endian composite buffer which wraps the specified
258      * arrays without copying them.  A modification on the specified arrays'
259      * content will be visible to the returned buffer.
260      */
261     public static ByteBuf wrappedBuffer(int maxNumComponents, byte[]... arrays) {
262         switch (arrays.length) {
263         case 0:
264             break;
265         case 1:
266             if (arrays[0].length != 0) {
267                 return wrappedBuffer(arrays[0]);
268             }
269             break;
270         default:
271             // Get the list of the component, while guessing the byte order.
272             final List<ByteBuf> components = new ArrayList<ByteBuf>(arrays.length);
273             for (byte[] a: arrays) {
274                 if (a == null) {
275                     break;
276                 }
277                 if (a.length > 0) {
278                     components.add(wrappedBuffer(a));
279                 }
280             }
281 
282             if (!components.isEmpty()) {
283                 return new CompositeByteBuf(ALLOC, false, maxNumComponents, components);
284             }
285         }
286 
287         return EMPTY_BUFFER;
288     }

290     /**
291      * Creates a new big-endian composite buffer which wraps the readable bytes of the
292      * specified buffers without copying them.  A modification on the content
293      * of the specified buffers will be visible to the returned buffer.
294      * @param maxNumComponents Advisement as to how many independent buffers are allowed to exist before
295      * consolidation occurs.
296      * @param buffers The buffers to wrap. Reference count ownership of all variables is transfered to this method.
297      * @return The readable portion of the {@code buffers}. The caller is responsible for releasing this buffer.
298      */
299     public static ByteBuf wrappedBuffer(int maxNumComponents, ByteBuf... buffers) {
300         switch (buffers.length) {
301         case 0:
302             break;
303         case 1:
304             ByteBuf buffer = buffers[0];
305             if (buffer.isReadable()) {
306                 return wrappedBuffer(buffer.order(BIG_ENDIAN));
307             } else {
308                 buffer.release();
309             }
310             break;
311         default:
312             for (ByteBuf b: buffers) {
313                 if (b.isReadable()) {
314                     return new CompositeByteBuf(ALLOC, false, maxNumComponents, buffers);
315                 } else {
316                     b.release();
317                 }
318             }
319         }
320         return EMPTY_BUFFER;
321     }

323     /**
324      * Creates a new big-endian composite buffer which wraps the slices of the specified
325      * NIO buffers without copying them.  A modification on the content of the
326      * specified buffers will be visible to the returned buffer.
327      */
328     public static ByteBuf wrappedBuffer(int maxNumComponents, ByteBuffer... buffers) {
329         switch (buffers.length) {
330         case 0:
331             break;
332         case 1:
333             if (buffers[0].hasRemaining()) {
334                 return wrappedBuffer(buffers[0].order(BIG_ENDIAN));
335             }
336             break;
337         default:
338             // Get the list of the component, while guessing the byte order.
339             final List<ByteBuf> components = new ArrayList<ByteBuf>(buffers.length);
340             for (ByteBuffer b: buffers) {
341                 if (b == null) {
342                     break;
343                 }
344                 if (b.remaining() > 0) {
345                     components.add(wrappedBuffer(b.order(BIG_ENDIAN)));
346                 }
347             }
348 
349             if (!components.isEmpty()) {
350                 return new CompositeByteBuf(ALLOC, false, maxNumComponents, components);
351             }
352         }
353 
354         return EMPTY_BUFFER;
355     }

398     /**
399      * Creates a new buffer whose content is a copy of the specified
400      * {@code buffer}'s current slice.  The new buffer's {@code readerIndex}
401      * and {@code writerIndex} are {@code 0} and {@code buffer.remaining}
402      * respectively.
403      */
404     public static ByteBuf copiedBuffer(ByteBuffer buffer) {
405         int length = buffer.remaining();
406         if (length == 0) {
407             return EMPTY_BUFFER;
408         }
409         byte[] copy = new byte[length];
410         // Duplicate the buffer so we not adjust the position during our get operation.
411         // See https://github.com/netty/netty/issues/3896
412         ByteBuffer duplicate = buffer.duplicate();
413         duplicate.get(copy);
414         return wrappedBuffer(copy).order(duplicate.order());
415     }

417     /**
418      * Creates a new buffer whose content is a copy of the specified
419      * {@code buffer}'s readable bytes.  The new buffer's {@code readerIndex}
420      * and {@code writerIndex} are {@code 0} and {@code buffer.readableBytes}
421      * respectively.
422      */
423     public static ByteBuf copiedBuffer(ByteBuf buffer) {
424         int readable = buffer.readableBytes();
425         if (readable > 0) {
426             ByteBuf copy = buffer(readable);
427             copy.writeBytes(buffer, buffer.readerIndex(), readable);
428             return copy;
429         } else {
430             return EMPTY_BUFFER;
431         }
432     }

434     /**
435      * Creates a new big-endian buffer whose content is a merged copy of
436      * the specified {@code arrays}.  The new buffer's {@code readerIndex}
437      * and {@code writerIndex} are {@code 0} and the sum of all arrays'
438      * {@code length} respectively.
439      */
440     public static ByteBuf copiedBuffer(byte[]... arrays) {
441         switch (arrays.length) {
442         case 0:
443             return EMPTY_BUFFER;
444         case 1:
445             if (arrays[0].length == 0) {
446                 return EMPTY_BUFFER;
447             } else {
448                 return copiedBuffer(arrays[0]);
449             }
450         }
451 
452         // Merge the specified arrays into one array.
453         int length = 0;
454         for (byte[] a: arrays) {
455             if (Integer.MAX_VALUE - length < a.length) {
456                 throw new IllegalArgumentException(
457                         "The total length of the specified arrays is too big.");
458             }
459             length += a.length;
460         }
461 
462         if (length == 0) {
463             return EMPTY_BUFFER;
464         }
465 
466         byte[] mergedArray = new byte[length];
467         for (int i = 0, j = 0; i < arrays.length; i ++) {
468             byte[] a = arrays[i];
469             System.arraycopy(a, 0, mergedArray, j, a.length);
470             j += a.length;
471         }
472 
473         return wrappedBuffer(mergedArray);
474     }

476     /**
477      * Creates a new buffer whose content is a merged copy of the specified
478      * {@code buffers}' readable bytes.  The new buffer's {@code readerIndex}
479      * and {@code writerIndex} are {@code 0} and the sum of all buffers'
480      * {@code readableBytes} respectively.
481      *
482      * @throws IllegalArgumentException
483      *         if the specified buffers' endianness are different from each
484      *         other
485      */
486     public static ByteBuf copiedBuffer(ByteBuf... buffers) {
487         switch (buffers.length) {
488         case 0:
489             return EMPTY_BUFFER;
490         case 1:
491             return copiedBuffer(buffers[0]);
492         }
493 
494         // Merge the specified buffers into one buffer.
495         ByteOrder order = null;
496         int length = 0;
497         for (ByteBuf b: buffers) {
498             int bLen = b.readableBytes();
499             if (bLen <= 0) {
500                 continue;
501             }
502             if (Integer.MAX_VALUE - length < bLen) {
503                 throw new IllegalArgumentException(
504                         "The total length of the specified buffers is too big.");
505             }
506             length += bLen;
507             if (order != null) {
508                 if (!order.equals(b.order())) {
509                     throw new IllegalArgumentException("inconsistent byte order");
510                 }
511             } else {
512                 order = b.order();
513             }
514         }
515 
516         if (length == 0) {
517             return EMPTY_BUFFER;
518         }
519 
520         byte[] mergedArray = new byte[length];
521         for (int i = 0, j = 0; i < buffers.length; i ++) {
522             ByteBuf b = buffers[i];
523             int bLen = b.readableBytes();
524             b.getBytes(b.readerIndex(), mergedArray, j, bLen);
525             j += bLen;
526         }
527 
528         return wrappedBuffer(mergedArray).order(order);
529     }

531     /**
532      * Creates a new buffer whose content is a merged copy of the specified
533      * {@code buffers}' slices.  The new buffer's {@code readerIndex} and
534      * {@code writerIndex} are {@code 0} and the sum of all buffers'
535      * {@code remaining} respectively.
536      *
537      * @throws IllegalArgumentException
538      *         if the specified buffers' endianness are different from each
539      *         other
540      */
541     public static ByteBuf copiedBuffer(ByteBuffer... buffers) {
542         switch (buffers.length) {
543         case 0:
544             return EMPTY_BUFFER;
545         case 1:
546             return copiedBuffer(buffers[0]);
547         }
548 
549         // Merge the specified buffers into one buffer.
550         ByteOrder order = null;
551         int length = 0;
552         for (ByteBuffer b: buffers) {
553             int bLen = b.remaining();
554             if (bLen <= 0) {
555                 continue;
556             }
557             if (Integer.MAX_VALUE - length < bLen) {
558                 throw new IllegalArgumentException(
559                         "The total length of the specified buffers is too big.");
560             }
561             length += bLen;
562             if (order != null) {
563                 if (!order.equals(b.order())) {
564                     throw new IllegalArgumentException("inconsistent byte order");
565                 }
566             } else {
567                 order = b.order();
568             }
569         }
570 
571         if (length == 0) {
572             return EMPTY_BUFFER;
573         }
574 
575         byte[] mergedArray = new byte[length];
576         for (int i = 0, j = 0; i < buffers.length; i ++) {
577             // Duplicate the buffer so we not adjust the position during our get operation.
578             // See https://github.com/netty/netty/issues/3896
579             ByteBuffer b = buffers[i].duplicate();
580             int bLen = b.remaining();
581             b.get(mergedArray, j, bLen);
582             j += bLen;
583         }
584 
585         return wrappedBuffer(mergedArray).order(order);
586     }

606     /**
607      * Creates a new big-endian buffer whose content is a subregion of
608      * the specified {@code string} encoded in the specified {@code charset}.
609      * The new buffer's {@code readerIndex} and {@code writerIndex} are
610      * {@code 0} and the length of the encoded string respectively.
611      */
612     public static ByteBuf copiedBuffer(
613             CharSequence string, int offset, int length, Charset charset) {
614         if (string == null) {
615             throw new NullPointerException("string");
616         }
617         if (length == 0) {
618             return EMPTY_BUFFER;
619         }
620 
621         if (string instanceof CharBuffer) {
622             CharBuffer buf = (CharBuffer) string;
623             if (buf.hasArray()) {
624                 return copiedBuffer(
625                         buf.array(),
626                         buf.arrayOffset() + buf.position() + offset,
627                         length, charset);
628             }
629 
630             buf = buf.slice();
631             buf.limit(length);
632             buf.position(offset);
633             return copiedBuffer(buf, charset);
634         }
635 
636         return copiedBuffer(CharBuffer.wrap(string, offset, offset + length), charset);
637     }
