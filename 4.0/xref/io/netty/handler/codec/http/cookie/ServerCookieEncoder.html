35  /**
36   * A <a href="http://tools.ietf.org/html/rfc6265">RFC6265</a> compliant cookie encoder to be used server side,
37   * so some fields are sent (Version is typically ignored).
38   *
39   * As Netty's Cookie merges Expires and MaxAge into one single field, only Max-Age field is sent.
40   *
41   * Note that multiple cookies are supposed to be sent at once in a single "Set-Cookie" header.
42   *
43   * <pre>
44   * // Example
45   * {@link HttpRequest} req = ...;
46   * res.setHeader("Cookie", {@link ServerCookieEncoder}.encode("JSESSIONID", "1234"));
47   * </pre>
48   *
49   * @see ServerCookieDecoder
50   */
51  public final class ServerCookieEncoder extends CookieEncoder {
52  
53      /**
54       * Strict encoder that validates that name and value chars are in the valid scope
55       * defined in RFC6265, and (for methods that accept multiple cookies) that only
56       * one cookie is encoded with any given name. (If multiple cookies have the same
57       * name, the last one is the one that is encoded.)
58       */
59      public static final ServerCookieEncoder STRICT = new ServerCookieEncoder(true);
60  
61      /**
62       * Lax instance that doesn't validate name and value, and that allows multiple
63       * cookies with the same name.
64       */
65      public static final ServerCookieEncoder LAX = new ServerCookieEncoder(false);
66  
67      private ServerCookieEncoder(boolean strict) {
68          super(strict);
69      }
70  
71      /**
72       * Encodes the specified cookie name-value pair into a Set-Cookie header value.
73       *
74       * @param name the cookie name
75       * @param value the cookie value
76       * @return a single Set-Cookie header value
77       */
78      public String encode(String name, String value) {
79          return encode(new DefaultCookie(name, value));
80      }

82      /**
83       * Encodes the specified cookie into a Set-Cookie header value.
84       *
85       * @param cookie the cookie
86       * @return a single Set-Cookie header value
87       */
88      public String encode(Cookie cookie) {
89          final String name = checkNotNull(cookie, "cookie").name();
90          final String value = cookie.value() != null ? cookie.value() : "";
91  
92          validateCookie(name, value);
93  
94          StringBuilder buf = stringBuilder();
95  
96          if (cookie.wrap()) {
97              addQuoted(buf, name, value);
98          } else {
99              add(buf, name, value);
100         }
101 
102         if (cookie.maxAge() != Long.MIN_VALUE) {
103             add(buf, CookieHeaderNames.MAX_AGE, cookie.maxAge());
104             Date expires = new Date(cookie.maxAge() * 1000 + System.currentTimeMillis());
105             add(buf, CookieHeaderNames.EXPIRES, HttpHeaderDateFormat.get().format(expires));
106         }
107 
108         if (cookie.path() != null) {
109             add(buf, CookieHeaderNames.PATH, cookie.path());
110         }
111 
112         if (cookie.domain() != null) {
113             add(buf, CookieHeaderNames.DOMAIN, cookie.domain());
114         }
115         if (cookie.isSecure()) {
116             add(buf, CookieHeaderNames.SECURE);
117         }
118         if (cookie.isHttpOnly()) {
119             add(buf, CookieHeaderNames.HTTPONLY);
120         }
121 
122         return stripTrailingSeparator(buf);
123     }

125     /** Deduplicate a list of encoded cookies by keeping only the last instance with a given name.
126      *
127      * @param encoded The list of encoded cookies.
128      * @param nameToLastIndex A map from cookie name to index of last cookie instance.
129      * @return The encoded list with all but the last instance of a named cookie.
130      */
131     private List<String> dedup(List<String> encoded, Map<String, Integer> nameToLastIndex) {
132         boolean[] isLastInstance = new boolean[encoded.size()];
133         for (int idx : nameToLastIndex.values()) {
134             isLastInstance[idx] = true;
135         }
136         List<String> dedupd = new ArrayList<String>(nameToLastIndex.size());
137         for (int i = 0, n = encoded.size(); i < n; i++) {
138             if (isLastInstance[i]) {
139                 dedupd.add(encoded.get(i));
140             }
141         }
142         return dedupd;
143     }

145     /**
146      * Batch encodes cookies into Set-Cookie header values.
147      *
148      * @param cookies a bunch of cookies
149      * @return the corresponding bunch of Set-Cookie headers
150      */
151     public List<String> encode(Cookie... cookies) {
152         if (checkNotNull(cookies, "cookies").length == 0) {
153             return Collections.emptyList();
154         }
155 
156         List<String> encoded = new ArrayList<String>(cookies.length);
157         Map<String, Integer> nameToIndex = strict && cookies.length > 1 ? new HashMap<String, Integer>() : null;
158         boolean hasDupdName = false;
159         for (int i = 0; i < cookies.length; i++) {
160             Cookie c = cookies[i];
161             encoded.add(encode(c));
162             if (nameToIndex != null) {
163                 hasDupdName |= nameToIndex.put(c.name(), i) != null;
164             }
165         }
166         return hasDupdName ? dedup(encoded, nameToIndex) : encoded;
167     }

169     /**
170      * Batch encodes cookies into Set-Cookie header values.
171      *
172      * @param cookies a bunch of cookies
173      * @return the corresponding bunch of Set-Cookie headers
174      */
175     public List<String> encode(Collection<? extends Cookie> cookies) {
176         if (checkNotNull(cookies, "cookies").isEmpty()) {
177             return Collections.emptyList();
178         }
179 
180         List<String> encoded = new ArrayList<String>(cookies.size());
181         Map<String, Integer> nameToIndex = strict && cookies.size() > 1 ? new HashMap<String, Integer>() : null;
182         int i = 0;
183         boolean hasDupdName = false;
184         for (Cookie c : cookies) {
185             encoded.add(encode(c));
186             if (nameToIndex != null) {
187                 hasDupdName |= nameToIndex.put(c.name(), i++) != null;
188             }
189         }
190         return hasDupdName ? dedup(encoded, nameToIndex) : encoded;
191     }

193     /**
194      * Batch encodes cookies into Set-Cookie header values.
195      *
196      * @param cookies a bunch of cookies
197      * @return the corresponding bunch of Set-Cookie headers
198      */
199     public List<String> encode(Iterable<? extends Cookie> cookies) {
200         Iterator<? extends Cookie> cookiesIt = checkNotNull(cookies, "cookies").iterator();
201         if (!cookiesIt.hasNext()) {
202             return Collections.emptyList();
203         }
204 
205         List<String> encoded = new ArrayList<String>();
206         Cookie firstCookie = cookiesIt.next();
207         Map<String, Integer> nameToIndex = strict && cookiesIt.hasNext() ? new HashMap<String, Integer>() : null;
208         int i = 0;
209         encoded.add(encode(firstCookie));
210         boolean hasDupdName = nameToIndex != null ? nameToIndex.put(firstCookie.name(), i++) != null : false;
211         while (cookiesIt.hasNext()) {
212             Cookie c = cookiesIt.next();
213             encoded.add(encode(c));
214             if (nameToIndex != null) {
215                 hasDupdName |= nameToIndex.put(c.name(), i++) != null;
216             }
217         }
218         return hasDupdName ? dedup(encoded, nameToIndex) : encoded;
219     }
