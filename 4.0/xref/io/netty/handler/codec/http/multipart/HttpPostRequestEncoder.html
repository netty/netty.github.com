57           * Legacy mode which should work for most. It is known to not work with OAUTH. For OAUTH use

160     /**
161      *
162      * @param factory
163      *            the factory used to create InterfaceHttpData
164      * @param request
165      *            the request to encode
166      * @param multipart
167      *            True if the FORM is a ENCTYPE="multipart/form-data"
168      * @param charset
169      *            the charset to use as default
170      * @param encoderMode
171      *            the mode for the encoder to use. See {@link EncoderMode} for the details.
172      * @throws NullPointerException
173      *             for request or charset or factory
174      * @throws ErrorDataEncoderException
175      *             if the request is not a POST
176      */
177     public HttpPostRequestEncoder(
178             HttpDataFactory factory, HttpRequest request, boolean multipart, Charset charset,
179             EncoderMode encoderMode)
180             throws ErrorDataEncoderException {
181         if (factory == null) {
182             throw new NullPointerException("factory");
183         }
184         if (request == null) {
185             throw new NullPointerException("request");
186         }
187         if (charset == null) {
188             throw new NullPointerException("charset");
189         }
190         if (request.getMethod() != HttpMethod.POST) {
191             throw new ErrorDataEncoderException("Cannot create a Encoder if not a POST");
192         }
193         this.request = request;
194         this.charset = charset;
195         this.factory = factory;
196         // Fill default values
197         bodyListDatas = new ArrayList<InterfaceHttpData>();
198         // default mode
199         isLastChunk = false;
200         isLastChunkSent = false;
201         isMultipart = multipart;
202         multipartHttpDatas = new ArrayList<InterfaceHttpData>();
203         this.encoderMode = encoderMode;
204         if (isMultipart) {
205             initDataMultipart();
206         }
207     }

291         globalBodySize = 0;
292         bodyListDatas.clear();
293         currentFileUpload = null;
294         duringMixedMode = false;
295         multipartHttpDatas.clear();
296         for (InterfaceHttpData data : datas) {

325     /**
326      * Add a file as a FileUpload
327      *
328      * @param name
329      *            the name of the parameter
330      * @param file
331      *            the file to be uploaded (if not Multipart mode, only the filename will be included)
332      * @param contentType
333      *            the associated contentType for the File
334      * @param isText
335      *            True if this file should be transmitted in Text format (else binary)
336      * @throws NullPointerException
337      *             for name and file
338      * @throws ErrorDataEncoderException
339      *             if the encoding is in error or if the finalize were already done
340      */
341     public void addBodyFileUpload(String name, File file, String contentType, boolean isText)
342             throws ErrorDataEncoderException {
343         if (name == null) {
344             throw new NullPointerException("name");
345         }
346         if (file == null) {
347             throw new NullPointerException("file");
348         }
349         String scontentType = contentType;
350         String contentTransferEncoding = null;
351         if (contentType == null) {
352             if (isText) {
353                 scontentType = HttpPostBodyUtil.DEFAULT_TEXT_CONTENT_TYPE;
354             } else {
355                 scontentType = HttpPostBodyUtil.DEFAULT_BINARY_CONTENT_TYPE;
356             }
357         }
358         if (!isText) {
359             contentTransferEncoding = HttpPostBodyUtil.TransferEncodingMechanism.BINARY.value();
360         }
361         FileUpload fileUpload = factory.createFileUpload(request, name, file.getName(), scontentType,
362                 contentTransferEncoding, null, file.length());
363         try {
364             fileUpload.setContent(file);
365         } catch (IOException e) {
366             throw new ErrorDataEncoderException(e);
367         }
368         addBodyHttpData(fileUpload);
369     }

371     /**
372      * Add a series of Files associated with one File parameter (implied Mixed mode in Multipart)
373      *
374      * @param name
375      *            the name of the parameter
376      * @param file
377      *            the array of files
378      * @param contentType
379      *            the array of content Types associated with each file
380      * @param isText
381      *            the array of isText attribute (False meaning binary mode) for each file
382      * @throws NullPointerException
383      *             also throws if array have different sizes
384      * @throws ErrorDataEncoderException
385      *             if the encoding is in error or if the finalize were already done
386      */
387     public void addBodyFileUploads(String name, File[] file, String[] contentType, boolean[] isText)
388             throws ErrorDataEncoderException {
389         if (file.length != contentType.length && file.length != isText.length) {
390             throw new NullPointerException("Different array length");
391         }
392         for (int i = 0; i < file.length; i++) {
393             addBodyFileUpload(name, file[i], contentType[i], isText[i]);
394         }
395     }

397     /**
398      * Add the InterfaceHttpData to the Body list
399      *
400      * @throws NullPointerException
401      *             for data
402      * @throws ErrorDataEncoderException
403      *             if the encoding is in error or if the finalize were already done
404      */
405     public void addBodyHttpData(InterfaceHttpData data) throws ErrorDataEncoderException {
406         if (headerFinalized) {
407             throw new ErrorDataEncoderException("Cannot add value once finalized");
408         }
409         if (data == null) {
410             throw new NullPointerException("data");
411         }
412         bodyListDatas.add(data);
413         if (!isMultipart) {
414             if (data instanceof Attribute) {
415                 Attribute attribute = (Attribute) data;
416                 try {
417                     // name=value& with encoded name and attribute
418                     String key = encodeAttribute(attribute.getName(), charset);
419                     String value = encodeAttribute(attribute.getValue(), charset);
420                     Attribute newattribute = factory.createAttribute(request, key, value);
421                     multipartHttpDatas.add(newattribute);
422                     globalBodySize += newattribute.getName().length() + 1 + newattribute.length() + 1;
423                 } catch (IOException e) {
424                     throw new ErrorDataEncoderException(e);
425                 }
426             } else if (data instanceof FileUpload) {
427                 // since not Multipart, only name=filename => Attribute
428                 FileUpload fileUpload = (FileUpload) data;
429                 // name=filename& with encoded name and filename
430                 String key = encodeAttribute(fileUpload.getName(), charset);
431                 String value = encodeAttribute(fileUpload.getFilename(), charset);
432                 Attribute newattribute = factory.createAttribute(request, key, value);
433                 multipartHttpDatas.add(newattribute);
434                 globalBodySize += newattribute.getName().length() + 1 + newattribute.length() + 1;
435             }
436             return;
437         }
438         /*
439          * Logic:
440          * if not Attribute:
441          *      add Data to body list
442          *      if (duringMixedMode)
443          *          add endmixedmultipart delimiter
444          *          currentFileUpload = null
445          *          duringMixedMode = false;
446          *      add multipart delimiter, multipart body header and Data to multipart list
447          *      reset currentFileUpload, duringMixedMode
448          * if FileUpload: take care of multiple file for one field => mixed mode
449          *      if (duringMixeMode)
450          *          if (currentFileUpload.name == data.name)
451          *              add mixedmultipart delimiter, mixedmultipart body header and Data to multipart list
452          *          else
453          *              add endmixedmultipart delimiter, multipart body header and Data to multipart list
454          *              currentFileUpload = data
455          *              duringMixedMode = false;
456          *      else
457          *          if (currentFileUpload.name == data.name)
458          *              change multipart body header of previous file into multipart list to
459          *                      mixedmultipart start, mixedmultipart body header
460          *              add mixedmultipart delimiter, mixedmultipart body header and Data to multipart list
461          *              duringMixedMode = true
462          *          else
463          *              add multipart delimiter, multipart body header and Data to multipart list
464          *              currentFileUpload = data
465          *              duringMixedMode = false;
466          * Do not add last delimiter! Could be:
467          * if duringmixedmode: endmixedmultipart + endmultipart
468          * else only endmultipart
469          */
470         if (data instanceof Attribute) {
471             if (duringMixedMode) {
472                 InternalAttribute internal = new InternalAttribute(charset);
473                 internal.addValue("\r\n--" + multipartMixedBoundary + "--");
474                 multipartHttpDatas.add(internal);
475                 multipartMixedBoundary = null;
476                 currentFileUpload = null;
477                 duringMixedMode = false;
478             }
479             InternalAttribute internal = new InternalAttribute(charset);
480             if (!multipartHttpDatas.isEmpty()) {
481                 // previously a data field so CRLF
482                 internal.addValue("\r\n");
483             }
484             internal.addValue("--" + multipartDataBoundary + "\r\n");
485             // content-disposition: form-data; name="field1"
486             Attribute attribute = (Attribute) data;
487             internal.addValue(HttpPostBodyUtil.CONTENT_DISPOSITION + ": " + HttpPostBodyUtil.FORM_DATA + "; "
488                     + HttpPostBodyUtil.NAME + "=\"" + attribute.getName() + "\"\r\n");
489             Charset localcharset = attribute.getCharset();
490             if (localcharset != null) {
491                 // Content-Type: text/plain; charset=charset
492                 internal.addValue(HttpHeaders.Names.CONTENT_TYPE + ": " +
493                         HttpPostBodyUtil.DEFAULT_TEXT_CONTENT_TYPE + "; " +
494                         HttpHeaders.Values.CHARSET + '='
495                         + localcharset.name() + "\r\n");
496             }
497             // CRLF between body header and data
498             internal.addValue("\r\n");
499             multipartHttpDatas.add(internal);
500             multipartHttpDatas.add(data);
501             globalBodySize += attribute.length() + internal.size();
502         } else if (data instanceof FileUpload) {
503             FileUpload fileUpload = (FileUpload) data;
504             InternalAttribute internal = new InternalAttribute(charset);
505             if (!multipartHttpDatas.isEmpty()) {
506                 // previously a data field so CRLF
507                 internal.addValue("\r\n");
508             }
509             boolean localMixed;
510             if (duringMixedMode) {
511                 if (currentFileUpload != null && currentFileUpload.getName().equals(fileUpload.getName())) {
512                     // continue a mixed mode
513 
514                     localMixed = true;
515                 } else {
516                     // end a mixed mode
517 
518                     // add endmixedmultipart delimiter, multipart body header
519                     // and
520                     // Data to multipart list
521                     internal.addValue("--" + multipartMixedBoundary + "--");
522                     multipartHttpDatas.add(internal);
523                     multipartMixedBoundary = null;
524                     // start a new one (could be replaced if mixed start again
525                     // from here
526                     internal = new InternalAttribute(charset);
527                     internal.addValue("\r\n");
528                     localMixed = false;
529                     // new currentFileUpload and no more in Mixed mode
530                     currentFileUpload = fileUpload;
531                     duringMixedMode = false;
532                 }
533             } else {
534                 if (currentFileUpload != null && currentFileUpload.getName().equals(fileUpload.getName())) {
535                     // create a new mixed mode (from previous file)
536 
537                     // change multipart body header of previous file into
538                     // multipart list to
539                     // mixedmultipart start, mixedmultipart body header
540 
541                     // change Internal (size()-2 position in multipartHttpDatas)
542                     // from (line starting with *)
543                     // --AaB03x
544                     // * Content-Disposition: form-data; name="files";
545                     // filename="file1.txt"
546                     // Content-Type: text/plain
547                     // to (lines starting with *)
548                     // --AaB03x
549                     // * Content-Disposition: form-data; name="files"
550                     // * Content-Type: multipart/mixed; boundary=BbC04y
551                     // *
552                     // * --BbC04y
553                     // * Content-Disposition: file; filename="file1.txt"
554                     // Content-Type: text/plain
555                     initMixedMultipart();
556                     InternalAttribute pastAttribute = (InternalAttribute) multipartHttpDatas.get(multipartHttpDatas
557                             .size() - 2);
558                     // remove past size
559                     globalBodySize -= pastAttribute.size();
560                     StringBuilder replacement = new StringBuilder(
561                             139 + multipartDataBoundary.length() + multipartMixedBoundary.length() * 2 +
562 
563                                     fileUpload.getFilename().length() + fileUpload.getName().length());
564 
565                     replacement.append("--")
566                     .append(multipartDataBoundary)
567                     .append("\r\n")
568 
569                     .append(HttpPostBodyUtil.CONTENT_DISPOSITION)
570                     .append(": ")
571                     .append(HttpPostBodyUtil.FORM_DATA)
572                     .append("; ")
573                     .append(HttpPostBodyUtil.NAME)
574                     .append("=\"")
575                     .append(fileUpload.getName())
576                     .append("\"\r\n")
577 
578                     .append(HttpHeaders.Names.CONTENT_TYPE)
579                     .append(": ")
580                     .append(HttpPostBodyUtil.MULTIPART_MIXED)
581                     .append("; ")
582                     .append(HttpHeaders.Values.BOUNDARY)
583                     .append('=')
584                     .append(multipartMixedBoundary)
585                     .append("\r\n\r\n")
586 
587                     .append("--")
588                     .append(multipartMixedBoundary)
589                     .append("\r\n")
590 
591                     .append(HttpPostBodyUtil.CONTENT_DISPOSITION)
592                     .append(": ")
593                     .append(HttpPostBodyUtil.ATTACHMENT)
594                     .append("; ")
595                     .append(HttpPostBodyUtil.FILENAME)
596                     .append("=\"")
597                     .append(fileUpload.getFilename())
598                     .append("\"\r\n");
599 
600                     pastAttribute.setValue(replacement.toString(), 1);
601                     pastAttribute.setValue("", 2);
602 
603                     // update past size
604                     globalBodySize += pastAttribute.size();
605 
606                     // now continue
607                     // add mixedmultipart delimiter, mixedmultipart body header
608                     // and
609                     // Data to multipart list
610                     localMixed = true;
611                     duringMixedMode = true;
612                 } else {
613                     // a simple new multipart
614                     // add multipart delimiter, multipart body header and Data
615                     // to multipart list
616                     localMixed = false;
617                     currentFileUpload = fileUpload;
618                     duringMixedMode = false;
619                 }
620             }
621 
622             if (localMixed) {
623                 // add mixedmultipart delimiter, mixedmultipart body header and
624                 // Data to multipart list
625                 internal.addValue("--" + multipartMixedBoundary + "\r\n");
626                 // Content-Disposition: file; filename="file1.txt"
627                 internal.addValue(HttpPostBodyUtil.CONTENT_DISPOSITION + ": " + HttpPostBodyUtil.ATTACHMENT + "; "
628                         + HttpPostBodyUtil.FILENAME + "=\"" + fileUpload.getFilename() + "\"\r\n");
629             } else {
630                 internal.addValue("--" + multipartDataBoundary + "\r\n");
631                 // Content-Disposition: form-data; name="files";
632                 // filename="file1.txt"
633                 internal.addValue(HttpPostBodyUtil.CONTENT_DISPOSITION + ": " + HttpPostBodyUtil.FORM_DATA + "; "
634                         + HttpPostBodyUtil.NAME + "=\"" + fileUpload.getName() + "\"; "
635                         + HttpPostBodyUtil.FILENAME + "=\"" + fileUpload.getFilename() + "\"\r\n");
636             }
637             // Content-Type: image/gif
638             // Content-Type: text/plain; charset=ISO-8859-1
639             // Content-Transfer-Encoding: binary
640             internal.addValue(HttpHeaders.Names.CONTENT_TYPE + ": " + fileUpload.getContentType());
641             String contentTransferEncoding = fileUpload.getContentTransferEncoding();
642             if (contentTransferEncoding != null
643                     && contentTransferEncoding.equals(HttpPostBodyUtil.TransferEncodingMechanism.BINARY.value())) {
644                 internal.addValue("\r\n" + HttpHeaders.Names.CONTENT_TRANSFER_ENCODING + ": "
645                         + HttpPostBodyUtil.TransferEncodingMechanism.BINARY.value() + "\r\n\r\n");
646             } else if (fileUpload.getCharset() != null) {
647                 internal.addValue("; " + HttpHeaders.Values.CHARSET + '='
648                                   + fileUpload.getCharset().name() + "\r\n\r\n");
649             } else {
650                 internal.addValue("\r\n\r\n");
651             }
652             multipartHttpDatas.add(internal);
653             multipartHttpDatas.add(data);
654             globalBodySize += fileUpload.length() + internal.size();
655         }
656     }

663     /**
664      * Finalize the request by preparing the Header in the request and returns the request ready to be sent.<br>
665      * Once finalized, no data must be added.<br>
666      * If the request does not need chunk (isChunked() == false), this request is the only object to send to the remote
667      * server.
668      *
669      * @return the request object (chunked or not according to size of body)
670      * @throws ErrorDataEncoderException
671      *             if the encoding is in error or if the finalize were already done
672      */
673     public HttpRequest finalizeRequest() throws ErrorDataEncoderException {
674         // Finalize the multipartHttpDatas
675         if (!headerFinalized) {
676             if (isMultipart) {
677                 InternalAttribute internal = new InternalAttribute(charset);
678                 if (duringMixedMode) {
679                     internal.addValue("\r\n--" + multipartMixedBoundary + "--");
680                 }
681                 internal.addValue("\r\n--" + multipartDataBoundary + "--\r\n");
682                 multipartHttpDatas.add(internal);
683                 multipartMixedBoundary = null;
684                 currentFileUpload = null;
685                 duringMixedMode = false;
686                 globalBodySize += internal.size();
687             }
688             headerFinalized = true;
689         } else {
690             throw new ErrorDataEncoderException("Header already encoded");
691         }
692 
693         HttpHeaders headers = request.headers();
694         List<String> contentTypes = headers.getAll(HttpHeaders.Names.CONTENT_TYPE);
695         List<String> transferEncoding = headers.getAll(HttpHeaders.Names.TRANSFER_ENCODING);
696         if (contentTypes != null) {
697             headers.remove(HttpHeaders.Names.CONTENT_TYPE);
698             for (String contentType : contentTypes) {
699                 // "multipart/form-data; boundary=--89421926422648"
700                 String lowercased = contentType.toLowerCase();
701                 if (lowercased.startsWith(HttpHeaders.Values.MULTIPART_FORM_DATA) ||
702                     lowercased.startsWith(HttpHeaders.Values.APPLICATION_X_WWW_FORM_URLENCODED)) {
703                     // ignore
704                 } else {
705                     headers.add(HttpHeaders.Names.CONTENT_TYPE, contentType);
706                 }
707             }
708         }
709         if (isMultipart) {
710             String value = HttpHeaders.Values.MULTIPART_FORM_DATA + "; " + HttpHeaders.Values.BOUNDARY + '='
711                     + multipartDataBoundary;
712             headers.add(HttpHeaders.Names.CONTENT_TYPE, value);
713         } else {
714             // Not multipart
715             headers.add(HttpHeaders.Names.CONTENT_TYPE, HttpHeaders.Values.APPLICATION_X_WWW_FORM_URLENCODED);
716         }
717         // Now consider size for chunk or not
718         long realSize = globalBodySize;
719         if (isMultipart) {
720             iterator = multipartHttpDatas.listIterator();
721         } else {
722             realSize -= 1; // last '&' removed
723             iterator = multipartHttpDatas.listIterator();
724         }
725         headers.set(HttpHeaders.Names.CONTENT_LENGTH, String.valueOf(realSize));
726         if (realSize > HttpPostBodyUtil.chunkSize || isMultipart) {
727             isChunked = true;
728             if (transferEncoding != null) {
729                 headers.remove(HttpHeaders.Names.TRANSFER_ENCODING);
730                 for (String v : transferEncoding) {
731                     if (v.equalsIgnoreCase(HttpHeaders.Values.CHUNKED)) {
732                         // ignore
733                     } else {
734                         headers.add(HttpHeaders.Names.TRANSFER_ENCODING, v);
735                     }
736                 }
737             }
738             HttpHeaders.setTransferEncodingChunked(request);
739 
740             // wrap to hide the possible content
741             return new WrappedHttpRequest(request);
742         } else {
743             // get the only one body and set it to the request
744             HttpContent chunk = nextChunk();
745             if (request instanceof FullHttpRequest) {
746                 FullHttpRequest fullRequest = (FullHttpRequest) request;
747                 ByteBuf chunkContent = chunk.content();
748                 if (fullRequest.content() != chunkContent) {
749                     fullRequest.content().clear().writeBytes(chunkContent);
750                     chunkContent.release();
751                 }
752                 return fullRequest;
753             } else {
754                 return new WrappedFullHttpRequest(request, chunk);
755             }
756         }
757     }

805      *
806      * @return the next ByteBuf to send as a HttpChunk and modifying currentBuffer accordingly
807      */
808     private ByteBuf fillByteBuf() {
809         int length = currentBuffer.readableBytes();
810         if (length > HttpPostBodyUtil.chunkSize) {
811             ByteBuf slice = currentBuffer.slice(currentBuffer.readerIndex(), HttpPostBodyUtil.chunkSize);
812             currentBuffer.retain();
813             currentBuffer.skipBytes(HttpPostBodyUtil.chunkSize);
814             return slice;
815         } else {
816             // to continue
817             ByteBuf slice = currentBuffer;
818             currentBuffer = null;
819             return slice;

823     /**
824      * From the current context (currentBuffer and currentData), returns the next HttpChunk (if possible) trying to get
825      * sizeleft bytes more into the currentBuffer. This is the Multipart version.
826      *
827      * @param sizeleft
828      *            the number of bytes to try to get from currentData
829      * @return the next HttpChunk or null if not enough bytes were found
830      * @throws ErrorDataEncoderException
831      *             if the encoding is in error
832      */
833     private HttpContent encodeNextChunkMultipart(int sizeleft) throws ErrorDataEncoderException {
834         if (currentData == null) {
835             return null;
836         }
837         ByteBuf buffer;
838         if (currentData instanceof InternalAttribute) {
839             buffer = ((InternalAttribute) currentData).toByteBuf();
840             currentData = null;
841         } else {
842             if (currentData instanceof Attribute) {
843                 try {
844                     buffer = ((Attribute) currentData).getChunk(sizeleft);
845                 } catch (IOException e) {
846                     throw new ErrorDataEncoderException(e);
847                 }
848             } else {
849                 try {
850                     buffer = ((HttpData) currentData).getChunk(sizeleft);
851                 } catch (IOException e) {
852                     throw new ErrorDataEncoderException(e);
853                 }
854             }
855             if (buffer.capacity() == 0) {
856                 // end for current InterfaceHttpData, need more data
857                 currentData = null;
858                 return null;
859             }
860         }
861         if (currentBuffer == null) {
862             currentBuffer = buffer;
863         } else {
864             currentBuffer = wrappedBuffer(currentBuffer, buffer);
865         }
866         if (currentBuffer.readableBytes() < HttpPostBodyUtil.chunkSize) {
867             currentData = null;
868             return null;
869         }
870         buffer = fillByteBuf();
871         return new DefaultHttpContent(buffer);
872     }

874     /**
875      * From the current context (currentBuffer and currentData), returns the next HttpChunk (if possible) trying to get
876      * sizeleft bytes more into the currentBuffer. This is the UrlEncoded version.
877      *
878      * @param sizeleft
879      *            the number of bytes to try to get from currentData
880      * @return the next HttpChunk or null if not enough bytes were found
881      * @throws ErrorDataEncoderException
882      *             if the encoding is in error
883      */
884     private HttpContent encodeNextChunkUrlEncoded(int sizeleft) throws ErrorDataEncoderException {
885         if (currentData == null) {
886             return null;
887         }
888         int size = sizeleft;
889         ByteBuf buffer;
890 
891         // Set name=
892         if (isKey) {
893             String key = currentData.getName();
894             buffer = wrappedBuffer(key.getBytes());
895             isKey = false;
896             if (currentBuffer == null) {
897                 currentBuffer = wrappedBuffer(buffer, wrappedBuffer("=".getBytes()));
898                 // continue
899                 size -= buffer.readableBytes() + 1;
900             } else {
901                 currentBuffer = wrappedBuffer(currentBuffer, buffer, wrappedBuffer("=".getBytes()));
902                 // continue
903                 size -= buffer.readableBytes() + 1;
904             }
905             if (currentBuffer.readableBytes() >= HttpPostBodyUtil.chunkSize) {
906                 buffer = fillByteBuf();
907                 return new DefaultHttpContent(buffer);
908             }
909         }
910 
911         // Put value into buffer
912         try {
913             buffer = ((HttpData) currentData).getChunk(size);
914         } catch (IOException e) {
915             throw new ErrorDataEncoderException(e);
916         }
917 
918         // Figure out delimiter
919         ByteBuf delimiter = null;
920         if (buffer.readableBytes() < size) {
921             isKey = true;
922             delimiter = iterator.hasNext() ? wrappedBuffer("&".getBytes()) : null;
923         }
924 
925         // End for current InterfaceHttpData, need potentially more data
926         if (buffer.capacity() == 0) {
927             currentData = null;
928             if (currentBuffer == null) {
929                 currentBuffer = delimiter;
930             } else {
931                 if (delimiter != null) {
932                     currentBuffer = wrappedBuffer(currentBuffer, delimiter);
933                 }
934             }
935             if (currentBuffer.readableBytes() >= HttpPostBodyUtil.chunkSize) {
936                 buffer = fillByteBuf();
937                 return new DefaultHttpContent(buffer);
938             }
939             return null;
940         }
941 
942         // Put it all together: name=value&
943         if (currentBuffer == null) {
944             if (delimiter != null) {
945                 currentBuffer = wrappedBuffer(buffer, delimiter);
946             } else {
947                 currentBuffer = buffer;
948             }
949         } else {
950             if (delimiter != null) {
951                 currentBuffer = wrappedBuffer(currentBuffer, buffer, delimiter);
952             } else {
953                 currentBuffer = wrappedBuffer(currentBuffer, buffer);
954             }
955         }
956 
957         // end for current InterfaceHttpData, need more data
958         if (currentBuffer.readableBytes() < HttpPostBodyUtil.chunkSize) {
959             currentData = null;
960             isKey = true;
961             return null;
962         }
963 
964         buffer = fillByteBuf();
965         return new DefaultHttpContent(buffer);
966     }

991     /**
992      * Returns the next available HttpChunk. The caller is responsible to test if this chunk is the last one (isLast()),
993      * in order to stop calling this getMethod.
994      *
995      * @return the next available HttpChunk
996      * @throws ErrorDataEncoderException
997      *             if the encoding is in error
998      */
999     private HttpContent nextChunk() throws ErrorDataEncoderException {
1000         if (isLastChunk) {
1001             isLastChunkSent = true;
1002             return LastHttpContent.EMPTY_LAST_CONTENT;
1003         }
1004         ByteBuf buffer;
1005         int size = HttpPostBodyUtil.chunkSize;
1006         // first test if previous buffer is not empty
1007         if (currentBuffer != null) {
1008             size -= currentBuffer.readableBytes();
1009         }
1010         if (size <= 0) {
1011             // NextChunk from buffer
1012             buffer = fillByteBuf();
1013             return new DefaultHttpContent(buffer);
1014         }
1015         // size > 0
1016         if (currentData != null) {
1017             // continue to read data
1018             if (isMultipart) {
1019                 HttpContent chunk = encodeNextChunkMultipart(size);
1020                 if (chunk != null) {
1021                     return chunk;
1022                 }
1023             } else {
1024                 HttpContent chunk = encodeNextChunkUrlEncoded(size);
1025                 if (chunk != null) {
1026                     // NextChunk Url from currentData
1027                     return chunk;
1028                 }
1029             }
1030             size = HttpPostBodyUtil.chunkSize - currentBuffer.readableBytes();
1031         }
1032         if (!iterator.hasNext()) {
1033             isLastChunk = true;
1034             // NextChunk as last non empty from buffer
1035             buffer = currentBuffer;
1036             currentBuffer = null;
1037             return new DefaultHttpContent(buffer);
1038         }
1039         while (size > 0 && iterator.hasNext()) {
1040             currentData = iterator.next();
1041             HttpContent chunk;
1042             if (isMultipart) {
1043                 chunk = encodeNextChunkMultipart(size);
1044             } else {
1045                 chunk = encodeNextChunkUrlEncoded(size);
1046             }
1047             if (chunk == null) {
1048                 // not enough
1049                 size = HttpPostBodyUtil.chunkSize - currentBuffer.readableBytes();
1050                 continue;
1051             }
1052             // NextChunk from data
1053             return chunk;
1054         }
1055         // end since no more data
1056         isLastChunk = true;
1057         if (currentBuffer == null) {
1058             isLastChunkSent = true;
1059             // LastChunk with no more data
1060             return LastHttpContent.EMPTY_LAST_CONTENT;
1061         }
1062         // Previous LastChunk with no more data
1063         buffer = currentBuffer;
1064         currentBuffer = null;
1065         return new DefaultHttpContent(buffer);
1066     }
