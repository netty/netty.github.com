39  /**
40   * A {@link ChannelHandler} that adds support for writing a large data stream
41   * asynchronously neither spending a lot of memory nor getting
42   * {@link OutOfMemoryError}.  Large data streaming such as file
43   * transfer requires complicated state management in a {@link ChannelHandler}
44   * implementation.  {@link ChunkedWriteHandler} manages such complicated states
45   * so that you can send a large data stream without difficulties.
46   * <p>
47   * To use {@link ChunkedWriteHandler} in your application, you have to insert
48   * a new {@link ChunkedWriteHandler} instance:
49   * <pre>
50   * {@link ChannelPipeline} p = ...;
51   * p.addLast("streamer", <b>new {@link ChunkedWriteHandler}()</b>);
52   * p.addLast("handler", new MyHandler());
53   * </pre>
54   * Once inserted, you can write a {@link ChunkedInput} so that the
55   * {@link ChunkedWriteHandler} can pick it up and fetch the content of the
56   * stream chunk by chunk and write the fetched chunk downstream:
57   * <pre>
58   * {@link Channel} ch = ...;
59   * ch.write(new {@link ChunkedFile}(new File("video.mkv"));
60   * </pre>
61   *
62   * <h3>Sending a stream which generates a chunk intermittently</h3>
63   *
64   * Some {@link ChunkedInput} generates a chunk on a certain event or timing.
65   * Such {@link ChunkedInput} implementation often returns {@code null} on
66   * {@link ChunkedInput#readChunk(ChannelHandlerContext)}, resulting in the indefinitely suspended
67   * transfer.  To resume the transfer when a new chunk is available, you have to
68   * call {@link #resumeTransfer()}.
69   */
70  public class ChunkedWriteHandler
71          extends ChannelDuplexHandler {

99      /**
100      * Continues to fetch the chunks from the input.
101      */
102     public void resumeTransfer() {
103         final ChannelHandlerContext ctx = this.ctx;
104         if (ctx == null) {
105             return;
106         }
107         if (ctx.executor().inEventLoop()) {
108             try {
109                 doFlush(ctx);
110             } catch (Exception e) {
111                 if (logger.isWarnEnabled()) {
112                     logger.warn("Unexpected exception while sending chunks.", e);
113                 }
114             }
115         } else {
116             // let the transfer resume on the next event loop round
117             ctx.executor().execute(new OneTimeTask() {
118 
119                 @Override
120                 public void run() {
121                     try {
122                         doFlush(ctx);
123                     } catch (Exception e) {
124                         if (logger.isWarnEnabled()) {
125                             logger.warn("Unexpected exception while sending chunks.", e);
126                         }
127                     }
128                 }
129             });
130         }
131     }

133     @Override

135         queue.add(new PendingWrite(msg, promise));

138     @Override
139     public void flush(ChannelHandlerContext ctx) throws Exception {
140         if (!doFlush(ctx)) {
141             // Make sure to flush at least once.
142             ctx.flush();
143         }
144     }

146     @Override

152     @Override
153     public void channelWritabilityChanged(ChannelHandlerContext ctx) throws Exception {
154         if (ctx.channel().isWritable()) {
155             // channel is writable again try to continue flushing
156             doFlush(ctx);
157         }
158         ctx.fireChannelWritabilityChanged();

161     private void discard(Throwable cause) {
162         for (;;) {
163             PendingWrite currentWrite = this.currentWrite;
164 
165             if (this.currentWrite == null) {
166                 currentWrite = queue.poll();
167             } else {
168                 this.currentWrite = null;
169             }
170 
171             if (currentWrite == null) {
172                 break;
173             }
174             Object message = currentWrite.msg;
175             if (message instanceof ChunkedInput) {
176                 ChunkedInput<?> in = (ChunkedInput<?>) message;
177                 try {
178                     if (!in.isEndOfInput()) {
179                         if (cause == null) {
180                             cause = new ClosedChannelException();
181                         }
182                         currentWrite.fail(cause);
183                     } else {
184                         currentWrite.success();
185                     }
186                     closeInput(in);
187                 } catch (Exception e) {
188                     currentWrite.fail(e);
189                     logger.warn(ChunkedInput.class.getSimpleName() + ".isEndOfInput() failed", e);
190                     closeInput(in);
191                 }
192             } else {
193                 if (cause == null) {
194                     cause = new ClosedChannelException();
195                 }
196                 currentWrite.fail(cause);
197             }
198         }
199     }

201     private boolean doFlush(final ChannelHandlerContext ctx) throws Exception {
202         final Channel channel = ctx.channel();
203         if (!channel.isActive()) {
204             discard(null);
205             return false;
206         }
207 
208         boolean flushed = false;
209         while (channel.isWritable()) {
210             if (currentWrite == null) {
211                 currentWrite = queue.poll();
212             }
213 
214             if (currentWrite == null) {
215                 break;
216             }
217             final PendingWrite currentWrite = this.currentWrite;
218             final Object pendingMessage = currentWrite.msg;
219 
220             if (pendingMessage instanceof ChunkedInput) {
221                 final ChunkedInput<?> chunks = (ChunkedInput<?>) pendingMessage;
222                 boolean endOfInput;
223                 boolean suspend;
224                 Object message = null;
225                 try {
226                     message = chunks.readChunk(ctx);
227                     endOfInput = chunks.isEndOfInput();
228 
229                     if (message == null) {
230                         // No need to suspend when reached at the end.
231                         suspend = !endOfInput;
232                     } else {
233                         suspend = false;
234                     }
235                 } catch (final Throwable t) {
236                     this.currentWrite = null;
237 
238                     if (message != null) {
239                         ReferenceCountUtil.release(message);
240                     }
241 
242                     currentWrite.fail(t);
243                     closeInput(chunks);
244                     break;
245                 }
246 
247                 if (suspend) {
248                     // ChunkedInput.nextChunk() returned null and it has
249                     // not reached at the end of input. Let's wait until
250                     // more chunks arrive. Nothing to write or notify.
251                     break;
252                 }
253 
254                 if (message == null) {
255                     // If message is null write an empty ByteBuf.
256                     // See https://github.com/netty/netty/issues/1671
257                     message = Unpooled.EMPTY_BUFFER;
258                 }
259 
260                 final int amount = amount(message);
261                 ChannelFuture f = ctx.write(message);
262                 if (endOfInput) {
263                     this.currentWrite = null;
264 
265                     // Register a listener which will close the input once the write is complete.
266                     // This is needed because the Chunk may have some resource bound that can not
267                     // be closed before its not written.
268                     //
269                     // See https://github.com/netty/netty/issues/303
270                     f.addListener(new ChannelFutureListener() {
271                         @Override
272                         public void operationComplete(ChannelFuture future) throws Exception {
273                             currentWrite.progress(amount);
274                             currentWrite.success();
275                             closeInput(chunks);
276                         }
277                     });
278                 } else if (channel.isWritable()) {
279                     f.addListener(new ChannelFutureListener() {
280                         @Override
281                         public void operationComplete(ChannelFuture future) throws Exception {
282                             if (!future.isSuccess()) {
283                                 closeInput((ChunkedInput<?>) pendingMessage);
284                                 currentWrite.fail(future.cause());
285                             } else {
286                                 currentWrite.progress(amount);
287                             }
288                         }
289                     });
290                 } else {
291                     f.addListener(new ChannelFutureListener() {
292                         @Override
293                         public void operationComplete(ChannelFuture future) throws Exception {
294                             if (!future.isSuccess()) {
295                                 closeInput((ChunkedInput<?>) pendingMessage);
296                                 currentWrite.fail(future.cause());
297                             } else {
298                                 currentWrite.progress(amount);
299                                 if (channel.isWritable()) {
300                                     resumeTransfer();
301                                 }
302                             }
303                         }
304                     });
305                 }
306             } else {
307                 ctx.write(pendingMessage, currentWrite.promise);
308                 this.currentWrite = null;
309             }
310 
311             // Always need to flush
312             ctx.flush();
313             flushed = true;
314 
315             if (!channel.isActive()) {
316                 discard(new ClosedChannelException());
317                 break;
318             }
319         }
320 
321         return flushed;
322     }

324     static void closeInput(ChunkedInput<?> chunks) {
325         try {
326             chunks.close();
327         } catch (Throwable t) {
328             if (logger.isWarnEnabled()) {
329                 logger.warn("Failed to close a chunked input.", t);
330             }
331         }
332     }

334     private static final class PendingWrite {
335         final Object msg;
336         final ChannelPromise promise;
337         private long progress;
338 
339         PendingWrite(Object msg, ChannelPromise promise) {
340             this.msg = msg;
341             this.promise = promise;
342         }
343 
344         void fail(Throwable cause) {
345             ReferenceCountUtil.release(msg);
346             promise.tryFailure(cause);
347         }
348 
349         void success() {
350             if (promise.isDone()) {
351                 // No need to notify the progress or fulfill the promise because it's done already.
352                 return;
353             }
354 
355             if (promise instanceof ChannelProgressivePromise) {
356                 // Now we know what the total is.
357                 ((ChannelProgressivePromise) promise).tryProgress(progress, progress);
358             }
359 
360             promise.trySuccess();
361         }
362 
363         void progress(int amount) {
364             progress += amount;
365             if (promise instanceof ChannelProgressivePromise) {
366                 ((ChannelProgressivePromise) promise).tryProgress(progress, -1);
367             }
368         }
369     }
370 
371     private static int amount(Object msg) {
372         if (msg instanceof ByteBuf) {
373             return ((ByteBuf) msg).readableBytes();
374         }
375         if (msg instanceof ByteBufHolder) {
376             return ((ByteBufHolder) msg).content().readableBytes();
377         }
378         return 1;
379     }
