64      public DefaultChannelPipeline(AbstractChannel channel) {
65          if (channel == null) {
66              throw new NullPointerException("channel");
67          }
68          this.channel = channel;
69  
70          tail = new TailContext(this);
71          tail.setHandlerAddedCalled();
72          head = new HeadContext(this);
73          head.setHandlerAddedCalled();
74  
75          head.next = tail;
76          tail.prev = head;
77      }

79      private AbstractChannelHandlerContext newContext(EventExecutorGroup group, String name, ChannelHandler handler) {
80          return new DefaultChannelHandlerContext(this, childExecutor(group), name, handler);
81      }

83      private EventExecutor childExecutor(EventExecutorGroup group) {
84          if (group == null) {
85              return null;
86          }
87          Map<EventExecutorGroup, EventExecutor> childExecutors = this.childExecutors;
88          if (childExecutors == null) {
89              // Use size of 4 as most people only use one extra EventExecutor.
90              childExecutors = this.childExecutors = new IdentityHashMap<EventExecutorGroup, EventExecutor>(4);
91          }
92          // Pin one of the child executors once and remember it so that the same child executor
93          // is used to fire events for the same channel.
94          EventExecutor childExecutor = childExecutors.get(group);
95          if (childExecutor == null) {
96              childExecutor = group.next();
97              childExecutors.put(group, childExecutor);
98          }
99          return childExecutor;
100     }

214     @Override
215     public ChannelPipeline addFirst(EventExecutorGroup executor, ChannelHandler... handlers) {
216         if (handlers == null) {
217             throw new NullPointerException("handlers");
218         }
219         if (handlers.length == 0 || handlers[0] == null) {
220             return this;
221         }
222 
223         int size;
224         for (size = 1; size < handlers.length; size ++) {
225             if (handlers[size] == null) {
226                 break;
227             }
228         }
229 
230         for (int i = size - 1; i >= 0; i --) {
231             ChannelHandler h = handlers[i];
232             addFirst(executor, generateName(h), h);
233         }
234 
235         return this;
236     }

259     private String generateName(ChannelHandler handler) {
260         Map<Class<?>, String> cache = nameCaches.get();
261         Class<?> handlerType = handler.getClass();
262         String name = cache.get(handlerType);
263         if (name == null) {
264             name = generateName0(handlerType);
265             cache.put(handlerType, name);
266         }
267 
268         synchronized (this) {
269             // It's not very likely for a user to put more than one handler of the same type, but make sure to avoid
270             // any name conflicts.  Note that we don't cache the names generated here.
271             if (context0(name) != null) {
272                 String baseName = name.substring(0, name.length() - 1); // Strip the trailing '0'.
273                 for (int i = 1;; i ++) {
274                     String newName = baseName + i;
275                     if (context0(newName) == null) {
276                         name = newName;
277                         break;
278                     }
279                 }
280             }
281         }
282 
283         return name;
284     }

307     private AbstractChannelHandlerContext remove(final AbstractChannelHandlerContext ctx) {
308         assert ctx != head && ctx != tail;
309 
310         AbstractChannelHandlerContext context;
311         Future<?> future;
312 
313         synchronized (this) {
314             if (!isExecuteLater(ctx)) {
315                 remove0(ctx);
316                 return ctx;
317             } else {
318                future = ctx.executor().submit(new OneTimeTask() {
319                    @Override
320                    public void run() {
321                        synchronized (DefaultChannelPipeline.this) {
322                            remove0(ctx);
323                        }
324                    }
325                });
326                context = ctx;
327             }
328         }
329 
330         // Run the following 'waiting' code outside of the above synchronized block
331         // in order to avoid deadlock
332 
333         waitForFuture(future);
334 
335         return context;
336     }

339         AbstractChannelHandlerContext prev = ctx.prev;
340         AbstractChannelHandlerContext next = ctx.next;
341         prev.next = next;
342         next.prev = prev;
343         callHandlerRemoved(ctx);
344     }

380     private ChannelHandler replace(
381             final AbstractChannelHandlerContext ctx, final String newName,
382             ChannelHandler newHandler) {
383 
384         assert ctx != head && ctx != tail;
385 
386         Future<?> future;
387         synchronized (this) {
388             boolean sameName = ctx.name().equals(newName);
389             if (!sameName) {
390                 checkDuplicateName(newName);
391             }
392 
393             final AbstractChannelHandlerContext newCtx = newContext(ctx.executor, newName, newHandler);
394 
395             if (!isExecuteLater(newCtx)) {
396                 replace0(ctx, newCtx);
397                 return ctx.handler();
398             } else {
399                 future = newCtx.executor().submit(new OneTimeTask() {
400                     @Override
401                     public void run() {
402                         synchronized (DefaultChannelPipeline.this) {
403                             replace0(ctx, newCtx);
404                         }
405                     }
406                 });
407             }
408         }
409 
410         // Run the following 'waiting' code outside of the above synchronized block
411         // in order to avoid deadlock
412 
413         waitForFuture(future);
414 
415         return ctx.handler();
416     }

418     private void replace0(AbstractChannelHandlerContext oldCtx, AbstractChannelHandlerContext newCtx) {
419         checkMultiplicity(newCtx);
420 
421         AbstractChannelHandlerContext prev = oldCtx.prev;
422         AbstractChannelHandlerContext next = oldCtx.next;
423         newCtx.prev = prev;
424         newCtx.next = next;
425 
426         // Finish the replacement of oldCtx with newCtx in the linked list.
427         // Note that this doesn't mean events will be sent to the new handler immediately
428         // because we are currently at the event handler thread and no more than one handler methods can be invoked
429         // at the same time (we ensured that in replace().)
430         prev.next = newCtx;
431         next.prev = newCtx;
432 
433         // update the reference to the replacement so forward of buffered content will work correctly
434         oldCtx.prev = newCtx;
435         oldCtx.next = newCtx;
436 
437         // Invoke newHandler.handlerAdded() first (i.e. before oldHandler.handlerRemoved() is invoked)
438         // because callHandlerRemoved() will trigger inboundBufferUpdated() or flush() on newHandler and those
439         // event handlers must be called after handlerAdded().
440         callHandlerAdded(newCtx);
441         callHandlerRemoved(oldCtx);
442     }

444     private static void checkMultiplicity(ChannelHandlerContext ctx) {
445         ChannelHandler handler = ctx.handler();
446         if (handler instanceof ChannelHandlerAdapter) {
447             ChannelHandlerAdapter h = (ChannelHandlerAdapter) handler;
448             if (!h.isSharable() && h.added) {
449                 throw new ChannelPipelineException(
450                         h.getClass().getName() +
451                         " is not a @Sharable handler, so can't be added or removed multiple times.");
452             }
453             h.added = true;
454         }
455     }

457     private void callHandlerAdded(final AbstractChannelHandlerContext ctx) {
458         if (isExecuteLater(ctx)) {
459             ctx.executor().execute(new OneTimeTask() {
460                 @Override
461                 public void run() {
462                     callHandlerAdded0(ctx);
463                 }
464             });
465             return;
466         }
467         callHandlerAdded0(ctx);
468     }
469 
470     private void callHandlerAdded0(final AbstractChannelHandlerContext ctx) {
471         try {
472             try {
473                 ctx.handler().handlerAdded(ctx);
474             } finally {
475                 // handlerAdded(...) method was called.
476                 ctx.setHandlerAddedCalled();
477             }
478         } catch (Throwable t) {
479             boolean removed = false;
480             try {
481                 remove((AbstractChannelHandlerContext) ctx);
482                 removed = true;
483             } catch (Throwable t2) {
484                 if (logger.isWarnEnabled()) {
485                     logger.warn("Failed to remove a handler: " + ctx.name(), t2);
486                 }
487             }
488 
489             if (removed) {
490                 fireExceptionCaught(new ChannelPipelineException(
491                         ctx.handler().getClass().getName() +
492                         ".handlerAdded() has thrown an exception; removed.", t));
493             } else {
494                 fireExceptionCaught(new ChannelPipelineException(
495                         ctx.handler().getClass().getName() +
496                         ".handlerAdded() has thrown an exception; also failed to remove.", t));
497             }
498         }
499     }

501     private void callHandlerRemoved(final AbstractChannelHandlerContext ctx) {
502         if (isExecuteLater(ctx)) {
503             ctx.executor().execute(new OneTimeTask() {
504                 @Override
505                 public void run() {
506                     callHandlerRemoved0(ctx);
507                  }
508             });
509             return;
510         }
511         callHandlerRemoved0(ctx);
512     }

514     private void callHandlerRemoved0(final AbstractChannelHandlerContext ctx) {
515         // Notify the complete removal.
516         try {
517             ctx.handler().handlerRemoved(ctx);
518             ctx.setRemoved();
519         } catch (Throwable t) {
520             fireExceptionCaught(new ChannelPipelineException(
521                     ctx.handler().getClass().getName() + ".handlerRemoved() has thrown an exception.", t));
522         }
523     }

529     /**
530      * Waits for a future to finish.  If the task is interrupted, then the current thread will be interrupted.
531      * It is expected that the task performs any appropriate locking.
532      * <p>
533      * If the internal call throws a {@link Throwable}, but it is not an instance of {@link Error} or
534      * {@link RuntimeException}, then it is wrapped inside a {@link ChannelPipelineException} and that is
535      * thrown instead.</p>
536      *
537      * @param future wait for this future
538      * @see Future#get()
539      * @throws Error if the task threw this.
540      * @throws RuntimeException if the task threw this.
541      * @throws ChannelPipelineException with a {@link Throwable} as a cause, if the task threw another type of
542      *         {@link Throwable}.
543      */
544     private static void waitForFuture(Future<?> future) {
545         try {
546             future.get();
547         } catch (ExecutionException ex) {
548             // In the arbitrary case, we can throw Error, RuntimeException, and Exception
549             PlatformDependent.throwException(ex.getCause());
550         } catch (InterruptedException ex) {
551             // Interrupt the calling thread (note that this method is not called from the event loop)
552             Thread.currentThread().interrupt();
553         }
554     }

740     /**
741      * Removes all handlers from the pipeline one by one from tail (exclusive) to head (exclusive) to trigger
742      * handlerRemoved().
743      *
744      * Note that we traverse up the pipeline ({@link #destroyUp(AbstractChannelHandlerContext, boolean)})
745      * before traversing down ({@link #destroyDown(Thread, AbstractChannelHandlerContext, boolean)}) so that
746      * the handlers are removed after all events are handled.
747      *
748      * See: https://github.com/netty/netty/issues/3156
749      */
750     private void destroy() {
751         destroyUp(head.next, false);
752     }
753 
754     private void destroyUp(AbstractChannelHandlerContext ctx, boolean inEventLoop) {
755         final Thread currentThread = Thread.currentThread();
756         final AbstractChannelHandlerContext tail = this.tail;
757         for (;;) {
758             if (ctx == tail) {
759                 destroyDown(currentThread, tail.prev, inEventLoop);
760                 break;
761             }
762 
763             final EventExecutor executor = ctx.executor();
764             if (!inEventLoop && !executor.inEventLoop(currentThread)) {
765                 final AbstractChannelHandlerContext finalCtx = ctx;
766                 executor.execute(new OneTimeTask() {
767                     @Override
768                     public void run() {
769                         destroyUp(finalCtx, true);
770                     }
771                 });
772                 break;
773             }
774 
775             ctx = ctx.next;
776             inEventLoop = false;
777         }
778     }
779 
780     private void destroyDown(Thread currentThread, AbstractChannelHandlerContext ctx, boolean inEventLoop) {
781         // We have reached at tail; now traverse backwards.
782         final AbstractChannelHandlerContext head = this.head;
783         for (;;) {
784             if (ctx == head) {
785                 break;
786             }
787 
788             final EventExecutor executor = ctx.executor();
789             if (inEventLoop || executor.inEventLoop(currentThread)) {
790                 synchronized (this) {
791                     remove0(ctx);
792                 }
793             } else {
794                 final AbstractChannelHandlerContext finalCtx = ctx;
795                 executor.execute(new OneTimeTask() {
796                     @Override
797                     public void run() {
798                         destroyDown(Thread.currentThread(), finalCtx, true);
799                     }
