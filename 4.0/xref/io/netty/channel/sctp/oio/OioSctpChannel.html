106     /**
107      * Create a new instance from the given {@link SctpChannel}.
108      *
109      * @param parent    the parent {@link Channel} which was used to create this instance. This can be null if the
110      *                  {@link} has no parent as it was created by your self.
111      * @param ch        the {@link SctpChannel} which is used by this instance
112      */
113     public OioSctpChannel(Channel parent, SctpChannel ch) {
114         super(parent);
115         this.ch = ch;
116         boolean success = false;
117         try {
118             ch.configureBlocking(false);
119             readSelector = Selector.open();
120             writeSelector = Selector.open();
121             connectSelector = Selector.open();
122 
123             ch.register(readSelector, SelectionKey.OP_READ);
124             ch.register(writeSelector, SelectionKey.OP_WRITE);
125             ch.register(connectSelector, SelectionKey.OP_CONNECT);
126 
127             config = new OioSctpChannelConfig(this, ch);
128             notificationHandler = new SctpNotificationHandler(this);
129             success = true;
130         } catch (Exception e) {
131             throw new ChannelException("failed to initialize a sctp channel", e);
132         } finally {
133             if (!success) {
134                 try {
135                     ch.close();
136                 } catch (IOException e) {
137                     logger.warn("Failed to close a sctp channel.", e);
138                 }
139             }
140         }
141     }

173     @Override
174     protected int doReadMessages(List<Object> msgs) throws Exception {
175         if (!readSelector.isOpen()) {
176             return 0;
177         }
178 
179         int readMessages = 0;
180 
181         final int selectedKeys = readSelector.select(SO_TIMEOUT);
182         final boolean keysSelected = selectedKeys > 0;
183 
184         if (!keysSelected) {
185             return readMessages;
186         }
187 
188         // We must clear the selectedKeys because the Selector will never do it. If we do not clear it, the selectionKey
189         // will always be returned even if there is no data can be read which causes performance issue. And in some
190         // implementation of Selector, the select method may return 0 if the selectionKey which is ready for process has
191         // already been in the selectedKeys and cause the keysSelected above to be false even if we actually have
192         // something to read.
193         readSelector.selectedKeys().clear();
194 
195         RecvByteBufAllocator.Handle allocHandle = this.allocHandle;
196         if (allocHandle == null) {
197             this.allocHandle = allocHandle = config().getRecvByteBufAllocator().newHandle();
198         }
199 
200         ByteBuf buffer = allocHandle.allocate(config().getAllocator());
201         boolean free = true;
202 
203         try {
204             ByteBuffer data = buffer.nioBuffer(buffer.writerIndex(), buffer.writableBytes());
205             MessageInfo messageInfo = ch.receive(data, null, notificationHandler);
206             if (messageInfo == null) {
207                 return readMessages;
208             }
209 
210             data.flip();
211             msgs.add(new SctpMessage(messageInfo, buffer.writerIndex(buffer.writerIndex() + data.remaining())));
212             free = false;
213             readMessages ++;
214         } catch (Throwable cause) {
215             PlatformDependent.throwException(cause);
216         }  finally {
217             int bytesRead = buffer.readableBytes();
218             allocHandle.record(bytesRead);
219             if (free) {
220                 buffer.release();
221             }
222         }
223         return readMessages;
224     }

226     @Override
227     protected void doWrite(ChannelOutboundBuffer in) throws Exception {
228         if (!writeSelector.isOpen()) {
229             return;
230         }
231         final int size = in.size();
232         final int selectedKeys = writeSelector.select(SO_TIMEOUT);
233         if (selectedKeys > 0) {
234             final Set<SelectionKey> writableKeys = writeSelector.selectedKeys();
235             if (writableKeys.isEmpty()) {
236                 return;
237             }
238             Iterator<SelectionKey> writableKeysIt = writableKeys.iterator();
239             int written = 0;
240             for (;;) {
241                 if (written == size) {
242                     // all written
243                     return;
244                 }
245                 writableKeysIt.next();
246                 writableKeysIt.remove();
247 
248                 SctpMessage packet = (SctpMessage) in.current();
249                 if (packet == null) {
250                     return;
251                 }
252 
253                 ByteBuf data = packet.content();
254                 int dataLen = data.readableBytes();
255                 ByteBuffer nioData;
256 
257                 if (data.nioBufferCount() != -1) {
258                     nioData = data.nioBuffer();
259                 } else {
260                     nioData = ByteBuffer.allocate(dataLen);
261                     data.getBytes(data.readerIndex(), nioData);
262                     nioData.flip();
263                 }
264 
265                 final MessageInfo mi = MessageInfo.createOutgoing(association(), null, packet.streamIdentifier());
266                 mi.payloadProtocolID(packet.protocolIdentifier());
267                 mi.streamNumber(packet.streamIdentifier());
268                 mi.unordered(packet.isUnordered());
269 
270                 ch.send(nioData, mi);
271                 written ++;
272                 in.remove();
273 
274                 if (!writableKeysIt.hasNext()) {
275                     return;
276                 }
277             }
278         }
279     }

364     @Override
365     protected void doConnect(SocketAddress remoteAddress,
366                              SocketAddress localAddress) throws Exception {
367         if (localAddress != null) {
368             ch.bind(localAddress);
369         }
370 
371         boolean success = false;
372         try {
373             ch.connect(remoteAddress);
374             boolean  finishConnect = false;
375             while (!finishConnect) {
376                 if (connectSelector.select(SO_TIMEOUT) >= 0) {
377                     final Set<SelectionKey> selectionKeys = connectSelector.selectedKeys();
378                     for (SelectionKey key : selectionKeys) {
379                        if (key.isConnectable()) {
380                            selectionKeys.clear();
381                            finishConnect = true;
382                            break;
383                        }
384                     }
385                     selectionKeys.clear();
386                 }
387             }
388             success = ch.finishConnect();
389         } finally {
390             if (!success) {
391                 doClose();
392             }
393         }
394     }
