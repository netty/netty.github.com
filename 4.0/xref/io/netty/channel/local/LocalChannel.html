168     @Override
169     protected void doRegister() throws Exception {
170         // Check if both peer and parent are non-null because this channel was created by a LocalServerChannel.
171         // This is needed as a peer may not be null also if a LocalChannel was connected before and
172         // deregistered / registered later again.
173         //
174         // See https://github.com/netty/netty/issues/2400
175         if (peer != null && parent() != null) {
176             // Store the peer in a local variable as it may be set to null if doClose() is called.
177             // Because of this we also set registerInProgress to true as we check for this in doClose() and make sure
178             // we delay the fireChannelInactive() to be fired after the fireChannelActive() and so keep the correct
179             // order of events.
180             //
181             // See https://github.com/netty/netty/issues/2144
182             final LocalChannel peer = this.peer;
183             registerInProgress = true;
184             state = 2;
185 
186             peer.remoteAddress = parent().localAddress();
187             peer.state = 2;
188 
189             // Always call peer.eventLoop().execute() even if peer.eventLoop().inEventLoop() is true.
190             // This ensures that if both channels are on the same event loop, the peer's channelActive
191             // event is triggered *after* this channel's channelRegistered event, so that this channel's
192             // pipeline is fully initialized by ChannelInitializer before any channelRead events.
193             peer.eventLoop().execute(new OneTimeTask() {
194                 @Override
195                 public void run() {
196                     registerInProgress = false;
197                     ChannelPromise promise = peer.connectPromise;
198 
199                     // Only trigger fireChannelActive() if the promise was not null and was not completed yet.
200                     // connectPromise may be set to null if doClose() was called in the meantime.
201                     if (promise != null && promise.trySuccess()) {
202                         peer.pipeline().fireChannelActive();
203                     }
204                 }
205             });
206         }
207         ((SingleThreadEventExecutor) eventLoop()).addShutdownHook(shutdownHook);
208     }

223     @Override
224     protected void doClose() throws Exception {
225         final LocalChannel peer = this.peer;
226         if (state <= 2) {
227             // Update all internal state before the closeFuture is notified.
228             if (localAddress != null) {
229                 if (parent() == null) {
230                     LocalChannelRegistry.unregister(localAddress);
231                 }
232                 localAddress = null;
233             }
234 
235             // State change must happen before finishPeerRead to ensure writes are released either in doWrite or
236             // channelRead.
237             state = 3;
238 
239             ChannelPromise promise = connectPromise;
240             if (promise != null) {
241                 // Use tryFailure() instead of setFailure() to avoid the race against cancel().
242                 promise.tryFailure(CLOSED_CHANNEL_EXCEPTION);
243                 connectPromise = null;
244             }
245 
246             // To preserve ordering of events we must process any pending reads
247             if (writeInProgress && peer != null) {
248                 finishPeerRead(peer);
249             }
250         }
251 
252         if (peer != null && peer.isActive()) {
253             // Need to execute the close in the correct EventLoop (see https://github.com/netty/netty/issues/1777).
254             // Also check if the registration was not done yet. In this case we submit the close to the EventLoop
255             // to make sure its run after the registration completes (see https://github.com/netty/netty/issues/2144).
256             if (peer.eventLoop().inEventLoop() && !registerInProgress) {
257                 doPeerClose(peer, peer.writeInProgress);
258             } else {
259                 // This value may change, and so we should save it before executing the Runnable.
260                 final boolean peerWriteInProgress = peer.writeInProgress;
261                 try {
262                     peer.eventLoop().execute(new OneTimeTask() {
263                         @Override
264                         public void run() {
265                             doPeerClose(peer, peerWriteInProgress);
266                         }
267                     });
268                 } catch (RuntimeException e) {
269                     // The peer close may attempt to drain this.inboundBuffers. If that fails make sure it is drained.
270                     releaseInboundBuffers();
271                     throw e;
272                 }
273             }
274             this.peer = null;
275         }
276     }

278     private void doPeerClose(LocalChannel peer, boolean peerWriteInProgress) {
279         if (peerWriteInProgress) {
280             finishPeerRead0(this);
281         }
282         peer.unsafe().close(peer.unsafe().voidPromise());
283     }

291     @Override
292     protected void doBeginRead() throws Exception {
293         if (readInProgress) {
294             return;
295         }
296 
297         ChannelPipeline pipeline = pipeline();
298         Queue<Object> inboundBuffer = this.inboundBuffer;
299         if (inboundBuffer.isEmpty()) {
300             readInProgress = true;
301             return;
302         }
303 
304         final InternalThreadLocalMap threadLocals = InternalThreadLocalMap.get();
305         final Integer stackDepth = threadLocals.localChannelReaderStackDepth();
306         if (stackDepth < MAX_READER_STACK_DEPTH) {
307             threadLocals.setLocalChannelReaderStackDepth(stackDepth + 1);
308             try {
309                 for (;;) {
310                     Object received = inboundBuffer.poll();
311                     if (received == null) {
312                         break;
313                     }
314                     pipeline.fireChannelRead(received);
315                 }
316                 pipeline.fireChannelReadComplete();
317             } finally {
318                 threadLocals.setLocalChannelReaderStackDepth(stackDepth);
319             }
320         } else {
321             try {
322                 eventLoop().execute(readTask);
323             } catch (RuntimeException e) {
324                 releaseInboundBuffers();
325                 throw e;
326             }
327         }
328     }

330     @Override
331     protected void doWrite(ChannelOutboundBuffer in) throws Exception {
332         if (state < 2) {
333             throw new NotYetConnectedException();
334         }
335         if (state > 2) {
336             throw CLOSED_CHANNEL_EXCEPTION;
337         }
338 
339         final LocalChannel peer = this.peer;
340 
341         writeInProgress = true;
342         try {
343             for (;;) {
344                 Object msg = in.current();
345                 if (msg == null) {
346                     break;
347                 }
348                 try {
349                     // It is possible the peer could have closed while we are writing, and in this case we should
350                     // simulate real socket behavior and ensure the write operation is failed.
351                     if (peer.state == 2) {
352                         peer.inboundBuffer.add(ReferenceCountUtil.retain(msg));
353                         in.remove();
354                     } else {
355                         in.remove(CLOSED_CHANNEL_EXCEPTION);
356                     }
357                 } catch (Throwable cause) {
358                     in.remove(cause);
359                 }
360             }
361         } finally {
362             // The following situation may cause trouble:
363             // 1. Write (with promise X)
364             // 2. promise X is completed when in.remove() is called, and a listener on this promise calls close()
365             // 3. Then the close event will be executed for the peer before the write events, when the write events
366             // actually happened before the close event.
367             writeInProgress = false;
368         }
369 
370         finishPeerRead(peer);
371     }

373     private void finishPeerRead(final LocalChannel peer) {
374         // If the peer is also writing, then we must schedule the event on the event loop to preserve read order.
375         if (peer.eventLoop() == eventLoop() && !peer.writeInProgress) {
376             finishPeerRead0(peer);
377         } else {
378             runFinishPeerReadTask(peer);
379         }
380     }
381 
382     private void runFinishPeerReadTask(final LocalChannel peer) {
383         // If the peer is writing, we must wait until after reads are completed for that peer before we can read. So
384         // we keep track of the task, and coordinate later that our read can't happen until the peer is done.
385         final Runnable finishPeerReadTask = new OneTimeTask() {
386             @Override
387             public void run() {
388                 finishPeerRead0(peer);
389             }
390         };
391         try {
392             if (peer.writeInProgress) {
393                 peer.finishReadFuture = peer.eventLoop().submit(finishPeerReadTask);
394             } else {
395                 peer.eventLoop().execute(finishPeerReadTask);
396             }
397         } catch (RuntimeException e) {
398             peer.releaseInboundBuffers();
399             throw e;
400         }
401     }
402 
403     private void releaseInboundBuffers() {
404         for (;;) {
405             Object o = inboundBuffer.poll();
406             if (o == null) {
407                 break;
408             }
409             ReferenceCountUtil.release(o);
410         }
411     }
412 
413     private void finishPeerRead0(LocalChannel peer) {
414         Future<?> peerFinishReadFuture = peer.finishReadFuture;
415         if (peerFinishReadFuture != null) {
416             if (!peerFinishReadFuture.isDone()) {
417                 runFinishPeerReadTask(peer);
418                 return;
419             } else {
420                 // Lazy unset to make sure we don't prematurely unset it while scheduling a new task.
421                 FINISH_READ_FUTURE_UPDATER.compareAndSet(peer, peerFinishReadFuture, null);
422             }
423         }
424         ChannelPipeline peerPipeline = peer.pipeline();
425         if (peer.readInProgress) {
426             peer.readInProgress = false;
427             for (;;) {
428                 Object received = peer.inboundBuffer.poll();
429                 if (received == null) {
430                     break;
431                 }
432                 peerPipeline.fireChannelRead(received);
433             }
434             peerPipeline.fireChannelReadComplete();
435         }

440         @Override
441         public void connect(final SocketAddress remoteAddress,
442                 SocketAddress localAddress, final ChannelPromise promise) {
443             if (!promise.setUncancellable() || !ensureOpen(promise)) {
444                 return;
445             }
446 
447             if (state == 2) {
448                 Exception cause = new AlreadyConnectedException();
449                 safeSetFailure(promise, cause);
450                 pipeline().fireExceptionCaught(cause);
451                 return;
452             }
453 
454             if (connectPromise != null) {
455                 throw new ConnectionPendingException();
456             }
457 
458             connectPromise = promise;
459 
460             if (state != 1) {
461                 // Not bound yet and no localAddress specified - get one.
462                 if (localAddress == null) {
463                     localAddress = new LocalAddress(LocalChannel.this);
464                 }
465             }
466 
467             if (localAddress != null) {
468                 try {
469                     doBind(localAddress);
470                 } catch (Throwable t) {
471                     safeSetFailure(promise, t);
472                     close(voidPromise());
473                     return;
474                 }
475             }
476 
477             Channel boundChannel = LocalChannelRegistry.get(remoteAddress);
478             if (!(boundChannel instanceof LocalServerChannel)) {
479                 Exception cause = new ChannelException("connection refused");
480                 safeSetFailure(promise, cause);
481                 close(voidPromise());
482                 return;
483             }
484 
485             LocalServerChannel serverChannel = (LocalServerChannel) boundChannel;
486             peer = serverChannel.serve(LocalChannel.this);
487         }
