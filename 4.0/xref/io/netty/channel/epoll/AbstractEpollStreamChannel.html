145     /**
146      * Splice from this {@link AbstractEpollStreamChannel} to another {@link AbstractEpollStreamChannel}.
147      * The {@code len} is the number of bytes to splice. If using {@link Integer#MAX_VALUE} it will
148      * splice until the {@link ChannelFuture} was canceled or it was failed.
149      *
150      * Please note:
151      * <ul>
152      *   <li>both channels need to be registered to the same {@link EventLoop}, otherwise an
153      *   {@link IllegalArgumentException} is thrown. </li>
154      *   <li>{@link EpollChannelConfig#getEpollMode()} must be {@link EpollMode#LEVEL_TRIGGERED} for this and the
155      *   target {@link AbstractEpollStreamChannel}</li>
156      * </ul>
157      *
158      */
159     public final ChannelFuture spliceTo(final AbstractEpollStreamChannel ch, final int len,
160                                         final ChannelPromise promise) {
161         if (ch.eventLoop() != eventLoop()) {
162             throw new IllegalArgumentException("EventLoops are not the same.");
163         }
164         if (len < 0) {
165             throw new IllegalArgumentException("len: " + len + " (expected: >= 0)");
166         }
167         if (ch.config().getEpollMode() != EpollMode.LEVEL_TRIGGERED
168                 || config().getEpollMode() != EpollMode.LEVEL_TRIGGERED) {
169             throw new IllegalStateException("spliceTo() supported only when using " + EpollMode.LEVEL_TRIGGERED);
170         }
171         checkNotNull(promise, "promise");
172         if (!isOpen()) {
173             promise.tryFailure(CLOSED_CHANNEL_EXCEPTION);
174         } else {
175             addToSpliceQueue(new SpliceInChannelTask(ch, len, promise));
176             failSpliceIfClosed(promise);
177         }
178         return promise;
179     }

200      * Splice from this {@link AbstractEpollStreamChannel} to another {@link FileDescriptor}.
201      * The {@code offset} is the offset for the {@link FileDescriptor} and {@code len} is the
202      * number of bytes to splice. If using {@link Integer#MAX_VALUE} it will splice until the
203      * {@link ChannelFuture} was canceled or it was failed.
204      *
205      * Please note:
206      * <ul>
207      *   <li>{@link EpollChannelConfig#getEpollMode()} must be {@link EpollMode#LEVEL_TRIGGERED} for this
208      *   {@link AbstractEpollStreamChannel}</li>
209      *   <li>the {@link FileDescriptor} will not be closed after the {@link ChannelPromise} is notified</li>
210      *   <li>this channel must be registered to an event loop or {@link IllegalStateException} will be thrown.</li>
211      * </ul>
212      */
213     public final ChannelFuture spliceTo(final FileDescriptor ch, final int offset, final int len,
214                                         final ChannelPromise promise) {
215         if (len < 0) {
216             throw new IllegalArgumentException("len: " + len + " (expected: >= 0)");
217         }
218         if (offset < 0) {
219             throw new IllegalArgumentException("offset must be >= 0 but was " + offset);
220         }
221         if (config().getEpollMode() != EpollMode.LEVEL_TRIGGERED) {
222             throw new IllegalStateException("spliceTo() supported only when using " + EpollMode.LEVEL_TRIGGERED);
223         }
224         checkNotNull(promise, "promise");
225         if (!isOpen()) {
226             promise.tryFailure(CLOSED_CHANNEL_EXCEPTION);
227         } else {
228             addToSpliceQueue(new SpliceFdTask(ch, offset, len, promise));
229             failSpliceIfClosed(promise);
230         }
231         return promise;
232     }

234     private void failSpliceIfClosed(ChannelPromise promise) {
235         if (!isOpen()) {
236             // Seems like the Channel was closed in the meantime try to fail the promise to prevent any
237             // cases where a future may not be notified otherwise.
238             if (promise.tryFailure(CLOSED_CHANNEL_EXCEPTION)) {
239                 eventLoop().execute(new OneTimeTask() {
240                     @Override
241                     public void run() {
242                         // Call this via the EventLoop as it is a MPSC queue.
243                         clearSpliceQueue();
244                     }
245                 });
246             }
247         }
248     }

250     /**
251      * Write bytes form the given {@link ByteBuf} to the underlying {@link java.nio.channels.Channel}.
252      * @param buf           the {@link ByteBuf} from which the bytes should be written
253      */
254     private boolean writeBytes(ChannelOutboundBuffer in, ByteBuf buf, int writeSpinCount) throws Exception {
255         int readableBytes = buf.readableBytes();
256         if (readableBytes == 0) {
257             in.remove();
258             return true;
259         }
260 
261         if (buf.hasMemoryAddress() || buf.nioBufferCount() == 1) {
262             int writtenBytes = doWriteBytes(buf, writeSpinCount);
263             in.removeBytes(writtenBytes);
264             return writtenBytes == readableBytes;
265         } else {
266             ByteBuffer[] nioBuffers = buf.nioBuffers();
267             return writeBytesMultiple(in, nioBuffers, nioBuffers.length, readableBytes, writeSpinCount);
268         }
269     }

271     private boolean writeBytesMultiple(
272             ChannelOutboundBuffer in, IovArray array, int writeSpinCount) throws IOException {
273 
274         long expectedWrittenBytes = array.size();
275         final long initialExpectedWrittenBytes = expectedWrittenBytes;
276 
277         int cnt = array.count();
278 
279         assert expectedWrittenBytes != 0;
280         assert cnt != 0;
281 
282         boolean done = false;
283         int offset = 0;
284         int end = offset + cnt;
285         for (int i = writeSpinCount - 1; i >= 0; i--) {
286             long localWrittenBytes = fd().writevAddresses(array.memoryAddress(offset), cnt);
287             if (localWrittenBytes == 0) {
288                 break;
289             }
290             expectedWrittenBytes -= localWrittenBytes;
291 
292             if (expectedWrittenBytes == 0) {
293                 // Written everything, just break out here (fast-path)
294                 done = true;
295                 break;
296             }
297 
298             do {
299                 long bytes = array.processWritten(offset, localWrittenBytes);
300                 if (bytes == -1) {
301                     // incomplete write
302                     break;
303                 } else {
304                     offset++;
305                     cnt--;
306                     localWrittenBytes -= bytes;
307                 }
308             } while (offset < end && localWrittenBytes > 0);
309         }
310         in.removeBytes(initialExpectedWrittenBytes - expectedWrittenBytes);
311         return done;
312     }

314     private boolean writeBytesMultiple(
315             ChannelOutboundBuffer in, ByteBuffer[] nioBuffers,
316             int nioBufferCnt, long expectedWrittenBytes, int writeSpinCount) throws IOException {
317 
318         assert expectedWrittenBytes != 0;
319         final long initialExpectedWrittenBytes = expectedWrittenBytes;
320 
321         boolean done = false;
322         int offset = 0;
323         int end = offset + nioBufferCnt;
324         for (int i = writeSpinCount - 1; i >= 0; i--) {
325             long localWrittenBytes = fd().writev(nioBuffers, offset, nioBufferCnt);
326             if (localWrittenBytes == 0) {
327                 break;
328             }
329             expectedWrittenBytes -= localWrittenBytes;
330 
331             if (expectedWrittenBytes == 0) {
332                 // Written everything, just break out here (fast-path)
333                 done = true;
334                 break;
335             }
336             do {
337                 ByteBuffer buffer = nioBuffers[offset];
338                 int pos = buffer.position();
339                 int bytes = buffer.limit() - pos;
340                 if (bytes > localWrittenBytes) {
341                     buffer.position(pos + (int) localWrittenBytes);
342                     // incomplete write
343                     break;
344                 } else {
345                     offset++;
346                     nioBufferCnt--;
347                     localWrittenBytes -= bytes;
348                 }
349             } while (offset < end && localWrittenBytes > 0);
350         }
351 
352         in.removeBytes(initialExpectedWrittenBytes - expectedWrittenBytes);
353         return done;
354     }

356     /**
357      * Write a {@link DefaultFileRegion}
358      *
359      * @param region        the {@link DefaultFileRegion} from which the bytes should be written
360      * @return amount       the amount of written bytes
361      */
362     private boolean writeFileRegion(
363             ChannelOutboundBuffer in, DefaultFileRegion region, int writeSpinCount) throws Exception {
364         final long regionCount = region.count();
365         if (region.transfered() >= regionCount) {
366             in.remove();
367             return true;
368         }
369 
370         final long baseOffset = region.position();
371         boolean done = false;
372         long flushedAmount = 0;
373 
374         for (int i = writeSpinCount - 1; i >= 0; i--) {
375             final long offset = region.transfered();
376             final long localFlushedAmount =
377                     Native.sendfile(fd().intValue(), region, baseOffset, offset, regionCount - offset);
378             if (localFlushedAmount == 0) {
379                 break;
380             }
381 
382             flushedAmount += localFlushedAmount;
383             if (region.transfered() >= regionCount) {
384                 done = true;
385                 break;
386             }
387         }
388 
389         if (flushedAmount > 0) {
390             in.progress(flushedAmount);
391         }
392 
393         if (done) {
394             in.remove();
395         }
396         return done;
397     }

399     @Override
400     protected void doWrite(ChannelOutboundBuffer in) throws Exception {
401         int writeSpinCount = config().getWriteSpinCount();
402         for (;;) {
403             final int msgCount = in.size();
404 
405             if (msgCount == 0) {
406                 // Wrote all messages.
407                 clearFlag(Native.EPOLLOUT);
408                 // Return here so we not set the EPOLLOUT flag.
409                 return;
410             }
411 
412             // Do gathering write if the outbounf buffer entries start with more than one ByteBuf.
413             if (msgCount > 1 && in.current() instanceof ByteBuf) {
414                 if (!doWriteMultiple(in, writeSpinCount)) {
415                     // Break the loop and so set EPOLLOUT flag.
416                     break;
417                 }
418 
419                 // We do not break the loop here even if the outbound buffer was flushed completely,
420                 // because a user might have triggered another write and flush when we notify his or her
421                 // listeners.
422             } else { // msgCount == 1
423                 if (!doWriteSingle(in, writeSpinCount)) {
424                     // Break the loop and so set EPOLLOUT flag.
425                     break;
426                 }
427             }
428         }
429         // Underlying descriptor can not accept all data currently, so set the EPOLLOUT flag to be woken up
430         // when it can accept more data.
431         setFlag(Native.EPOLLOUT);
432     }

434     protected boolean doWriteSingle(ChannelOutboundBuffer in, int writeSpinCount) throws Exception {
435         // The outbound buffer contains only one message or it contains a file region.
436         Object msg = in.current();
437         if (msg instanceof ByteBuf) {
438             ByteBuf buf = (ByteBuf) msg;
439             if (!writeBytes(in, buf, writeSpinCount)) {
440                 // was not able to write everything so break here we will get notified later again once
441                 // the network stack can handle more writes.
442                 return false;
443             }
444         } else if (msg instanceof DefaultFileRegion) {
445             DefaultFileRegion region = (DefaultFileRegion) msg;
446             if (!writeFileRegion(in, region, writeSpinCount)) {
447                 // was not able to write everything so break here we will get notified later again once
448                 // the network stack can handle more writes.
449                 return false;
450             }
451         } else if (msg instanceof SpliceOutTask) {
452             if (!((SpliceOutTask) msg).spliceOut()) {
453                 return false;
454             }
455             in.remove();
456         } else {
457             // Should never reach here.
458             throw new Error();
459         }
460 
461         return true;
462     }

464     private boolean doWriteMultiple(ChannelOutboundBuffer in, int writeSpinCount) throws Exception {
465         if (PlatformDependent.hasUnsafe()) {
466             // this means we can cast to IovArray and write the IovArray directly.
467             IovArray array = ((EpollEventLoop) eventLoop()).cleanArray();
468             in.forEachFlushedMessage(array);
469 
470             int cnt = array.count();
471             if (cnt >= 1) {
472                 // TODO: Handle the case where cnt == 1 specially.
473                 if (!writeBytesMultiple(in, array, writeSpinCount)) {
474                     // was not able to write everything so break here we will get notified later again once
475                     // the network stack can handle more writes.
476                     return false;
477                 }
478             } else { // cnt == 0, which means the outbound buffer contained empty buffers only.
479                 in.removeBytes(0);
480             }
481         } else {
482             ByteBuffer[] buffers = in.nioBuffers();
483             int cnt = in.nioBufferCount();
484             if (cnt >= 1) {
485                 // TODO: Handle the case where cnt == 1 specially.
486                 if (!writeBytesMultiple(in, buffers, cnt, in.nioBufferSize(), writeSpinCount)) {
487                     // was not able to write everything so break here we will get notified later again once
488                     // the network stack can handle more writes.
489                     return false;
490                 }
491             } else { // cnt == 0, which means the outbound buffer contained empty buffers only.
492                 in.removeBytes(0);
493             }
494         }
495 
496         return true;
497     }

499     @Override
500     protected Object filterOutboundMessage(Object msg) {
501         if (msg instanceof ByteBuf) {
502             ByteBuf buf = (ByteBuf) msg;
503             if (!buf.hasMemoryAddress() && (PlatformDependent.hasUnsafe() || !buf.isDirect())) {
504                 if (buf instanceof CompositeByteBuf) {
505                     // Special handling of CompositeByteBuf to reduce memory copies if some of the Components
506                     // in the CompositeByteBuf are backed by a memoryAddress.
507                     CompositeByteBuf comp = (CompositeByteBuf) buf;
508                     if (!comp.isDirect() || comp.nioBufferCount() > Native.IOV_MAX) {
509                         // more then 1024 buffers for gathering writes so just do a memory copy.
510                         buf = newDirectBuffer(buf);
511                         assert buf.hasMemoryAddress();
512                     }
513                 } else {
514                     // We can only handle buffers with memory address so we need to copy if a non direct is
515                     // passed to write.
516                     buf = newDirectBuffer(buf);
517                     assert buf.hasMemoryAddress();
518                 }
519             }
520             return buf;
521         }
522 
523         if (msg instanceof DefaultFileRegion || msg instanceof SpliceOutTask) {
524             return msg;
525         }
526 
527         throw new UnsupportedOperationException(
528                 "unsupported message type: " + StringUtil.simpleClassName(msg) + EXPECTED_TYPES);
529     }

540     @Override
541     protected void doClose() throws Exception {
542         try {
543             ChannelPromise promise = connectPromise;
544             if (promise != null) {
545                 // Use tryFailure() instead of setFailure() to avoid the race against cancel().
546                 promise.tryFailure(CLOSED_CHANNEL_EXCEPTION);
547                 connectPromise = null;
548             }
549 
550             ScheduledFuture<?> future = connectTimeoutFuture;
551             if (future != null) {
552                 future.cancel(false);
553                 connectTimeoutFuture = null;
554             }
555             // Calling super.doClose() first so splceTo(...) will fail on next call.
556             super.doClose();
557         } finally {
558             safeClosePipe(pipeIn);
559             safeClosePipe(pipeOut);
560             clearSpliceQueue();
561         }
562     }
563 
564     private void clearSpliceQueue() {
565         if (spliceQueue == null) {
566             return;
567         }
568         for (;;) {
569             SpliceInTask task = spliceQueue.poll();
570             if (task == null) {
571                 break;
572             }
573             task.promise.tryFailure(CLOSED_CHANNEL_EXCEPTION);
574         }
575     }

577     /**
578      * Connect to the remote peer
579      */
580     protected boolean doConnect(SocketAddress remoteAddress, SocketAddress localAddress) throws Exception {
581         if (localAddress != null) {
582             fd().bind(localAddress);
583         }
584 
585         boolean success = false;
586         try {
587             boolean connected = fd().connect(remoteAddress);
588             if (!connected) {
589                 setFlag(Native.EPOLLOUT);
590             }
591             success = true;
592             return connected;
593         } finally {
594             if (!success) {
595                 doClose();
596             }
597         }
598     }

600     private void safeClosePipe(FileDescriptor fd) {
601         if (fd != null) {
602             try {
603                 fd.close();
604             } catch (IOException e) {
605                 if (logger.isWarnEnabled()) {
606                     logger.warn("Error while closing a pipe", e);
607                 }
608             }
609         }

616         private boolean handleReadException(ChannelPipeline pipeline, ByteBuf byteBuf, Throwable cause, boolean close) {
617             if (byteBuf != null) {
618                 if (byteBuf.isReadable()) {
619                     readPending = false;
620                     pipeline.fireChannelRead(byteBuf);
621                 } else {
622                     byteBuf.release();
623                 }
624             }
625             pipeline.fireChannelReadComplete();
626             pipeline.fireExceptionCaught(cause);
627             if (close || cause instanceof IOException) {
628                 shutdownInput();
629                 return true;
630             }
631             return false;
632         }

634         @Override
635         public void connect(
636                 final SocketAddress remoteAddress, final SocketAddress localAddress, final ChannelPromise promise) {
637             if (!promise.setUncancellable() || !ensureOpen(promise)) {
638                 return;
639             }
640 
641             try {
642                 if (connectPromise != null) {
643                     throw new IllegalStateException("connection attempt already made");
644                 }
645 
646                 boolean wasActive = isActive();
647                 if (doConnect(remoteAddress, localAddress)) {
648                     fulfillConnectPromise(promise, wasActive);
649                 } else {
650                     connectPromise = promise;
651                     requestedRemoteAddress = remoteAddress;
652 
653                     // Schedule connect timeout.
654                     int connectTimeoutMillis = config().getConnectTimeoutMillis();
655                     if (connectTimeoutMillis > 0) {
656                         connectTimeoutFuture = eventLoop().schedule(new OneTimeTask() {
657                             @Override
658                             public void run() {
659                                 ChannelPromise connectPromise = AbstractEpollStreamChannel.this.connectPromise;
660                                 ConnectTimeoutException cause =
661                                         new ConnectTimeoutException("connection timed out: " + remoteAddress);
662                                 if (connectPromise != null && connectPromise.tryFailure(cause)) {
663                                     close(voidPromise());
664                                 }
665                             }
666                         }, connectTimeoutMillis, TimeUnit.MILLISECONDS);
667                     }
668 
669                     promise.addListener(new ChannelFutureListener() {
670                         @Override
671                         public void operationComplete(ChannelFuture future) throws Exception {
672                             if (future.isCancelled()) {
673                                 if (connectTimeoutFuture != null) {
674                                     connectTimeoutFuture.cancel(false);
675                                 }
676                                 connectPromise = null;
677                                 close(voidPromise());
678                             }
679                         }
680                     });
681                 }
682             } catch (Throwable t) {
683                 closeIfClosed();
684                 promise.tryFailure(annotateConnectException(t, remoteAddress));
685             }
686         }

688         private void fulfillConnectPromise(ChannelPromise promise, boolean wasActive) {
689             if (promise == null) {
690                 // Closed via cancellation and the promise has been notified already.
691                 return;
692             }
693             active = true;
694 
695             // trySuccess() will return false if a user cancelled the connection attempt.
696             boolean promiseSet = promise.trySuccess();
697 
698             // Regardless if the connection attempt was cancelled, channelActive() event should be triggered,
699             // because what happened is what happened.
700             if (!wasActive && isActive()) {
701                 pipeline().fireChannelActive();
702             }
703 
704             // If a user cancelled the connection attempt, close the channel, which is followed by channelInactive().
705             if (!promiseSet) {
706                 close(voidPromise());
707             }

710         private void fulfillConnectPromise(ChannelPromise promise, Throwable cause) {
711             if (promise == null) {
712                 // Closed via cancellation and the promise has been notified already.
713                 return;
714             }
715 
716             // Use tryFailure() instead of setFailure() to avoid the race against cancel().
717             promise.tryFailure(cause);
718             closeIfClosed();

721         private void finishConnect() {
722             // Note this method is invoked by the event loop only if the connection attempt was
723             // neither cancelled nor timed out.
724 
725             assert eventLoop().inEventLoop();
726 
727             boolean connectStillInProgress = false;
728             try {
729                 boolean wasActive = isActive();
730                 if (!doFinishConnect()) {
731                     connectStillInProgress = true;
732                     return;
733                 }
734                 fulfillConnectPromise(connectPromise, wasActive);
735             } catch (Throwable t) {
736                 fulfillConnectPromise(connectPromise, annotateConnectException(t, requestedRemoteAddress));
737             } finally {
738                 if (!connectStillInProgress) {
739                     // Check for null as the connectTimeoutFuture is only created if a connectTimeoutMillis > 0 is used
740                     // See https://github.com/netty/netty/issues/1770
741                     if (connectTimeoutFuture != null) {
742                         connectTimeoutFuture.cancel(false);
743                     }
744                     connectPromise = null;
745                 }
746             }
747         }

772         @Override
773         void epollInReady() {
774             if (fd().isInputShutdown()) {
775                 return;
776             }
777             final ChannelConfig config = config();
778             boolean edgeTriggered = isFlagSet(Native.EPOLLET);
779 
780             if (!readPending && !edgeTriggered && !config.isAutoRead()) {
781                 // ChannelConfig.setAutoRead(false) was called in the meantime
782                 clearEpollIn0();
783                 return;
784             }
785 
786             final ChannelPipeline pipeline = pipeline();
787             final ByteBufAllocator allocator = config.getAllocator();
788             RecvByteBufAllocator.Handle allocHandle = this.allocHandle;
789             if (allocHandle == null) {
790                 this.allocHandle = allocHandle = config.getRecvByteBufAllocator().newHandle();
791             }
792 
793             ByteBuf byteBuf = null;
794             boolean close = false;
795             try {
796                 // if edgeTriggered is used we need to read all messages as we are not notified again otherwise.
797                 final int maxMessagesPerRead = edgeTriggered
798                         ? Integer.MAX_VALUE : config.getMaxMessagesPerRead();
799                 int messages = 0;
800                 int totalReadAmount = 0;
801                 do {
802                     if (spliceQueue != null) {
803                         SpliceInTask spliceTask = spliceQueue.peek();
804                         if (spliceTask != null) {
805                             if (spliceTask.spliceIn(allocHandle)) {
806                                 // We need to check if it is still active as if not we removed all SpliceTasks in
807                                 // doClose(...)
808                                 if (isActive()) {
809                                     spliceQueue.remove();
810                                 }
811                                 continue;
812                             } else {
813                                 break;
814                             }
815                         }
816                     }
817 
818                     // we use a direct buffer here as the native implementations only be able
819                     // to handle direct buffers.
820                     byteBuf = allocHandle.allocate(allocator);
821                     int writable = byteBuf.writableBytes();
822                     int localReadAmount = doReadBytes(byteBuf);
823                     if (localReadAmount <= 0) {
824                         // not was read release the buffer
825                         byteBuf.release();
826                         close = localReadAmount < 0;
827                         break;
828                     }
829                     readPending = false;
830                     pipeline.fireChannelRead(byteBuf);
831                     byteBuf = null;
832 
833                     if (totalReadAmount >= Integer.MAX_VALUE - localReadAmount) {
834                         allocHandle.record(totalReadAmount);
835 
836                         // Avoid overflow.
837                         totalReadAmount = localReadAmount;
838                     } else {
839                         totalReadAmount += localReadAmount;
840                     }
841 
842                     if (localReadAmount < writable) {
843                         // Read less than what the buffer can hold,
844                         // which might mean we drained the recv buffer completely.
845                         break;
846                     }
847                     if (!edgeTriggered && !config.isAutoRead()) {
848                         // This is not using EPOLLET so we can stop reading
849                         // ASAP as we will get notified again later with
850                         // pending data
851                         break;
852                     }
853                 } while (++ messages < maxMessagesPerRead || isRdHup());
854 
855                 pipeline.fireChannelReadComplete();
856                 allocHandle.record(totalReadAmount);
857 
858                 if (close) {
859                     shutdownInput();
860                     close = false;
861                 }
862             } catch (Throwable t) {
863                 boolean closed = handleReadException(pipeline, byteBuf, t, close);
864                 if (!closed) {
865                     // trigger a read again as there may be something left to read and because of epoll ET we
866                     // will not get notified again until we read everything from the socket
867                     eventLoop().execute(new OneTimeTask() {
868                         @Override
869                         public void run() {
870                             epollInReady();
871                         }
872                     });
873                 }
874             } finally {
875                 // Check if there is a readPending which was not processed yet.
876                 // This could be for two reasons:
877                 // * The user called Channel.read() or ChannelHandlerContext.read() in channelRead(...) method
878                 // * The user called Channel.read() or ChannelHandlerContext.read() in channelReadComplete(...) method
879                 //
880                 // See https://github.com/netty/netty/issues/2254
881                 if (!readPending && !config.isAutoRead()) {
882                     clearEpollIn0();
883                 }
884             }
885         }
886     }

910         final ChannelPromise promise;
911         int len;

913         protected SpliceInTask(int len, ChannelPromise promise) {
914             this.promise = promise;
915             this.len = len;

924 
925         protected final int spliceIn(FileDescriptor pipeOut, RecvByteBufAllocator.Handle handle) throws IOException {
926             // calculate the maximum amount of data we are allowed to splice
927             int length = Math.min(handle.guess(), len);
928             int splicedIn = 0;
929             for (;;) {
930                 // Splicing until there is nothing left to splice.
931                 int localSplicedIn = Native.splice(fd().intValue(), -1, pipeOut.intValue(), -1, length);
932                 if (localSplicedIn == 0) {
933                     break;
934                 }
935                 splicedIn += localSplicedIn;
936                 length -= localSplicedIn;
937             }
938 
939             // record the number of bytes we spliced before
940             handle.record(splicedIn);
941             return splicedIn;
942         }

946     private final class SpliceInChannelTask extends SpliceInTask implements ChannelFutureListener {
947         private final AbstractEpollStreamChannel ch;
948 
949         SpliceInChannelTask(AbstractEpollStreamChannel ch, int len, ChannelPromise promise) {
950             super(len, promise);
951             this.ch = ch;
952         }
953 
954         @Override
955         public void operationComplete(ChannelFuture future) throws Exception {
956             if (!future.isSuccess()) {
957                 promise.setFailure(future.cause());
958             }
959         }
960 
961         @Override
962         public boolean spliceIn(RecvByteBufAllocator.Handle handle) throws IOException {
963             assert ch.eventLoop().inEventLoop();
964             if (len == 0) {
965                 promise.setSuccess();
966                 return true;
967             }
968             try {
969                 // We create the pipe on the target channel as this will allow us to just handle pending writes
970                 // later in a correct fashion without get into any ordering issues when spliceTo(...) is called
971                 // on multiple Channels pointing to one target Channel.
972                 FileDescriptor pipeOut = ch.pipeOut;
973                 if (pipeOut == null) {
974                     // Create a new pipe as non was created before.
975                     FileDescriptor[] pipe = pipe();
976                     ch.pipeIn = pipe[0];
977                     pipeOut = ch.pipeOut = pipe[1];
978                 }
979 
980                 int splicedIn = spliceIn(pipeOut, handle);
981                 if (splicedIn > 0) {
982                     // Integer.MAX_VALUE is a special value which will result in splice forever.
983                     if (len != Integer.MAX_VALUE) {
984                         len -= splicedIn;
985                     }
986 
987                     // Depending on if we are done with splicing inbound data we set the right promise for the
988                     // outbound splicing.
989                     final ChannelPromise splicePromise;
990                     if (len == 0) {
991                         splicePromise = promise;
992                     } else {
993                         splicePromise = ch.newPromise().addListener(this);
994                     }
995 
996                     boolean autoRead = config().isAutoRead();
997 
998                     // Just call unsafe().write(...) and flush() as we not want to traverse the whole pipeline for this
999                     // case.
1000                     ch.unsafe().write(new SpliceOutTask(ch, splicedIn, autoRead), splicePromise);
1001                     ch.unsafe().flush();
1002                     if (autoRead && !splicePromise.isDone()) {
1003                         // Write was not done which means the target channel was not writable. In this case we need to
1004                         // disable reading until we are done with splicing to the target channel because:
1005                         //
1006                         // - The user may want to to trigger another splice operation once the splicing was complete.
1007                         config().setAutoRead(false);
1008                     }
1009                 }
1010 
1011                 return len == 0;
1012             } catch (Throwable cause) {
1013                 promise.setFailure(cause);
1014                 return true;
1015             }
1016         }
