34  /**
35   * Checks if a thread is alive periodically and runs a task when a thread dies.
36   * <p>
37   * This thread starts a daemon thread to check the state of the threads being watched and to invoke their
38   * associated {@link Runnable}s.  When there is no thread to watch (i.e. all threads are dead), the daemon thread
39   * will terminate itself, and a new daemon thread will be started again when a new watch is added.
40   * </p>
41   */
42  public final class ThreadDeathWatcher {
43  
44      private static final InternalLogger logger = InternalLoggerFactory.getInstance(ThreadDeathWatcher.class);
45      private static final ThreadFactory threadFactory;
46  
47      private static final Queue<Entry> pendingEntries = PlatformDependent.newMpscQueue();
48      private static final Watcher watcher = new Watcher();
49      private static final AtomicBoolean started = new AtomicBoolean();
50      private static volatile Thread watcherThread;
51  
52      static {
53          String poolName = "threadDeathWatcher";
54          String serviceThreadPrefix = SystemPropertyUtil.get("io.netty.serviceThreadPrefix");
55          if (!StringUtil.isNullOrEmpty(serviceThreadPrefix)) {
56              poolName = serviceThreadPrefix + poolName;
57          }
58          threadFactory = new DefaultThreadFactory(poolName, true, Thread.MIN_PRIORITY);
59      }
60  
61      /**
62       * Schedules the specified {@code task} to run when the specified {@code thread} dies.
63       *
64       * @param thread the {@link Thread} to watch
65       * @param task the {@link Runnable} to run when the {@code thread} dies
66       *
67       * @throws IllegalArgumentException if the specified {@code thread} is not alive
68       */
69      public static void watch(Thread thread, Runnable task) {
70          if (thread == null) {
71              throw new NullPointerException("thread");
72          }
73          if (task == null) {
74              throw new NullPointerException("task");
75          }
76          if (!thread.isAlive()) {
77              throw new IllegalArgumentException("thread must be alive.");
78          }
79  
80          schedule(thread, task, true);
81      }
82  
83      /**
84       * Cancels the task scheduled via {@link #watch(Thread, Runnable)}.
85       */
86      public static void unwatch(Thread thread, Runnable task) {
87          if (thread == null) {
88              throw new NullPointerException("thread");
89          }
90          if (task == null) {
91              throw new NullPointerException("task");
92          }
93  
94          schedule(thread, task, false);
95      }
96  
97      private static void schedule(Thread thread, Runnable task, boolean isWatch) {
98          pendingEntries.add(new Entry(thread, task, isWatch));
99  
100         if (started.compareAndSet(false, true)) {
101             Thread watcherThread = threadFactory.newThread(watcher);
102             watcherThread.start();
103             ThreadDeathWatcher.watcherThread = watcherThread;
104         }
105     }
106 
107     /**
108      * Waits until the thread of this watcher has no threads to watch and terminates itself.
109      * Because a new watcher thread will be started again on {@link #watch(Thread, Runnable)},
110      * this operation is only useful when you want to ensure that the watcher thread is terminated
111      * <strong>after</strong> your application is shut down and there's no chance of calling
112      * {@link #watch(Thread, Runnable)} afterwards.
113      *
114      * @return {@code true} if and only if the watcher thread has been terminated
115      */
116     public static boolean awaitInactivity(long timeout, TimeUnit unit) throws InterruptedException {
117         if (unit == null) {
118             throw new NullPointerException("unit");
119         }
120 
121         Thread watcherThread = ThreadDeathWatcher.watcherThread;
122         if (watcherThread != null) {
123             watcherThread.join(unit.toMillis(timeout));
124             return !watcherThread.isAlive();
125         } else {
126             return true;
127         }
128     }
129 
130     private ThreadDeathWatcher() { }
131 
132     private static final class Watcher implements Runnable {
133 
134         private final List<Entry> watchees = new ArrayList<Entry>();
135 
136         @Override
137         public void run() {
138             for (;;) {
139                 fetchWatchees();
140                 notifyWatchees();
141 
142                 // Try once again just in case notifyWatchees() triggered watch() or unwatch().
143                 fetchWatchees();
144                 notifyWatchees();
145 
146                 try {
147                     Thread.sleep(1000);
148                 } catch (InterruptedException ignore) {
149                     // Ignore the interrupt; do not terminate until all tasks are run.
150                 }
151 
152                 if (watchees.isEmpty() && pendingEntries.isEmpty()) {
153 
154                     // Mark the current worker thread as stopped.
155                     // The following CAS must always success and must be uncontended,
156                     // because only one watcher thread should be running at the same time.
157                     boolean stopped = started.compareAndSet(true, false);
158                     assert stopped;
159 
160                     // Check if there are pending entries added by watch() while we do CAS above.
161                     if (pendingEntries.isEmpty()) {
162                         // A) watch() was not invoked and thus there's nothing to handle
163                         //    -> safe to terminate because there's nothing left to do
164                         // B) a new watcher thread started and handled them all
165                         //    -> safe to terminate the new watcher thread will take care the rest
166                         break;
167                     }
168 
169                     // There are pending entries again, added by watch()
170                     if (!started.compareAndSet(false, true)) {
171                         // watch() started a new watcher thread and set 'started' to true.
172                         // -> terminate this thread so that the new watcher reads from pendingEntries exclusively.
173                         break;
174                     }
175 
176                     // watch() added an entry, but this worker was faster to set 'started' to true.
177                     // i.e. a new watcher thread was not started
178                     // -> keep this thread alive to handle the newly added entries.
179                 }
180             }
181         }
182 
183         private void fetchWatchees() {
184             for (;;) {
185                 Entry e = pendingEntries.poll();
186                 if (e == null) {
187                     break;
188                 }
189 
190                 if (e.isWatch) {
191                     watchees.add(e);
192                 } else {
193                     watchees.remove(e);
194                 }
195             }
196         }
197 
198         private void notifyWatchees() {
199             List<Entry> watchees = this.watchees;
200             for (int i = 0; i < watchees.size();) {
201                 Entry e = watchees.get(i);
202                 if (!e.thread.isAlive()) {
203                     watchees.remove(i);
204                     try {
205                         e.task.run();
206                     } catch (Throwable t) {
207                         logger.warn("Thread death watcher task raised an exception:", t);
208                     }
209                 } else {
210                     i ++;
211                 }
212             }
213         }
214     }
215 
216     private static final class Entry extends MpscLinkedQueueNode<Entry> {
217         final Thread thread;
218         final Runnable task;
219         final boolean isWatch;
220 
221         Entry(Thread thread, Runnable task, boolean isWatch) {
222             this.thread = thread;
223             this.task = task;
224             this.isWatch = isWatch;
225         }
226 
227         @Override
228         public Entry value() {
229             return this;
230         }
231 
232         @Override
233         public int hashCode() {
234             return thread.hashCode() ^ task.hashCode();
235         }
236 
237         @Override
238         public boolean equals(Object obj) {
239             if (obj == this) {
240                 return true;
241             }
242 
243             if (!(obj instanceof Entry)) {
244                 return false;
245             }
246 
247             Entry that = (Entry) obj;
248             return thread == that.thread && task == that.task;
249         }
250     }
