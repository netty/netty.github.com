539     private static boolean isRoot0() {
540         if (isWindows()) {
541             return false;
542         }
543 
544         String[] ID_COMMANDS = { "/usr/bin/id", "/bin/id", "/usr/xpg4/bin/id", "id"};
545         Pattern UID_PATTERN = Pattern.compile("^(?:0|[1-9][0-9]*)$");
546         for (String idCmd: ID_COMMANDS) {
547             Process p = null;
548             BufferedReader in = null;
549             String uid = null;
550             try {
551                 p = Runtime.getRuntime().exec(new String[] { idCmd, "-u" });
552                 in = new BufferedReader(new InputStreamReader(p.getInputStream(), CharsetUtil.US_ASCII));
553                 uid = in.readLine();
554                 in.close();
555 
556                 for (;;) {
557                     try {
558                         int exitCode = p.waitFor();
559                         if (exitCode != 0) {
560                             uid = null;
561                         }
562                         break;
563                     } catch (InterruptedException e) {
564                         // Ignore
565                     }
566                 }
567             } catch (Throwable ignored) {
568                 // Failed to run the command.
569                 uid = null;
570             } finally {
571                 if (in != null) {
572                     try {
573                         in.close();
574                     } catch (IOException e) {
575                         // Ignore
576                     }
577                 }
578                 if (p != null) {
579                     try {
580                         p.destroy();
581                     } catch (Exception e) {
582                         // Android sometimes triggers an ErrnoException.
583                     }
584                 }
585             }
586 
587             if (uid != null && UID_PATTERN.matcher(uid).matches()) {
588                 logger.debug("UID: {}", uid);
589                 return "0".equals(uid);
590             }
591         }
592 
593         logger.debug("Could not determine the current UID using /usr/bin/id; attempting to bind at privileged ports.");
594 
595         Pattern PERMISSION_DENIED = Pattern.compile(".*(?:denied|not.*permitted).*");
596         for (int i = 1023; i > 0; i --) {
597             ServerSocket ss = null;
598             try {
599                 ss = new ServerSocket();
600                 ss.setReuseAddress(true);
601                 ss.bind(new InetSocketAddress(i));
602                 if (logger.isDebugEnabled()) {
603                     logger.debug("UID: 0 (succeded to bind at port {})", i);
604                 }
605                 return true;
606             } catch (Exception e) {
607                 // Failed to bind.
608                 // Check the error message so that we don't always need to bind 1023 times.
609                 String message = e.getMessage();
610                 if (message == null) {
611                     message = "";
612                 }
613                 message = message.toLowerCase();
614                 if (PERMISSION_DENIED.matcher(message).matches()) {
615                     break;
616                 }
617             } finally {
618                 if (ss != null) {
619                     try {
620                         ss.close();
621                     } catch (Exception e) {
622                         // Ignore.
623                     }
624                 }
625             }
626         }
627 
628         logger.debug("UID: non-root (failed to bind at any privileged ports)");
629         return false;
630     }

715     private static long maxDirectMemory0() {
716         long maxDirectMemory = 0;
717         try {
718             // Try to get from sun.misc.VM.maxDirectMemory() which should be most accurate.
719             Class<?> vmClass = Class.forName("sun.misc.VM", true, getSystemClassLoader());
720             Method m = vmClass.getDeclaredMethod("maxDirectMemory");
721             maxDirectMemory = ((Number) m.invoke(null)).longValue();
722         } catch (Throwable ignored) {
723             // Ignore
724         }
725 
726         if (maxDirectMemory > 0) {
727             return maxDirectMemory;
728         }
