282         synchronized (this) {
283             while (!isDone()) {
284                 checkDeadLock();
285                 incWaiters();
286                 try {

324     private boolean await0(long timeoutNanos, boolean interruptable) throws InterruptedException {
325         if (isDone()) {
326             return true;
327         }
328 
329         if (timeoutNanos <= 0) {
330             return isDone();
331         }
332 
333         if (interruptable && Thread.interrupted()) {
334             throw new InterruptedException(toString());
335         }
336 
337         long startTime = System.nanoTime();
338         long waitTime = timeoutNanos;
339         boolean interrupted = false;
340 
341         try {
342             synchronized (this) {
343                 if (isDone()) {
344                     return true;
345                 }
346 
347                 if (waitTime <= 0) {
348                     return isDone();
349                 }
350 
351                 checkDeadLock();
352                 incWaiters();
353                 try {
354                     for (;;) {
355                         try {
356                             wait(waitTime / 1000000, (int) (waitTime % 1000000));
357                         } catch (InterruptedException e) {
358                             if (interruptable) {
359                                 throw e;
360                             } else {
361                                 interrupted = true;
362                             }
363                         }
364 
365                         if (isDone()) {
366                             return true;
367                         } else {
368                             waitTime = timeoutNanos - (System.nanoTime() - startTime);
369                             if (waitTime <= 0) {
370                                 return isDone();
371                             }
372                         }
373                     }
374                 } finally {
375                     decWaiters();
376                 }
377             }
378         } finally {
379             if (interrupted) {
380                 Thread.currentThread().interrupt();
381             }
382         }
383     }

544     private void notifyListeners() {
545         // This method doesn't need synchronization because:
546         // 1) This method is always called after synchronized (this) block.
547         //    Hence any listener list modification happens-before this method.
548         // 2) This method is called only when 'done' is true.  Once 'done'
549         //    becomes true, the listener list is never modified - see add/removeListener()
550 
551         Object listeners = this.listeners;
552         if (listeners == null) {
553             return;
554         }
555 
556         EventExecutor executor = executor();
557         if (executor.inEventLoop()) {
558             final InternalThreadLocalMap threadLocals = InternalThreadLocalMap.get();
559             final int stackDepth = threadLocals.futureListenerStackDepth();
560             if (stackDepth < MAX_LISTENER_STACK_DEPTH) {
561                 threadLocals.setFutureListenerStackDepth(stackDepth + 1);
562                 try {
563                     if (listeners instanceof DefaultFutureListeners) {
564                         notifyListeners0(this, (DefaultFutureListeners) listeners);
565                     } else {
566                         final GenericFutureListener<? extends Future<V>> l =
567                                 (GenericFutureListener<? extends Future<V>>) listeners;
568                         notifyListener0(this, l);
569                     }
570                 } finally {
571                     this.listeners = null;
572                     threadLocals.setFutureListenerStackDepth(stackDepth);
573                 }
574                 return;
575             }
576         }
577 
578         if (listeners instanceof DefaultFutureListeners) {
579             final DefaultFutureListeners dfl = (DefaultFutureListeners) listeners;
580             execute(executor, new OneTimeTask() {
581                 @Override
582                 public void run() {
583                     notifyListeners0(DefaultPromise.this, dfl);
584                     DefaultPromise.this.listeners = null;
585                 }
586             });
587         } else {
588             final GenericFutureListener<? extends Future<V>> l =
589                     (GenericFutureListener<? extends Future<V>>) listeners;
590             execute(executor, new OneTimeTask() {
591                 @Override
592                 public void run() {
593                     notifyListener0(DefaultPromise.this, l);
594                     DefaultPromise.this.listeners = null;
595                 }
596             });
597         }
598     }

608     /**
609      * Notifies the specified listener which were added after this promise is already done.
610      * This method ensures that the specified listener is not notified until {@link #listeners} becomes {@code null}
611      * to avoid the case where the late listeners are notified even before the early listeners are notified.
612      */
613     private void notifyLateListener(final GenericFutureListener<?> l) {
614         final EventExecutor executor = executor();
615         if (executor.inEventLoop()) {
616             // Execute immediately if late listeners is empty. This allows subsequent late listeners
617             // that are added after completion to be notified immediately and preserver order.
618             if (listeners == null && (lateListeners == null || lateListeners.isEmpty())) {
619                 final InternalThreadLocalMap threadLocals = InternalThreadLocalMap.get();
620                 final int stackDepth = threadLocals.futureListenerStackDepth();
621                 if (stackDepth < MAX_LISTENER_STACK_DEPTH) {
622                     threadLocals.setFutureListenerStackDepth(stackDepth + 1);
623                     try {
624                         notifyListener0(this, l);
625                     } finally {
626                         threadLocals.setFutureListenerStackDepth(stackDepth);
627                     }
628                     return;
629                 }
630             } else {
631                 LateListeners lateListeners = this.lateListeners;
632                 if (lateListeners == null) {
633                     this.lateListeners = lateListeners = new LateListeners();
634                 }
635                 lateListeners.add(l);
636                 execute(executor, lateListeners);
637                 return;
638             }
639         }
640 
641         // Add the late listener to lateListeners in the executor thread for thread safety.
642         // We could just make LateListeners extend ConcurrentLinkedQueue, but it's an overkill considering
643         // that most asynchronous applications won't execute this code path.
644         execute(executor, new LateListenerNotifier(l));
645     }

647     protected static void notifyListener(
648             final EventExecutor eventExecutor, final Future<?> future, final GenericFutureListener<?> l) {
649 
650         if (eventExecutor.inEventLoop()) {
651             final InternalThreadLocalMap threadLocals = InternalThreadLocalMap.get();
652             final int stackDepth = threadLocals.futureListenerStackDepth();
653             if (stackDepth < MAX_LISTENER_STACK_DEPTH) {
654                 threadLocals.setFutureListenerStackDepth(stackDepth + 1);
655                 try {
656                     notifyListener0(future, l);
657                 } finally {
658                     threadLocals.setFutureListenerStackDepth(stackDepth);
659                 }
660                 return;
661             }
662         }
663 
664         execute(eventExecutor, new OneTimeTask() {
665             @Override
666             public void run() {
667                 notifyListener0(future, l);
668             }
669         });
670     }

691     /**
692      * Returns a {@link GenericProgressiveFutureListener}, an array of {@link GenericProgressiveFutureListener}, or
693      * {@code null}.
694      */
695     private synchronized Object progressiveListeners() {
696         Object listeners = this.listeners;
697         if (listeners == null) {
698             // No listeners added
699             return null;
700         }
701 
702         if (listeners instanceof DefaultFutureListeners) {
703             // Copy DefaultFutureListeners into an array of listeners.
704             DefaultFutureListeners dfl = (DefaultFutureListeners) listeners;
705             int progressiveSize = dfl.progressiveSize();
706             switch (progressiveSize) {
707                 case 0:
708                     return null;
709                 case 1:
710                     for (GenericFutureListener<?> l: dfl.listeners()) {
711                         if (l instanceof GenericProgressiveFutureListener) {
712                             return l;
713                         }
714                     }
715                     return null;
716             }
717 
718             GenericFutureListener<?>[] array = dfl.listeners();
719             GenericProgressiveFutureListener<?>[] copy = new GenericProgressiveFutureListener[progressiveSize];
720             for (int i = 0, j = 0; j < progressiveSize; i ++) {
721                 GenericFutureListener<?> l = array[i];
722                 if (l instanceof GenericProgressiveFutureListener) {
723                     copy[j ++] = (GenericProgressiveFutureListener<?>) l;
724                 }
725             }
726 
727             return copy;
728         } else if (listeners instanceof GenericProgressiveFutureListener) {
729             return listeners;
730         } else {
731             // Only one listener was added and it's not a progressive listener.
732             return null;
733         }
734     }

738         final Object listeners = progressiveListeners();
739         if (listeners == null) {

741         }
742

754         } else {

764             } else {

766                         (GenericProgressiveFutureListener<ProgressiveFuture<V>>) listeners;

811     protected StringBuilder toStringBuilder() {
812         StringBuilder buf = new StringBuilder(64)
813             .append(StringUtil.simpleClassName(this))
814             .append('@')
815             .append(Integer.toHexString(hashCode()));
816 
817         Object result = this.result;
818         if (result == SUCCESS) {
819             buf.append("(success)");
820         } else if (result == UNCANCELLABLE) {
821             buf.append("(uncancellable)");
822         } else if (result instanceof CauseHolder) {
823             buf.append("(failure: ")
824                .append(((CauseHolder) result).cause)
825                .append(')');
