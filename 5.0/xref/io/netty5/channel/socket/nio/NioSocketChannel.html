52  public class NioSocketChannel
53          extends AbstractNioByteChannel<NioServerSocketChannel, InetSocketAddress, InetSocketAddress>

55      private static final SelectorProvider DEFAULT_SELECTOR_PROVIDER = SelectorProvider.provider();

57      private static final Method OPEN_SOCKET_CHANNEL_WITH_FAMILY =
58              SelectorProviderUtil.findOpenMethod("openSocketChannel");
59  
60      private static SocketChannel newChannel(SelectorProvider provider, InternetProtocolFamily family) {
61          try {
62              SocketChannel channel = SelectorProviderUtil.newChannel(OPEN_SOCKET_CHANNEL_WITH_FAMILY, provider, family);
63              return channel == null ? provider.openSocketChannel() : channel;
64          } catch (IOException e) {
65              throw new ChannelException("Failed to open a socket.", e);
66          }
67      }

127     @Override
128     public boolean isShutdown(ChannelShutdownDirection direction) {
129         if (!isActive()) {
130             return true;
131         }
132         switch (direction) {
133             case Outbound:
134                 return javaChannel().socket().isOutputShutdown();
135             case Inbound:
136                 return javaChannel().socket().isInputShutdown();
137             default:
138                 throw new AssertionError();
139         }
140     }

142     @Override
143     protected void doShutdown(ChannelShutdownDirection direction) throws Exception {
144         switch (direction) {
145             case Inbound:
146                 javaChannel().shutdownInput();
147                 break;
148             case Outbound:
149                 javaChannel().shutdownOutput();
150                 break;
151             default:
152                 throw new AssertionError();
153         }
154     }

171     private void doBind0(SocketAddress localAddress) throws Exception {
172         SocketUtils.bind(javaChannel(), localAddress);
173     }

175     @Override
176     protected boolean doConnect(SocketAddress remoteAddress, SocketAddress localAddress) throws Exception {
177         if (localAddress != null) {
178             doBind0(localAddress);
179         }
180 
181         boolean success = false;
182         try {
183             boolean connected = SocketUtils.connect(javaChannel(), remoteAddress);
184             if (!connected) {
185                 selectionKey().interestOps(SelectionKey.OP_CONNECT);
186             }
187             success = true;
188             return connected;
189         } finally {
190             if (!success) {
191                 doClose();
192             }
193         }
194     }
195 
196     @Override
197     protected boolean doFinishConnect(InetSocketAddress requestedRemoteAddress) throws Exception {
198         return javaChannel().finishConnect();
199     }

202     protected void doDisconnect() throws Exception {
203         doClose();

206     @Override
207     protected int doReadBytes(Buffer buffer) throws Exception {
208         final RecvBufferAllocator.Handle allocHandle = recvBufAllocHandle();
209         allocHandle.attemptedBytesRead(buffer.writableBytes());
210         return buffer.transferFrom(javaChannel(), allocHandle.attemptedBytesRead());
211     }

213     @Override
214     protected int doWriteBytes(Buffer buf) throws Exception {
215         final int expectedWrittenBytes = buf.readableBytes();
216         return buf.transferTo(javaChannel(), expectedWrittenBytes);
217     }

219     @Override
220     protected long doWriteFileRegion(FileRegion region) throws Exception {
221         final long position = region.transferred();
222         return region.transferTo(javaChannel(), position);
223     }

225     private void adjustMaxBytesPerGatheringWrite(int attempted, int written, int oldMaxBytesPerGatheringWrite) {
226         // By default we track the SO_SNDBUF when ever it is explicitly set. However some OSes may dynamically change
227         // SO_SNDBUF (and other characteristics that determine how much data can be written at once) so we should try
228         // make a best effort to adjust as OS behavior changes.
229         if (attempted == written) {
230             if (attempted << 1 > oldMaxBytesPerGatheringWrite) {
231                 ((NioSocketChannelConfig) config).setMaxBytesPerGatheringWrite(attempted << 1);
232             }
233         } else if (attempted > MAX_BYTES_PER_GATHERING_WRITE_ATTEMPTED_LOW_THRESHOLD && written < attempted >>> 1) {
234             ((NioSocketChannelConfig) config).setMaxBytesPerGatheringWrite(attempted >>> 1);
235         }
236     }
237 
238     @Override
239     protected void doWrite(ChannelOutboundBuffer in) throws Exception {
240         SocketChannel ch = javaChannel();
241         int writeSpinCount = config().getWriteSpinCount();
242         do {
243             if (in.isEmpty()) {
244                 // All written so clear OP_WRITE
245                 clearOpWrite();
246                 // Directly return here so incompleteWrite(...) is not called.
247                 return;
248             }
249 
250             // Ensure the pending writes are made of ByteBufs only.
251             int maxBytesPerGatheringWrite = ((NioSocketChannelConfig) config).getMaxBytesPerGatheringWrite();
252             ByteBuffer[] nioBuffers = in.nioBuffers(1024, maxBytesPerGatheringWrite);
253             int nioBufferCnt = in.nioBufferCount();
254 
255             // Always use nioBuffers() to workaround data-corruption.
256             // See https://github.com/netty/netty/issues/2761
257             switch (nioBufferCnt) {
258                 case 0:
259                     // We have something else beside ByteBuffers to write so fallback to normal writes.
260                     writeSpinCount -= doWrite0(in);
261                     break;
262                 case 1: {
263                     // Only one ByteBuf so use non-gathering write
264                     // Zero length buffers are not added to nioBuffers by ChannelOutboundBuffer, so there is no need
265                     // to check if the total size of all the buffers is non-zero.
266                     ByteBuffer buffer = nioBuffers[0];
267                     int attemptedBytes = buffer.remaining();
268                     final int localWrittenBytes = ch.write(buffer);
269                     if (localWrittenBytes <= 0) {
270                         incompleteWrite(true);
271                         return;
272                     }
273                     adjustMaxBytesPerGatheringWrite(attemptedBytes, localWrittenBytes, maxBytesPerGatheringWrite);
274                     in.removeBytes(localWrittenBytes);
275                     --writeSpinCount;
276                     break;
277                 }
278                 default: {
279                     // Zero length buffers are not added to nioBuffers by ChannelOutboundBuffer, so there is no need
280                     // to check if the total size of all the buffers is non-zero.
281                     // We limit the max amount to int above so cast is safe
282                     long attemptedBytes = in.nioBufferSize();
283                     final long localWrittenBytes = ch.write(nioBuffers, 0, nioBufferCnt);
284                     if (localWrittenBytes <= 0) {
285                         incompleteWrite(true);
286                         return;
287                     }
288                     // Casting to int is safe because we limit the total amount of data in the nioBuffers to int above.
289                     adjustMaxBytesPerGatheringWrite((int) attemptedBytes, (int) localWrittenBytes,
290                             maxBytesPerGatheringWrite);
291                     in.removeBytes(localWrittenBytes);
292                     --writeSpinCount;
293                     break;
294                 }
295             }
296         } while (writeSpinCount > 0);
297 
298         incompleteWrite(writeSpinCount < 0);
299     }
300 
301     @Override
302     protected Future<Executor> prepareToClose() {
303         try {
304             if (javaChannel().isOpen() && config().getSoLinger() > 0) {
305                 // We need to cancel this key of the channel so we may not end up in a eventloop spin
306                 // because we try to read or write until the actual close happens which may be later due
307                 // SO_LINGER handling.
308                 // See https://github.com/netty/netty/issues/4449
309                 return executor().deregisterForIo(this).map(v -> GlobalEventExecutor.INSTANCE);
310             }
311         } catch (Throwable ignore) {
312             // Ignore the error as the underlying channel may be closed in the meantime and so
313             // getSoLinger() may produce an exception. In this case we just return null.
314             // See https://github.com/netty/netty/issues/4449
315         }
316         return null;
317     }

319     private final class NioSocketChannelConfig extends DefaultSocketChannelConfig {
320         private volatile int maxBytesPerGatheringWrite = Integer.MAX_VALUE;
321         private NioSocketChannelConfig(NioSocketChannel channel, Socket javaSocket) {
322             super(channel, javaSocket);
323             calculateMaxBytesPerGatheringWrite();
324         }
325 
326         @Override
327         protected void autoReadCleared() {
328             clearReadPending();
329         }
330 
331         @Override
332         public NioSocketChannelConfig setSendBufferSize(int sendBufferSize) {
333             super.setSendBufferSize(sendBufferSize);
334             calculateMaxBytesPerGatheringWrite();
335             return this;
336         }
337 
338         @Override
339         public <T> boolean setOption(ChannelOption<T> option, T value) {
340             if (option instanceof NioChannelOption) {
341                 return NioChannelOption.setOption(jdkChannel(), (NioChannelOption<T>) option, value);
342             }
343             return super.setOption(option, value);
344         }
345 
346         @Override
347         public <T> T getOption(ChannelOption<T> option) {
348             if (option instanceof NioChannelOption) {
349                 return NioChannelOption.getOption(jdkChannel(), (NioChannelOption<T>) option);
350             }
351             return super.getOption(option);
352         }
353 
354         @Override
355         public Map<ChannelOption<?>, Object> getOptions() {
356             return getOptions(super.getOptions(), NioChannelOption.getOptions(jdkChannel()));
357         }
358 
359         void setMaxBytesPerGatheringWrite(int maxBytesPerGatheringWrite) {
360             this.maxBytesPerGatheringWrite = maxBytesPerGatheringWrite;
361         }
362 
363         int getMaxBytesPerGatheringWrite() {
364             return maxBytesPerGatheringWrite;
365         }
366 
367         private void calculateMaxBytesPerGatheringWrite() {
368             // Multiply by 2 to give some extra space in case the OS can process write data faster than we can provide.
369             int newSendBufferSize = getSendBufferSize() << 1;
370             if (newSendBufferSize > 0) {
371                 setMaxBytesPerGatheringWrite(newSendBufferSize);
372             }
373         }
374 
375         private SocketChannel jdkChannel() {
376             return ((NioSocketChannel) channel).javaChannel();
377         }
