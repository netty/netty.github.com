241     /**
242      * Constructor with full custom configuration.
243      *
244      * @param compressionLevel
245      *            Compression level between 0 and 9 (default is 6).
246      * @param allowServerWindowSize
247      *            allows WebSocket client to customize the server inflater window size
248      *            (default is false).
249      * @param preferredClientWindowSize
250      *            indicates the preferred client window size to use if client inflater is customizable.
251      * @param allowServerNoContext
252      *            allows WebSocket client to activate server_no_context_takeover
253      *            (default is false).
254      * @param preferredClientNoContext
255      *            indicates if server prefers to activate client_no_context_takeover
256      *            if client is compatible with (default is false).
257      * @param serverWindowSize
258      *            upper bound (in bits, 8-15) on the server-side deflate window. If the client offers
259      *            {@code server_max_window_bits=N}, the negotiated value is {@code min(N, serverWindowSize)}.
260      *            Per <a href="https://tools.ietf.org/html/rfc7692#section-7.1.2.1">RFC 7692 §7.1.2.1</a>
261      *            the server may include {@code server_max_window_bits} with the same or smaller value than the
262      *            offer; this handshaker also advertises it unilaterally when {@code serverWindowSize} is less
263      *            than {@link #MAX_WINDOW_SIZE}.
264      * @param memLevel
265      *            zlib memory level for the server-side deflater (1-9). Lower values reduce per-connection
266      *            memory at the cost of compression ratio.
267      * @param extensionFilterProvider
268      *            provides server extension filters for per message deflate encoder and decoder.
269      * @param maxAllocation
270      *            Maximum size of the decompression buffer. Must be &gt;= 0. If zero, maximum size is not limited.
271      */
272     public PerMessageDeflateServerExtensionHandshaker(int compressionLevel, boolean allowServerWindowSize,
273             int preferredClientWindowSize,
274             boolean allowServerNoContext, boolean preferredClientNoContext,
275             int serverWindowSize, int memLevel,
276             WebSocketExtensionFilterProvider extensionFilterProvider,
277             int maxAllocation) {
278         if (preferredClientWindowSize > MAX_WINDOW_SIZE || preferredClientWindowSize < MIN_WINDOW_SIZE) {
279             throw new IllegalArgumentException(
280                     "preferredServerWindowSize: " + preferredClientWindowSize + " (expected: 8-15)");
281         }
282         if (compressionLevel < 0 || compressionLevel > 9) {
283             throw new IllegalArgumentException(
284                     "compressionLevel: " + compressionLevel + " (expected: 0-9)");
285         }
286         this.compressionLevel = compressionLevel;
287         this.allowServerWindowSize = allowServerWindowSize;
288         this.preferredClientWindowSize = preferredClientWindowSize;
289         this.allowServerNoContext = allowServerNoContext;
290         this.preferredClientNoContext = preferredClientNoContext;
291         this.serverWindowSize = checkInRange(serverWindowSize, MIN_WINDOW_SIZE, MAX_WINDOW_SIZE, "serverWindowSize");
292         this.memLevel = checkInRange(memLevel, MIN_MEM_LEVEL, MAX_MEM_LEVEL, "memLevel");
293         this.extensionFilterProvider = checkNotNull(extensionFilterProvider, "extensionFilterProvider");
294         this.maxAllocation = checkPositiveOrZero(maxAllocation, "maxAllocation");
295     }

297     @Override
298     public WebSocketServerExtension handshakeExtension(WebSocketExtensionData extensionData) {
299         if (!PERMESSAGE_DEFLATE_EXTENSION.equals(extensionData.name())) {
300             return null;
301         }
302 
303         boolean deflateEnabled = true;
304         int clientWindowSize = MAX_WINDOW_SIZE;
305         int negotiatedServerWindowSize = this.serverWindowSize;
306         boolean serverNoContext = false;
307         boolean clientNoContext = false;
308 
309         Iterator<Entry<String, String>> parametersIterator =
310                 extensionData.parameters().entrySet().iterator();
311         while (deflateEnabled && parametersIterator.hasNext()) {
312             Entry<String, String> parameter = parametersIterator.next();
313 
314             if (CLIENT_MAX_WINDOW.equalsIgnoreCase(parameter.getKey())) {
315                 // RFC 7692: client_max_window_bits may have a value or no value
316                 String value = parameter.getValue();
317                 if (value != null) {
318                     // Let NumberFormatException bubble up if value is invalid
319                     clientWindowSize = Integer.parseInt(value);
320                     if (clientWindowSize > MAX_WINDOW_SIZE || clientWindowSize < MIN_WINDOW_SIZE) {
321                         deflateEnabled = false;
322                     }
323                 } else {
324                     // No value specified, use preferred client window size
325                     clientWindowSize = preferredClientWindowSize;
326                 }
327             } else if (SERVER_MAX_WINDOW.equalsIgnoreCase(parameter.getKey())) {
328                 // use provided windowSize if it is allowed
329                 if (allowServerWindowSize) {
330                     int clientOfferedServerWindowSize = Integer.parseInt(parameter.getValue());
331                     if (clientOfferedServerWindowSize > MAX_WINDOW_SIZE
332                             || clientOfferedServerWindowSize < MIN_WINDOW_SIZE) {
333                         deflateEnabled = false;
334                     } else {
335                         // RFC 7692 §7.1.2.1: server accepts with the same or smaller value than the offer.
336                         // Cap at the configured serverWindowSize so the server's memory bound is respected.
337                         negotiatedServerWindowSize = Math.min(clientOfferedServerWindowSize, this.serverWindowSize);
338                     }
339                 } else {
340                     deflateEnabled = false;
341                 }
342             } else if (CLIENT_NO_CONTEXT.equalsIgnoreCase(parameter.getKey())) {
343                 // use preferred clientNoContext because client is compatible with customization
344                 clientNoContext = preferredClientNoContext;
345             } else if (SERVER_NO_CONTEXT.equalsIgnoreCase(parameter.getKey())) {
346                 // use server no context if allowed
347                 if (allowServerNoContext) {
348                     serverNoContext = true;
349                 } else {
350                     deflateEnabled = false;
351                 }
352             } else {
353                 // unknown parameter
354                 deflateEnabled = false;
355             }
356         }
357 
358         if (deflateEnabled) {
359             return new PermessageDeflateExtension(compressionLevel, serverNoContext,
360                     negotiatedServerWindowSize, memLevel, clientNoContext, clientWindowSize,
361                     extensionFilterProvider, maxAllocation);
362         } else {
363             return null;
364         }
365     }

367     private static class PermessageDeflateExtension implements WebSocketServerExtension {
368 
369         private final int compressionLevel;
370         private final boolean serverNoContext;
371         private final int serverWindowSize;
372         private final int memLevel;
373         private final boolean clientNoContext;
374         private final int clientWindowSize;
375         private final WebSocketExtensionFilterProvider extensionFilterProvider;
376         private final int maxAllocation;
377 
378         PermessageDeflateExtension(int compressionLevel, boolean serverNoContext,
379                 int serverWindowSize, int memLevel, boolean clientNoContext, int clientWindowSize,
380                 WebSocketExtensionFilterProvider extensionFilterProvider, int maxAllocation) {
381             this.compressionLevel = compressionLevel;
382             this.serverNoContext = serverNoContext;
383             this.serverWindowSize = serverWindowSize;
384             this.memLevel = memLevel;
385             this.clientNoContext = clientNoContext;
386             this.clientWindowSize = clientWindowSize;
387             this.extensionFilterProvider = extensionFilterProvider;
388             this.maxAllocation = maxAllocation;
389         }
390 
391         @Override
392         public int rsv() {
393             return RSV1;
394         }
395 
396         @Override
397         public WebSocketExtensionEncoder newExtensionEncoder() {
398             return new PerMessageDeflateEncoder(compressionLevel, serverWindowSize, memLevel, serverNoContext,
399                                                 extensionFilterProvider.encoderFilter());
400         }
401 
402         @Override
403         public WebSocketExtensionDecoder newExtensionDecoder() {
404             return new PerMessageDeflateDecoder(clientNoContext, extensionFilterProvider.decoderFilter(),
405                                                 maxAllocation);
406         }
407 
408         @Override
409         public WebSocketExtensionData newReponseData() {
410             HashMap<String, String> parameters = new HashMap<String, String>(4);
411             if (serverNoContext) {
412                 parameters.put(SERVER_NO_CONTEXT, null);
413             }
414             if (clientNoContext) {
415                 parameters.put(CLIENT_NO_CONTEXT, null);
416             }
417             if (serverWindowSize != MAX_WINDOW_SIZE) {
418                 parameters.put(SERVER_MAX_WINDOW, Integer.toString(serverWindowSize));
419             }
420             if (clientWindowSize != MAX_WINDOW_SIZE) {
421                 parameters.put(CLIENT_MAX_WINDOW, Integer.toString(clientWindowSize));
422             }
423             return new WebSocketExtensionData(PERMESSAGE_DEFLATE_EXTENSION, parameters);
424         }
425     }
