17  package io.netty.resolver;
18  
19  import io.netty.util.concurrent.EventExecutor;
20  import io.netty.util.concurrent.Future;
21  import io.netty.util.concurrent.FutureListener;
22  import io.netty.util.internal.logging.InternalLogger;
23  import io.netty.util.internal.logging.InternalLoggerFactory;
24  
25  import java.io.Closeable;
26  import java.net.SocketAddress;
27  import java.util.IdentityHashMap;
28  import java.util.Map;
29  import java.util.concurrent.ConcurrentMap;
30  
31  /**
32   * Creates and manages {@link NameResolver}s so that each {@link EventExecutor} has its own resolver instance.
33   */
34  public abstract class AddressResolverGroup<T extends SocketAddress> implements Closeable {
35  
36      private static final InternalLogger logger = InternalLoggerFactory.getInstance(AddressResolverGroup.class);
37  
38      /**
39       * Note that we do not use a {@link ConcurrentMap} here because it is usually expensive to instantiate a resolver.
40       */
41      private final Map<EventExecutor, AddressResolver<T>> resolvers =
42              new IdentityHashMap<EventExecutor, AddressResolver<T>>();
43  
44      protected AddressResolverGroup() { }
45  
46      /**
47       * Returns the {@link AddressResolver} associated with the specified {@link EventExecutor}. If there's no associated
48       * resolved found, this method creates and returns a new resolver instance created by
49       * {@link #newResolver(EventExecutor)} so that the new resolver is reused on another
50       * {@link #getResolver(EventExecutor)} call with the same {@link EventExecutor}.
51       */
52      public AddressResolver<T> getResolver(final EventExecutor executor) {
53          if (executor == null) {
54              throw new NullPointerException("executor");
55          }
56  
57          if (executor.isShuttingDown()) {
58              throw new IllegalStateException("executor not accepting a task");
59          }
60  
61          AddressResolver<T> r;
62          synchronized (resolvers) {
63              r = resolvers.get(executor);
64              if (r == null) {
65                  final AddressResolver<T> newResolver;
66                  try {
67                      newResolver = newResolver(executor);
68                  } catch (Exception e) {
69                      throw new IllegalStateException("failed to create a new resolver", e);
70                  }
71  
72                  resolvers.put(executor, newResolver);
73                  executor.terminationFuture().addListener(new FutureListener<Object>() {
74                      @Override
75                      public void operationComplete(Future<Object> future) throws Exception {
76                          synchronized (resolvers) {
77                              resolvers.remove(executor);
78                          }
79                          newResolver.close();
80                      }
81                  });
82  
83                  r = newResolver;
84              }
85          }
86  
87          return r;
88      }
89  
90      /**
91       * Invoked by {@link #getResolver(EventExecutor)} to create a new {@link AddressResolver}.
92       */
93      protected abstract AddressResolver<T> newResolver(EventExecutor executor) throws Exception;
94  
95      /**
96       * Closes all {@link NameResolver}s created by this group.
97       */
98      @Override
99      @SuppressWarnings({ "unchecked", "SuspiciousToArrayCall" })
100     public void close() {
101         final AddressResolver<T>[] rArray;
102         synchronized (resolvers) {
103             rArray = (AddressResolver<T>[]) resolvers.values().toArray(new AddressResolver[resolvers.size()]);
104             resolvers.clear();
105         }
106 
107         for (AddressResolver<T> r: rArray) {
108             try {
109                 r.close();
110             } catch (Throwable t) {
111                 logger.warn("Failed to close a resolver:", t);
112             }
113         }
114     }
115 }
