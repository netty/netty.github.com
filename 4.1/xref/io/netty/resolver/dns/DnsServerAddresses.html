38      private static final InternalLogger logger = InternalLoggerFactory.getInstance(DnsServerAddresses.class);

40      private static final List<InetSocketAddress> DEFAULT_NAME_SERVER_LIST;
41      private static final InetSocketAddress[] DEFAULT_NAME_SERVER_ARRAY;
42      private static final DnsServerAddresses DEFAULT_NAME_SERVERS;
43  
44      static {
45          final int DNS_PORT = 53;
46          final List<InetSocketAddress> defaultNameServers = new ArrayList<InetSocketAddress>(2);
47          try {
48              Class<?> configClass = Class.forName("sun.net.dns.ResolverConfiguration");
49              Method open = configClass.getMethod("open");
50              Method nameservers = configClass.getMethod("nameservers");
51              Object instance = open.invoke(null);
52  
53              @SuppressWarnings("unchecked")
54              final List<String> list = (List<String>) nameservers.invoke(instance);
55              for (String a: list) {
56                  if (a != null) {
57                      defaultNameServers.add(new InetSocketAddress(InetAddress.getByName(a), DNS_PORT));
58                  }
59              }
60          } catch (Exception ignore) {
61              // Failed to get the system name server list.
62              // Will add the default name servers afterwards.
63          }
64  
65          if (!defaultNameServers.isEmpty()) {
66              if (logger.isDebugEnabled()) {
67                  logger.debug(
68                          "Default DNS servers: {} (sun.net.dns.ResolverConfiguration)", defaultNameServers);
69              }
70          } else {
71              Collections.addAll(
72                      defaultNameServers,
73                      new InetSocketAddress("8.8.8.8", DNS_PORT),
74                      new InetSocketAddress("8.8.4.4", DNS_PORT));
75  
76              if (logger.isWarnEnabled()) {
77                  logger.warn(
78                          "Default DNS servers: {} (Google Public DNS as a fallback)", defaultNameServers);
79              }
80          }
81  
82          DEFAULT_NAME_SERVER_LIST = Collections.unmodifiableList(defaultNameServers);
83          DEFAULT_NAME_SERVER_ARRAY = defaultNameServers.toArray(new InetSocketAddress[defaultNameServers.size()]);
84          DEFAULT_NAME_SERVERS = sequential(DEFAULT_NAME_SERVER_ARRAY);
85      }

211     private static InetSocketAddress[] sanitize(Iterable<? extends InetSocketAddress> addresses) {
212         if (addresses == null) {
213             throw new NullPointerException("addresses");
214         }
215 
216         final List<InetSocketAddress> list;
217         if (addresses instanceof Collection) {
218             list = new ArrayList<InetSocketAddress>(((Collection<?>) addresses).size());
219         } else {
220             list = new ArrayList<InetSocketAddress>(4);
221         }
222 
223         for (InetSocketAddress a : addresses) {
224             if (a == null) {
225                 break;
226             }
227             if (a.isUnresolved()) {
228                 throw new IllegalArgumentException("cannot use an unresolved DNS server address: " + a);
229             }
230             list.add(a);
231         }
232 
233         if (list.isEmpty()) {
234             throw new IllegalArgumentException("empty addresses");
235         }
236 
237         return list.toArray(new InetSocketAddress[list.size()]);
238     }
239 
240     private static InetSocketAddress[] sanitize(InetSocketAddress[] addresses) {
241         if (addresses == null) {
242             throw new NullPointerException("addresses");
243         }
244 
245         List<InetSocketAddress> list = InternalThreadLocalMap.get().arrayList(addresses.length);
246         for (InetSocketAddress a: addresses) {
247             if (a == null) {
248                 break;
249             }
250             if (a.isUnresolved()) {
251                 throw new IllegalArgumentException("cannot use an unresolved DNS server address: " + a);
252             }
253             list.add(a);
254         }
255 
256         if (list.isEmpty()) {
257             return DEFAULT_NAME_SERVER_ARRAY;
258         }
259 
260         return list.toArray(new InetSocketAddress[list.size()]);
261     }
