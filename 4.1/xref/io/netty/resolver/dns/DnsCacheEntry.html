19  import static io.netty.util.internal.ObjectUtil.checkNotNull;
20  
21  import io.netty.channel.EventLoop;
22  import io.netty.util.concurrent.ScheduledFuture;
23  
24  import java.net.InetAddress;
25  import java.util.concurrent.TimeUnit;
26  
27  /**
28   * Entry in {@link DnsCache}.
29   */
30  public final class DnsCacheEntry {
31  
32      private final String hostname;
33      private final InetAddress address;
34      private final Throwable cause;
35      private volatile ScheduledFuture<?> expirationFuture;
36  
37      public DnsCacheEntry(String hostname, InetAddress address) {
38          this.hostname = checkNotNull(hostname, "hostname");
39          this.address = checkNotNull(address, "address");
40          cause = null;
41      }
42  
43      public DnsCacheEntry(String hostname, Throwable cause) {
44          this.hostname = checkNotNull(hostname, "hostname");
45          this.cause = checkNotNull(cause, "cause");
46          address = null;
47      }
48  
49      public String hostname() {
50          return hostname;
51      }
52  
53      public InetAddress address() {
54          return address;
55      }
56  
57      public Throwable cause() {
58          return cause;
59      }
60  
61      void scheduleExpiration(EventLoop loop, Runnable task, long delay, TimeUnit unit) {
62          assert expirationFuture == null: "expiration task scheduled already";
63          expirationFuture = loop.schedule(task, delay, unit);
64      }
65  
66      void cancelExpiration() {
67          ScheduledFuture<?> expirationFuture = this.expirationFuture;
68          if (expirationFuture != null) {
69              expirationFuture.cancel(false);
70          }
71      }
72  
73      @Override
74      public String toString() {
75          if (cause != null) {
76              return hostname + '/' + cause;
77          } else {
78              return address.toString();
79          }
80      }
81  }
