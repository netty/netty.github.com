36  /**
37   * Sets up the Netty pipeline for the example server. Depending on the endpoint config, sets up the
38   * pipeline for NPN or cleartext HTTP upgrade to HTTP/2.
39   */
40  public class Http2ServerInitializer extends ChannelInitializer<SocketChannel> {
41  
42      private static final UpgradeCodecFactory upgradeCodecFactory = new UpgradeCodecFactory() {
43          @Override
44          public UpgradeCodec newUpgradeCodec(CharSequence protocol) {
45              if (AsciiString.contentEquals(Http2CodecUtil.HTTP_UPGRADE_PROTOCOL_NAME, protocol)) {
46                  return new Http2ServerUpgradeCodec(new HelloWorldHttp2HandlerBuilder().build());
47              } else {
48                  return null;
49              }
50          }
51      };
52  
53      private final SslContext sslCtx;
54      private final int maxHttpContentLength;
55  
56      public Http2ServerInitializer(SslContext sslCtx) {
57          this(sslCtx, 16 * 1024);
58      }
59  
60      public Http2ServerInitializer(SslContext sslCtx, int maxHttpContentLength) {
61          if (maxHttpContentLength < 0) {
62              throw new IllegalArgumentException("maxHttpContentLength (expected >= 0): " + maxHttpContentLength);
63          }
64          this.sslCtx = sslCtx;
65          this.maxHttpContentLength = maxHttpContentLength;
66      }
67  
68      @Override
69      public void initChannel(SocketChannel ch) {
70          if (sslCtx != null) {
71              configureSsl(ch);
72          } else {
73              configureClearText(ch);
74          }
75      }
76  
77      /**
78       * Configure the pipeline for TLS NPN negotiation to HTTP/2.
79       */
80      private void configureSsl(SocketChannel ch) {
81          ch.pipeline().addLast(sslCtx.newHandler(ch.alloc()), new Http2OrHttpHandler());
82      }
83  
84      /**
85       * Configure the pipeline for a cleartext upgrade from HTTP to HTTP/2.0
86       */
87      private void configureClearText(SocketChannel ch) {
88          final ChannelPipeline p = ch.pipeline();
89          final HttpServerCodec sourceCodec = new HttpServerCodec();
90  
91          p.addLast(sourceCodec);
92          p.addLast(new HttpServerUpgradeHandler(sourceCodec, upgradeCodecFactory));
93          p.addLast(new SimpleChannelInboundHandler<HttpMessage>() {
94              @Override
95              protected void channelRead0(ChannelHandlerContext ctx, HttpMessage msg) throws Exception {
96                  // If this handler is hit then no upgrade has been attempted and the client is just talking HTTP.
97                  System.err.println("Directly talking: " + msg.protocolVersion() + " (no upgrade was attempted)");
98                  ChannelPipeline pipeline = ctx.pipeline();
99                  ChannelHandlerContext thisCtx = pipeline.context(this);
100                 pipeline.addAfter(thisCtx.name(), null, new HelloWorldHttp1Handler("Direct. No Upgrade Attempted."));
101                 pipeline.replace(this, null, new HttpObjectAggregator(maxHttpContentLength));
102                 ctx.fireChannelRead(msg);
103             }
104         });
105 
106         p.addLast(new UserEventLogger());
107     }
108 
109     /**
110      * Class that logs any User Events triggered on this channel.
111      */
112     private static class UserEventLogger extends ChannelInboundHandlerAdapter {
113         @Override
114         public void userEventTriggered(ChannelHandlerContext ctx, Object evt) {
115             System.out.println("User Event Triggered: " + evt);
116             ctx.fireUserEventTriggered(evt);
117         }
118     }
