125         @Override
126         protected void onRemoval(Stack<T> value) {
127             // Let us remove the WeakOrderQueue from the WeakHashMap directly if its safe to remove some overhead
128             if (value.threadRef.get() == Thread.currentThread()) {

154     protected Recycler(int maxCapacityPerThread, int maxSharedCapacityFactor,
155                        int ratio, int maxDelayedQueuesPerThread, int delayedQueueRatio) {
156         interval = max(0, ratio);
157         delayedQueueInterval = max(0, delayedQueueRatio);
158         if (maxCapacityPerThread <= 0) {
159             this.maxCapacityPerThread = 0;
160             this.maxSharedCapacityFactor = 1;
161             this.maxDelayedQueuesPerThread = 0;
162         } else {
163             this.maxCapacityPerThread = maxCapacityPerThread;

165             this.maxDelayedQueuesPerThread = max(0, maxDelayedQueuesPerThread);
166         }

169     @SuppressWarnings("unchecked")
170     public final T get() {
171         if (maxCapacityPerThread == 0) {
172             return newObject((Handle<T>) NOOP_HANDLE);
173         }
174         Stack<T> stack = threadLocal.get();
175         DefaultHandle<T> handle = stack.pop();
176         if (handle == null) {
177             handle = stack.newHandle();
178             handle.value = newObject(handle);
179         }
180         return (T) handle.value;
181     }

213     @SuppressWarnings("unchecked")
214     private static final class DefaultHandle<T> implements Handle<T> {
215         private static final AtomicIntegerFieldUpdater<DefaultHandle<?>> LAST_RECYCLED_ID_UPDATER;
216         static {
217             AtomicIntegerFieldUpdater<?> updater = AtomicIntegerFieldUpdater.newUpdater(
218                     DefaultHandle.class, "lastRecycledId");
219             LAST_RECYCLED_ID_UPDATER = (AtomicIntegerFieldUpdater<DefaultHandle<?>>) updater;
220         }
221 
222         volatile int lastRecycledId;
223         int recycleId;
224 
225         boolean hasBeenRecycled;
226 
227         Stack<?> stack;
228         Object value;
229 
230         DefaultHandle(Stack<?> stack) {
231             this.stack = stack;
232         }
233 
234         @Override
235         public void recycle(Object object) {
236             if (object != value) {
237                 throw new IllegalArgumentException("object does not belong to handle");
238             }
239 
240             Stack<?> stack = this.stack;
241             if (lastRecycledId != recycleId || stack == null) {
242                 throw new IllegalStateException("recycled already");
243             }
244 
245             stack.push(this);
246         }
247 
248         public boolean compareAndSetLastRecycledId(int expectLastRecycledId, int updateLastRecycledId) {
249             // Use "weak…" because we do not need synchronize-with ordering, only atomicity.
250             // Also, spurious failures are fine, since no code should rely on recycling for correctness.
251             return LAST_RECYCLED_ID_UPDATER.weakCompareAndSet(this, expectLastRecycledId, updateLastRecycledId);
252         }

256             new FastThreadLocal<Map<Stack<?>, WeakOrderQueue>>() {

263     // a queue that makes only moderate guarantees about visibility: items are seen in the correct order,
264     // but we aren't absolutely guaranteed to ever see anything at all, thereby keeping the queue cheap to maintain
265     private static final class WeakOrderQueue extends WeakReference<Thread> {
266 
267         static final WeakOrderQueue DUMMY = new WeakOrderQueue();
268 
269         // Let Link extend AtomicInteger for intrinsics. The Link itself will be used as writerIndex.
270         @SuppressWarnings("serial")
271         static final class Link extends AtomicInteger {
272             final DefaultHandle<?>[] elements = new DefaultHandle[LINK_CAPACITY];
273 
274             int readIndex;
275             Link next;
276         }

278         // Its important this does not hold any reference to either Stack or WeakOrderQueue.
279         private static final class Head {
280             private final AtomicInteger availableSharedCapacity;
281 
282             Link link;
283 
284             Head(AtomicInteger availableSharedCapacity) {
285                 this.availableSharedCapacity = availableSharedCapacity;
286             }
287 
288             /**
289              * Reclaim all used space and also unlink the nodes to prevent GC nepotism.
290              */
291             void reclaimAllSpaceAndUnlink() {
292                 Link head = link;
293                 link = null;
294                 int reclaimSpace = 0;
295                 while (head != null) {
296                     reclaimSpace += LINK_CAPACITY;
297                     Link next = head.next;
298                     // Unlink to help GC and guard against GC nepotism.
299                     head.next = null;
300                     head = next;
301                 }
302                 if (reclaimSpace > 0) {
303                     reclaimSpace(reclaimSpace);
304                 }
305             }
306 
307             private void reclaimSpace(int space) {
308                 availableSharedCapacity.addAndGet(space);
309             }
310 
311             void relink(Link link) {
312                 reclaimSpace(LINK_CAPACITY);
313                 this.link = link;
314             }
315 
316             /**
317              * Creates a new {@link} and returns it if we can reserve enough space for it, otherwise it
318              * returns {@code null}.
319              */
320             Link newLink() {
321                 return reserveSpaceForLink(availableSharedCapacity) ? new Link() : null;
322             }
323 
324             static boolean reserveSpaceForLink(AtomicInteger availableSharedCapacity) {
325                 for (;;) {
326                     int available = availableSharedCapacity.get();
327                     if (available < LINK_CAPACITY) {
328                         return false;
329                     }
330                     if (availableSharedCapacity.compareAndSet(available, available - LINK_CAPACITY)) {
331                         return true;
332                     }
333                 }
334             }
335         }
336 
337         // chain of data items
338         private final Head head;
339         private Link tail;
340         // pointer to another queue of delayed items for the same stack
341         private WeakOrderQueue next;
342         private final int id = ID_GENERATOR.getAndIncrement();
343         private final int interval;
344         private int handleRecycleCount;
345 
346         private WeakOrderQueue() {
347             super(null);
348             head = new Head(null);
349             interval = 0;
350         }
351 
352         private WeakOrderQueue(Stack<?> stack, Thread thread) {
353             super(thread);
354             tail = new Link();
355 
356             // Its important that we not store the Stack itself in the WeakOrderQueue as the Stack also is used in
357             // the WeakHashMap as key. So just store the enclosed AtomicInteger which should allow to have the
358             // Stack itself GCed.
359             head = new Head(stack.availableSharedCapacity);
360             head.link = tail;
361             interval = stack.delayedQueueInterval;
362             handleRecycleCount = interval; // Start at interval so the first one will be recycled.
363         }
364 
365         static WeakOrderQueue newQueue(Stack<?> stack, Thread thread) {
366             // We allocated a Link so reserve the space
367             if (!Head.reserveSpaceForLink(stack.availableSharedCapacity)) {
368                 return null;
369             }
370             final WeakOrderQueue queue = new WeakOrderQueue(stack, thread);
371             // Done outside of the constructor to ensure WeakOrderQueue.this does not escape the constructor and so
372             // may be accessed while its still constructed.
373             stack.setHead(queue);
374 
375             return queue;
376         }
377 
378         WeakOrderQueue getNext() {
379             return next;
380         }
381 
382         void setNext(WeakOrderQueue next) {
383             assert next != this;
384             this.next = next;
385         }
386 
387         void reclaimAllSpaceAndUnlink() {
388             head.reclaimAllSpaceAndUnlink();
389             next = null;
390         }

392         void add(DefaultHandle<?> handle) {
393             if (!handle.compareAndSetLastRecycledId(0, id)) {
394                 // Separate threads could be racing to add the handle to each their own WeakOrderQueue.
395                 // We only add the handle to the queue if we win the race and observe that lastRecycledId is zero.
396                 return;
397             }
398 
399             // While we also enforce the recycling ratio when we transfer objects from the WeakOrderQueue to the Stack
400             // we better should enforce it as well early. Missing to do so may let the WeakOrderQueue grow very fast
401             // without control
402             if (handleRecycleCount < interval) {
403                 handleRecycleCount++;
404                 // Drop the item to prevent recycling to aggressive.
405                 return;
406             }
407             handleRecycleCount = 0;
408 
409             Link tail = this.tail;
410             int writeIndex;
411             if ((writeIndex = tail.get()) == LINK_CAPACITY) {
412                 Link link = head.newLink();
413                 if (link == null) {
414                     // Drop it.
415                     return;
416                 }
417                 // We allocate a Link so reserve the space
418                 this.tail = tail = tail.next = link;
419 
420                 writeIndex = tail.get();
421             }
422             tail.elements[writeIndex] = handle;
423             handle.stack = null;
424             // we lazy set to ensure that setting stack to null appears before we unnull it in the owning thread;
425             // this also means we guarantee visibility of an element in the queue if we see the index updated
426             tail.lazySet(writeIndex + 1);
427         }
428 
429         boolean hasFinalData() {
430             return tail.readIndex != tail.get();
431         }
432 
433         // transfer as many items as we can from this queue to the stack, returning true if any were transferred
434         @SuppressWarnings("rawtypes")
435         boolean transfer(Stack<?> dst) {
436             Link head = this.head.link;
437             if (head == null) {
438                 return false;
439             }
440 
441             if (head.readIndex == LINK_CAPACITY) {
442                 if (head.next == null) {
443                     return false;
444                 }
445                 head = head.next;
446                 this.head.relink(head);
447             }
448 
449             final int srcStart = head.readIndex;
450             int srcEnd = head.get();
451             final int srcSize = srcEnd - srcStart;
452             if (srcSize == 0) {
453                 return false;
454             }
455 
456             final int dstSize = dst.size;
457             final int expectedCapacity = dstSize + srcSize;
458 
459             if (expectedCapacity > dst.elements.length) {
460                 final int actualCapacity = dst.increaseCapacity(expectedCapacity);
461                 srcEnd = min(srcStart + actualCapacity - dstSize, srcEnd);
462             }
463 
464             if (srcStart != srcEnd) {
465                 final DefaultHandle[] srcElems = head.elements;
466                 final DefaultHandle[] dstElems = dst.elements;
467                 int newDstSize = dstSize;
468                 for (int i = srcStart; i < srcEnd; i++) {
469                     DefaultHandle<?> element = srcElems[i];
470                     if (element.recycleId == 0) {
471                         element.recycleId = element.lastRecycledId;
472                     } else if (element.recycleId != element.lastRecycledId) {
473                         throw new IllegalStateException("recycled already");
474                     }
475                     srcElems[i] = null;
476 
477                     if (dst.dropHandle(element)) {
478                         // Drop the object.
479                         continue;
480                     }
481                     element.stack = dst;
482                     dstElems[newDstSize ++] = element;
483                 }
484 
485                 if (srcEnd == LINK_CAPACITY && head.next != null) {
486                     // Add capacity back as the Link is GCed.
487                     this.head.relink(head.next);
488                 }
489 
490                 head.readIndex = srcEnd;
491                 if (dst.size == newDstSize) {
492                     return false;
493                 }
494                 dst.size = newDstSize;
495                 return true;
496             } else {
497                 // The destination stack is full already.
498                 return false;
499             }
500         }
501     }

503     private static final class Stack<T> {
504 
505         // we keep a queue of per-thread queues, which is appended to once only, each time a new thread other
506         // than the stack owner recycles: when we run out of items in our stack we iterate this collection
507         // to scavenge those that can be reused. this permits us to incur minimal thread synchronisation whilst
508         // still recycling all items.
509         final Recycler<T> parent;
510 
511         // We store the Thread in a WeakReference as otherwise we may be the only ones that still hold a strong
512         // Reference to the Thread itself after it died because DefaultHandle will hold a reference to the Stack.
513         //
514         // The biggest issue is if we do not use a WeakReference the Thread may not be able to be collected at all if
515         // the user will store a reference to the DefaultHandle somewhere and never clear this reference (or not clear
516         // it in a timely manner).
517         final WeakReference<Thread> threadRef;
518         final AtomicInteger availableSharedCapacity;
519         private final int maxDelayedQueues;
520 
521         private final int maxCapacity;
522         private final int interval;
523         private final int delayedQueueInterval;
524         DefaultHandle<?>[] elements;
525         int size;
526         private int handleRecycleCount;
527         private WeakOrderQueue cursor, prev;
528         private volatile WeakOrderQueue head;
529 
530         Stack(Recycler<T> parent, Thread thread, int maxCapacity, int maxSharedCapacityFactor,
531               int interval, int maxDelayedQueues, int delayedQueueInterval) {
532             this.parent = parent;
533             threadRef = new WeakReference<Thread>(thread);
534             this.maxCapacity = maxCapacity;
535             availableSharedCapacity = new AtomicInteger(max(maxCapacity / maxSharedCapacityFactor, LINK_CAPACITY));
536             elements = new DefaultHandle[min(INITIAL_CAPACITY, maxCapacity)];
537             this.interval = interval;
538             this.delayedQueueInterval = delayedQueueInterval;
539             handleRecycleCount = interval; // Start at interval so the first one will be recycled.
540             this.maxDelayedQueues = maxDelayedQueues;
541         }
542 
543         // Marked as synchronized to ensure this is serialized.
544         synchronized void setHead(WeakOrderQueue queue) {
545             queue.setNext(head);
546             head = queue;
547         }
548 
549         int increaseCapacity(int expectedCapacity) {
550             int newCapacity = elements.length;
551             int maxCapacity = this.maxCapacity;
552             do {
553                 newCapacity <<= 1;
554             } while (newCapacity < expectedCapacity && newCapacity < maxCapacity);
555 
556             newCapacity = min(newCapacity, maxCapacity);
557             if (newCapacity != elements.length) {
558                 elements = Arrays.copyOf(elements, newCapacity);
559             }
560 
561             return newCapacity;
562         }
563 
564         @SuppressWarnings({ "unchecked", "rawtypes" })
565         DefaultHandle<T> pop() {
566             int size = this.size;
567             if (size == 0) {
568                 if (!scavenge()) {
569                     return null;
570                 }
571                 size = this.size;
572                 if (size <= 0) {
573                     // double check, avoid races
574                     return null;
575                 }
576             }
577             size --;
578             DefaultHandle ret = elements[size];
579             elements[size] = null;
580             // As we already set the element[size] to null we also need to store the updated size before we do
581             // any validation. Otherwise we may see a null value when later try to pop again without a new element
582             // added before.
583             this.size = size;
584 
585             if (ret.lastRecycledId != ret.recycleId) {
586                 throw new IllegalStateException("recycled multiple times");
587             }
588             ret.recycleId = 0;
589             ret.lastRecycledId = 0;
590             return ret;
591         }
592 
593         private boolean scavenge() {
594             // continue an existing scavenge, if any
595             if (scavengeSome()) {
596                 return true;
597             }
598 
599             // reset our scavenge cursor
600             prev = null;
601             cursor = head;
602             return false;
603         }
604 
605         private boolean scavengeSome() {
606             WeakOrderQueue prev;
607             WeakOrderQueue cursor = this.cursor;
608             if (cursor == null) {
609                 prev = null;
610                 cursor = head;
611                 if (cursor == null) {
612                     return false;
613                 }
614             } else {
615                 prev = this.prev;
616             }
617 
618             boolean success = false;
619             do {
620                 if (cursor.transfer(this)) {
621                     success = true;
622                     break;
623                 }
624                 WeakOrderQueue next = cursor.getNext();
625                 if (cursor.get() == null) {
626                     // If the thread associated with the queue is gone, unlink it, after
627                     // performing a volatile read to confirm there is no data left to collect.
628                     // We never unlink the first queue, as we don't want to synchronize on updating the head.
629                     if (cursor.hasFinalData()) {
630                         for (;;) {
631                             if (cursor.transfer(this)) {
632                                 success = true;
633                             } else {
634                                 break;
635                             }
636                         }
637                     }
638 
639                     if (prev != null) {
640                         // Ensure we reclaim all space before dropping the WeakOrderQueue to be GC'ed.
641                         cursor.reclaimAllSpaceAndUnlink();
642                         prev.setNext(next);
643                     }
644                 } else {
645                     prev = cursor;
646                 }
647 
648                 cursor = next;
649 
650             } while (cursor != null && !success);
651 
652             this.prev = prev;
653             this.cursor = cursor;
654             return success;
655         }
656 
657         void push(DefaultHandle<?> item) {
658             Thread currentThread = Thread.currentThread();
659             if (threadRef.get() == currentThread) {
660                 // The current Thread is the thread that belongs to the Stack, we can try to push the object now.
661                 pushNow(item);
662             } else {
663                 // The current Thread is not the one that belongs to the Stack
664                 // (or the Thread that belonged to the Stack was collected already), we need to signal that the push
665                 // happens later.
666                 pushLater(item, currentThread);
667             }
668         }
669 
670         private void pushNow(DefaultHandle<?> item) {
671             if (item.recycleId != 0 || !item.compareAndSetLastRecycledId(0, OWN_THREAD_ID)) {
672                 throw new IllegalStateException("recycled already");
673             }
674             item.recycleId = OWN_THREAD_ID;
675 
676             int size = this.size;
677             if (size >= maxCapacity || dropHandle(item)) {
678                 // Hit the maximum capacity or should drop - drop the possibly youngest object.
679                 return;
680             }
681             if (size == elements.length) {
682                 elements = Arrays.copyOf(elements, min(size << 1, maxCapacity));
683             }
684 
685             elements[size] = item;
686             this.size = size + 1;
687         }
688 
689         private void pushLater(DefaultHandle<?> item, Thread thread) {
690             if (maxDelayedQueues == 0) {
691                 // We don't support recycling across threads and should just drop the item on the floor.
692                 return;
693             }
694 
695             // we don't want to have a ref to the queue as the value in our weak map
696             // so we null it out; to ensure there are no races with restoring it later
697             // we impose a memory ordering here (no-op on x86)
698             Map<Stack<?>, WeakOrderQueue> delayedRecycled = DELAYED_RECYCLED.get();
699             WeakOrderQueue queue = delayedRecycled.get(this);
700             if (queue == null) {
701                 if (delayedRecycled.size() >= maxDelayedQueues) {
702                     // Add a dummy queue so we know we should drop the object
703                     delayedRecycled.put(this, WeakOrderQueue.DUMMY);
704                     return;
705                 }
706                 // Check if we already reached the maximum number of delayed queues and if we can allocate at all.
707                 if ((queue = newWeakOrderQueue(thread)) == null) {
708                     // drop object
709                     return;
710                 }
711                 delayedRecycled.put(this, queue);
712             } else if (queue == WeakOrderQueue.DUMMY) {
713                 // drop object
714                 return;
715             }
716 
717             queue.add(item);
718         }
719 
720         /**
721          * Allocate a new {@link WeakOrderQueue} or return {@code null} if not possible.
722          */
723         private WeakOrderQueue newWeakOrderQueue(Thread thread) {
724             return WeakOrderQueue.newQueue(this, thread);
725         }
726 
727         boolean dropHandle(DefaultHandle<?> handle) {
728             if (!handle.hasBeenRecycled) {
729                 if (handleRecycleCount < interval) {
730                     handleRecycleCount++;
731                     // Drop the object.
732                     return true;
733                 }
734                 handleRecycleCount = 0;
735                 handle.hasBeenRecycled = true;
736             }
737             return false;
738         }
739 
740         DefaultHandle<T> newHandle() {
741             return new DefaultHandle<T>(this);
742         }
743     }
