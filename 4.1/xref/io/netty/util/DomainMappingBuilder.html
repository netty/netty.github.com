17  package io.netty.util;

19  import java.util.LinkedHashMap;
20  import java.util.Map;
21  import java.util.Set;

23  import static io.netty.util.internal.ObjectUtil.checkNotNull;

25  /**
26   * Builder for immutable {@link DomainNameMapping} instances.
27   *
28   * @param <V> concrete type of value objects
29   */
30  public final class DomainMappingBuilder<V> {
31  
32      private final V defaultValue;
33      private final Map<String, V> map;
34  
35      /**
36       * Constructor with default initial capacity of the map holding the mappings
37       *
38       * @param defaultValue the default value for {@link DomainNameMapping#map(String)} to return
39       *                     when nothing matches the input
40       */
41      public DomainMappingBuilder(V defaultValue) {
42          this(4, defaultValue);
43      }
44  
45      /**
46       * Constructor with initial capacity of the map holding the mappings
47       *
48       * @param initialCapacity initial capacity for the internal map
49       * @param defaultValue    the default value for {@link DomainNameMapping#map(String)} to return
50       *                        when nothing matches the input
51       */
52      public DomainMappingBuilder(int initialCapacity, V defaultValue) {
53          this.defaultValue = checkNotNull(defaultValue, "defaultValue");
54          this.map = new LinkedHashMap<String, V>(initialCapacity);
55      }
56  
57      /**
58       * Adds a mapping that maps the specified (optionally wildcard) host name to the specified output value.
59       * Null values are forbidden for both hostnames and values.
60       * <p>
61       * <a href="http://en.wikipedia.org/wiki/Wildcard_DNS_record">DNS wildcard</a> is supported as hostname.
62       * For example, you can use {@code *.netty.io} to match {@code netty.io} and {@code downloads.netty.io}.
63       * </p>
64       *
65       * @param hostname the host name (optionally wildcard)
66       * @param output   the output value that will be returned by {@link DomainNameMapping#map(String)}
67       *                 when the specified host name matches the specified input host name
68       */
69      public DomainMappingBuilder<V> add(String hostname, V output) {
70          map.put(checkNotNull(hostname, "hostname"), checkNotNull(output, "output"));
71          return this;
72      }
73  
74      /**
75       * Creates a new instance of immutable {@link DomainNameMapping}
76       * Attempts to add new mappings to the result object will cause {@link UnsupportedOperationException} to be thrown
77       *
78       * @return new {@link DomainNameMapping} instance
79       */
80      public DomainNameMapping<V> build() {
81          return new ImmutableDomainNameMapping<V>(this.defaultValue, this.map);
82      }
83  
84      /**
85       * Immutable mapping from domain name pattern to its associated value object.
86       * Mapping is represented by two arrays: keys and values. Key domainNamePatterns[i] is associated with values[i].
87       *
88       * @param <V> concrete type of value objects
89       */
90      private static final class ImmutableDomainNameMapping<V> extends DomainNameMapping<V> {
91          private static final String REPR_HEADER = "ImmutableDomainNameMapping(default: ";
92          private static final String REPR_MAP_OPENING = ", map: {";
93          private static final String REPR_MAP_CLOSING = "})";
94          private static final int REPR_CONST_PART_LENGTH =
95              REPR_HEADER.length() + REPR_MAP_OPENING.length() + REPR_MAP_CLOSING.length();
96  
97          private final String[] domainNamePatterns;
98          private final V[] values;
99  
100         @SuppressWarnings("unchecked")
101         private ImmutableDomainNameMapping(V defaultValue, Map<String, V> map) {
102             super(null, defaultValue);
103 
104             Set<Map.Entry<String, V>> mappings = map.entrySet();
105             int numberOfMappings = mappings.size();
106             domainNamePatterns = new String[numberOfMappings];
107             values = (V[]) new Object[numberOfMappings];
108 
109             int index = 0;
110             for (Map.Entry<String, V> mapping : mappings) {
111                 domainNamePatterns[index] = normalizeHostname(mapping.getKey());
112                 values[index] = mapping.getValue();
113                 ++index;
114             }
115         }
116 
117         @Override
118         @Deprecated
119         public DomainNameMapping<V> add(String hostname, V output) {
120             throw new UnsupportedOperationException(
121                 "Immutable DomainNameMapping does not support modification after initial creation");
122         }
123 
124         @Override
125         public V map(String hostname) {
126             if (hostname != null) {
127                 hostname = normalizeHostname(hostname);
128 
129                 int length = domainNamePatterns.length;
130                 for (int index = 0; index < length; ++index) {
131                     if (matches(domainNamePatterns[index], hostname)) {
132                         return values[index];
133                     }
134                 }
135             }
136 
137             return defaultValue;
138         }
139 
140         @Override
141         public String toString() {
142             String defaultValueStr = defaultValue.toString();
143 
144             int numberOfMappings = domainNamePatterns.length;
145             if (numberOfMappings == 0) {
146                 return REPR_HEADER + defaultValueStr + REPR_MAP_OPENING + REPR_MAP_CLOSING;
147             }
148 
149             String pattern0 = domainNamePatterns[0];
150             String value0 = values[0].toString();
151             int oneMappingLength = pattern0.length() + value0.length() + 3; // 2 for separator ", " and 1 for '='
152             int estimatedBufferSize = estimateBufferSize(defaultValueStr.length(), numberOfMappings, oneMappingLength);
153 
154             StringBuilder sb = new StringBuilder(estimatedBufferSize)
155                 .append(REPR_HEADER).append(defaultValueStr).append(REPR_MAP_OPENING);
156 
157             appendMapping(sb, pattern0, value0);
158             for (int index = 1; index < numberOfMappings; ++index) {
159                 sb.append(", ");
160                 appendMapping(sb, index);
161             }
162 
163             return sb.append(REPR_MAP_CLOSING).toString();
164         }
165 
166         /**
167          * Estimates the length of string representation of the given instance:
168          * est = lengthOfConstantComponents + defaultValueLength + (estimatedMappingLength * numOfMappings) * 1.10
169          *
170          * @param defaultValueLength     length of string representation of {@link #defaultValue}
171          * @param numberOfMappings       number of mappings the given instance holds,
172          *                               e.g. {@link #domainNamePatterns#length}
173          * @param estimatedMappingLength estimated size taken by one mapping
174          * @return estimated length of string returned by {@link #toString()}
175          */
176         private static int estimateBufferSize(int defaultValueLength,
177                                               int numberOfMappings,
178                                               int estimatedMappingLength) {
179             return REPR_CONST_PART_LENGTH + defaultValueLength
180                 + (int) (estimatedMappingLength * numberOfMappings * 1.10);
181         }
182 
183         private StringBuilder appendMapping(StringBuilder sb, int mappingIndex) {
184             return appendMapping(sb, domainNamePatterns[mappingIndex], values[mappingIndex].toString());
185         }
186 
187         private static StringBuilder appendMapping(StringBuilder sb, String domainNamePattern, String value) {
188             return sb.append(domainNamePattern).append('=').append(value);
189         }
190     }
