54      private static final byte[] UNIQUE_ID_BYTES =
55              "0123456789abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ".getBytes(CharsetUtil.US_ASCII);

57      static {
58          String workdir = SystemPropertyUtil.get("io.netty.native.workdir");
59          if (workdir != null) {
60              File f = new File(workdir);
61              f.mkdirs();
62  
63              try {
64                  f = f.getAbsoluteFile();
65              } catch (Exception ignored) {
66                  // Good to have an absolute path, but it's OK.
67              }
68  
69              WORKDIR = f;
70              logger.debug("-Dio.netty.native.workdir: " + WORKDIR);
71          } else {
72              WORKDIR = PlatformDependent.tmpdir();
73              logger.debug("-Dio.netty.native.workdir: " + WORKDIR + " (io.netty.tmpdir)");
74          }
75  
76          DELETE_NATIVE_LIB_AFTER_LOADING = SystemPropertyUtil.getBoolean(
77                  "io.netty.native.deleteLibAfterLoading", true);
78          logger.debug("-Dio.netty.native.deleteLibAfterLoading: {}", DELETE_NATIVE_LIB_AFTER_LOADING);
79  
80          TRY_TO_PATCH_SHADED_ID = SystemPropertyUtil.getBoolean(
81                  "io.netty.native.tryPatchShadedId", true);
82          logger.debug("-Dio.netty.native.tryPatchShadedId: {}", TRY_TO_PATCH_SHADED_ID);
83      }

85      /**
86       * Loads the first available library in the collection with the specified
87       * {@link ClassLoader}.
88       *
89       * @throws IllegalArgumentException
90       *         if none of the given libraries load successfully.
91       */
92      public static void loadFirstAvailable(ClassLoader loader, String... names) {
93          List<Throwable> suppressed = new ArrayList<Throwable>();
94          for (String name : names) {
95              try {
96                  load(name, loader);
97                  return;
98              } catch (Throwable t) {
99                  suppressed.add(t);
100             }
101         }
102 
103         IllegalArgumentException iae =
104                 new IllegalArgumentException("Failed to load any of the given libraries: " + Arrays.toString(names));
105         ThrowableUtil.addSuppressedAndClear(iae, suppressed);
106         throw iae;
107     }

109     /**
110      * The shading prefix added to this class's full name.
111      *
112      * @throws UnsatisfiedLinkError if the shader used something other than a prefix
113      */
114     private static String calculatePackagePrefix() {
115         String maybeShaded = NativeLibraryLoader.class.getName();
116         // Use ! instead of . to avoid shading utilities from modifying the string
117         String expected = "io!netty!util!internal!NativeLibraryLoader".replace('!', '.');
118         if (!maybeShaded.endsWith(expected)) {
119             throw new UnsatisfiedLinkError(String.format(
120                     "Could not find prefix added to %s to get %s. When shading, only adding a "
121                     + "package prefix is supported", expected, maybeShaded));
122         }
123         return maybeShaded.substring(0, maybeShaded.length() - expected.length());
124     }

126     /**
127      * Load the given library with the specified {@link ClassLoader}
128      */
129     public static void load(String originalName, ClassLoader loader) {
130         // Adjust expected name to support shading of native libraries.
131         String packagePrefix = calculatePackagePrefix().replace('.', '_');
132         String name = packagePrefix + originalName;
133         List<Throwable> suppressed = new ArrayList<Throwable>();
134         try {
135             // first try to load from java.library.path
136             loadLibrary(loader, name, false);
137             return;
138         } catch (Throwable ex) {
139             suppressed.add(ex);
140         }
141 
142         String libname = System.mapLibraryName(name);
143         String path = NATIVE_RESOURCE_HOME + libname;
144 
145         InputStream in = null;
146         OutputStream out = null;
147         File tmpFile = null;
148         URL url;
149         if (loader == null) {
150             url = ClassLoader.getSystemResource(path);
151         } else {
152             url = loader.getResource(path);
153         }
154         try {
155             if (url == null) {
156                 if (PlatformDependent.isOsx()) {
157                     String fileName = path.endsWith(".jnilib") ? NATIVE_RESOURCE_HOME + "lib" + name + ".dynlib" :
158                             NATIVE_RESOURCE_HOME + "lib" + name + ".jnilib";
159                     if (loader == null) {
160                         url = ClassLoader.getSystemResource(fileName);
161                     } else {
162                         url = loader.getResource(fileName);
163                     }
164                     if (url == null) {
165                         FileNotFoundException fnf = new FileNotFoundException(fileName);
166                         ThrowableUtil.addSuppressedAndClear(fnf, suppressed);
167                         throw fnf;
168                     }
169                 } else {
170                     FileNotFoundException fnf = new FileNotFoundException(path);
171                     ThrowableUtil.addSuppressedAndClear(fnf, suppressed);
172                     throw fnf;
173                 }
174             }
175 
176             int index = libname.lastIndexOf('.');
177             String prefix = libname.substring(0, index);
178             String suffix = libname.substring(index);
179 
180             tmpFile = PlatformDependent.createTempFile(prefix, suffix, WORKDIR);
181             in = url.openStream();
182             out = new FileOutputStream(tmpFile);
183 
184             byte[] buffer = new byte[8192];
185             int length;
186             while ((length = in.read(buffer)) > 0) {
187                 out.write(buffer, 0, length);
188             }
189             out.flush();
190 
191             if (shouldShadedLibraryIdBePatched(packagePrefix)) {
192                 // Let's try to patch the id and re-sign it. This is a best-effort and might fail if a
193                 // SecurityManager is setup or the right executables are not installed :/
194                 tryPatchShadedLibraryIdAndSign(tmpFile, originalName);
195             }
196 
197             // Close the output stream before loading the unpacked library,
198             // because otherwise Windows will refuse to load it when it's in use by other process.
199             closeQuietly(out);
200             out = null;
201 
202             loadLibrary(loader, tmpFile.getPath(), true);
203         } catch (UnsatisfiedLinkError e) {
204             try {
205                 if (tmpFile != null && tmpFile.isFile() && tmpFile.canRead() &&
206                     !NoexecVolumeDetector.canExecuteExecutable(tmpFile)) {
207                     // Pass "io.netty.native.workdir" as an argument to allow shading tools to see
208                     // the string. Since this is printed out to users to tell them what to do next,
209                     // we want the value to be correct even when shading.
210                     logger.info("{} exists but cannot be executed even when execute permissions set; " +
211                                 "check volume for \"noexec\" flag; use -D{}=[path] " +
212                                 "to set native working directory separately.",
213                                 tmpFile.getPath(), "io.netty.native.workdir");
214                 }
215             } catch (Throwable t) {
216                 suppressed.add(t);
217                 logger.debug("Error checking if {} is on a file store mounted with noexec", tmpFile, t);
218             }
219             // Re-throw to fail the load
220             ThrowableUtil.addSuppressedAndClear(e, suppressed);
221             throw e;
222         } catch (Exception e) {
223             UnsatisfiedLinkError ule = new UnsatisfiedLinkError("could not load a native library: " + name);
224             ule.initCause(e);
225             ThrowableUtil.addSuppressedAndClear(ule, suppressed);
226             throw ule;
227         } finally {
228             closeQuietly(in);
229             closeQuietly(out);
230             // After we load the library it is safe to delete the file.
231             // We delete the file immediately to free up resources as soon as possible,
232             // and if this fails fallback to deleting on JVM exit.
233             if (tmpFile != null && (!DELETE_NATIVE_LIB_AFTER_LOADING || !tmpFile.delete())) {
234                 tmpFile.deleteOnExit();
235             }
236         }
237     }

239     static void tryPatchShadedLibraryIdAndSign(File libraryFile, String originalName) {
240         String newId = new String(generateUniqueId(originalName.length()), CharsetUtil.UTF_8);
241         if (!tryExec("install_name_tool -id " + newId + " " + libraryFile.getAbsolutePath())) {
242             return;
243         }
244 
245         tryExec("codesign -s - " + libraryFile.getAbsolutePath());
246     }

248     private static boolean tryExec(String cmd) {
249         try {
250             int exitValue = Runtime.getRuntime().exec(cmd).waitFor();
251             if (exitValue != 0) {
252                 logger.debug("Execution of '{}' failed: {}", cmd, exitValue);
253                 return false;
254             }
255             logger.debug("Execution of '{}' succeed: {}", cmd, exitValue);
256             return true;
257         } catch (InterruptedException e) {
258             Thread.currentThread().interrupt();
259         } catch (IOException e) {
260             logger.info("Execution of '{}' failed.", cmd, e);
261         } catch (SecurityException e) {
262             logger.error("Execution of '{}' failed.", cmd, e);
263         }
264         return false;
265     }
266 
267     private static boolean shouldShadedLibraryIdBePatched(String packagePrefix) {
268         return TRY_TO_PATCH_SHADED_ID && PlatformDependent.isOsx() && !packagePrefix.isEmpty();
269     }
270 
271     private static byte[] generateUniqueId(int length) {
272         byte[] idBytes = new byte[length];
273         for (int i = 0; i < idBytes.length; i++) {
274             // We should only use bytes as replacement that are in our UNIQUE_ID_BYTES array.
275             idBytes[i] = UNIQUE_ID_BYTES[PlatformDependent.threadLocalRandom()
276                     .nextInt(UNIQUE_ID_BYTES.length)];
277         }
278         return idBytes;
279     }

281     /**
282      * Loading the native library into the specified {@link ClassLoader}.
283      * @param loader - The {@link ClassLoader} where the native library will be loaded into
284      * @param name - The native library path or name
285      * @param absolute - Whether the native library will be loaded by path or by name
286      */
287     private static void loadLibrary(final ClassLoader loader, final String name, final boolean absolute) {
288         Throwable suppressed = null;
289         try {
290             try {
291                 // Make sure the helper is belong to the target ClassLoader.
292                 final Class<?> newHelper = tryToLoadClass(loader, NativeLibraryUtil.class);
293                 loadLibraryByHelper(newHelper, name, absolute);
294                 logger.debug("Successfully loaded the library {}", name);
295                 return;
296             } catch (UnsatisfiedLinkError e) { // Should by pass the UnsatisfiedLinkError here!
297                 suppressed = e;
298             } catch (Exception e) {
299                 suppressed = e;
300             }
301             NativeLibraryUtil.loadLibrary(name, absolute);  // Fallback to local helper class.
302             logger.debug("Successfully loaded the library {}", name);
303         } catch (NoSuchMethodError nsme) {
304             if (suppressed != null) {
305                 ThrowableUtil.addSuppressed(nsme, suppressed);
306             }
307             rethrowWithMoreDetailsIfPossible(name, nsme);
308         } catch (UnsatisfiedLinkError ule) {
309             if (suppressed != null) {
310                 ThrowableUtil.addSuppressed(ule, suppressed);
311             }
312             throw ule;
313         }
314     }

316     @SuppressJava6Requirement(reason = "Guarded by version check")
317     private static void rethrowWithMoreDetailsIfPossible(String name, NoSuchMethodError error) {
318         if (PlatformDependent.javaVersion() >= 7) {
319             throw new LinkageError(
320                     "Possible multiple incompatible native libraries on the classpath for '" + name + "'?", error);
321         }
322         throw error;
323     }

325     private static void loadLibraryByHelper(final Class<?> helper, final String name, final boolean absolute)
326             throws UnsatisfiedLinkError {
327         Object ret = AccessController.doPrivileged(new PrivilegedAction<Object>() {
328             @Override
329             public Object run() {
330                 try {
331                     // Invoke the helper to load the native library, if succeed, then the native
332                     // library belong to the specified ClassLoader.
333                     Method method = helper.getMethod("loadLibrary", String.class, boolean.class);
334                     method.setAccessible(true);
335                     return method.invoke(null, name, absolute);
336                 } catch (Exception e) {
337                     return e;
338                 }
339             }
340         });
341         if (ret instanceof Throwable) {
342             Throwable t = (Throwable) ret;
343             assert !(t instanceof UnsatisfiedLinkError) : t + " should be a wrapper throwable";
344             Throwable cause = t.getCause();
345             if (cause instanceof UnsatisfiedLinkError) {
346                 throw (UnsatisfiedLinkError) cause;
347             }
348             UnsatisfiedLinkError ule = new UnsatisfiedLinkError(t.getMessage());
349             ule.initCause(t);
350             throw ule;
351         }
352     }

354     /**
355      * Try to load the helper {@link Class} into specified {@link ClassLoader}.
356      * @param loader - The {@link ClassLoader} where to load the helper {@link Class}
357      * @param helper - The helper {@link Class}
358      * @return A new helper Class defined in the specified ClassLoader.
359      * @throws ClassNotFoundException Helper class not found or loading failed
360      */
361     private static Class<?> tryToLoadClass(final ClassLoader loader, final Class<?> helper)
362             throws ClassNotFoundException {
363         try {
364             return Class.forName(helper.getName(), false, loader);
365         } catch (ClassNotFoundException e1) {
366             if (loader == null) {
367                 // cannot defineClass inside bootstrap class loader
368                 throw e1;
369             }
370             try {
371                 // The helper class is NOT found in target ClassLoader, we have to define the helper class.
372                 final byte[] classBinary = classToByteArray(helper);
373                 return AccessController.doPrivileged(new PrivilegedAction<Class<?>>() {
374                     @Override
375                     public Class<?> run() {
376                         try {
377                             // Define the helper class in the target ClassLoader,
378                             //  then we can call the helper to load the native library.
379                             Method defineClass = ClassLoader.class.getDeclaredMethod("defineClass", String.class,
380                                     byte[].class, int.class, int.class);
381                             defineClass.setAccessible(true);
382                             return (Class<?>) defineClass.invoke(loader, helper.getName(), classBinary, 0,
383                                     classBinary.length);
384                         } catch (Exception e) {
385                             throw new IllegalStateException("Define class failed!", e);
386                         }
387                     }
388                 });
389             } catch (ClassNotFoundException e2) {
390                 ThrowableUtil.addSuppressed(e2, e1);
391                 throw e2;
392             } catch (RuntimeException e2) {
393                 ThrowableUtil.addSuppressed(e2, e1);
394                 throw e2;
395             } catch (Error e2) {
396                 ThrowableUtil.addSuppressed(e2, e1);
397                 throw e2;
398             }
399         }
400     }

402     /**
403      * Load the helper {@link Class} as a byte array, to be redefined in specified {@link ClassLoader}.
404      * @param clazz - The helper {@link Class} provided by this bundle
405      * @return The binary content of helper {@link Class}.
406      * @throws ClassNotFoundException Helper class not found or loading failed
407      */
408     private static byte[] classToByteArray(Class<?> clazz) throws ClassNotFoundException {
409         String fileName = clazz.getName();
410         int lastDot = fileName.lastIndexOf('.');
411         if (lastDot > 0) {
412             fileName = fileName.substring(lastDot + 1);
413         }
414         URL classUrl = clazz.getResource(fileName + ".class");
415         if (classUrl == null) {
416             throw new ClassNotFoundException(clazz.getName());
417         }
418         byte[] buf = new byte[1024];
419         ByteArrayOutputStream out = new ByteArrayOutputStream(4096);
420         InputStream in = null;
421         try {
422             in = classUrl.openStream();
423             for (int r; (r = in.read(buf)) != -1;) {
424                 out.write(buf, 0, r);
425             }
426             return out.toByteArray();
427         } catch (IOException ex) {
428             throw new ClassNotFoundException(clazz.getName(), ex);
429         } finally {
430             closeQuietly(in);
431             closeQuietly(out);
432         }
433     }

435     private static void closeQuietly(Closeable c) {
436         if (c != null) {
437             try {
438                 c.close();
439             } catch (IOException ignore) {
440                 // ignore
441             }
442         }
443     }

449     private static final class NoexecVolumeDetector {
450 
451         @SuppressJava6Requirement(reason = "Usage guarded by java version check")
452         private static boolean canExecuteExecutable(File file) throws IOException {
453             if (PlatformDependent.javaVersion() < 7) {
454                 // Pre-JDK7, the Java API did not directly support POSIX permissions; instead of implementing a custom
455                 // work-around, assume true, which disables the check.
456                 return true;
457             }
458 
459             // If we can already execute, there is nothing to do.
460             if (file.canExecute()) {
461                 return true;
462             }
463 
464             // On volumes, with noexec set, even files with the executable POSIX permissions will fail to execute.
465             // The File#canExecute() method honors this behavior, probaby via parsing the noexec flag when initializing
466             // the UnixFileStore, though the flag is not exposed via a public API.  To find out if library is being
467             // loaded off a volume with noexec, confirm or add executalbe permissions, then check File#canExecute().
468 
469             // Note: We use FQCN to not break when netty is used in java6
470             Set<java.nio.file.attribute.PosixFilePermission> existingFilePermissions =
471                     java.nio.file.Files.getPosixFilePermissions(file.toPath());
472             Set<java.nio.file.attribute.PosixFilePermission> executePermissions =
473                     EnumSet.of(java.nio.file.attribute.PosixFilePermission.OWNER_EXECUTE,
474                             java.nio.file.attribute.PosixFilePermission.GROUP_EXECUTE,
475                             java.nio.file.attribute.PosixFilePermission.OTHERS_EXECUTE);
476             if (existingFilePermissions.containsAll(executePermissions)) {
477                 return false;
478             }
479 
480             Set<java.nio.file.attribute.PosixFilePermission> newPermissions = EnumSet.copyOf(existingFilePermissions);
481             newPermissions.addAll(executePermissions);
482             java.nio.file.Files.setPosixFilePermissions(file.toPath(), newPermissions);
483             return file.canExecute();
484         }

486         private NoexecVolumeDetector() {
487             // Utility
488         }
