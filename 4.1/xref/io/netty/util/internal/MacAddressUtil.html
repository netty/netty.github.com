32  public final class MacAddressUtil {
33  
34      /**
35       * Length of a valid MAC address.
36       */
37      public static final int MAC_ADDRESS_LENGTH = 8;
38  
39      private static final byte[] NOT_FOUND = { -1 };
40  
41      private static final InternalLogger logger = InternalLoggerFactory.getInstance(MacAddressUtil.class);
42  
43      /**
44       * Obtains the best MAC address found on local network interfaces.
45       * Generally speaking, an active network interface used on public
46       * networks is better than a local network interface.
47       *
48       * @return byte array containing a MAC. null if no MAC can be found.
49       */
50      public static byte[] bestAvailableMac() {
51          // Find the best MAC address available.
52          byte[] bestMacAddr = NOT_FOUND;
53          InetAddress bestInetAddr = NetUtil.LOCALHOST4;
54  
55          // Retrieve the list of available network interfaces.
56          Map<NetworkInterface, InetAddress> ifaces = new LinkedHashMap<NetworkInterface, InetAddress>();
57          try {
58              for (Enumeration<NetworkInterface> i = NetworkInterface.getNetworkInterfaces(); i.hasMoreElements();) {
59                  NetworkInterface iface = i.nextElement();
60                  // Use the interface with proper INET addresses only.
61                  Enumeration<InetAddress> addrs = iface.getInetAddresses();
62                  if (addrs.hasMoreElements()) {
63                      InetAddress a = addrs.nextElement();
64                      if (!a.isLoopbackAddress()) {
65                          ifaces.put(iface, a);
66                      }
67                  }
68              }
69          } catch (SocketException e) {
70              logger.warn("Failed to retrieve the list of available network interfaces", e);
71          }
72  
73          for (Entry<NetworkInterface, InetAddress> entry: ifaces.entrySet()) {
74              NetworkInterface iface = entry.getKey();
75              InetAddress inetAddr = entry.getValue();
76              if (iface.isVirtual()) {
77                  continue;
78              }
79  
80              byte[] macAddr;
81              try {
82                  macAddr = iface.getHardwareAddress();
83              } catch (SocketException e) {
84                  logger.debug("Failed to get the hardware address of a network interface: {}", iface, e);
85                  continue;
86              }
87  
88              boolean replace = false;
89              int res = compareAddresses(bestMacAddr, macAddr);
90              if (res < 0) {
91                  // Found a better MAC address.
92                  replace = true;
93              } else if (res == 0) {
94                  // Two MAC addresses are of pretty much same quality.
95                  res = compareAddresses(bestInetAddr, inetAddr);
96                  if (res < 0) {
97                      // Found a MAC address with better INET address.
98                      replace = true;
99                  } else if (res == 0) {
100                     // Cannot tell the difference.  Choose the longer one.
101                     if (bestMacAddr.length < macAddr.length) {
102                         replace = true;
103                     }
104                 }
105             }
106 
107             if (replace) {
108                 bestMacAddr = macAddr;
109                 bestInetAddr = inetAddr;
110             }
111         }
112 
113         if (bestMacAddr == NOT_FOUND) {
114             return null;
115         }
116 
117         switch (bestMacAddr.length) {
118             case 6: // EUI-48 - convert to EUI-64
119                 byte[] newAddr = new byte[MAC_ADDRESS_LENGTH];
120                 System.arraycopy(bestMacAddr, 0, newAddr, 0, 3);
121                 newAddr[3] = (byte) 0xFF;
122                 newAddr[4] = (byte) 0xFE;
123                 System.arraycopy(bestMacAddr, 3, newAddr, 5, 3);
124                 bestMacAddr = newAddr;
125                 break;
126             default: // Unknown
127                 bestMacAddr = Arrays.copyOf(bestMacAddr, MAC_ADDRESS_LENGTH);
128         }
129 
130         return bestMacAddr;
131     }

133     /**
134      * @param addr byte array of a MAC address.
135      * @return hex formatted MAC address.
136      */
137     public static String formatAddress(byte[] addr) {
138         StringBuilder buf = new StringBuilder(24);
139         for (byte b: addr) {
140             buf.append(String.format("%02x:", b & 0xff));
141         }
142         return buf.substring(0, buf.length() - 1);
143     }

145     /**
146      * @return positive - current is better, 0 - cannot tell from MAC addr, negative - candidate is better.
147      */
148     private static int compareAddresses(byte[] current, byte[] candidate) {
149         if (candidate == null) {
150             return 1;
151         }
152 
153         // Must be EUI-48 or longer.
154         if (candidate.length < 6) {
155             return 1;
156         }
157 
158         // Must not be filled with only 0 and 1.
159         boolean onlyZeroAndOne = true;
160         for (byte b: candidate) {
161             if (b != 0 && b != 1) {
162                 onlyZeroAndOne = false;
163                 break;
164             }
165         }
166 
167         if (onlyZeroAndOne) {
168             return 1;
169         }
170 
171         // Must not be a multicast address
172         if ((candidate[0] & 1) != 0) {
173             return 1;
174         }
175 
176         // Prefer globally unique address.
177         if ((current[0] & 2) == 0) {
178             if ((candidate[0] & 2) == 0) {
179                 // Both current and candidate are globally unique addresses.
180                 return 0;
181             } else {
182                 // Only current is globally unique.
183                 return 1;
184             }
185         } else {
186             if ((candidate[0] & 2) == 0) {
187                 // Only candidate is globally unique.
188                 return -1;
189             } else {
190                 // Both current and candidate are non-unique.
191                 return 0;
192             }
193         }
194     }
195 
196     /**
197      * @return positive - current is better, 0 - cannot tell, negative - candidate is better
198      */
199     private static int compareAddresses(InetAddress current, InetAddress candidate) {
200         return scoreAddress(current) - scoreAddress(candidate);
201     }
202 
203     private static int scoreAddress(InetAddress addr) {
204         if (addr.isAnyLocalAddress() || addr.isLoopbackAddress()) {
205             return 0;
206         }
207         if (addr.isMulticastAddress()) {
208             return 1;
209         }
210         if (addr.isLinkLocalAddress()) {
211             return 2;
212         }
213         if (addr.isSiteLocalAddress()) {
214             return 3;
215         }
216 
217         return 4;
218     }
