651     static void copyMemory(long srcAddr, long dstAddr, long length) {
652         // Manual safe-point polling is only needed prior Java9:
653         // See https://bugs.openjdk.java.net/browse/JDK-8149596
654         if (javaVersion() <= 8) {
655             copyMemoryWithSafePointPolling(srcAddr, dstAddr, length);
656         } else {
657             UNSAFE.copyMemory(srcAddr, dstAddr, length);
658         }

661     private static void copyMemoryWithSafePointPolling(long srcAddr, long dstAddr, long length) {
662         while (length > 0) {
663             long size = Math.min(length, UNSAFE_COPY_THRESHOLD);
664             UNSAFE.copyMemory(srcAddr, dstAddr, size);

780     static int hashCodeAscii(byte[] bytes, int startPos, int length) {
781         int hash = HASH_CODE_ASCII_SEED;
782         long baseOffset = BYTE_ARRAY_BASE_OFFSET + startPos;
783         final int remainingBytes = length & 7;
784         final long end = baseOffset + remainingBytes;
785         for (long i = baseOffset - 8 + length; i >= end; i -= 8) {
786             hash = hashCodeAsciiCompute(UNSAFE.getLong(bytes, i), hash);
787         }
788         if (remainingBytes == 0) {
789             return hash;
790         }
791         int hcConst = HASH_CODE_C1;
792         if (remainingBytes != 2 & remainingBytes != 4 & remainingBytes != 6) { // 1, 3, 5, 7
793             hash = hash * HASH_CODE_C1 + hashCodeAsciiSanitize(UNSAFE.getByte(bytes, baseOffset));
794             hcConst = HASH_CODE_C2;
795             baseOffset++;
796         }
797         if (remainingBytes != 1 & remainingBytes != 4 & remainingBytes != 5) { // 2, 3, 6, 7
798             hash = hash * hcConst + hashCodeAsciiSanitize(UNSAFE.getShort(bytes, baseOffset));
799             hcConst = hcConst == HASH_CODE_C1 ? HASH_CODE_C2 : HASH_CODE_C1;
800             baseOffset += 2;
801         }
802         if (remainingBytes >= 4) { // 4, 5, 6, 7
803             return hash * hcConst + hashCodeAsciiSanitize(UNSAFE.getInt(bytes, baseOffset));
804         }
805         return hash;
806     }

808     static int hashCodeAsciiCompute(long value, int hash) {
809         // masking with 0x1f reduces the number of overall bits that impact the hash code but makes the hash
810         // code the same regardless of character case (upper case or lower case hash is the same).
811         return hash * HASH_CODE_C1 +
812                 // Low order int
813                 hashCodeAsciiSanitize((int) value) * HASH_CODE_C2 +
814                 // High order int
815                 (int) ((value & 0x1f1f1f1f00000000L) >>> 32);
816     }
