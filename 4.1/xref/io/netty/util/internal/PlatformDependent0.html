303     static void copyMemory(long srcAddr, long dstAddr, long length) {
304         //UNSAFE.copyMemory(srcAddr, dstAddr, length);
305         while (length > 0) {
306             long size = Math.min(length, UNSAFE_COPY_THRESHOLD);
307             UNSAFE.copyMemory(srcAddr, dstAddr, size);
308             length -= size;
309             srcAddr += size;
310             dstAddr += size;
311         }

325     static boolean equals(byte[] bytes1, int startPos1, byte[] bytes2, int startPos2, int length) {
326         if (length == 0) {
327             return true;
328         }
329         final long baseOffset1 = BYTE_ARRAY_BASE_OFFSET + startPos1;
330         final long baseOffset2 = BYTE_ARRAY_BASE_OFFSET + startPos2;
331         final int remainingBytes = length & 7;
332         final long end = baseOffset1 + remainingBytes;
333         for (long i = baseOffset1 - 8 + length, j = baseOffset2 - 8 + length; i >= end; i -= 8, j -= 8) {
334             if (UNSAFE.getLong(bytes1, i) != UNSAFE.getLong(bytes2, j)) {
335                 return false;
336             }
337         }
338         switch (remainingBytes) {
339         case 7:
340             return UNSAFE.getInt(bytes1, baseOffset1 + 3) == UNSAFE.getInt(bytes2, baseOffset2 + 3) &&
341                    UNSAFE.getChar(bytes1, baseOffset1 + 1) == UNSAFE.getChar(bytes2, baseOffset2 + 1) &&
342                    UNSAFE.getByte(bytes1, baseOffset1) == UNSAFE.getByte(bytes2, baseOffset2);
343         case 6:
344             return UNSAFE.getInt(bytes1, baseOffset1 + 2) == UNSAFE.getInt(bytes2, baseOffset2 + 2) &&
345                    UNSAFE.getChar(bytes1, baseOffset1) == UNSAFE.getChar(bytes2, baseOffset2);
346         case 5:
347             return UNSAFE.getInt(bytes1, baseOffset1 + 1) == UNSAFE.getInt(bytes2, baseOffset2 + 1) &&
348                    UNSAFE.getByte(bytes1, baseOffset1) == UNSAFE.getByte(bytes2, baseOffset2);
349         case 4:
350             return UNSAFE.getInt(bytes1, baseOffset1) == UNSAFE.getInt(bytes2, baseOffset2);
351         case 3:
352             return UNSAFE.getChar(bytes1, baseOffset1 + 1) == UNSAFE.getChar(bytes2, baseOffset2 + 1) &&
353                    UNSAFE.getByte(bytes1, baseOffset1) == UNSAFE.getByte(bytes2, baseOffset2);
354         case 2:
355             return UNSAFE.getChar(bytes1, baseOffset1) == UNSAFE.getChar(bytes2, baseOffset2);
356         case 1:
357             return UNSAFE.getByte(bytes1, baseOffset1) == UNSAFE.getByte(bytes2, baseOffset2);
358         default:
359             return true;
360         }
361     }

367     /**
368      * This must remain consistent with {@link #hashCodeAscii(char[])}.
369      */
370     static int hashCodeAscii(byte[] bytes, int startPos, int length) {
371         int hash = HASH_CODE_ASCII_SEED;
372         final long baseOffset = BYTE_ARRAY_BASE_OFFSET + startPos;
373         final int remainingBytes = length & 7;
374         if (length > 7) { // Fast path for small sized inputs. Benchmarking shows this is beneficial.
375             final long end = baseOffset + remainingBytes;
376             for (long i = baseOffset - 8 + length; i >= end; i -= 8) {
377                 hash = hashCodeAsciiCompute(UNSAFE.getLong(bytes, i), hash);
378             }
379         }
380         switch(remainingBytes) {
381         case 7:
382             return ((hash * 31 + Integer.rotateLeft(hashCodeAsciiSanitize(UNSAFE.getInt(bytes, baseOffset + 3)), 13))
383                      * 31 + hashCodeAsciiSanitize(UNSAFE.getShort(bytes, baseOffset + 1)))
384                        * 31 + hashCodeAsciiSanitize(UNSAFE.getByte(bytes, baseOffset));
385         case 6:
386             return (hash * 31 + Integer.rotateLeft(hashCodeAsciiSanitize(UNSAFE.getInt(bytes, baseOffset + 2)), 13))
387                     * 31 + hashCodeAsciiSanitize(UNSAFE.getShort(bytes, baseOffset));
388         case 5:
389             return (hash * 31 + Integer.rotateLeft(hashCodeAsciiSanitize(UNSAFE.getInt(bytes, baseOffset + 1)), 13))
390                     * 31 + hashCodeAsciiSanitize(UNSAFE.getByte(bytes, baseOffset));
391         case 4:
392             return hash * 31 + hashCodeAsciiSanitize(UNSAFE.getInt(bytes, baseOffset));
393         case 3:
394             return (hash * 31 + hashCodeAsciiSanitize(UNSAFE.getShort(bytes, baseOffset + 1)))
395                     * 31 + hashCodeAsciiSanitize(UNSAFE.getByte(bytes, baseOffset));
396         case 2:
397             return hash * 31 + hashCodeAsciiSanitize(UNSAFE.getShort(bytes, baseOffset));
398         case 1:
399             return hash * 31 + hashCodeAsciiSanitize(UNSAFE.getByte(bytes, baseOffset));
400         default:
401             return hash;
402         }
403     }

405     /**
406      * This method assumes that {@code bytes} is equivalent to a {@code byte[]} but just using {@code char[]}
407      * for storage. The MSB of each {@code char} from {@code bytes} is ignored.
408      * <p>
409      * This must remain consistent with {@link #hashCodeAscii(byte[], int, int)}.
410      */
411     static int hashCodeAscii(char[] bytes) {
412         int hash = HASH_CODE_ASCII_SEED;
413         final int remainingBytes = bytes.length & 7;
414         for (int i = bytes.length - 8; i >= remainingBytes; i -= 8) {
415             hash = hashCodeAsciiComputeFromChar(
416                                   UNSAFE.getLong(bytes, CHAR_ARRAY_BASE_OFFSET + i * CHAR_ARRAY_INDEX_SCALE),
417                                   UNSAFE.getLong(bytes, CHAR_ARRAY_BASE_OFFSET + (i + 4) * CHAR_ARRAY_INDEX_SCALE),
418                                   hash);
419         }
420         switch(remainingBytes) {
421         case 7:
422             return ((hash * 31 + Integer.rotateLeft(hashCodeAsciiSanitizeFromChar(
423                             UNSAFE.getLong(bytes, CHAR_ARRAY_BASE_OFFSET + 3 * CHAR_ARRAY_INDEX_SCALE)), 13))
424                      * 31 + hashCodeAsciiSanitizeFromChar(
425                              UNSAFE.getInt(bytes, CHAR_ARRAY_BASE_OFFSET + CHAR_ARRAY_INDEX_SCALE)))
426                        * 31 + hashCodeAsciiSanitizeFromChar(
427                                UNSAFE.getShort(bytes, CHAR_ARRAY_BASE_OFFSET));
428         case 6:
429             return (hash * 31 + Integer.rotateLeft(hashCodeAsciiSanitizeFromChar(
430                             UNSAFE.getLong(bytes, CHAR_ARRAY_BASE_OFFSET + 2 * CHAR_ARRAY_INDEX_SCALE)), 13))
431                     * 31 + hashCodeAsciiSanitizeFromChar(
432                             UNSAFE.getInt(bytes, CHAR_ARRAY_BASE_OFFSET));
433         case 5:
434             return (hash * 31 + Integer.rotateLeft(hashCodeAsciiSanitizeFromChar(
435                             UNSAFE.getLong(bytes, CHAR_ARRAY_BASE_OFFSET + CHAR_ARRAY_INDEX_SCALE)), 13))
436                     * 31 + hashCodeAsciiSanitizeFromChar(
437                             UNSAFE.getShort(bytes, CHAR_ARRAY_BASE_OFFSET));
438         case 4:
439             return hash * 31 + hashCodeAsciiSanitizeFromChar(
440                             UNSAFE.getLong(bytes, CHAR_ARRAY_BASE_OFFSET));
441         case 3:
442             return (hash * 31 + hashCodeAsciiSanitizeFromChar(
443                             UNSAFE.getInt(bytes, CHAR_ARRAY_BASE_OFFSET + CHAR_ARRAY_INDEX_SCALE)))
444                     * 31 + hashCodeAsciiSanitizeFromChar(
445                             UNSAFE.getShort(bytes, CHAR_ARRAY_BASE_OFFSET));
446         case 2:
447             return hash * 31 + hashCodeAsciiSanitizeFromChar(
448                             UNSAFE.getInt(bytes, CHAR_ARRAY_BASE_OFFSET));
449         case 1:
450             return hash * 31 + hashCodeAsciiSanitizeFromChar(
451                             UNSAFE.getShort(bytes, CHAR_ARRAY_BASE_OFFSET));
452         default:
453             return hash;
454         }

479     static int hashCodeAsciiCompute(long value, int hash) {
480         // masking with 0x1f reduces the number of overall bits that impact the hash code but makes the hash
481         // code the same regardless of character case (upper case or lower case hash is the same).
482         return (hash * 31 +
483                 // High order int
484                 (int) ((value & 0x1f1f1f1f00000000L) >>> 32)) * 31 +
485                 // Low order int
486                 hashCodeAsciiSanitize((int) value);
487     }

489     static int hashCodeAsciiComputeFromChar(long high, long low, int hash) {
490         // masking with 0x1f reduces the number of overall bits that impact the hash code but makes the hash
491         // code the same regardless of character case (upper case or lower case hash is the same).
492         return (hash * 31 +
493                 // High order int (which is low order for char)
494                 hashCodeAsciiSanitizeFromChar(low)) * 31 +
495                 // Low order int (which is high order for char)
496                 hashCodeAsciiSanitizeFromChar(high);
497     }

503     private static int hashCodeAsciiSanitizeFromChar(long value) {
504         return (int) (((value & 0x1f000000000000L) >>> 24) |
505                       ((value & 0x1f00000000L) >>> 16) |
506                       ((value & 0x1f0000) >>> 8) |
507                       (value & 0x1f));
