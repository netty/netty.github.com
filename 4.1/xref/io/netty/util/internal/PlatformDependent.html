153         // Here is how the system property is used:
154         //
155         // * <  0  - Don't use cleaner, and inherit max direct memory from java. In this case the
156         //           "practical max direct memory" would be 2 * max memory as defined by the JDK.
157         // * == 0  - Use cleaner, Netty will not enforce max memory, and instead will defer to JDK.
158         // * >  0  - Don't use cleaner. This will limit Netty's total direct memory
159         //           (note: that JDK's direct memory limit is independent of this).
160         long maxDirectMemory = SystemPropertyUtil.getLong("io.netty.maxDirectMemory", -1);
161 
162         if (maxDirectMemory == 0 || !hasUnsafe() || !PlatformDependent0.hasDirectBufferNoCleanerConstructor()) {
163             USE_DIRECT_BUFFER_NO_CLEANER = false;
164             DIRECT_MEMORY_COUNTER = null;
165         } else {
166             USE_DIRECT_BUFFER_NO_CLEANER = true;
167             if (maxDirectMemory < 0) {
168                 maxDirectMemory = MAX_DIRECT_MEMORY;
169                 if (maxDirectMemory <= 0) {
170                     DIRECT_MEMORY_COUNTER = null;
171                 } else {
172                     DIRECT_MEMORY_COUNTER = new AtomicLong();
173                 }
174             } else {
175                 DIRECT_MEMORY_COUNTER = new AtomicLong();
176             }

179         DIRECT_MEMORY_LIMIT = maxDirectMemory >= 1 ? maxDirectMemory : MAX_DIRECT_MEMORY;

182                 SystemPropertyUtil.getInt("io.netty.uninitializedArrayAllocationThreshold", 1024);
183         UNINITIALIZED_ARRAY_ALLOCATION_THRESHOLD = javaVersion() >= 9 && PlatformDependent0.hasAllocateArrayMethod() ?
184                 tryAllocateUninitializedArray : -1;
185         logger.debug("-Dio.netty.uninitializedArrayAllocationThreshold: {}", UNINITIALIZED_ARRAY_ALLOCATION_THRESHOLD);
186 
187         MAYBE_SUPER_USER = maybeSuperUser0();
188 
189         if (!isAndroid()) {
190             // only direct to method if we are not running on android.
191             // See https://github.com/netty/netty/issues/2604
192             if (javaVersion() >= 9) {
193                 CLEANER = CleanerJava9.isSupported() ? new CleanerJava9() : NOOP;
194             } else {

196             }
197         } else {

202         DIRECT_BUFFER_PREFERRED = CLEANER != NOOP

220                 new HashSet<String>(Arrays.asList(ALLOWED_LINUX_OS_CLASSIFIERS)));
221         final Set<String> availableClassifiers = new LinkedHashSet<String>();
222         for (final String osReleaseFileName : OS_RELEASE_FILES) {
223             final File file = new File(osReleaseFileName);
224             boolean found = AccessController.doPrivileged(new PrivilegedAction<Boolean>() {
225                 @Override
226                 public Boolean run() {
227                     try {
228                         if (file.exists()) {
229                             BufferedReader reader = null;
230                             try {
231                                 reader = new BufferedReader(
232                                         new InputStreamReader(
233                                                 new FileInputStream(file), CharsetUtil.UTF_8));
234 
235                                 String line;
236                                 while ((line = reader.readLine()) != null) {
237                                     if (line.startsWith(LINUX_ID_PREFIX)) {
238                                         String id = normalizeOsReleaseVariableValue(
239                                                 line.substring(LINUX_ID_PREFIX.length()));
240                                         addClassifier(allowedClassifiers, availableClassifiers, id);
241                                     } else if (line.startsWith(LINUX_ID_LIKE_PREFIX)) {
242                                         line = normalizeOsReleaseVariableValue(
243                                                 line.substring(LINUX_ID_LIKE_PREFIX.length()));
244                                         addClassifier(allowedClassifiers, availableClassifiers, line.split("[ ]+"));
245                                     }
246                                 }
247                             } catch (SecurityException e) {
248                                 logger.debug("Unable to read {}", osReleaseFileName, e);
249                             } catch (IOException e) {
250                                 logger.debug("Error while reading content of {}", osReleaseFileName, e);
251                             } finally {
252                                 if (reader != null) {
253                                     try {
254                                         reader.close();
255                                     } catch (IOException ignored) {
256                                         // Ignore
257                                     }
258                                 }
259                             }
260                             // specification states we should only fall back if /etc/os-release does not exist
261                             return true;
262                         }
263                     } catch (SecurityException e) {
264                         logger.debug("Unable to check if {} exists", osReleaseFileName, e);
265                     }
266                     return false;
267                 }
268             });
269 
270             if (found) {
271                 break;
272             }
273         }
274         LINUX_OS_CLASSIFIERS = Collections.unmodifiableSet(availableClassifiers);
275     }

559     private static long getLongSafe(byte[] bytes, int offset) {
560         if (BIG_ENDIAN_NATIVE_ORDER) {
561             return (long) bytes[offset] << 56 |
562                     ((long) bytes[offset + 1] & 0xff) << 48 |
563                     ((long) bytes[offset + 2] & 0xff) << 40 |
564                     ((long) bytes[offset + 3] & 0xff) << 32 |
565                     ((long) bytes[offset + 4] & 0xff) << 24 |
566                     ((long) bytes[offset + 5] & 0xff) << 16 |
567                     ((long) bytes[offset + 6] & 0xff) <<  8 |
568                     (long) bytes[offset + 7] & 0xff;
569         }
570         return (long) bytes[offset] & 0xff |
571                 ((long) bytes[offset + 1] & 0xff) << 8 |
572                 ((long) bytes[offset + 2] & 0xff) << 16 |
573                 ((long) bytes[offset + 3] & 0xff) << 24 |
574                 ((long) bytes[offset + 4] & 0xff) << 32 |
575                 ((long) bytes[offset + 5] & 0xff) << 40 |
576                 ((long) bytes[offset + 6] & 0xff) << 48 |
577                 (long) bytes[offset + 7] << 56;
578     }

580     private static int getIntSafe(byte[] bytes, int offset) {
581         if (BIG_ENDIAN_NATIVE_ORDER) {
582             return bytes[offset] << 24 |
583                     (bytes[offset + 1] & 0xff) << 16 |
584                     (bytes[offset + 2] & 0xff) << 8 |
585                     bytes[offset + 3] & 0xff;
586         }
587         return bytes[offset] & 0xff |
588                 (bytes[offset + 1] & 0xff) << 8 |
589                 (bytes[offset + 2] & 0xff) << 16 |
590                 bytes[offset + 3] << 24;
591     }

603     private static int hashCodeAsciiCompute(CharSequence value, int offset, int hash) {
604         if (BIG_ENDIAN_NATIVE_ORDER) {
605             return hash * HASH_CODE_C1 +

614                 // High order int
615                 hashCodeAsciiSanitizeInt(value, offset + 4);

618     /**
619      * Identical to {@link PlatformDependent0#hashCodeAsciiSanitize(int)} but for {@link CharSequence}.
620      */
621     private static int hashCodeAsciiSanitizeInt(CharSequence value, int offset) {
622         if (BIG_ENDIAN_NATIVE_ORDER) {
623             // mimic a unsafe.getInt call on a big endian machine

892     /**
893      * Calculate a hash code of a byte array assuming ASCII character encoding.
894      * The resulting hash code will be case insensitive.
895      * <p>
896      * This method assumes that {@code bytes} is equivalent to a {@code byte[]} but just using {@link CharSequence}
897      * for storage. The upper most byte of each {@code char} from {@code bytes} is ignored.
898      * @param bytes The array which contains the data to hash (assumed to be equivalent to a {@code byte[]}).
899      * @return The hash code of {@code bytes} assuming ASCII character encoding.
900      * The resulting hash code will be case insensitive.
901      */
902     public static int hashCodeAscii(CharSequence bytes) {
903         final int length = bytes.length();
904         final int remainingBytes = length & 7;
905         int hash = HASH_CODE_ASCII_SEED;
906         // Benchmarking shows that by just naively looping for inputs 8~31 bytes long we incur a relatively large
907         // performance penalty (only achieve about 60% performance of loop which iterates over each char). So because
908         // of this we take special provisions to unroll the looping for these conditions.
909         if (length >= 32) {
910             for (int i = length - 8; i >= remainingBytes; i -= 8) {
911                 hash = hashCodeAsciiCompute(bytes, i, hash);
912             }
913         } else if (length >= 8) {
914             hash = hashCodeAsciiCompute(bytes, length - 8, hash);
915             if (length >= 16) {
916                 hash = hashCodeAsciiCompute(bytes, length - 16, hash);
917                 if (length >= 24) {
918                     hash = hashCodeAsciiCompute(bytes, length - 24, hash);
919                 }
920             }
921         }
922         if (remainingBytes == 0) {
923             return hash;
924         }
925         int offset = 0;
926         if (remainingBytes != 2 & remainingBytes != 4 & remainingBytes != 6) { // 1, 3, 5, 7
927             hash = hash * HASH_CODE_C1 + hashCodeAsciiSanitizeByte(bytes.charAt(0));
928             offset = 1;
929         }
930         if (remainingBytes != 1 & remainingBytes != 4 & remainingBytes != 5) { // 2, 3, 6, 7
931             hash = hash * (offset == 0 ? HASH_CODE_C1 : HASH_CODE_C2)
932                     + hashCodeAsciiSanitize(hashCodeAsciiSanitizeShort(bytes, offset));
933             offset += 2;
934         }
935         if (remainingBytes >= 4) { // 4, 5, 6, 7
936             return hash * ((offset == 0 | offset == 3) ? HASH_CODE_C1 : HASH_CODE_C2)
937                     + hashCodeAsciiSanitizeInt(bytes, offset);
938         }
939         return hash;
940     }
