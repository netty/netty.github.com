31  /**
32   * A lock-free concurrent single-consumer multi-producer {@link Queue}.
33   * It allows multiple producer threads to perform the following operations simultaneously:
34   * <ul>
35   *     <li>{@link #offer(Object)}, {@link #add(Object)}, {@link #addAll(Collection)}</li>
36   *     <li>{@link #isEmpty()}</li>
37   * </ul>
38   * .. while only one consumer thread is allowed to perform the following operations exclusively:
39   * <ul>
40   *     <li>{@link #poll()} and {@link #remove()}</li>
41   *     <li>{@link #element()}, {@link #peek()}</li>
42   *     <li>{@link #remove(Object)}, {@link #removeAll(Collection)}, and {@link #retainAll(Collection)}</li>
43   *     <li>{@link #clear()}</li>
44   *     <li>{@link #iterator()}</li>
45   *     <li>{@link #toArray()} and {@link #toArray(Object[])}</li>
46   *     <li>{@link #contains(Object)} and {@link #containsAll(Collection)}</li>
47   *     <li>{@link #size()}</li>
48   * </ul>
49   *
50   * <strong>The behavior of this implementation is undefined if you perform the operations for a consumer thread only
51   * from multiple threads.</strong>
52   *
53   * The initial implementation is based on:
54   * <ul>
55   *   <li><a href="http://netty.io/s/mpsc-1024c">Non-intrusive MPSC node based queue</a> from 1024cores.net</li>
56   *   <li><a href="http://netty.io/s/mpsc-akka">AbstractNodeQueue</a> from Akka</li>
57   * </ul>
58   * and adopted padded head node changes from:
59   * <ul>
60   * <li><a href="http://netty.io/s/mpsc-rxjava">MpscPaddedQueue</a> from RxJava</li>
61   * </ul>
62   * data structure modified to avoid false sharing between head and tail Ref as per implementation of MpscLinkedQueue
63   * on <a href="https://github.com/JCTools/JCTools">JCTools project</a>.
64   */
65  final class MpscLinkedQueue<E> extends MpscLinkedQueueTailRef<E> implements Queue<E> {

72      // offer() occurs at the tail of the linked list.
73      // poll() occurs at the head of the linked list.
74      //
75      // Resulting layout is:
76      //
77      //   head --next--> 1st element --next--> 2nd element --next--> ... tail (last element)
78      //
79      // where the head is a dummy node whose value is null.
80      //
81      // offer() appends a new node next to the tail using AtomicReference.getAndSet()
82      // poll() removes head from the linked list and promotes the 1st element to the head,
83      // setting its value to null if possible.
84      //
85      // Also note that this class extends AtomicReference for the "tail" slot (which is the one that is appended to)
86      // since Unsafe does not expose XCHG operation intrinsically.
87      MpscLinkedQueue() {
88          MpscLinkedQueueNode<E> tombstone = new DefaultNode<E>(null);
89          setHeadRef(tombstone);
90          setTailRef(tombstone);
91      }

93      /**
94       * Returns the node right next to the head, which contains the first element of this queue.
95       */
96      private MpscLinkedQueueNode<E> peekNode() {
97          MpscLinkedQueueNode<E> head = headRef();
98          MpscLinkedQueueNode<E> next = head.next();
99          if (next == null && head != tailRef()) {
100             // if tail != head this is not going to change until consumer makes progress
101             // we can avoid reading the head and just spin on next until it shows up
102             //
103             // See https://github.com/akka/akka/pull/15596
104             do {
105                 next = head.next();
106             } while (next == null);
107         }
108         return next;
109     }

111     @Override
112     @SuppressWarnings("unchecked")
113     public boolean offer(E value) {
114         if (value == null) {
115             throw new NullPointerException("value");
116         }
117 
118         final MpscLinkedQueueNode<E> newTail;
119         if (value instanceof MpscLinkedQueueNode) {
120             newTail = (MpscLinkedQueueNode<E>) value;
121             newTail.setNext(null);
122         } else {
123             newTail = new DefaultNode<E>(value);
124         }
125 
126         MpscLinkedQueueNode<E> oldTail = getAndSetTailRef(newTail);
127         oldTail.setNext(newTail);
128         return true;
129     }
130 
131     @Override
132     public E poll() {
133         final MpscLinkedQueueNode<E> next = peekNode();
134         if (next == null) {
135             return null;
136         }
137 
138         // next becomes a new head.
139         MpscLinkedQueueNode<E> oldHead = headRef();
140         // Similar to 'headRef.node = next', but slightly faster (storestore vs loadstore)
141         // See: http://robsjava.blogspot.com/2013/06/a-faster-volatile.html
142         // See: http://psy-lob-saw.blogspot.com/2012/12/atomiclazyset-is-performance-win-for.html
143         lazySetHeadRef(next);
144 
145         // Break the linkage between the old head and the new head.
146         oldHead.unlink();
147 
148         return next.clearMaybe();
149     }

151     @Override
152     public E peek() {
153         final MpscLinkedQueueNode<E> next = peekNode();
154         if (next == null) {
155             return null;
156         }
157         return next.value();
158     }

160     @Override
161     public int size() {
162         int count = 0;
163         MpscLinkedQueueNode<E> n = peekNode();
164          for (;;) {
165             // If value == null it means that clearMaybe() was called on the MpscLinkedQueueNode.
166             if (n == null || n.value() == null) {
167                 break;
168             }
169             MpscLinkedQueueNode<E> next = n.next();
170             if (n == next) {
171                 break;
172             }
173             n = next;
174             if (++ count == Integer.MAX_VALUE) {
175                 // Guard against overflow of integer.
176                 break;
177             }
178         }
179         return count;
180     }

182     @Override
183     public boolean isEmpty() {
184         return headRef() == tailRef();
185     }
186 
187     @Override
188     public boolean contains(Object o) {
189         MpscLinkedQueueNode<E> n = peekNode();
190         for (;;) {
191             if (n == null) {
192                 break;
193             }
194             E value = n.value();
195             // If value == null it means that clearMaybe() was called on the MpscLinkedQueueNode.
196             if (value == null) {
197                 return false;
198             }
199             if (value == o) {
200                 return true;
201             }
202             MpscLinkedQueueNode<E> next = n.next();
203             if (n == next) {
204                 break;
205             }
206             n = next;
207         }
208         return false;
209     }

216     @Override

218         if (offer(e)) {
219             return true;
220         }
221         throw new IllegalStateException("queue full");
222     }

225     public E remove() {
226         E e = poll();

229         }
230         throw new NoSuchElementException();

250     private List<E> toList(List<E> elements) {
251         MpscLinkedQueueNode<E> n = peekNode();
252         for (;;) {
253             if (n == null) {
254                 break;
255             }
256             E value = n.value();
257             if (value == null) {
258                 break;
259             }
260             if (!elements.add(value)) {
261                 // Seems like there is no space left, break here.
262                 break;
263             }
264             MpscLinkedQueueNode<E> next = n.next();
265             if (n == next) {
266                 break;
267             }
268             n = next;
269         }
270         return elements;
271     }

299     @Override
300     public boolean addAll(Collection<? extends E> c) {
301         if (c == null) {
302             throw new NullPointerException("c");
303         }
304         if (c == this) {
305             throw new IllegalArgumentException("c == this");
306         }
307 
308         boolean modified = false;
309         for (E e: c) {
310             add(e);
311             modified = true;

333     private void writeObject(ObjectOutputStream out) throws IOException {
334         out.defaultWriteObject();
335         for (E e: this) {
336             out.writeObject(e);
337         }
338         out.writeObject(null);
339     }

341     private void readObject(ObjectInputStream in) throws IOException, ClassNotFoundException {
342         in.defaultReadObject();
343 
344         final MpscLinkedQueueNode<E> tombstone = new DefaultNode<E>(null);
345         setHeadRef(tombstone);
346         setTailRef(tombstone);
347 
348         for (;;) {
349             @SuppressWarnings("unchecked")
350             E e = (E) in.readObject();
351             if (e == null) {
352                 break;
353             }
354             add(e);
355         }
356     }

358     private static final class DefaultNode<T> extends MpscLinkedQueueNode<T> {

373             T value = this.value;
