49      AbstractChannelHandlerContext(
50              DefaultChannelPipeline pipeline, ChannelHandlerInvoker invoker,
51              String name, boolean inbound, boolean outbound) {

299     private AbstractChannelHandlerContext findContextInbound() {
300         AbstractChannelHandlerContext ctx = this;
301         do {
302             ctx = ctx.next;
303         } while (!ctx.inbound);
304         return ctx;
305     }

307     private AbstractChannelHandlerContext findContextOutbound() {
308         AbstractChannelHandlerContext ctx = this;
309         do {
310             ctx = ctx.prev;
311         } while (!ctx.outbound);
312         return ctx;
313     }
