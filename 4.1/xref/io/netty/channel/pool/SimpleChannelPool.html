36  /**
37   * Simple {@link ChannelPool} implementation which will create new {@link Channel}s if someone tries to acquire
38   * a {@link Channel} but none is in the pool atm. No limit on the maximal concurrent {@link Channel}s is enforced.
39   *
40   * This implementation uses LIFO order for {@link Channel}s in the {@link ChannelPool}.
41   *
42   */
43  public class SimpleChannelPool implements ChannelPool {
44      private static final AttributeKey<SimpleChannelPool> POOL_KEY = AttributeKey.newInstance("channelPool");
45      private static final IllegalStateException FULL_EXCEPTION = new IllegalStateException("ChannelPool full");
46      private static final IllegalStateException UNHEALTHY_NON_OFFERED_TO_POOL =
47              new IllegalStateException("Channel is unhealthy not offering it back to pool");
48      static {
49          FULL_EXCEPTION.setStackTrace(EmptyArrays.EMPTY_STACK_TRACE);
50          UNHEALTHY_NON_OFFERED_TO_POOL.setStackTrace(EmptyArrays.EMPTY_STACK_TRACE);
51      }
52      private final Deque<Channel> deque = PlatformDependent.newConcurrentDeque();
53      private final ChannelPoolHandler handler;
54      private final ChannelHealthChecker healthCheck;
55      private final Bootstrap bootstrap;
56      private final boolean releaseHealthCheck;

80      /**
81       * Creates a new instance.
82       *
83       * @param bootstrap          theÂ {@link Bootstrap} that is used for connections
84       * @param handler            the {@link ChannelPoolHandler} that will be notified for the different pool actions
85       * @param healthCheck        the {@link ChannelHealthChecker} that will be used to check if a {@link Channel} is
86       *                           still healthy when obtain from the {@link ChannelPool}
87       * @param releaseHealthCheck will offercheck channel health before offering back if this parameter set to
88       *                           {@code true}.
89       */
90      public SimpleChannelPool(Bootstrap bootstrap, final ChannelPoolHandler handler, ChannelHealthChecker healthCheck,
91                               boolean releaseHealthCheck) {
92          this.handler = checkNotNull(handler, "handler");
93          this.healthCheck = checkNotNull(healthCheck, "healthCheck");
94          this.releaseHealthCheck = releaseHealthCheck;
95          // Clone the original Bootstrap as we want to set our own handler
96          this.bootstrap = checkNotNull(bootstrap, "bootstrap").clone();
97          this.bootstrap.handler(new ChannelInitializer<Channel>() {
98              @Override
99              protected void initChannel(Channel ch) throws Exception {
100                 assert ch.eventLoop().inEventLoop();
101                 handler.channelCreated(ch);
102             }
103         });
104     }

113         checkNotNull(promise, "promise");
114         return acquireHealthyFromPoolOrNew(promise);

117     /**
118      * Tries to retrieve healthy channel from the pool if any or creates a new channel otherwise.
119      * @param promise the promise to provide acquire result.
120      * @return future for acquiring a channel.
121      */
122     private Future<Channel> acquireHealthyFromPoolOrNew(final Promise<Channel> promise) {
123         try {
124             final Channel ch = pollChannel();
125             if (ch == null) {
126                 // No Channel left in the pool bootstrap a new Channel
127                 Bootstrap bs = bootstrap.clone();
128                 bs.attr(POOL_KEY, this);
129                 ChannelFuture f = connectChannel(bs);
130                 if (f.isDone()) {
131                     notifyConnect(f, promise);
132                 } else {
133                     f.addListener(new ChannelFutureListener() {
134                         @Override
135                         public void operationComplete(ChannelFuture future) throws Exception {
136                             notifyConnect(future, promise);
137                         }
138                     });
139                 }
140                 return promise;
141             }
142             EventLoop loop = ch.eventLoop();
143             if (loop.inEventLoop()) {
144                 doHealthCheck(ch, promise);
145             } else {
146                 loop.execute(new OneTimeTask() {
147                     @Override
148                     public void run() {
149                         doHealthCheck(ch, promise);
150                     }
151                 });
152             }
153         } catch (Throwable cause) {
154             promise.setFailure(cause);
155         }
156         return promise;
157     }
158 
159     private static void notifyConnect(ChannelFuture future, Promise<Channel> promise) {
160         if (future.isSuccess()) {
161             promise.setSuccess(future.channel());
162         } else {
163             promise.setFailure(future.cause());
164         }
165     }

167     private void doHealthCheck(final Channel ch, final Promise<Channel> promise) {
168         assert ch.eventLoop().inEventLoop();
169 
170         Future<Boolean> f = healthCheck.isHealthy(ch);
171         if (f.isDone()) {
172             notifyHealthCheck(f, ch, promise);
173         } else {
174             f.addListener(new FutureListener<Boolean>() {
175                 @Override
176                 public void operationComplete(Future<Boolean> future) throws Exception {
177                     notifyHealthCheck(future, ch, promise);
178                 }
179             });
180         }
181     }
182 
183     private void notifyHealthCheck(Future<Boolean> future, Channel ch, Promise<Channel> promise) {
184         assert ch.eventLoop().inEventLoop();
185 
186         if (future.isSuccess()) {
187             if (future.getNow()) {
188                 try {
189                     ch.attr(POOL_KEY).set(this);
190                     handler.channelAcquired(ch);
191                     promise.setSuccess(ch);
192                 } catch (Throwable cause) {
193                     closeAndFail(ch, cause, promise);
194                 }
195             } else {
196                 closeChannel(ch);
197                 acquireHealthyFromPoolOrNew(promise);
198             }
199         } else {
200             closeChannel(ch);
201             acquireHealthyFromPoolOrNew(promise);
202         }
203     }

220     @Override
221     public Future<Void> release(final Channel channel, final Promise<Void> promise) {
222         checkNotNull(channel, "channel");
223         checkNotNull(promise, "promise");
224         try {
225             EventLoop loop = channel.eventLoop();
226             if (loop.inEventLoop()) {
227                 doReleaseChannel(channel, promise);
228             } else {
229                 loop.execute(new OneTimeTask() {
230                     @Override
231                     public void run() {
232                         doReleaseChannel(channel, promise);
233                     }
234                 });
235             }
236         } catch (Throwable cause) {
237             closeAndFail(channel, cause, promise);
238         }
239         return promise;
240     }
241 
242     private void doReleaseChannel(Channel channel, Promise<Void> promise) {
243         assert channel.eventLoop().inEventLoop();
244         // Remove the POOL_KEY attribute from the Channel and check if it was acquired from this pool, if not fail.
245         if (channel.attr(POOL_KEY).getAndSet(null) != this) {
246             closeAndFail(channel,
247                          // Better include a stracktrace here as this is an user error.
248                          new IllegalArgumentException(
249                                  "Channel " + channel + " was not acquired from this ChannelPool"),
250                          promise);
251         } else {
252             try {
253                 if (releaseHealthCheck) {
254                     doHealthCheckOnRelease(channel, promise);
255                 } else {
256                     releaseAndOffer(channel, promise);
257                 }
258             } catch (Throwable cause) {
259                 closeAndFail(channel, cause, promise);
260             }
261         }
262     }
263 
264     private void doHealthCheckOnRelease(final Channel channel, final Promise<Void> promise) throws Exception {
265         final Future<Boolean> f = healthCheck.isHealthy(channel);
266         if (f.isDone()) {
267             releaseAndOfferIfHealthy(channel, promise, f);
268         } else {
269             f.addListener(new FutureListener<Boolean>() {
270                 @Override
271                 public void operationComplete(Future<Boolean> future) throws Exception {
272                     releaseAndOfferIfHealthy(channel, promise, f);
273                 }
274             });
275         }
276     }

278     /**
279      * Adds the channel back to the pool only if the channel is healty.
280      * @param channel the channel to put back to the pool
281      * @param promise offer operation promise.
282      * @param future the future that contains information fif channel is healthy or not.
283      * @throws Exception in case when failed to notify handler about release operation.
284      */
285     private void releaseAndOfferIfHealthy(Channel channel, Promise<Void> promise, Future<Boolean> future)
286             throws Exception {
287         if (future.getNow()) { //channel turns out to be healthy, offering and releasing it.
288             releaseAndOffer(channel, promise);
289         } else { //channel ont healthy, just releasing it.
290             handler.channelReleased(channel);
291             closeAndFail(channel, UNHEALTHY_NON_OFFERED_TO_POOL, promise);
292         }
293     }
294 
295     private void releaseAndOffer(Channel channel, Promise<Void> promise) throws Exception {
296         if (offerChannel(channel)) {
297             handler.channelReleased(channel);
298             promise.setSuccess(null);
299         } else {
300             closeAndFail(channel, FULL_EXCEPTION, promise);
301         }
302     }
303 
304     private static void closeChannel(Channel channel) {
305         channel.attr(POOL_KEY).getAndSet(null);
306         channel.close();
307     }
308 
309     private static void closeAndFail(Channel channel, Throwable cause, Promise<?> promise) {
310         closeChannel(channel);
311         promise.setFailure(cause);
312     }

314     /**
315      * Poll a {@link Channel} out of the internal storage to reuse it. This will return {@code null} if no
316      * {@link Channel} is ready to be reused.
317      *
318      * Sub-classes may override {@link #pollChannel()} and {@link #offerChannel(Channel)}. Be aware that
319      * implementations of these methods needs to be thread-safe!
320      */
321     protected Channel pollChannel() {
322         return deque.pollLast();
323     }
324 
325     /**
326      * Offer a {@link Channel} back to the internal storage. This will return {@code true} if the {@link Channel}
327      * could be added, {@code false} otherwise.
328      *
329      * Sub-classes may override {@link #pollChannel()} and {@link #offerChannel(Channel)}. Be aware that
330      * implementations of these methods needs to be thread-safe!
331      */
332     protected boolean offerChannel(Channel channel) {
333         return deque.offer(channel);
334     }

336     @Override
337     public void close() {
338         for (;;) {
339             Channel channel = pollChannel();
340             if (channel == null) {
341                 break;
342             }
343             channel.close();
344         }
345     }
