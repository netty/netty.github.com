42      private static final Pattern MACHINE_ID_PATTERN = Pattern.compile("^(?:[0-9a-fA-F][:-]?){6,8}$");
43      private static final int MACHINE_ID_LEN = MacAddressUtil.MAC_ADDRESS_LENGTH;
44      private static final byte[] MACHINE_ID;
45      private static final int PROCESS_ID_LEN = 4;
46      // Maximal value for 64bit systems is 2^22.  See man 5 proc.
47      // See https://github.com/netty/netty/issues/2706
48      private static final int MAX_PROCESS_ID = 4194304;
49      private static final int PROCESS_ID;
50      private static final int SEQUENCE_LEN = 4;
51      private static final int TIMESTAMP_LEN = 8;
52      private static final int RANDOM_LEN = 4;

54      private static final AtomicInteger nextSequence = new AtomicInteger();

56      static ChannelId newInstance() {
57          DefaultChannelId id = new DefaultChannelId();
58          id.init();
59          return id;
60      }

62      static {
63          int processId = -1;
64          String customProcessId = SystemPropertyUtil.get("io.netty.processId");
65          if (customProcessId != null) {
66              try {
67                  processId = Integer.parseInt(customProcessId);
68              } catch (NumberFormatException e) {
69                  // Malformed input.
70              }
71  
72              if (processId < 0 || processId > MAX_PROCESS_ID) {
73                  processId = -1;
74                  logger.warn("-Dio.netty.processId: {} (malformed)", customProcessId);
75              } else if (logger.isDebugEnabled()) {
76                  logger.debug("-Dio.netty.processId: {} (user-set)", processId);
77              }
78          }
79  
80          if (processId < 0) {
81              processId = defaultProcessId();
82              if (logger.isDebugEnabled()) {
83                  logger.debug("-Dio.netty.processId: {} (auto-detected)", processId);
84              }
85          }
86  
87          PROCESS_ID = processId;
88  
89          byte[] machineId = null;
90          String customMachineId = SystemPropertyUtil.get("io.netty.machineId");
91          if (customMachineId != null) {
92              if (MACHINE_ID_PATTERN.matcher(customMachineId).matches()) {
93                  machineId = parseMachineId(customMachineId);
94                  logger.debug("-Dio.netty.machineId: {} (user-set)", customMachineId);
95              } else {
96                  logger.warn("-Dio.netty.machineId: {} (malformed)", customMachineId);
97              }
98          }
99  
100         if (machineId == null) {
101             machineId = defaultMachineId();
102             if (logger.isDebugEnabled()) {
103                 logger.debug("-Dio.netty.machineId: {} (auto-detected)", MacAddressUtil.formatAddress(machineId));
104             }
105         }
106 
107         MACHINE_ID = machineId;
108     }

110     @SuppressWarnings("DynamicRegexReplaceableByCompiledPattern")
111     private static byte[] parseMachineId(String value) {
112         // Strip separators.
113         value = value.replaceAll("[:-]", "");
114 
115         byte[] machineId = new byte[MACHINE_ID_LEN];
116         for (int i = 0; i < value.length(); i += 2) {
117             machineId[i] = (byte) Integer.parseInt(value.substring(i, i + 2), 16);
118         }
119 
120         return machineId;
121     }

123     private static byte[] defaultMachineId() {
124         byte[] bestMacAddr = MacAddressUtil.bestAvailableMac();
125         if (bestMacAddr == null) {
126             bestMacAddr = new byte[MacAddressUtil.MAC_ADDRESS_LENGTH];
127             ThreadLocalRandom.current().nextBytes(bestMacAddr);
128             logger.warn(
129                     "Failed to find a usable hardware address from the network interfaces; using random bytes: {}",
130                     MacAddressUtil.formatAddress(bestMacAddr));
131         }
132         return bestMacAddr;
133     }

135     private static int defaultProcessId() {
136         final ClassLoader loader = PlatformDependent.getSystemClassLoader();
137         String value;
138         try {
139             // Invoke java.lang.management.ManagementFactory.getRuntimeMXBean().getName()
140             Class<?> mgmtFactoryType = Class.forName("java.lang.management.ManagementFactory", true, loader);
141             Class<?> runtimeMxBeanType = Class.forName("java.lang.management.RuntimeMXBean", true, loader);
142 
143             Method getRuntimeMXBean = mgmtFactoryType.getMethod("getRuntimeMXBean", EmptyArrays.EMPTY_CLASSES);
144             Object bean = getRuntimeMXBean.invoke(null, EmptyArrays.EMPTY_OBJECTS);
145             Method getName = runtimeMxBeanType.getDeclaredMethod("getName", EmptyArrays.EMPTY_CLASSES);
146             value = (String) getName.invoke(bean, EmptyArrays.EMPTY_OBJECTS);
147         } catch (Exception e) {
148             logger.debug("Could not invoke ManagementFactory.getRuntimeMXBean().getName(); Android?", e);
149             try {
150                 // Invoke android.os.Process.myPid()
151                 Class<?> processType = Class.forName("android.os.Process", true, loader);
152                 Method myPid = processType.getMethod("myPid", EmptyArrays.EMPTY_CLASSES);
153                 value = myPid.invoke(null, EmptyArrays.EMPTY_OBJECTS).toString();
154             } catch (Exception e2) {
155                 logger.debug("Could not invoke Process.myPid(); not Android?", e2);
156                 value = "";
157             }
158         }
159 
160         int atIndex = value.indexOf('@');
161         if (atIndex >= 0) {
162             value = value.substring(0, atIndex);
163         }
164 
165         int pid;
166         try {
167             pid = Integer.parseInt(value);
168         } catch (NumberFormatException e) {
169             // value did not contain an integer.
170             pid = -1;
171         }
172 
173         if (pid < 0 || pid > MAX_PROCESS_ID) {
174             pid = ThreadLocalRandom.current().nextInt(MAX_PROCESS_ID + 1);
175             logger.warn("Failed to find the current process ID from '{}'; using a random value: {}",  value, pid);
176         }
177 
178         return pid;
179     }

181     private final byte[] data = new byte[MACHINE_ID_LEN + PROCESS_ID_LEN + SEQUENCE_LEN + TIMESTAMP_LEN + RANDOM_LEN];
182     private int hashCode;
183 
184     private transient String shortValue;
185     private transient String longValue;

187     private void init() {
188         int i = 0;
189 
190         // machineId
191         System.arraycopy(MACHINE_ID, 0, data, i, MACHINE_ID_LEN);
192         i += MACHINE_ID_LEN;
193 
194         // processId
195         i = writeInt(i, PROCESS_ID);
196 
197         // sequence
198         i = writeInt(i, nextSequence.getAndIncrement());
199 
200         // timestamp (kind of)
201         i = writeLong(i, Long.reverse(System.nanoTime()) ^ System.currentTimeMillis());
202 
203         // random
204         int random = ThreadLocalRandom.current().nextInt();
205         hashCode = random;
206         i = writeInt(i, random);
207 
208         assert i == data.length;
209     }
210 
211     private int writeInt(int i, int value) {
212         data[i ++] = (byte) (value >>> 24);
213         data[i ++] = (byte) (value >>> 16);
214         data[i ++] = (byte) (value >>> 8);
215         data[i ++] = (byte) value;
216         return i;
217     }
218 
219     private int writeLong(int i, long value) {
220         data[i ++] = (byte) (value >>> 56);
221         data[i ++] = (byte) (value >>> 48);
222         data[i ++] = (byte) (value >>> 40);
223         data[i ++] = (byte) (value >>> 32);
224         data[i ++] = (byte) (value >>> 24);
225         data[i ++] = (byte) (value >>> 16);
226         data[i ++] = (byte) (value >>> 8);
227         data[i ++] = (byte) value;
228         return i;
229     }

241     @Override
242     public String asLongText() {
243         String longValue = this.longValue;
244         if (longValue == null) {
245             this.longValue = longValue = newLongValue();
246         }
247         return longValue;
248     }
249 
250     private String newLongValue() {
251         StringBuilder buf = new StringBuilder(2 * data.length + 5);
252         int i = 0;
253         i = appendHexDumpField(buf, i, MACHINE_ID_LEN);
254         i = appendHexDumpField(buf, i, PROCESS_ID_LEN);
255         i = appendHexDumpField(buf, i, SEQUENCE_LEN);
256         i = appendHexDumpField(buf, i, TIMESTAMP_LEN);
257         i = appendHexDumpField(buf, i, RANDOM_LEN);
258         assert i == data.length;
259         return buf.substring(0, buf.length() - 1);
260     }
261 
262     private int appendHexDumpField(StringBuilder buf, int i, int length) {
263         buf.append(ByteBufUtil.hexDump(data, i, length));
264         buf.append('-');
265         i += length;
266         return i;
267     }
