227     @Override
228     public ChannelPipeline addFirst(EventExecutorGroup group, ChannelHandler... handlers) {
229         if (handlers == null) {
230             throw new NullPointerException("handlers");
231         }
232         if (handlers.length == 0 || handlers[0] == null) {
233             return this;
234         }
235 
236         int size;
237         for (size = 1; size < handlers.length; size ++) {
238             if (handlers[size] == null) {
239                 break;
240             }
241         }
242 
243         for (int i = size - 1; i >= 0; i --) {
244             ChannelHandler h = handlers[i];
245             addFirst(group, generateName(h), h);
246         }
247 
248         return this;
249     }

312     // No need for synchronization because it is always executed in a synchronized(this) block.
313     private ChannelHandlerInvoker findInvoker(EventExecutorGroup group) {
314         if (group == null) {
315             return null;
316         }
317 
318         // Lazily initialize the data structure that maps an EventExecutorGroup to a ChannelHandlerInvoker.
319         Map<EventExecutorGroup, ChannelHandlerInvoker> childInvokers = this.childInvokers;
320         if (childInvokers == null) {
321             childInvokers = this.childInvokers = new IdentityHashMap<EventExecutorGroup, ChannelHandlerInvoker>(4);
322         }
323 
324         // Pick one of the child executors and remember its invoker
325         // so that the same invoker is used to fire events for the same channel.
326         ChannelHandlerInvoker  invoker = childInvokers.get(group);
327         if (invoker == null) {
328             EventExecutor executor = group.next();
329             if (executor instanceof EventLoop) {
330                 invoker = ((EventLoop) executor).asInvoker();
331             } else {
332                 invoker = new DefaultChannelHandlerInvoker(executor);
333             }
334             childInvokers.put(group, invoker);
335         }
336 
337         return invoker;
338     }

340     private String generateName(ChannelHandler handler) {
341         Map<Class<?>, String> cache = nameCaches.get();
342         Class<?> handlerType = handler.getClass();
343         String name = cache.get(handlerType);
344         if (name == null) {
345             name = generateName0(handlerType);
346             cache.put(handlerType, name);
347         }
348 
349         synchronized (this) {
350             // It's not very likely for a user to put more than one handler of the same type, but make sure to avoid
351             // any name conflicts.  Note that we don't cache the names generated here.
352             if (context0(name) != null) {
353                 String baseName = name.substring(0, name.length() - 1); // Strip the trailing '0'.
354                 for (int i = 1;; i ++) {
355                     String newName = baseName + i;
356                     if (context0(newName) == null) {
357                         name = newName;
358                         break;
359                     }
360                 }
361             }
362         }
363 
364         return name;
365     }

388     private AbstractChannelHandlerContext remove(final AbstractChannelHandlerContext ctx) {
389         assert ctx != head && ctx != tail;
390 
391         AbstractChannelHandlerContext context;
392         Future<?> future;
393 
394         synchronized (this) {
395             if (!ctx.channel().isRegistered() || ctx.executor().inEventLoop()) {
396                 remove0(ctx);
397                 return ctx;
398             } else {
399                future = ctx.executor().submit(new OneTimeTask() {
400                    @Override
401                    public void run() {
402                        synchronized (DefaultChannelPipeline.this) {
403                            remove0(ctx);
404                        }
405                    }
406                });
407                context = ctx;
408             }
409         }
410 
411         // Run the following 'waiting' code outside of the above synchronized block
412         // in order to avoid deadlock
413 
414         waitForFuture(future);
415 
416         return context;
417     }

461     private ChannelHandler replace(
462             final AbstractChannelHandlerContext ctx, String newName, ChannelHandler newHandler) {
463 
464         assert ctx != head && ctx != tail;
465 
466         Future<?> future;
467         synchronized (this) {
468             if (newName == null) {
469                 newName = ctx.name();
470             } else if (!ctx.name().equals(newName)) {
471                 newName = filterName(newName, newHandler);
472             }
473 
474             final AbstractChannelHandlerContext newCtx =
475                     new DefaultChannelHandlerContext(this, ctx.invoker, newName, newHandler);
476 
477             if (!newCtx.channel().isRegistered() || newCtx.executor().inEventLoop()) {
478                 replace0(ctx, newCtx);
479                 return ctx.handler();
480             } else {
481                 future = newCtx.executor().submit(new OneTimeTask() {
482                     @Override
483                     public void run() {
484                         synchronized (DefaultChannelPipeline.this) {
485                             replace0(ctx, newCtx);
486                         }
487                     }
488                 });
489             }
490         }
491 
492         // Run the following 'waiting' code outside of the above synchronized block
493         // in order to avoid deadlock
494 
495         waitForFuture(future);
496 
497         return ctx.handler();
498     }

500     private void replace0(AbstractChannelHandlerContext oldCtx, AbstractChannelHandlerContext newCtx) {
501         checkMultiplicity(newCtx);
502 
503         AbstractChannelHandlerContext prev = oldCtx.prev;
504         AbstractChannelHandlerContext next = oldCtx.next;
505         newCtx.prev = prev;
506         newCtx.next = next;
507 
508         // Finish the replacement of oldCtx with newCtx in the linked list.
509         // Note that this doesn't mean events will be sent to the new handler immediately
510         // because we are currently at the event handler thread and no more than one handler methods can be invoked
511         // at the same time (we ensured that in replace().)
512         prev.next = newCtx;
513         next.prev = newCtx;
514 
515         // update the reference to the replacement so forward of buffered content will work correctly
516         oldCtx.prev = newCtx;
517         oldCtx.next = newCtx;
518 
519         // Invoke newHandler.handlerAdded() first (i.e. before oldHandler.handlerRemoved() is invoked)
520         // because callHandlerRemoved() will trigger inboundBufferUpdated() or flush() on newHandler and those
521         // event handlers must be called after handlerAdded().
522         callHandlerAdded(newCtx);
523         callHandlerRemoved(oldCtx);
524     }

526     private static void checkMultiplicity(ChannelHandlerContext ctx) {
527         ChannelHandler handler = ctx.handler();
528         if (handler instanceof ChannelHandlerAdapter) {
529             ChannelHandlerAdapter h = (ChannelHandlerAdapter) handler;
530             if (!h.isSharable() && h.added) {
531                 throw new ChannelPipelineException(
532                         h.getClass().getName() +
533                         " is not a @Sharable handler, so can't be added or removed multiple times.");
534             }
535             h.added = true;
536         }
537     }

539     private void callHandlerAdded(final AbstractChannelHandlerContext ctx) {
540         if (ctx.channel().isRegistered() && !ctx.executor().inEventLoop()) {
541             ctx.executor().execute(new OneTimeTask() {
542                 @Override
543                 public void run() {
544                     callHandlerAdded0(ctx);
545                 }
546             });
547             return;
548         }
549         callHandlerAdded0(ctx);
550     }
551 
552     private void callHandlerAdded0(final AbstractChannelHandlerContext ctx) {
553         try {
554             ctx.handler().handlerAdded(ctx);
555         } catch (Throwable t) {
556             boolean removed = false;
557             try {
558                 remove(ctx);
559                 removed = true;
560             } catch (Throwable t2) {
561                 if (logger.isWarnEnabled()) {
562                     logger.warn("Failed to remove a handler: " + ctx.name(), t2);
563                 }
564             }
565 
566             if (removed) {
567                 fireExceptionCaught(new ChannelPipelineException(
568                         ctx.handler().getClass().getName() +
569                         ".handlerAdded() has thrown an exception; removed.", t));
570             } else {
571                 fireExceptionCaught(new ChannelPipelineException(
572                         ctx.handler().getClass().getName() +
573                         ".handlerAdded() has thrown an exception; also failed to remove.", t));
574             }
575         }
576     }

578     private void callHandlerRemoved(final AbstractChannelHandlerContext ctx) {
579         if (ctx.channel().isRegistered() && !ctx.executor().inEventLoop()) {
580             ctx.executor().execute(new OneTimeTask() {
581                 @Override
582                 public void run() {
583                     callHandlerRemoved0(ctx);
584                  }
585             });

602     /**
603      * Waits for a future to finish.  If the task is interrupted, then the current thread will be interrupted.
604      * It is expected that the task performs any appropriate locking.
605      * <p>
606      * If the internal call throws a {@link Throwable}, but it is not an instance of {@link Error} or
607      * {@link RuntimeException}, then it is wrapped inside a {@link ChannelPipelineException} and that is
608      * thrown instead.</p>
609      *
610      * @param future wait for this future
611      * @see Future#get()
612      * @throws Error if the task threw this.
613      * @throws RuntimeException if the task threw this.
614      * @throws ChannelPipelineException with a {@link Throwable} as a cause, if the task threw another type of
615      *         {@link Throwable}.
616      */
617     private static void waitForFuture(Future<?> future) {
618         try {
619             future.get();
620         } catch (ExecutionException ex) {
621             // In the arbitrary case, we can throw Error, RuntimeException, and Exception
622             PlatformDependent.throwException(ex.getCause());
623         } catch (InterruptedException ex) {
624             // Interrupt the calling thread (note that this method is not called from the event loop)
625             Thread.currentThread().interrupt();
626         }
627     }

768     @Override
769     public String toString() {
770         StringBuilder buf = new StringBuilder()
771             .append(StringUtil.simpleClassName(this))
772             .append('{');
773         AbstractChannelHandlerContext ctx = head.next;
774         for (;;) {
775             if (ctx == tail) {
776                 break;
777             }
778 
779             buf.append('(')
780                .append(ctx.name())
781                .append(" = ")
782                .append(ctx.handler().getClass().getName())
783                .append(')');

813     /**
814      * Removes all handlers from the pipeline one by one from tail (exclusive) to head (exclusive) to trigger
815      * handlerRemoved().
816      *
817      * Note that we traverse up the pipeline ({@link #destroyUp(AbstractChannelHandlerContext, boolean)})
818      * before traversing down ({@link #destroyDown(Thread, AbstractChannelHandlerContext, boolean)}) so that
819      * the handlers are removed after all events are handled.
820      *
821      * See: https://github.com/netty/netty/issues/3156
822      */

825     }

827     private void destroyUp(AbstractChannelHandlerContext ctx, boolean inEventLoop) {
828         final Thread currentThread = Thread.currentThread();
829         final AbstractChannelHandlerContext tail = this.tail;
830         for (;;) {
831             if (ctx == tail) {
832                 destroyDown(currentThread, tail.prev, inEventLoop);
833                 break;
834             }
835 
836             final EventExecutor executor = ctx.executor();
837             if (!inEventLoop && !executor.inEventLoop(currentThread)) {
838                 final AbstractChannelHandlerContext finalCtx = ctx;
839                 executor.execute(new OneTimeTask() {
840                     @Override
841                     public void run() {
842                         destroyUp(finalCtx, true);
843                     }
844                 });
845                 break;
846             }
847 
848             ctx = ctx.next;
849             inEventLoop = false;
850         }
851     }

853     private void destroyDown(Thread currentThread, AbstractChannelHandlerContext ctx, boolean inEventLoop) {
854         // We have reached at tail; now traverse backwards.
855         final AbstractChannelHandlerContext head = this.head;
856         for (;;) {
857             if (ctx == head) {
858                 break;
859             }
860 
861             final EventExecutor executor = ctx.executor();
862             if (inEventLoop || executor.inEventLoop(currentThread)) {
863                 synchronized (this) {
864                     remove0(ctx);
865                 }
866             } else {
867                 final AbstractChannelHandlerContext finalCtx = ctx;
868                 executor.execute(new OneTimeTask() {
869                     @Override
870                     public void run() {
871                         destroyDown(Thread.currentThread(), finalCtx, true);
872                     }
873                 });
874                 break;
875             }
876 
877             ctx = ctx.prev;
878             inEventLoop = false;
879         }
