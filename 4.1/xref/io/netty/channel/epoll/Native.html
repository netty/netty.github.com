53      static {
54          Selector selector = null;
55          try {
56              // We call Selector.open() as this will under the hood cause IOUtil to be loaded.
57              // This is a workaround for a possible classloader deadlock that could happen otherwise:
58              //
59              // See https://github.com/netty/netty/issues/10187
60              selector = Selector.open();
61          } catch (IOException ignore) {
62              // Just ignore
63          }
64          try {
65              // First, try calling a side-effect free JNI method to see if the library was already
66              // loaded by the application.
67              offsetofEpollData();
68          } catch (UnsatisfiedLinkError ignore) {
69              // The library was not previously loaded, load it now.
70              loadNativeLibrary();
71          } finally {
72              try {
73                  if (selector != null) {
74                      selector.close();
75                  }
76              } catch (IOException ignore) {
77                  // Just ignore
78              }
79          }
80          Unix.registerInternal(new Runnable() {
81              @Override
82              public void run() {
83                  registerUnix();
84              }
85          });
86      }

150     public static int epollWait(FileDescriptor epollFd, EpollEventArray events, FileDescriptor timerFd,
151                                 int timeoutSec, int timeoutNs) throws IOException {
152         if (timeoutSec == 0 && timeoutNs == 0) {
153             // Zero timeout => poll (aka return immediately)
154             return epollWait(epollFd, events, 0);
155         }
156         if (timeoutSec == Integer.MAX_VALUE) {
157             // Max timeout => wait indefinitely: disarm timerfd first
158             timeoutSec = 0;
159             timeoutNs = 0;
160         }
161         int ready = epollWait0(epollFd.intValue(), events.memoryAddress(), events.length(), timerFd.intValue(),
162                                timeoutSec, timeoutNs);
163         if (ready < 0) {
164             throw newIOException("epoll_wait", ready);
165         }
166         return ready;
167     }

281     private static void loadNativeLibrary() {
282         String name = SystemPropertyUtil.get("os.name").toLowerCase(Locale.UK).trim();
283         if (!name.startsWith("linux")) {
284             throw new IllegalStateException("Only supported on Linux");
285         }
286         String staticLibName = "netty_transport_native_epoll";
287         String sharedLibName = staticLibName + '_' + PlatformDependent.normalizedArch();
288         ClassLoader cl = PlatformDependent.getClassLoader(Native.class);
289         try {
290             NativeLibraryLoader.load(sharedLibName, cl);
291         } catch (UnsatisfiedLinkError e1) {
292             try {
293                 NativeLibraryLoader.load(staticLibName, cl);
294                 logger.debug("Failed to load {}", sharedLibName, e1);
295             } catch (UnsatisfiedLinkError e2) {
296                 ThrowableUtil.addSuppressed(e1, e2);
297                 throw e1;
298             }
299         }
300     }
