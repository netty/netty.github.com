51      EpollSocketChannel(Channel parent, Socket fd, InetSocketAddress remote) {
52          super(parent, fd);
53          config = new EpollSocketChannelConfig(this);
54          // Directly cache the remote and local addresses
55          // See https://github.com/netty/netty/issues/2359
56          this.remote = remote;
57          local = fd.localAddress();
58  
59          if (parent instanceof EpollServerSocketChannel) {
60              tcpMd5SigAddresses = ((EpollServerSocketChannel) parent).tcpMd5SigAddresses();
61          }
62      }

152     private static InetSocketAddress computeRemoteAddr(InetSocketAddress remoteAddr, InetSocketAddress osRemoteAddr) {
153         if (osRemoteAddr != null) {
154             if (PlatformDependent.javaVersion() >= 7) {
155                 try {
156                     // Only try to construct a new InetSocketAddress if we using java >= 7 as getHostString() does not
157                     // exists in earlier releases and so the retrieval of the hostname could block the EventLoop if a
158                     // reverse lookup would be needed.
159                     return new InetSocketAddress(InetAddress.getByAddress(remoteAddr.getHostString(),
160                             osRemoteAddr.getAddress().getAddress()),
161                             osRemoteAddr.getPort());
162                 } catch (UnknownHostException ignore) {
163                     // Should never happen but fallback to osRemoteAddr anyway.
164                 }
165             }
166             return osRemoteAddr;
167         }
168         return remoteAddr;
169     }

171     @Override
172     protected boolean doConnect(SocketAddress remoteAddress, SocketAddress localAddress) throws Exception {
173         if (localAddress != null) {
174             checkResolvable((InetSocketAddress) localAddress);
175         }
176         // We always need to set the localAddress even if not connected yet
177         //
178         // See https://github.com/netty/netty/issues/3463
179         local = fd().localAddress();
180 
181         InetSocketAddress remoteAddr = (InetSocketAddress) remoteAddress;
182         checkResolvable(remoteAddr);
183         if (super.doConnect(remoteAddress, localAddress)) {
184             remote = computeRemoteAddr(remoteAddr, fd().remoteAddress());
185             return true;
186         }
187 
188         // Store for later usage in doFinishConnect()
189         requestedRemote = remoteAddr;
190         return false;
191     }

193     private final class EpollSocketChannelUnsafe extends EpollStreamUnsafe {
194         @Override
195         protected Executor prepareToClose() {
196             try {
197                 // Check isOpen() first as otherwise it will throw a RuntimeException
198                 // when call getSoLinger() as the fd is not valid anymore.
199                 if (isOpen() && config().getSoLinger() > 0) {
200                     // We need to cancel this key of the channel so we may not end up in a eventloop spin
201                     // because we try to read or write until the actual close happens which may be later due
202                     // SO_LINGER handling.
203                     // See https://github.com/netty/netty/issues/4449
204                     ((EpollEventLoop) eventLoop()).remove(EpollSocketChannel.this);
205                     return GlobalEventExecutor.INSTANCE;
206                 }
207             } catch (Throwable ignore) {
208                 // Ignore the error as the underlying channel may be closed in the meantime and so
209                 // getSoLinger() may produce an exception. In this case we just return null.
210                 // See https://github.com/netty/netty/issues/4449
211             }
212             return null;
213         }
214 
215         @Override
216         boolean doFinishConnect() throws Exception {
217             if (super.doFinishConnect()) {
218                 remote = computeRemoteAddr(requestedRemote, fd().remoteAddress());
219                 requestedRemote = null;
220                 return true;
221             }
222             return false;
223         }
224     }
