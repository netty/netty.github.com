75      void setNetworkInterface(NetworkInterface netInterface) throws IOException {
76          InetAddress address = deriveInetAddress(netInterface, family() == InternetProtocolFamily.IPv6);
77          if (address.equals(family() == InternetProtocolFamily.IPv4 ? INET_ANY : INET6_ANY)) {
78              throw new IOException("NetworkInterface does not support " + family());
79          }
80          final NativeInetAddress nativeAddress = NativeInetAddress.newInstance(address);
81          setInterface(intValue(), ipv6, nativeAddress.address(), nativeAddress.scopeId(), interfaceIndex(netInterface));
82      }

84      InetAddress getInterface() throws IOException {
85          NetworkInterface inf = getNetworkInterface();
86          if (inf != null) {
87              Enumeration<InetAddress> addresses = SocketUtils.addressesFromNetworkInterface(inf);
88              if (addresses.hasMoreElements()) {
89                  return addresses.nextElement();
90              }
91          }
92          return null;
93      }

95      NetworkInterface getNetworkInterface() throws IOException {
96          int ret = getInterface(intValue(), ipv6);
97          if (ipv6) {
98              return PlatformDependent.javaVersion() >= 7 ? NetworkInterface.getByIndex(ret) : null;
99          }
100         InetAddress address = inetAddress(ret);
101         return address != null ? NetworkInterface.getByInetAddress(address) : null;
102     }

104     private static InetAddress inetAddress(int value) {
105         byte[] var1 = {
106                 (byte) (value >>> 24 & 255),
107                 (byte) (value >>> 16 & 255),
108                 (byte) (value >>> 8 & 255),
109                 (byte) (value & 255)
110         };
111 
112         try {
113             return InetAddress.getByAddress(var1);
114         } catch (UnknownHostException ignore) {
115             return null;
116         }
117     }

119     void joinGroup(InetAddress group, NetworkInterface netInterface, InetAddress source) throws IOException {
120         final NativeInetAddress g = NativeInetAddress.newInstance(group);
121         final boolean isIpv6 = group instanceof Inet6Address;
122         final NativeInetAddress i = NativeInetAddress.newInstance(deriveInetAddress(netInterface, isIpv6));
123         if (source != null) {
124             if (source.getClass() != group.getClass()) {
125                 throw new IllegalArgumentException("Source address is different type to group");
126             }
127             final NativeInetAddress s = NativeInetAddress.newInstance(source);
128             joinSsmGroup(intValue(), ipv6 && isIpv6, g.address(), i.address(),
129                     g.scopeId(), interfaceIndex(netInterface), s.address());
130         } else {
131             joinGroup(intValue(), ipv6 && isIpv6, g.address(), i.address(), g.scopeId(), interfaceIndex(netInterface));
132         }
133     }

313     private static InetAddress deriveInetAddress(NetworkInterface netInterface, boolean ipv6) {
314         final InetAddress ipAny = ipv6 ? INET6_ANY : INET_ANY;
315         if (netInterface != null) {
316             final Enumeration<InetAddress> ias = netInterface.getInetAddresses();
317             while (ias.hasMoreElements()) {
318                 final InetAddress ia = ias.nextElement();
319                 final boolean isV6 = ia instanceof Inet6Address;
320                 if (isV6 == ipv6) {
321                     return ia;
322                 }
323             }
324         }
325         return ipAny;
326     }
