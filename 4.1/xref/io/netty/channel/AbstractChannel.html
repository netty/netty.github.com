347     /**
348      * Returns the {@link String} representation of this channel.  The returned
349      * string contains the {@linkplain #hashCode()} ID}, {@linkplain #localAddress() local address},
350      * and {@linkplain #remoteAddress() remote address} of this channel for
351      * easier identification.
352      */
353     @Override
354     public String toString() {
355         boolean active = isActive();
356         if (strValActive == active && strVal != null) {
357             return strVal;
358         }
359 
360         SocketAddress remoteAddr = remoteAddress();
361         SocketAddress localAddr = localAddress();
362         if (remoteAddr != null) {
363             SocketAddress srcAddr;
364             SocketAddress dstAddr;
365             if (parent == null) {
366                 srcAddr = localAddr;
367                 dstAddr = remoteAddr;
368             } else {
369                 srcAddr = remoteAddr;
370                 dstAddr = localAddr;
371             }
372 
373             StringBuilder buf = new StringBuilder(96)
374                 .append("[id: 0x")
375                 .append(id.asShortText())
376                 .append(", ")
377                 .append(srcAddr)
378                 .append(active? " => " : " :> ")
379                 .append(dstAddr)
380                 .append(']');
381             strVal = buf.toString();
382         } else if (localAddr != null) {
383             StringBuilder buf = new StringBuilder(64)
384                 .append("[id: 0x")
385                 .append(id.asShortText())
386                 .append(", ")
387                 .append(localAddr)
388                 .append(']');
389             strVal = buf.toString();
390         } else {
391             StringBuilder buf = new StringBuilder(16)
392                 .append("[id: 0x")
393                 .append(id.asShortText())
394                 .append(']');
395             strVal = buf.toString();
396         }
397 
398         strValActive = active;
399         return strVal;
400     }

453         @Override
454         public final void register(EventLoop eventLoop, final ChannelPromise promise) {
455             if (eventLoop == null) {
456                 throw new NullPointerException("eventLoop");
457             }
458             if (isRegistered()) {
459                 promise.setFailure(new IllegalStateException("registered to an event loop already"));
460                 return;
461             }
462             if (!isCompatible(eventLoop)) {
463                 promise.setFailure(
464                         new IllegalStateException("incompatible event loop type: " + eventLoop.getClass().getName()));
465                 return;
466             }
467 
468             AbstractChannel.this.eventLoop = eventLoop;
469 
470             if (eventLoop.inEventLoop()) {
471                 register0(promise);
472             } else {
473                 try {
474                     eventLoop.execute(new OneTimeTask() {
475                         @Override
476                         public void run() {
477                             register0(promise);
478                         }
479                     });
480                 } catch (Throwable t) {
481                     logger.warn(
482                             "Force-closing a channel whose registration task was not accepted by an event loop: {}",
483                             AbstractChannel.this, t);
484                     closeForcibly();
485                     closeFuture.setClosed();
486                     safeSetFailure(promise, t);
487                 }
488             }
489         }
490 
491         private void register0(ChannelPromise promise) {
492             try {
493                 // check if the channel is still open as it could be closed in the mean time when the register
494                 // call was outside of the eventLoop
495                 if (!promise.setUncancellable() || !ensureOpen(promise)) {
496                     return;
497                 }
498                 boolean firstRegistration = neverRegistered;
499                 doRegister();
500                 neverRegistered = false;
501                 registered = true;
502                 safeSetSuccess(promise);
503                 pipeline.fireChannelRegistered();
504                 // Only fire a channelActive if the channel has never been registered. This prevents firing
505                 // multiple channel actives if the channel is deregistered and re-registered.
506                 if (firstRegistration && isActive()) {
507                     pipeline.fireChannelActive();
508                 }
509             } catch (Throwable t) {
510                 // Close the channel directly to avoid FD leak.
511                 closeForcibly();
512                 closeFuture.setClosed();
513                 safeSetFailure(promise, t);
514             }
515         }

517         @Override
518         public final void bind(final SocketAddress localAddress, final ChannelPromise promise) {
519             if (!promise.setUncancellable() || !ensureOpen(promise)) {
520                 return;
521             }
522 
523             // See: https://github.com/netty/netty/issues/576
524             if (Boolean.TRUE.equals(config().getOption(ChannelOption.SO_BROADCAST)) &&
525                 localAddress instanceof InetSocketAddress &&
526                 !((InetSocketAddress) localAddress).getAddress().isAnyLocalAddress() &&
527                 !PlatformDependent.isWindows() && !PlatformDependent.isRoot()) {
528                 // Warn a user about the fact that a non-root user can't receive a
529                 // broadcast packet on *nix if the socket is bound on non-wildcard address.
530                 logger.warn(
531                         "A non-root user can't receive a broadcast packet if the socket " +
532                         "is not bound to a wildcard address; binding to a non-wildcard " +
533                         "address (" + localAddress + ") anyway as requested.");
534             }
535 
536             boolean wasActive = isActive();
537             try {
538                 doBind(localAddress);
539             } catch (Throwable t) {
540                 safeSetFailure(promise, t);
541                 closeIfClosed();
542                 return;
543             }
544 
545             if (!wasActive && isActive()) {
546                 invokeLater(new OneTimeTask() {
547                     @Override
548                     public void run() {
549                         pipeline.fireChannelActive();
550                     }
551                 });
552             }
553 
554             safeSetSuccess(promise);
555         }

565                 doDisconnect();
566             } catch (Throwable t) {
567                 safeSetFailure(promise, t);
568                 closeIfClosed();
569                 return;
570             }

573                 invokeLater(new OneTimeTask() {
574                     @Override
575                     public void run() {
576                         pipeline.fireChannelInactive();
577                     }
578                 });
579             }

582             closeIfClosed(); // doDisconnect() might have closed the channel
583         }

590         private void close(final ChannelPromise promise, final Throwable cause, final boolean notify) {
591             if (!promise.setUncancellable()) {
592                 return;
593             }
594 
595             final ChannelOutboundBuffer outboundBuffer = this.outboundBuffer;
596             if (outboundBuffer == null) {
597                 // Only needed if no VoidChannelPromise.
598                 if (!(promise instanceof VoidChannelPromise)) {
599                     // This means close() was called before so we just register a listener and return
600                     closeFuture.addListener(new ChannelFutureListener() {
601                         @Override
602                         public void operationComplete(ChannelFuture future) throws Exception {
603                             promise.setSuccess();
604                         }
605                     });
606                 }
607                 return;
608             }
609 
610             if (closeFuture.isDone()) {
611                 // Closed already.
612                 safeSetSuccess(promise);
613                 return;
614             }
615 
616             final boolean wasActive = isActive();
617             this.outboundBuffer = null; // Disallow adding any messages and flushes to outboundBuffer.
618             Executor closeExecutor = prepareToClose();
619             if (closeExecutor != null) {
620                 closeExecutor.execute(new OneTimeTask() {
621                     @Override
622                     public void run() {
623                         try {
624                             // Execute the close.
625                             doClose0(promise);
626                         } finally {
627                             // Call invokeLater so closeAndDeregister is executed in the EventLoop again!
628                             invokeLater(new OneTimeTask() {
629                                 @Override
630                                 public void run() {
631                                     // Fail all the queued messages
632                                     outboundBuffer.failFlushed(cause, notify);
633                                     outboundBuffer.close(CLOSED_CHANNEL_EXCEPTION);
634                                     fireChannelInactiveAndDeregister(wasActive);
635                                 }
636                             });
637                         }
638                     }
639                 });
640             } else {
641                 try {
642                     // Close the channel and fail the queued messages in all cases.
643                     doClose0(promise);
644                 } finally {
645                     // Fail all the queued messages.
646                     outboundBuffer.failFlushed(cause, notify);
647                     outboundBuffer.close(CLOSED_CHANNEL_EXCEPTION);
648                 }
649                 if (inFlush0) {
650                     invokeLater(new OneTimeTask() {
651                         @Override
652                         public void run() {
653                             fireChannelInactiveAndDeregister(wasActive);
654                         }
655                     });
656                 } else {
657                     fireChannelInactiveAndDeregister(wasActive);
658                 }
659             }
660         }
661 
662         private void doClose0(ChannelPromise promise) {
663             try {
664                 doClose();
665                 closeFuture.setClosed();
666                 safeSetSuccess(promise);
667             } catch (Throwable t) {
668                 closeFuture.setClosed();
669                 safeSetFailure(promise, t);
670             }
671         }

691         private void deregister(final ChannelPromise promise, final boolean fireChannelInactive) {
692             if (!promise.setUncancellable()) {
693                 return;
694             }
695 
696             if (!registered) {
697                 safeSetSuccess(promise);
698                 return;
699             }
700 
701             // As a user may call deregister() from within any method while doing processing in the ChannelPipeline,
702             // we need to ensure we do the actual deregister operation later. This is needed as for example,
703             // we may be in the ByteToMessageDecoder.callDecode(...) method and so still try to do processing in
704             // the old EventLoop while the user already registered the Channel to a new EventLoop. Without delay,
705             // the deregister operation this could lead to have a handler invoked by different EventLoop and so
706             // threads.
707             //
708             // See:
709             // https://github.com/netty/netty/issues/4435
710             invokeLater(new OneTimeTask() {
711                 @Override
712                 public void run() {
713                     try {
714                         doDeregister();
715                     } catch (Throwable t) {
716                         logger.warn("Unexpected exception occurred while deregistering a channel.", t);
717                     } finally {
718                         if (fireChannelInactive) {
719                             pipeline.fireChannelInactive();
720                         }
721                         // Some transports like local and AIO does not allow the deregistration of
722                         // an open channel.  Their doDeregister() calls close(). Consequently,
723                         // close() calls deregister() again - no need to fire channelUnregistered, so check
724                         // if it was registered.
725                         if (registered) {
726                             registered = false;
727                             pipeline.fireChannelUnregistered();
728                         }
729                         safeSetSuccess(promise);
730                     }
731                 }
732             });
733         }

754         @Override
755         public final void write(Object msg, ChannelPromise promise) {
756             ChannelOutboundBuffer outboundBuffer = this.outboundBuffer;
757             if (outboundBuffer == null) {
758                 // If the outboundBuffer is null we know the channel was closed and so
759                 // need to fail the future right away. If it is not null the handling of the rest
760                 // will be done in flush0()
761                 // See https://github.com/netty/netty/issues/2362
762                 safeSetFailure(promise, CLOSED_CHANNEL_EXCEPTION);
763                 // release message now to prevent resource-leak
764                 ReferenceCountUtil.release(msg);
765                 return;
766             }
767 
768             int size;
769             try {
770                 msg = filterOutboundMessage(msg);
771                 size = estimatorHandle().size(msg);
772                 if (size < 0) {
773                     size = 0;
774                 }
775             } catch (Throwable t) {
776                 safeSetFailure(promise, t);
777                 ReferenceCountUtil.release(msg);
778                 return;
779             }
780 
781             outboundBuffer.addMessage(msg, size, promise);
782         }

784         @Override
785         public final void flush() {
786             ChannelOutboundBuffer outboundBuffer = this.outboundBuffer;
787             if (outboundBuffer == null) {
788                 return;
789             }
790 
791             outboundBuffer.addFlush();
792             flush0();
793         }

795         @SuppressWarnings("deprecation")
796         protected void flush0() {
797             if (inFlush0) {
798                 // Avoid re-entrance
799                 return;
800             }
801 
802             final ChannelOutboundBuffer outboundBuffer = this.outboundBuffer;
803             if (outboundBuffer == null || outboundBuffer.isEmpty()) {
804                 return;
805             }
806 
807             inFlush0 = true;
808 
809             // Mark all pending write requests as failure if the channel is inactive.
810             if (!isActive()) {
811                 try {
812                     if (isOpen()) {
813                         outboundBuffer.failFlushed(NOT_YET_CONNECTED_EXCEPTION, true);
814                     } else {
815                         // Do not trigger channelWritabilityChanged because the channel is closed already.
816                         outboundBuffer.failFlushed(CLOSED_CHANNEL_EXCEPTION, false);
817                     }
818                 } finally {
819                     inFlush0 = false;
820                 }
821                 return;
822             }
823 
824             try {
825                 doWrite(outboundBuffer);
826             } catch (Throwable t) {
827                 if (t instanceof IOException && config().isAutoClose()) {
828                     /**
829                      * Just call {@link #close(ChannelPromise, Throwable, boolean)} here which will take care of
830                      * failing all flushed messages and also ensure the actual close of the underlying transport
831                      * will happen before the promises are notified.

833                      * This is needed as otherwise {@link #isActive()} , {@link #isOpen()} and {@link #isWritable()}
834                      * may still return {@code true} even if the channel should be closed as result of the exception.
835                      */
836                     close(voidPromise(), t, false);
837                 } else {
838                     outboundBuffer.failFlushed(t, true);
839                 }
840             } finally {
841                 inFlush0 = false;
842             }

884         private void invokeLater(Runnable task) {
885             try {
886                 // This method is used by outbound operation implementations to trigger an inbound event later.
887                 // They do not trigger an inbound event immediately because an outbound operation might have been
888                 // triggered by another inbound event handler method.  If fired immediately, the call stack
889                 // will look like this for example:
890                 //
891                 //   handlerA.inboundBufferUpdated() - (1) an inbound handler method closes a connection.
892                 //   -> handlerA.ctx.close()
893                 //      -> channel.unsafe.close()
894                 //         -> handlerA.channelInactive() - (2) another inbound handler method called while in (1) yet
895                 //
896                 // which means the execution of two inbound handler methods of the same handler overlap undesirably.
897                 eventLoop().execute(task);
898             } catch (RejectedExecutionException e) {
899                 logger.warn("Can't invoke task later as EventLoop rejected it", e);
900             }
901         }
