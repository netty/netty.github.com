16  package io.netty.channel;
17  
18  import java.util.ArrayList;
19  import java.util.List;
20  
21  /**
22   * The {@link RecvByteBufAllocator} that automatically increases and
23   * decreases the predicted buffer size on feed back.
24   * <p>
25   * It gradually increases the expected number of readable bytes if the previous
26   * read fully filled the allocated buffer.  It gradually decreases the expected
27   * number of readable bytes if the read operation was not able to fill a certain
28   * amount of the allocated buffer two times consecutively.  Otherwise, it keeps
29   * returning the same prediction.
30   */
31  public class AdaptiveRecvByteBufAllocator extends DefaultMaxMessagesRecvByteBufAllocator {
32  
33      static final int DEFAULT_MINIMUM = 64;
34      static final int DEFAULT_INITIAL = 1024;
35      static final int DEFAULT_MAXIMUM = 65536;
36  
37      private static final int INDEX_INCREMENT = 4;
38      private static final int INDEX_DECREMENT = 1;
39  
40      private static final int[] SIZE_TABLE;
41  
42      static {
43          List<Integer> sizeTable = new ArrayList<Integer>();
44          for (int i = 16; i < 512; i += 16) {
45              sizeTable.add(i);
46          }
47  
48          for (int i = 512; i > 0; i <<= 1) {
49              sizeTable.add(i);
50          }
51  
52          SIZE_TABLE = new int[sizeTable.size()];
53          for (int i = 0; i < SIZE_TABLE.length; i ++) {
54              SIZE_TABLE[i] = sizeTable.get(i);
55          }
56      }
57  
58      /**
59       * @deprecated There is state for {@link #maxMessagesPerRead()} which is typically based upon channel type.
60       */
61      @Deprecated
62      public static final AdaptiveRecvByteBufAllocator DEFAULT = new AdaptiveRecvByteBufAllocator();
63  
64      private static int getSizeTableIndex(final int size) {
65          for (int low = 0, high = SIZE_TABLE.length - 1;;) {
66              if (high < low) {
67                  return low;
68              }
69              if (high == low) {
70                  return high;
71              }
72  
73              int mid = low + high >>> 1;
74              int a = SIZE_TABLE[mid];
75              int b = SIZE_TABLE[mid + 1];
76              if (size > b) {
77                  low = mid + 1;
78              } else if (size < a) {
79                  high = mid - 1;
80              } else if (size == a) {
81                  return mid;
82              } else {
83                  return mid + 1;
84              }
85          }
86      }
87  
88      private final class HandleImpl extends MaxMessageHandle {
89          private final int minIndex;
90          private final int maxIndex;
91          private int index;
92          private int nextReceiveBufferSize;
93          private boolean decreaseNow;
94  
95          public HandleImpl(int minIndex, int maxIndex, int initial) {
96              this.minIndex = minIndex;
97              this.maxIndex = maxIndex;
98  
99              index = getSizeTableIndex(initial);
100             nextReceiveBufferSize = SIZE_TABLE[index];
101         }
102 
103         @Override
104         public int guess() {
105             return nextReceiveBufferSize;
106         }
107 
108         private void record(int actualReadBytes) {
109             if (actualReadBytes <= SIZE_TABLE[Math.max(0, index - INDEX_DECREMENT - 1)]) {
110                 if (decreaseNow) {
111                     index = Math.max(index - INDEX_DECREMENT, minIndex);
112                     nextReceiveBufferSize = SIZE_TABLE[index];
113                     decreaseNow = false;
114                 } else {
115                     decreaseNow = true;
116                 }
117             } else if (actualReadBytes >= nextReceiveBufferSize) {
118                 index = Math.min(index + INDEX_INCREMENT, maxIndex);
119                 nextReceiveBufferSize = SIZE_TABLE[index];
120                 decreaseNow = false;
121             }
122         }
123 
124         @Override
125         public void readComplete() {
126             record(totalBytesRead());
127         }
128     }
129 
130     private final int minIndex;
131     private final int maxIndex;
132     private final int initial;
133 
134     /**
135      * Creates a new predictor with the default parameters.  With the default
136      * parameters, the expected buffer size starts from {@code 1024}, does not
137      * go down below {@code 64}, and does not go up above {@code 65536}.
138      */
139     public AdaptiveRecvByteBufAllocator() {
140         this(DEFAULT_MINIMUM, DEFAULT_INITIAL, DEFAULT_MAXIMUM);
141     }
142 
143     /**
144      * Creates a new predictor with the specified parameters.
145      *
146      * @param minimum  the inclusive lower bound of the expected buffer size
147      * @param initial  the initial buffer size when no feed back was received
148      * @param maximum  the inclusive upper bound of the expected buffer size
149      */
150     public AdaptiveRecvByteBufAllocator(int minimum, int initial, int maximum) {
151         if (minimum <= 0) {
152             throw new IllegalArgumentException("minimum: " + minimum);
153         }
154         if (initial < minimum) {
155             throw new IllegalArgumentException("initial: " + initial);
156         }
157         if (maximum < initial) {
158             throw new IllegalArgumentException("maximum: " + maximum);
159         }
160 
161         int minIndex = getSizeTableIndex(minimum);
162         if (SIZE_TABLE[minIndex] < minimum) {
163             this.minIndex = minIndex + 1;
164         } else {
165             this.minIndex = minIndex;
166         }
167 
168         int maxIndex = getSizeTableIndex(maximum);
169         if (SIZE_TABLE[maxIndex] > maximum) {
170             this.maxIndex = maxIndex - 1;
171         } else {
172             this.maxIndex = maxIndex;
173         }
174 
175         this.initial = initial;
176     }
177 
178     @Override
179     public Handle newHandle() {
180         return new HandleImpl(minIndex, maxIndex, initial);
181     }
