69      public final void shutdown(boolean read, boolean write) throws IOException {
70          for (;;) {
71              // We need to only shutdown what has not been shutdown yet, and if there is no change we should not
72              // shutdown anything. This is because if the underlying FD is reused and we still have an object which
73              // represents the previous incarnation of the FD we need to be sure we don't inadvertently shutdown the
74              // "new" FD without explicitly having a change.
75              final int oldState = this.state;
76              if (isClosed(oldState)) {
77                  throw new ClosedChannelException();
78              }
79              int newState = oldState;
80              if (read && !isInputShutdown(newState)) {
81                  newState = inputShutdown(newState);
82              }
83              if (write && !isOutputShutdown(newState)) {
84                  newState = outputShutdown(newState);
85              }
86  
87              // If there is no change in state, then we should not take any action.
88              if (newState == oldState) {
89                  return;
90              }
91              if (casState(oldState, newState)) {
92                  break;
93              }
94          }
95          int res = shutdown(fd, read, write);
96          if (res < 0) {
97              ioResult("shutdown", res);
98          }
99      }

118     public final int sendTo(ByteBuffer buf, int pos, int limit, InetAddress addr, int port, boolean fastOpen)
119             throws IOException {
120         // just duplicate the toNativeInetAddress code here to minimize object creation as this method is expected
121         // to be called frequently
122         byte[] address;
123         int scopeId;
124         if (addr instanceof Inet6Address) {
125             address = addr.getAddress();
126             scopeId = ((Inet6Address) addr).getScopeId();
127         } else {
128             // convert to ipv4 mapped ipv6 address;
129             scopeId = 0;
130             address = ipv4MappedIpv6Address(addr.getAddress());
131         }
132         int flags = fastOpen ? msgFastopen() : 0;
133         int res = sendTo(fd, useIpv6(addr), buf, pos, limit, address, scopeId, port, flags);
134         if (res >= 0) {
135             return res;
136         }
137         if (res == ERRNO_EINPROGRESS_NEGATIVE && fastOpen) {
138             // This happens when we (as a client) have no pre-existing cookie for doing a fast-open connection.
139             // In this case, our TCP connection will be established normally, but no data was transmitted at this time.
140             // We'll just transmit the data with normal writes later.
141             return 0;
142         }
143         if (res == ERROR_ECONNREFUSED_NEGATIVE) {
144             throw new PortUnreachableException("sendTo failed");
145         }
146         return ioResult("sendTo", res);
147     }

229         int res = recvFd(fd);

233         if (res == 0) {

256     public final boolean connect(SocketAddress socketAddress) throws IOException {
257         int res;
258         if (socketAddress instanceof InetSocketAddress) {
259             InetSocketAddress inetSocketAddress = (InetSocketAddress) socketAddress;
260             InetAddress inetAddress = inetSocketAddress.getAddress();
261             NativeInetAddress address = NativeInetAddress.newInstance(inetAddress);
262             res = connect(fd, useIpv6(inetAddress), address.address, address.scopeId, inetSocketAddress.getPort());
263         } else if (socketAddress instanceof DomainSocketAddress) {
264             DomainSocketAddress unixDomainSocketAddress = (DomainSocketAddress) socketAddress;
265             res = connectDomainSocket(fd, unixDomainSocketAddress.path().getBytes(CharsetUtil.UTF_8));
266         } else {
267             throw new Error("Unexpected SocketAddress implementation " + socketAddress);
268         }
269         if (res < 0) {
270             return handleConnectErrno("connect", res);
271         }
272         return true;
273     }

290     public final void bind(SocketAddress socketAddress) throws IOException {
291         if (socketAddress instanceof InetSocketAddress) {
292             InetSocketAddress addr = (InetSocketAddress) socketAddress;
293             InetAddress inetAddress = addr.getAddress();
294             NativeInetAddress address = NativeInetAddress.newInstance(inetAddress);
295             int res = bind(fd, useIpv6(inetAddress), address.address, address.scopeId, addr.getPort());
296             if (res < 0) {
297                 throw newIOException("bind", res);
298             }
299         } else if (socketAddress instanceof DomainSocketAddress) {
300             DomainSocketAddress addr = (DomainSocketAddress) socketAddress;
301             int res = bindDomainSocket(fd, addr.path().getBytes(CharsetUtil.UTF_8));
302             if (res < 0) {
303                 throw newIOException("bind", res);
304             }
305         } else {
306             throw new Error("Unexpected SocketAddress implementation " + socketAddress);
