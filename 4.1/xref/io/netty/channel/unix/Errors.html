33  /**
34   * <strong>Internal usage only!</strong>
35   * <p>Static members which call JNI methods must be defined in {@link ErrorsStaticallyReferencedJniMethods}.
36   */
37  public final class Errors {
38      // As all our JNI methods return -errno on error we need to compare with the negative errno codes.
39      public static final int ERRNO_ENOTCONN_NEGATIVE = -errnoENOTCONN();
40      public static final int ERRNO_EBADF_NEGATIVE = -errnoEBADF();
41      public static final int ERRNO_EPIPE_NEGATIVE = -errnoEPIPE();
42      public static final int ERRNO_ECONNRESET_NEGATIVE = -errnoECONNRESET();
43      public static final int ERRNO_EAGAIN_NEGATIVE = -errnoEAGAIN();
44      public static final int ERRNO_EWOULDBLOCK_NEGATIVE = -errnoEWOULDBLOCK();
45      public static final int ERRNO_EINPROGRESS_NEGATIVE = -errnoEINPROGRESS();
46  
47      /**
48       * Holds the mappings for errno codes to String messages.
49       * This eliminates the need to call back into JNI to get the right String message on an exception
50       * and thus is faster.
51       *
52       * The array length of 512 should be more then enough because errno.h only holds < 200 codes.
53       */
54      private static final String[] ERRORS = new String[512];
55  
56      // Pre-instantiated exceptions which does not need any stacktrace and
57      // can be thrown multiple times for performance reasons.
58      static final ClosedChannelException CLOSED_CHANNEL_EXCEPTION;
59      static final NativeIoException CONNECTION_NOT_CONNECTED_SHUTDOWN_EXCEPTION;
60      static final NativeIoException CONNECTION_RESET_EXCEPTION_WRITE;
61      static final NativeIoException CONNECTION_RESET_EXCEPTION_WRITEV;
62      static final NativeIoException CONNECTION_RESET_EXCEPTION_READ;
63      static final NativeIoException CONNECTION_RESET_EXCEPTION_SENDTO;
64      static final NativeIoException CONNECTION_RESET_EXCEPTION_SENDMSG;
65  
66      /**
67       * <strong>Internal usage only!</strong>
68       */
69      public static final class NativeIoException extends IOException {
70          private static final long serialVersionUID = 8222160204268655526L;
71          private final int expectedErr;
72          public NativeIoException(String method, int expectedErr) {
73              super(method);
74              this.expectedErr = expectedErr;
75          }
76  
77          public int expectedErr() {
78              return expectedErr;
79          }
80      }
81  
82      static {
83          for (int i = 0; i < ERRORS.length; i++) {
84              // This is ok as strerror returns 'Unknown error i' when the message is not known.
85              ERRORS[i] = strError(i);
86          }
87  
88          CONNECTION_RESET_EXCEPTION_READ = newConnectionResetException("syscall:read(...)",
89                  ERRNO_ECONNRESET_NEGATIVE);
90          CONNECTION_RESET_EXCEPTION_WRITE = newConnectionResetException("syscall:write(...)",
91                  ERRNO_EPIPE_NEGATIVE);
92          CONNECTION_RESET_EXCEPTION_WRITEV = newConnectionResetException("syscall:writev(...)",
93                  ERRNO_EPIPE_NEGATIVE);
94          CONNECTION_RESET_EXCEPTION_SENDTO = newConnectionResetException("syscall:sendto(...)",
95                  ERRNO_EPIPE_NEGATIVE);
96          CONNECTION_RESET_EXCEPTION_SENDMSG = newConnectionResetException("syscall:sendmsg(...)",
97                  ERRNO_EPIPE_NEGATIVE);
98          CONNECTION_NOT_CONNECTED_SHUTDOWN_EXCEPTION = newConnectionResetException("syscall:shutdown(...)",
99                  ERRNO_ENOTCONN_NEGATIVE);
100         CLOSED_CHANNEL_EXCEPTION = new ClosedChannelException();
101         CLOSED_CHANNEL_EXCEPTION.setStackTrace(EmptyArrays.EMPTY_STACK_TRACE);
102     }
103 
104     static ConnectException newConnectException(String method, int err) {
105         return new ConnectException(method + "() failed: " + ERRORS[-err]);
106     }
107 
108     public static NativeIoException newConnectionResetException(String method, int errnoNegative) {
109         NativeIoException exception = newIOException(method, errnoNegative);
110         exception.setStackTrace(EmptyArrays.EMPTY_STACK_TRACE);
111         return exception;
112     }
113 
114     public static NativeIoException newIOException(String method, int err) {
115         return new NativeIoException(method + "() failed: " + ERRORS[-err], err);
116     }
117 
118     public static int ioResult(String method, int err, NativeIoException resetCause) throws IOException {
119         // network stack saturated... try again later
120         if (err == ERRNO_EAGAIN_NEGATIVE || err == ERRNO_EWOULDBLOCK_NEGATIVE) {
121             return 0;
122         }
123         if (err == resetCause.expectedErr()) {
124             throw resetCause;
125         }
126         if (err == ERRNO_EBADF_NEGATIVE || err == ERRNO_ENOTCONN_NEGATIVE) {
127             throw CLOSED_CHANNEL_EXCEPTION;
128         }
129         // TODO: We could even go further and use a pre-instantiated IOException for the other error codes, but for
130         //       all other errors it may be better to just include a stack trace.
131         throw newIOException(method, err);
132     }
133 
134     private Errors() { }
135 }
