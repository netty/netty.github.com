130     private void allocate(PoolThreadCache cache, PooledByteBuf<T> buf, final int reqCapacity) {
131         final int sizeIdx = size2SizeIdx(reqCapacity);
132 
133         if (sizeIdx <= smallMaxSizeIdx) {
134             tcacheAllocateSmall(cache, buf, reqCapacity, sizeIdx);
135         } else if (sizeIdx < nSizes) {
136             tcacheAllocateNormal(cache, buf, reqCapacity, sizeIdx);
137         } else {
138             int normCapacity = directMemoryCacheAlignment > 0
139                     ? normalizeSize(reqCapacity) : reqCapacity;
140             // Huge allocations are never served via the cache so just call allocateHuge
141             allocateHuge(buf, normCapacity);
142         }
143     }

145     private void tcacheAllocateSmall(PoolThreadCache cache, PooledByteBuf<T> buf, final int reqCapacity,
146                                      final int sizeIdx) {
147 
148         if (cache.allocateSmall(this, buf, reqCapacity, sizeIdx)) {
149             // was able to allocate out of the cache so move on
150             return;
151         }
152 
153         /*
154          * Synchronize on the head. This is needed as {@link PoolChunk#allocateSubpage(int)} and
155          * {@link PoolChunk#free(long)} may modify the doubly linked list as well.
156          */
157         final PoolSubpage<T> head = smallSubpagePools[sizeIdx];
158         final boolean needsNormalAllocation;
159         synchronized (head) {
160             final PoolSubpage<T> s = head.next;
161             needsNormalAllocation = s == head;
162             if (!needsNormalAllocation) {
163                 assert s.doNotDestroy && s.elemSize == sizeIdx2size(sizeIdx);
164                 long handle = s.allocate();
165                 assert handle >= 0;
166                 s.chunk.initBufWithSubpage(buf, null, handle, reqCapacity, cache);
167             }
168         }
169 
170         if (needsNormalAllocation) {
171             synchronized (this) {
172                 allocateNormal(buf, reqCapacity, sizeIdx, cache);
173             }
174         }
175 
176         incSmallAllocation();
177     }

179     private void tcacheAllocateNormal(PoolThreadCache cache, PooledByteBuf<T> buf, final int reqCapacity,
180                                       final int sizeIdx) {
181         if (cache.allocateNormal(this, buf, reqCapacity, sizeIdx)) {
182             // was able to allocate out of the cache so move on
183             return;
184         }
185         synchronized (this) {
186             allocateNormal(buf, reqCapacity, sizeIdx, cache);
187             ++allocationsNormal;
188         }
189     }

191     // Method must be called inside synchronized(this) { ... } block
192     private void allocateNormal(PooledByteBuf<T> buf, int reqCapacity, int sizeIdx, PoolThreadCache threadCache) {
193         if (q050.allocate(buf, reqCapacity, sizeIdx, threadCache) ||
194             q025.allocate(buf, reqCapacity, sizeIdx, threadCache) ||
195             q000.allocate(buf, reqCapacity, sizeIdx, threadCache) ||
196             qInit.allocate(buf, reqCapacity, sizeIdx, threadCache) ||
197             q075.allocate(buf, reqCapacity, sizeIdx, threadCache)) {
198             return;
199         }
200 
201         // Add a new chunk.
202         PoolChunk<T> c = newChunk(pageSize, nPSizes, pageShifts, chunkSize);
203         boolean success = c.allocate(buf, reqCapacity, sizeIdx, threadCache);
204         assert success;
205         qInit.add(c);
206     }

212     private void allocateHuge(PooledByteBuf<T> buf, int reqCapacity) {
213         PoolChunk<T> chunk = newUnpooledChunk(reqCapacity);
214         activeBytesHuge.add(chunk.chunkSize());
215         buf.initUnpooled(chunk, reqCapacity);
216         allocationsHuge.increment();
217     }

219     void free(PoolChunk<T> chunk, ByteBuffer nioBuffer, long handle, int normCapacity, PoolThreadCache cache) {
220         if (chunk.unpooled) {
221             int size = chunk.chunkSize();
222             destroyChunk(chunk);
223             activeBytesHuge.add(-size);
224             deallocationsHuge.increment();
225         } else {
226             SizeClass sizeClass = sizeClass(handle);
227             if (cache != null && cache.add(this, chunk, nioBuffer, handle, normCapacity, sizeClass)) {
228                 // cached so not free it.
229                 return;
230             }
231 
232             freeChunk(chunk, handle, normCapacity, sizeClass, nioBuffer, false);
233         }
234     }

236     private static SizeClass sizeClass(long handle) {
237         return isSubpage(handle) ? SizeClass.Small : SizeClass.Normal;

240     void freeChunk(PoolChunk<T> chunk, long handle, int normCapacity, SizeClass sizeClass, ByteBuffer nioBuffer,
241                    boolean finalizer) {
242         final boolean destroyChunk;
243         synchronized (this) {
244             // We only call this if freeChunk is not called because of the PoolThreadCache finalizer as otherwise this
245             // may fail due lazy class-loading in for example tomcat.
246             if (!finalizer) {
247                 switch (sizeClass) {
248                     case Normal:
249                         ++deallocationsNormal;
250                         break;
251                     case Small:
252                         ++deallocationsSmall;
253                         break;
254                     default:
255                         throw new Error();
256                 }
257             }
258             destroyChunk = !chunk.parent.free(chunk, handle, normCapacity, nioBuffer);
259         }
260         if (destroyChunk) {
261             // destroyChunk not need to be called while holding the synchronized lock.
262             destroyChunk(chunk);
263         }
264     }

270     void reallocate(PooledByteBuf<T> buf, int newCapacity, boolean freeOldMemory) {
271         assert newCapacity >= 0 && newCapacity <= buf.maxCapacity();
272 
273         int oldCapacity = buf.length;
274         if (oldCapacity == newCapacity) {
275             return;
276         }
277 
278         PoolChunk<T> oldChunk = buf.chunk;
279         ByteBuffer oldNioBuffer = buf.tmpNioBuf;
280         long oldHandle = buf.handle;
281         T oldMemory = buf.memory;
282         int oldOffset = buf.offset;
283         int oldMaxLength = buf.maxLength;
284 
285         // This does not touch buf's reader/writer indices
286         allocate(parent.threadCache(), buf, newCapacity);
287         int bytesToCopy;
288         if (newCapacity > oldCapacity) {
289             bytesToCopy = oldCapacity;
290         } else {
291             buf.trimIndicesToCapacity(newCapacity);
292             bytesToCopy = newCapacity;
293         }
294         memoryCopy(oldMemory, oldOffset, buf, bytesToCopy);
295         if (freeOldMemory) {
296             free(oldChunk, oldNioBuffer, oldHandle, oldMaxLength, buf.cache);
297         }
298     }

335     private static List<PoolSubpageMetric> subPageMetricList(PoolSubpage<?>[] pages) {
336         List<PoolSubpageMetric> metrics = new ArrayList<PoolSubpageMetric>();
337         for (PoolSubpage<?> head : pages) {
338             if (head.next == head) {
339                 continue;
340             }
341             PoolSubpage<?> s = head.next;
342             for (;;) {
343                 metrics.add(s);
344                 s = s.next;
345                 if (s == head) {
346                     break;
347                 }
348             }
349         }
350         return metrics;
351     }

498     private static void appendPoolSubPages(StringBuilder buf, PoolSubpage<?>[] subpages) {
499         for (int i = 0; i < subpages.length; i ++) {
500             PoolSubpage<?> head = subpages[i];
501             if (head.next == head) {
502                 continue;
503             }
504 
505             buf.append(StringUtil.NEWLINE)
506                     .append(i)
507                     .append(": ");
508             PoolSubpage<?> s = head.next;
509             for (;;) {
510                 buf.append(s);
511                 s = s.next;
512                 if (s == head) {
513                     break;
514                 }
515             }
516         }
517     }

590     static final class DirectArena extends PoolArena<ByteBuffer> {
591 
592         DirectArena(PooledByteBufAllocator parent, int pageSize, int pageShifts,
593                     int chunkSize, int directMemoryCacheAlignment) {
594             super(parent, pageSize, pageShifts, chunkSize,
595                   directMemoryCacheAlignment);
596         }
597 
598         @Override
599         boolean isDirect() {
600             return true;
601         }
602 
603         @Override
604         protected PoolChunk<ByteBuffer> newChunk(int pageSize, int maxPageIdx,
605             int pageShifts, int chunkSize) {
606             if (directMemoryCacheAlignment == 0) {
607                 ByteBuffer memory = allocateDirect(chunkSize);
608                 return new PoolChunk<ByteBuffer>(this, memory, memory, pageSize, pageShifts,
609                         chunkSize, maxPageIdx);
610             }
611 
612             final ByteBuffer base = allocateDirect(chunkSize + directMemoryCacheAlignment);
613             final ByteBuffer memory = PlatformDependent.alignDirectBuffer(base, directMemoryCacheAlignment);
614             return new PoolChunk<ByteBuffer>(this, base, memory, pageSize,
615                     pageShifts, chunkSize, maxPageIdx);
616         }
617 
618         @Override
619         protected PoolChunk<ByteBuffer> newUnpooledChunk(int capacity) {
620             if (directMemoryCacheAlignment == 0) {
621                 ByteBuffer memory = allocateDirect(capacity);
622                 return new PoolChunk<ByteBuffer>(this, memory, memory, capacity);
623             }
624 
625             final ByteBuffer base = allocateDirect(capacity + directMemoryCacheAlignment);
626             final ByteBuffer memory = PlatformDependent.alignDirectBuffer(base, directMemoryCacheAlignment);
627             return new PoolChunk<ByteBuffer>(this, base, memory, capacity);
628         }
629 
630         private static ByteBuffer allocateDirect(int capacity) {
631             return PlatformDependent.useDirectBufferNoCleaner() ?
632                     PlatformDependent.allocateDirectNoCleaner(capacity) : ByteBuffer.allocateDirect(capacity);
633         }
634 
635         @Override
636         protected void destroyChunk(PoolChunk<ByteBuffer> chunk) {
637             if (PlatformDependent.useDirectBufferNoCleaner()) {
638                 PlatformDependent.freeDirectNoCleaner((ByteBuffer) chunk.base);
639             } else {
640                 PlatformDependent.freeDirectBuffer((ByteBuffer) chunk.base);
641             }
642         }
643 
644         @Override
645         protected PooledByteBuf<ByteBuffer> newByteBuf(int maxCapacity) {
646             if (HAS_UNSAFE) {
647                 return PooledUnsafeDirectByteBuf.newInstance(maxCapacity);
648             } else {
649                 return PooledDirectByteBuf.newInstance(maxCapacity);
650             }
651         }
652 
653         @Override
654         protected void memoryCopy(ByteBuffer src, int srcOffset, PooledByteBuf<ByteBuffer> dstBuf, int length) {
655             if (length == 0) {
656                 return;
657             }
658 
659             if (HAS_UNSAFE) {
660                 PlatformDependent.copyMemory(
661                         PlatformDependent.directBufferAddress(src) + srcOffset,
662                         PlatformDependent.directBufferAddress(dstBuf.memory) + dstBuf.offset, length);
663             } else {
664                 // We must duplicate the NIO buffers because they may be accessed by other Netty buffers.
665                 src = src.duplicate();
666                 ByteBuffer dst = dstBuf.internalNioBuffer();
667                 src.position(srcOffset).limit(srcOffset + length);
668                 dst.position(dstBuf.offset);
669                 dst.put(src);
670             }
671         }
