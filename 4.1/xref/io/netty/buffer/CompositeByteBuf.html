190     /**
191      * Precondition is that {@code buffer != null}.
192      */
193     private int addComponent0(int cIndex, ByteBuf buffer) {
194         assert buffer != null;
195         boolean wasAdded = false;
196         try {
197             checkComponentIndex(cIndex);
198 
199             int readableBytes = buffer.readableBytes();
200 
201             // No need to consolidate - just add a component to the list.
202             @SuppressWarnings("deprecation")
203             Component c = new Component(buffer.order(ByteOrder.BIG_ENDIAN).slice());
204             if (cIndex == components.size()) {
205                 wasAdded = components.add(c);
206                 if (cIndex == 0) {
207                     c.endOffset = readableBytes;
208                 } else {
209                     Component prev = components.get(cIndex - 1);
210                     c.offset = prev.endOffset;
211                     c.endOffset = c.offset + readableBytes;
212                 }
213             } else {
214                 components.add(cIndex, c);
215                 wasAdded = true;
216                 if (readableBytes != 0) {
217                     updateComponentOffsets(cIndex);
218                 }
219             }
220             return cIndex;
221         } finally {
222             if (!wasAdded) {
223                 buffer.release();
224             }
225         }
226     }

248     private int addComponents0(int cIndex, ByteBuf... buffers) {
249         checkNotNull(buffers, "buffers");
250         int i = 0;
251         try {
252             checkComponentIndex(cIndex);
253 
254             // No need for consolidation
255             while (i < buffers.length) {
256                 // Increment i now to prepare for the next iteration and prevent a duplicate release (addComponent0
257                 // will release if an exception occurs, and we also release in the finally block here).
258                 ByteBuf b = buffers[i++];
259                 if (b == null) {
260                     break;
261                 }
262                 cIndex = addComponent0(cIndex, b) + 1;
263                 int size = components.size();
264                 if (cIndex > size) {
265                     cIndex = size;
266                 }
267             }
268             return cIndex;
269         } finally {
270             for (; i < buffers.length; ++i) {
271                 ByteBuf b = buffers[i];
272                 if (b != null) {
273                     try {
274                         b.release();
275                     } catch (Throwable ignored) {
276                         // ignore
277                     }
278                 }
279             }
280         }
281     }

302     private int addComponents0(int cIndex, Iterable<ByteBuf> buffers) {
303         if (buffers instanceof ByteBuf) {
304             // If buffers also implements ByteBuf (e.g. CompositeByteBuf), it has to go to addComponent(ByteBuf).
305             return addComponent0(cIndex, (ByteBuf) buffers);
306         }
307         checkNotNull(buffers, "buffers");
308 
309         if (!(buffers instanceof Collection)) {
310             List<ByteBuf> list = new ArrayList<ByteBuf>();
311             try {
312                 for (ByteBuf b: buffers) {
313                     list.add(b);
314                 }
315                 buffers = list;
316             } finally {
317                 if (buffers != list) {
318                     for (ByteBuf b: buffers) {
319                         if (b != null) {
320                             try {
321                                 b.release();
322                             } catch (Throwable ignored) {
323                                 // ignore
324                             }
325                         }
326                     }
327                 }
328             }
329         }
330 
331         Collection<ByteBuf> col = (Collection<ByteBuf>) buffers;
332         return addComponents0(cIndex, col.toArray(new ByteBuf[col.size()]));
333     }

335     /**
336      * This should only be called as last operation from a method as this may adjust the underlying
337      * array of components and so affect the index etc.
338      */
339     private void consolidateIfNeeded() {
340         // Consolidate if the number of components will exceed the allowed maximum by the current
341         // operation.
342         final int numComponents = components.size();
343         if (numComponents > maxNumComponents) {
344             final int capacity = components.get(numComponents - 1).endOffset;
345 
346             ByteBuf consolidated = allocBuffer(capacity);
347 
348             // We're not using foreach to avoid creating an iterator.
349             for (int i = 0; i < numComponents; i ++) {
350                 Component c = components.get(i);
351                 ByteBuf b = c.buf;
352                 consolidated.writeBytes(b);
353                 c.freeIfNecessary();
354             }
355             Component c = new Component(consolidated);
356             c.endOffset = c.length;
357             components.clear();
358             components.add(c);
359         }
360     }

363         ensureAccessible();
364         if (cIndex < 0 || cIndex > components.size()) {

371     private void checkComponentIndex(int cIndex, int numComponents) {
372         ensureAccessible();
373         if (cIndex < 0 || cIndex + numComponents > components.size()) {
374             throw new IndexOutOfBoundsException(String.format(
375                     "cIndex: %d, numComponents: %d " +
376                     "(expected: cIndex >= 0 && cIndex + numComponents <= totalNumComponents(%d))",
377                     cIndex, numComponents, components.size()));
378         }

381     private void updateComponentOffsets(int cIndex) {
382         int size = components.size();
383         if (size <= cIndex) {
384             return;
385         }
386 
387         Component c = components.get(cIndex);
388         if (cIndex == 0) {
389             c.offset = 0;
390             c.endOffset = c.length;
391             cIndex ++;
392         }
393 
394         for (int i = cIndex; i < size; i ++) {
395             Component prev = components.get(i - 1);
396             Component cur = components.get(i);
397             cur.offset = prev.endOffset;
398             cur.endOffset = cur.offset + cur.length;
399         }
400     }

419      * Remove the number of {@link ByteBuf}s starting from the given index.
420      *
421      * @param cIndex the index on which the {@link ByteBuf}s will be started to removed
422      * @param numComponents the number of components to remove
423      */
424     public CompositeByteBuf removeComponents(int cIndex, int numComponents) {
425         checkComponentIndex(cIndex, numComponents);
426 
427         if (numComponents == 0) {
428             return this;
429         }
430         List<Component> toRemove = components.subList(cIndex, cIndex + numComponents);
431         boolean needsUpdate = false;
432         for (Component c: toRemove) {
433             if (c.length > 0) {
434                 needsUpdate = true;
435             }
436             c.freeIfNecessary();
437         }
438         toRemove.clear();
439 
440         if (needsUpdate) {
441             // Only need to call updateComponentOffsets if the length was > 0
442             updateComponentOffsets(cIndex);
443         }
444         return this;
445     }

456     /**
457      * Same with {@link #slice(int, int)} except that this method returns a list.
458      */
459     public List<ByteBuf> decompose(int offset, int length) {
460         checkIndex(offset, length);
461         if (length == 0) {
462             return Collections.emptyList();
463         }
464 
465         int componentId = toComponentIndex(offset);
466         List<ByteBuf> slice = new ArrayList<ByteBuf>(components.size());
467 
468         // The first component
469         Component firstC = components.get(componentId);
470         ByteBuf first = firstC.buf.duplicate();
471         first.readerIndex(offset - firstC.offset);
472 
473         ByteBuf buf = first;
474         int bytesToSlice = length;
475         do {
476             int readableBytes = buf.readableBytes();
477             if (bytesToSlice <= readableBytes) {
478                 // Last component
479                 buf.writerIndex(buf.readerIndex() + bytesToSlice);
480                 slice.add(buf);
481                 break;
482             } else {
483                 // Not the last component
484                 slice.add(buf);
485                 bytesToSlice -= readableBytes;
486                 componentId ++;
487 
488                 // Fetch the next component.
489                 buf = components.get(componentId).buf.duplicate();
490             }
491         } while (bytesToSlice > 0);
492 
493         // Slice all components because only readable bytes are interesting.
494         for (int i = 0; i < slice.size(); i ++) {
495             slice.set(i, slice.get(i).slice());
496         }
497 
498         return slice;
499     }

584     @Override
585     public CompositeByteBuf capacity(int newCapacity) {
586         ensureAccessible();
587         if (newCapacity < 0 || newCapacity > maxCapacity()) {
588             throw new IllegalArgumentException("newCapacity: " + newCapacity);
589         }
590 
591         int oldCapacity = capacity();
592         if (newCapacity > oldCapacity) {
593             final int paddingLength = newCapacity - oldCapacity;
594             ByteBuf padding;
595             int nComponents = components.size();
596             if (nComponents < maxNumComponents) {
597                 padding = allocBuffer(paddingLength);
598                 padding.setIndex(0, paddingLength);
599                 addComponent0(components.size(), padding);
600             } else {
601                 padding = allocBuffer(paddingLength);
602                 padding.setIndex(0, paddingLength);
603                 // FIXME: No need to create a padding buffer and consolidate.
604                 // Just create a big single buffer and put the current content there.
605                 addComponent0(components.size(), padding);
606                 consolidateIfNeeded();
607             }
608         } else if (newCapacity < oldCapacity) {
609             int bytesToTrim = oldCapacity - newCapacity;
610             for (ListIterator<Component> i = components.listIterator(components.size()); i.hasPrevious();) {
611                 Component c = i.previous();
612                 if (bytesToTrim >= c.length) {
613                     bytesToTrim -= c.length;
614                     i.remove();
615                     continue;
616                 }
617 
618                 // Replace the last component with the trimmed slice.
619                 Component newC = new Component(c.buf.slice(0, c.length - bytesToTrim));
620                 newC.offset = c.offset;
621                 newC.endOffset = newC.offset + newC.length;
622                 i.set(newC);
623                 break;
624             }
625 
626             if (readerIndex() > newCapacity) {
627                 setIndex(newCapacity, newCapacity);
628             } else if (writerIndex() > newCapacity) {
629                 writerIndex(newCapacity);
630             }
631         }
632         return this;
633     }

662     public int toComponentIndex(int offset) {
663         checkIndex(offset);
664 
665         for (int low = 0, high = components.size(); low <= high;) {
666             int mid = low + high >>> 1;
667             Component c = components.get(mid);
668             if (offset >= c.endOffset) {
669                 low = mid + 1;
670             } else if (offset < c.offset) {
671                 high = mid - 1;
672             } else {
673                 return mid;
674             }
675         }
676 
677         throw new Error("should not reach here");
678     }
