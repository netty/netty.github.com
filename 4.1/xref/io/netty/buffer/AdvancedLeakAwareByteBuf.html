37      private static final String PROP_ACQUIRE_AND_RELEASE_ONLY = "io.netty.leakDetection.acquireAndReleaseOnly";
38      private static final boolean ACQUIRE_AND_RELEASE_ONLY;
39  
40      private static final InternalLogger logger = InternalLoggerFactory.getInstance(AdvancedLeakAwareByteBuf.class);
41  
42      static {
43          ACQUIRE_AND_RELEASE_ONLY = SystemPropertyUtil.getBoolean(PROP_ACQUIRE_AND_RELEASE_ONLY, false);
44  
45          if (logger.isDebugEnabled()) {
46              logger.debug("-D{}: {}", PROP_ACQUIRE_AND_RELEASE_ONLY, ACQUIRE_AND_RELEASE_ONLY);
47          }
48      }
49  
50      private final ResourceLeak leak;
51  
52      AdvancedLeakAwareByteBuf(ByteBuf buf, ResourceLeak leak) {
53          super(buf);
54          this.leak = leak;
55      }
56  
57      static void recordLeakNonRefCountingOperation(ResourceLeak leak) {
58          if (!ACQUIRE_AND_RELEASE_ONLY) {
59              leak.record();
60          }
61      }
62  
63      @Override
64      public ByteBuf order(ByteOrder endianness) {
65          recordLeakNonRefCountingOperation(leak);
66          if (order() == endianness) {
67              return this;
68          } else {
69              return new AdvancedLeakAwareByteBuf(super.order(endianness), leak);
70          }
71      }
