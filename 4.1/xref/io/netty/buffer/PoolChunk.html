23  /**
24   * Description of algorithm for PageRun/PoolSubpage allocation from PoolChunk
25   *
26   * Notation: The following terms are important to understand the code
27   * > page  - a page is the smallest unit of memory chunk that can be allocated
28   * > run   - a run is a collection of pages
29   * > chunk - a chunk is a collection of runs
30   * > in this code chunkSize = maxPages * pageSize
31   *
32   * To begin we allocate a byte array of size = chunkSize
33   * Whenever a ByteBuf of given size needs to be created we search for the first position
34   * in the byte array that has enough empty space to accommodate the requested size and
35   * return a (long) handle that encodes this offset information, (this memory segment is then
36   * marked as reserved so it is always used by exactly one ByteBuf and no more)
37   *
38   * For simplicity all sizes are normalized according to {@link PoolArena#size2SizeIdx(int)} method.
39   * This ensures that when we request for memory segments of size > pageSize the normalizedCapacity
40   * equals the next nearest size in {@link SizeClasses}.
41   *
42   *
43   *  A chunk has the following layout:
44   *
45   *     /-----------------\
46   *     | run             |
47   *     |                 |
48   *     |                 |
49   *     |-----------------|
50   *     | run             |
51   *     |                 |
52   *     |-----------------|
53   *     | unalloctated    |
54   *     | (freed)         |
55   *     |                 |
56   *     |-----------------|
57   *     | subpage         |
58   *     |-----------------|
59   *     | unallocated     |
60   *     | (freed)         |
61   *     | ...             |
62   *     | ...             |
63   *     | ...             |
64   *     |                 |
65   *     |                 |
66   *     |                 |
67   *     \-----------------/
68   *
69   *
70   * handle:
71   * -------
72   * a handle is a long number, the bit layout of a run looks like:
73   *
74   * oooooooo ooooooos ssssssss ssssssue bbbbbbbb bbbbbbbb bbbbbbbb bbbbbbbb
75   *
76   * o: runOffset (page offset in the chunk), 15bit
77   * s: size (number of pages) of this run, 15bit
78   * u: isUsed?, 1bit
79   * e: isSubpage?, 1bit
80   * b: bitmapIdx of subpage, zero if it's not subpage, 32bit
81   *
82   * runsAvailMap:
83   * ------
84   * a map which manages all runs (used and not in used).
85   * For each run, the first runOffset and last runOffset are stored in runsAvailMap.
86   * key: runOffset
87   * value: handle
88   *
89   * runsAvail:
90   * ----------
91   * an array of {@link PriorityQueue}.
92   * Each queue manages same size of runs.
93   * Runs are sorted by offset, so that we always allocate runs with smaller offset.
94   *
95   *
96   * Algorithm:
97   * ----------
98   *
99   *   As we allocate runs, we update values stored in runsAvailMap and runsAvail so that the property is maintained.
100  *
101  * Initialization -
102  *  In the beginning we store the initial run which is the whole chunk.
103  *  The initial run:
104  *  runOffset = 0
105  *  size = chunkSize
106  *  isUsed = no
107  *  isSubpage = no
108  *  bitmapIdx = 0
109  *
110  *
111  * Algorithm: [allocateRun(size)]
112  * ----------
113  * 1) find the first avail run using in runsAvails according to size
114  * 2) if pages of run is larger than request pages then split it, and save the tailing run
115  *    for later using
116  *
117  * Algorithm: [allocateSubpage(size)]
118  * ----------
119  * 1) find a not full subpage according to size.
120  *    if it already exists just return, otherwise allocate a new PoolSubpage and call init()
121  *    note that this subpage object is added to subpagesPool in the PoolArena when we init() it
122  * 2) call subpage.allocate()
123  *
124  * Algorithm: [free(handle, length, nioBuffer)]
125  * ----------
126  * 1) if it is a subpage, return the slab back into this subpage
127  * 2) if the subpage is not used or it is a run, then start free this run
128  * 3) merge continuous avail runs
129  * 4) save the merged run
130  *
131  */

184     PoolChunk(PoolArena<T> arena, Object base, T memory, int pageSize, int pageShifts, int chunkSize, int maxPageIdx) {

199         int pages = chunkSize >> pageShifts;

229     private void insertAvailRun(int runOffset, int pages, long handle) {
230         int pageIdxFloor = arena.pages2pageIdxFloor(pages);
231         LongPriorityQueue queue = runsAvail[pageIdxFloor];
232         queue.offer(handle);
233 
234         //insert first page of run
235         insertAvailRun0(runOffset, handle);
236         if (pages > 1) {
237             //insert last page of run
238             insertAvailRun0(lastPage(runOffset, pages), handle);
239         }
240     }

242     private void insertAvailRun0(int runOffset, long handle) {
243         long pre = runsAvailMap.put(runOffset, handle);
244         assert pre == -1;
245     }

247     private void removeAvailRun(long handle) {
248         int pageIdxFloor = arena.pages2pageIdxFloor(runPages(handle));
249         LongPriorityQueue queue = runsAvail[pageIdxFloor];
250         removeAvailRun(queue, handle);
251     }

253     private void removeAvailRun(LongPriorityQueue queue, long handle) {
254         queue.remove(handle);
255 
256         int runOffset = runOffset(handle);
257         int pages = runPages(handle);
258         //remove first page of run
259         runsAvailMap.remove(runOffset);
260         if (pages > 1) {
261             //remove last page of run
262             runsAvailMap.remove(lastPage(runOffset, pages));
263         }
264     }

284         if (freeBytes == 0) {
285             return 100;

295     boolean allocate(PooledByteBuf<T> buf, int reqCapacity, int sizeIdx, PoolThreadCache cache) {
296         final long handle;
297         if (sizeIdx <= arena.smallMaxSizeIdx) {
298             // small
299             handle = allocateSubpage(sizeIdx);
300             if (handle < 0) {
301                 return false;
302             }
303             assert isSubpage(handle);
304         } else {
305             // normal
306             // runSize must be multiple of pageSize
307             int runSize = arena.sizeIdx2size(sizeIdx);
308             handle = allocateRun(runSize);
309             if (handle < 0) {
310                 return false;
311             }
312         }
313 
314         ByteBuffer nioBuffer = cachedNioBuffers != null? cachedNioBuffers.pollLast() : null;
315         initBuf(buf, nioBuffer, handle, reqCapacity, cache);
316         return true;
317     }

319     private long allocateRun(int runSize) {
320         int pages = runSize >> pageShifts;
321         int pageIdx = arena.pages2pageIdx(pages);
322 
323         synchronized (runsAvail) {
324             //find first queue which has at least one big enough run
325             int queueIdx = runFirstBestFit(pageIdx);
326             if (queueIdx == -1) {
327                 return -1;
328             }
329 
330             //get run with min offset in this queue
331             LongPriorityQueue queue = runsAvail[queueIdx];
332             long handle = queue.poll();
333 
334             assert handle != LongPriorityQueue.NO_VALUE && !isUsed(handle) : "invalid handle: " + handle;
335 
336             removeAvailRun(queue, handle);
337 
338             if (handle != -1) {
339                 handle = splitLargeRun(handle, pages);
340             }
341 
342             freeBytes -= runSize(pageShifts, handle);
343             return handle;
344         }
345     }

347     private int calculateRunSize(int sizeIdx) {
348         int maxElements = 1 << pageShifts - SizeClasses.LOG2_QUANTUM;
349         int runSize = 0;
350         int nElements;
351 
352         final int elemSize = arena.sizeIdx2size(sizeIdx);
353 
354         //find lowest common multiple of pageSize and elemSize
355         do {
356             runSize += pageSize;
357             nElements = runSize / elemSize;
358         } while (nElements < maxElements && runSize != nElements * elemSize);
359 
360         while (nElements > maxElements) {
361             runSize -= pageSize;
362             nElements = runSize / elemSize;
363         }
364 
365         assert nElements > 0;
366         assert runSize <= chunkSize;
367         assert runSize >= elemSize;
368 
369         return runSize;
370     }

372     private int runFirstBestFit(int pageIdx) {
373         if (freeBytes == chunkSize) {
374             return arena.nPSizes - 1;
375         }
376         for (int i = pageIdx; i < arena.nPSizes; i++) {
377             LongPriorityQueue queue = runsAvail[i];
378             if (queue != null && !queue.isEmpty()) {
379                 return i;
380             }
381         }
382         return -1;
383     }

385     private long splitLargeRun(long handle, int needPages) {
386         assert needPages > 0;
387 
388         int totalPages = runPages(handle);
389         assert needPages <= totalPages;
390 
391         int remPages = totalPages - needPages;
392 
393         if (remPages > 0) {
394             int runOffset = runOffset(handle);
395 
396             // keep track of trailing unused pages for later use
397             int availOffset = runOffset + needPages;
398             long availRun = toRunHandle(availOffset, remPages, 0);
399             insertAvailRun(availOffset, remPages, availRun);
400 
401             // not avail
402             return toRunHandle(runOffset, needPages, 1);
403         }
404 
405         //mark it as used
406         handle |= 1L << IS_USED_SHIFT;
407         return handle;
408     }

410     /**
411      * Create / initialize a new PoolSubpage of normCapacity. Any PoolSubpage created / initialized here is added to
412      * subpage pool in the PoolArena that owns this PoolChunk
413      *
414      * @param sizeIdx sizeIdx of normalized size
415      *
416      * @return index in memoryMap
417      */
418     private long allocateSubpage(int sizeIdx) {
419         // Obtain the head of the PoolSubPage pool that is owned by the PoolArena and synchronize on it.
420         // This is need as we may add it back and so alter the linked-list structure.
421         PoolSubpage<T> head = arena.findSubpagePoolHead(sizeIdx);
422         synchronized (head) {
423             //allocate a new run
424             int runSize = calculateRunSize(sizeIdx);
425             //runSize must be multiples of pageSize
426             long runHandle = allocateRun(runSize);
427             if (runHandle < 0) {
428                 return -1;
429             }
430 
431             int runOffset = runOffset(runHandle);
432             assert subpages[runOffset] == null;
433             int elemSize = arena.sizeIdx2size(sizeIdx);
434 
435             PoolSubpage<T> subpage = new PoolSubpage<T>(head, this, pageShifts, runOffset,
436                                runSize(pageShifts, runHandle), elemSize);
437 
438             subpages[runOffset] = subpage;
439             return subpage.allocate();
440         }
441     }

443     /**
444      * Free a subpage or a run of pages When a subpage is freed from PoolSubpage, it might be added back to subpage pool
445      * of the owning PoolArena. If the subpage pool in PoolArena has at least one other PoolSubpage of given elemSize,
446      * we can completely free the owning Page so it is available for subsequent allocations
447      *
448      * @param handle handle to free
449      */
450     void free(long handle, int normCapacity, ByteBuffer nioBuffer) {
451         if (isSubpage(handle)) {
452             int sizeIdx = arena.size2SizeIdx(normCapacity);
453             PoolSubpage<T> head = arena.findSubpagePoolHead(sizeIdx);
454 
455             int sIdx = runOffset(handle);
456             PoolSubpage<T> subpage = subpages[sIdx];
457             assert subpage != null && subpage.doNotDestroy;
458 
459             // Obtain the head of the PoolSubPage pool that is owned by the PoolArena and synchronize on it.
460             // This is need as we may add it back and so alter the linked-list structure.
461             synchronized (head) {
462                 if (subpage.free(head, bitmapIdx(handle))) {
463                     //the subpage is still used, do not free it
464                     return;
465                 }
466                 assert !subpage.doNotDestroy;
467                 // Null out slot in the array as it was freed and we should not use it anymore.
468                 subpages[sIdx] = null;
469             }
470         }
471 
472         //start free run
473         int pages = runPages(handle);
474 
475         synchronized (runsAvail) {
476             // collapse continuous runs, successfully collapsed runs
477             // will be removed from runsAvail and runsAvailMap
478             long finalRun = collapseRuns(handle);
479 
480             //set run as not used
481             finalRun &= ~(1L << IS_USED_SHIFT);
482             //if it is a subpage, set it to run
483             finalRun &= ~(1L << IS_SUBPAGE_SHIFT);
484 
485             insertAvailRun(runOffset(finalRun), runPages(finalRun), finalRun);
486             freeBytes += pages << pageShifts;
487         }
488 
489         if (nioBuffer != null && cachedNioBuffers != null &&
490             cachedNioBuffers.size() < PooledByteBufAllocator.DEFAULT_MAX_CACHED_BYTEBUFFERS_PER_CHUNK) {
491             cachedNioBuffers.offer(nioBuffer);
492         }
493     }

495     private long collapseRuns(long handle) {
496         return collapseNext(collapsePast(handle));
497     }

499     private long collapsePast(long handle) {
500         for (;;) {
501             int runOffset = runOffset(handle);
502             int runPages = runPages(handle);
503 
504             long pastRun = getAvailRunByOffset(runOffset - 1);
505             if (pastRun == -1) {
506                 return handle;
507             }
508 
509             int pastOffset = runOffset(pastRun);
510             int pastPages = runPages(pastRun);
511 
512             //is continuous
513             if (pastRun != handle && pastOffset + pastPages == runOffset) {
514                 //remove past run
515                 removeAvailRun(pastRun);
516                 handle = toRunHandle(pastOffset, pastPages + runPages, 0);
517             } else {
518                 return handle;
519             }
520         }
521     }
522 
523     private long collapseNext(long handle) {
524         for (;;) {
525             int runOffset = runOffset(handle);
526             int runPages = runPages(handle);
527 
528             long nextRun = getAvailRunByOffset(runOffset + runPages);
529             if (nextRun == -1) {
530                 return handle;
531             }
532 
533             int nextOffset = runOffset(nextRun);
534             int nextPages = runPages(nextRun);
535 
536             //is continuous
537             if (nextRun != handle && runOffset + runPages == nextOffset) {
538                 //remove next run
539                 removeAvailRun(nextRun);
540                 handle = toRunHandle(runOffset, runPages + nextPages, 0);
541             } else {
542                 return handle;
543             }
544         }
545     }

547     private static long toRunHandle(int runOffset, int runPages, int inUsed) {
548         return (long) runOffset << RUN_OFFSET_SHIFT
549                | (long) runPages << SIZE_SHIFT
550                | (long) inUsed << IS_USED_SHIFT;
551     }

553     void initBuf(PooledByteBuf<T> buf, ByteBuffer nioBuffer, long handle, int reqCapacity,
554                  PoolThreadCache threadCache) {
555         if (isRun(handle)) {
556             buf.init(this, nioBuffer, handle, runOffset(handle) << pageShifts,
557                      reqCapacity, runSize(pageShifts, handle), arena.parent.threadCache());
558         } else {
559             initBufWithSubpage(buf, nioBuffer, handle, reqCapacity, threadCache);
560         }
561     }

563     void initBufWithSubpage(PooledByteBuf<T> buf, ByteBuffer nioBuffer, long handle, int reqCapacity,
564                             PoolThreadCache threadCache) {
565         int runOffset = runOffset(handle);
566         int bitmapIdx = bitmapIdx(handle);
567 
568         PoolSubpage<T> s = subpages[runOffset];
569         assert s.doNotDestroy;
570         assert reqCapacity <= s.elemSize;
571 
572         int offset = (runOffset << pageShifts) + bitmapIdx * s.elemSize;
573         buf.init(this, nioBuffer, handle, offset, reqCapacity, s.elemSize, threadCache);
574     }
