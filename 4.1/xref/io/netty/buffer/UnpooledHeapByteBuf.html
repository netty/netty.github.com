59      private UnpooledHeapByteBuf(
60              ByteBufAllocator alloc, byte[] initialArray, int readerIndex, int writerIndex, int maxCapacity) {
61  
62          super(maxCapacity);
63  
64          if (alloc == null) {
65              throw new NullPointerException("alloc");
66          }
67          if (initialArray == null) {
68              throw new NullPointerException("initialArray");
69          }
70          if (initialArray.length > maxCapacity) {
71              throw new IllegalArgumentException(String.format(
72                      "initialCapacity(%d) > maxCapacity(%d)", initialArray.length, maxCapacity));
73          }
74  
75          this.alloc = alloc;
76          setArray(initialArray);
77          setIndex(readerIndex, writerIndex);
78      }
79  
80      private void setArray(byte[] initialArray) {
81          array = initialArray;
82          tmpNioBuf = null;
83      }

106     @Override
107     public ByteBuf capacity(int newCapacity) {
108         ensureAccessible();
109         if (newCapacity < 0 || newCapacity > maxCapacity()) {
110             throw new IllegalArgumentException("newCapacity: " + newCapacity);
111         }
112 
113         int oldCapacity = array.length;
114         if (newCapacity > oldCapacity) {
115             byte[] newArray = new byte[newCapacity];
116             System.arraycopy(array, 0, newArray, 0, array.length);
117             setArray(newArray);
118         } else if (newCapacity < oldCapacity) {
119             byte[] newArray = new byte[newCapacity];
120             int readerIndex = readerIndex();
121             if (readerIndex < newCapacity) {
122                 int writerIndex = writerIndex();
123                 if (writerIndex > newCapacity) {
124                     writerIndex(writerIndex = newCapacity);
125                 }
126                 System.arraycopy(array, readerIndex, newArray, readerIndex, writerIndex - readerIndex);
127             } else {
128                 setIndex(newCapacity, newCapacity);
129             }
130             setArray(newArray);
131         }
132         return this;
133     }

161     @Override
162     public ByteBuf getBytes(int index, ByteBuf dst, int dstIndex, int length) {
163         checkDstIndex(index, length, dstIndex, dst.capacity());
164         if (dst.hasMemoryAddress()) {
165             PlatformDependent.copyMemory(array, index, dst.memoryAddress() + dstIndex, length);
166         } else if (dst.hasArray()) {
167             getBytes(index, dst.array(), dst.arrayOffset() + dstIndex, length);
168         } else {
169             dst.setBytes(dstIndex, array, index, length);
170         }
171         return this;
172     }

207     private int getBytes(int index, GatheringByteChannel out, int length, boolean internal) throws IOException {
208         ensureAccessible();
209         ByteBuffer tmpBuf;
210         if (internal) {
211             tmpBuf = internalNioBuffer();
212         } else {
213             tmpBuf = ByteBuffer.wrap(array);
214         }
215         return out.write((ByteBuffer) tmpBuf.clear().position(index).limit(index + length));
216     }

218     private int getBytes(int index, FileChannel out, long position, int length, boolean internal) throws IOException {
219         ensureAccessible();
220         ByteBuffer tmpBuf = internal ? internalNioBuffer() : ByteBuffer.wrap(array);
221         return out.write((ByteBuffer) tmpBuf.clear().position(index).limit(index + length), position);
222     }

478     private ByteBuffer internalNioBuffer() {
479         ByteBuffer tmpNioBuf = this.tmpNioBuf;
480         if (tmpNioBuf == null) {
481             this.tmpNioBuf = tmpNioBuf = ByteBuffer.wrap(array);
482         }
483         return tmpNioBuf;
