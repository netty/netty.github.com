65      static {
66          int defaultAlignment = SystemPropertyUtil.getInt(
67                  "io.netty.allocator.directMemoryCacheAlignment", 0);
68          int defaultPageSize = SystemPropertyUtil.getInt("io.netty.allocator.pageSize", 8192);
69          Throwable pageSizeFallbackCause = null;
70          try {
71              validateAndCalculatePageShifts(defaultPageSize, defaultAlignment);
72          } catch (Throwable t) {
73              pageSizeFallbackCause = t;
74              defaultPageSize = 8192;
75              defaultAlignment = 0;
76          }
77          DEFAULT_PAGE_SIZE = defaultPageSize;
78          DEFAULT_DIRECT_MEMORY_CACHE_ALIGNMENT = defaultAlignment;
79  
80          int defaultMaxOrder = SystemPropertyUtil.getInt("io.netty.allocator.maxOrder", 11);
81          Throwable maxOrderFallbackCause = null;
82          try {
83              validateAndCalculateChunkSize(DEFAULT_PAGE_SIZE, defaultMaxOrder);
84          } catch (Throwable t) {
85              maxOrderFallbackCause = t;
86              defaultMaxOrder = 11;
87          }
88          DEFAULT_MAX_ORDER = defaultMaxOrder;
89  
90          // Determine reasonable default for nHeapArena and nDirectArena.
91          // Assuming each arena has 3 chunks, the pool should not consume more than 50% of max memory.
92          final Runtime runtime = Runtime.getRuntime();
93  
94          /*
95           * We use 2 * available processors by default to reduce contention as we use 2 * available processors for the
96           * number of EventLoops in NIO and EPOLL as well. If we choose a smaller number we will run into hot spots as
97           * allocation and de-allocation needs to be synchronized on the PoolArena.
98           *
99           * See https://github.com/netty/netty/issues/3888.
100          */
101         final int defaultMinNumArena = NettyRuntime.availableProcessors() * 2;
102         final int defaultChunkSize = DEFAULT_PAGE_SIZE << DEFAULT_MAX_ORDER;
103         DEFAULT_NUM_HEAP_ARENA = Math.max(0,
104                 SystemPropertyUtil.getInt(
105                         "io.netty.allocator.numHeapArenas",
106                         (int) Math.min(
107                                 defaultMinNumArena,
108                                 runtime.maxMemory() / defaultChunkSize / 2 / 3)));
109         DEFAULT_NUM_DIRECT_ARENA = Math.max(0,
110                 SystemPropertyUtil.getInt(
111                         "io.netty.allocator.numDirectArenas",
112                         (int) Math.min(
113                                 defaultMinNumArena,
114                                 PlatformDependent.maxDirectMemory() / defaultChunkSize / 2 / 3)));
115 
116         // cache sizes
117         DEFAULT_SMALL_CACHE_SIZE = SystemPropertyUtil.getInt("io.netty.allocator.smallCacheSize", 256);
118         DEFAULT_NORMAL_CACHE_SIZE = SystemPropertyUtil.getInt("io.netty.allocator.normalCacheSize", 64);
119 
120         // 32 kb is the default maximum capacity of the cached buffer. Similar to what is explained in
121         // 'Scalable memory allocation using jemalloc'
122         DEFAULT_MAX_CACHED_BUFFER_CAPACITY = SystemPropertyUtil.getInt(
123                 "io.netty.allocator.maxCachedBufferCapacity", 32 * 1024);
124 
125         // the number of threshold of allocations when cached entries will be freed up if not frequently used
126         DEFAULT_CACHE_TRIM_INTERVAL = SystemPropertyUtil.getInt(
127                 "io.netty.allocator.cacheTrimInterval", 8192);
128 
129         if (SystemPropertyUtil.contains("io.netty.allocation.cacheTrimIntervalMillis")) {
130             logger.warn("-Dio.netty.allocation.cacheTrimIntervalMillis is deprecated," +
131                     " use -Dio.netty.allocator.cacheTrimIntervalMillis");
132 
133             if (SystemPropertyUtil.contains("io.netty.allocator.cacheTrimIntervalMillis")) {
134                 // Both system properties are specified. Use the non-deprecated one.
135                 DEFAULT_CACHE_TRIM_INTERVAL_MILLIS = SystemPropertyUtil.getLong(
136                         "io.netty.allocator.cacheTrimIntervalMillis", 0);
137             } else {
138                 DEFAULT_CACHE_TRIM_INTERVAL_MILLIS = SystemPropertyUtil.getLong(
139                         "io.netty.allocation.cacheTrimIntervalMillis", 0);
140             }
141         } else {
142             DEFAULT_CACHE_TRIM_INTERVAL_MILLIS = SystemPropertyUtil.getLong(
143                     "io.netty.allocator.cacheTrimIntervalMillis", 0);
144         }
145 
146         DEFAULT_USE_CACHE_FOR_ALL_THREADS = SystemPropertyUtil.getBoolean(
147                 "io.netty.allocator.useCacheForAllThreads", true);
148 
149         // Use 1023 by default as we use an ArrayDeque as backing storage which will then allocate an internal array
150         // of 1024 elements. Otherwise we would allocate 2048 and only use 1024 which is wasteful.
151         DEFAULT_MAX_CACHED_BYTEBUFFERS_PER_CHUNK = SystemPropertyUtil.getInt(
152                 "io.netty.allocator.maxCachedByteBuffersPerChunk", 1023);
153 
154         if (logger.isDebugEnabled()) {
155             logger.debug("-Dio.netty.allocator.numHeapArenas: {}", DEFAULT_NUM_HEAP_ARENA);
156             logger.debug("-Dio.netty.allocator.numDirectArenas: {}", DEFAULT_NUM_DIRECT_ARENA);
157             if (pageSizeFallbackCause == null) {
158                 logger.debug("-Dio.netty.allocator.pageSize: {}", DEFAULT_PAGE_SIZE);
159             } else {
160                 logger.debug("-Dio.netty.allocator.pageSize: {}", DEFAULT_PAGE_SIZE, pageSizeFallbackCause);
161             }
162             if (maxOrderFallbackCause == null) {
163                 logger.debug("-Dio.netty.allocator.maxOrder: {}", DEFAULT_MAX_ORDER);
164             } else {
165                 logger.debug("-Dio.netty.allocator.maxOrder: {}", DEFAULT_MAX_ORDER, maxOrderFallbackCause);
166             }
167             logger.debug("-Dio.netty.allocator.chunkSize: {}", DEFAULT_PAGE_SIZE << DEFAULT_MAX_ORDER);
168             logger.debug("-Dio.netty.allocator.smallCacheSize: {}", DEFAULT_SMALL_CACHE_SIZE);
169             logger.debug("-Dio.netty.allocator.normalCacheSize: {}", DEFAULT_NORMAL_CACHE_SIZE);
170             logger.debug("-Dio.netty.allocator.maxCachedBufferCapacity: {}", DEFAULT_MAX_CACHED_BUFFER_CAPACITY);
171             logger.debug("-Dio.netty.allocator.cacheTrimInterval: {}", DEFAULT_CACHE_TRIM_INTERVAL);
172             logger.debug("-Dio.netty.allocator.cacheTrimIntervalMillis: {}", DEFAULT_CACHE_TRIM_INTERVAL_MILLIS);
173             logger.debug("-Dio.netty.allocator.useCacheForAllThreads: {}", DEFAULT_USE_CACHE_FOR_ALL_THREADS);
174             logger.debug("-Dio.netty.allocator.maxCachedByteBuffersPerChunk: {}",
175                     DEFAULT_MAX_CACHED_BYTEBUFFERS_PER_CHUNK);
176         }
177     }

263     public PooledByteBufAllocator(boolean preferDirect, int nHeapArena, int nDirectArena, int pageSize, int maxOrder,
264                                   int smallCacheSize, int normalCacheSize,
265                                   boolean useCacheForAllThreads, int directMemoryCacheAlignment) {
266         super(preferDirect);
267         threadCache = new PoolThreadLocalCache(useCacheForAllThreads);
268         this.smallCacheSize = smallCacheSize;
269         this.normalCacheSize = normalCacheSize;
270 
271         if (directMemoryCacheAlignment != 0) {
272             if (!PlatformDependent.hasAlignDirectByteBuffer()) {
273                 throw new UnsupportedOperationException("Buffer alignment is not supported. " +
274                         "Either Unsafe or ByteBuffer.alignSlice() must be available.");
275             }
276 
277             // Ensure page size is a whole multiple of the alignment, or bump it to the next whole multiple.
278             pageSize = (int) PlatformDependent.align(pageSize, directMemoryCacheAlignment);
279         }
280 
281         chunkSize = validateAndCalculateChunkSize(pageSize, maxOrder);
282 
283         checkPositiveOrZero(nHeapArena, "nHeapArena");
284         checkPositiveOrZero(nDirectArena, "nDirectArena");
285 
286         checkPositiveOrZero(directMemoryCacheAlignment, "directMemoryCacheAlignment");
287         if (directMemoryCacheAlignment > 0 && !isDirectMemoryCacheAlignmentSupported()) {
288             throw new IllegalArgumentException("directMemoryCacheAlignment is not supported");
289         }
290 
291         if ((directMemoryCacheAlignment & -directMemoryCacheAlignment) != directMemoryCacheAlignment) {
292             throw new IllegalArgumentException("directMemoryCacheAlignment: "
293                     + directMemoryCacheAlignment + " (expected: power of two)");
294         }
295 
296         int pageShifts = validateAndCalculatePageShifts(pageSize, directMemoryCacheAlignment);
297 
298         if (nHeapArena > 0) {
299             heapArenas = newArenaArray(nHeapArena);
300             List<PoolArenaMetric> metrics = new ArrayList<PoolArenaMetric>(heapArenas.length);
301             for (int i = 0; i < heapArenas.length; i ++) {
302                 PoolArena.HeapArena arena = new PoolArena.HeapArena(this,
303                         pageSize, pageShifts, chunkSize,
304                         directMemoryCacheAlignment);
305                 heapArenas[i] = arena;
306                 metrics.add(arena);
307             }
308             heapArenaMetrics = Collections.unmodifiableList(metrics);
309         } else {
310             heapArenas = null;
311             heapArenaMetrics = Collections.emptyList();
312         }
313 
314         if (nDirectArena > 0) {
315             directArenas = newArenaArray(nDirectArena);
316             List<PoolArenaMetric> metrics = new ArrayList<PoolArenaMetric>(directArenas.length);
317             for (int i = 0; i < directArenas.length; i ++) {
318                 PoolArena.DirectArena arena = new PoolArena.DirectArena(
319                         this, pageSize, pageShifts, chunkSize, directMemoryCacheAlignment);
320                 directArenas[i] = arena;
321                 metrics.add(arena);
322             }
323             directArenaMetrics = Collections.unmodifiableList(metrics);
324         } else {
325             directArenas = null;
326             directArenaMetrics = Collections.emptyList();
327         }
328         metric = new PooledByteBufAllocatorMetric(this);
329     }

336     private static int validateAndCalculatePageShifts(int pageSize, int alignment) {
337         if (pageSize < MIN_PAGE_SIZE) {
338             throw new IllegalArgumentException("pageSize: " + pageSize + " (expected: " + MIN_PAGE_SIZE + ')');
339         }
340 
341         if ((pageSize & pageSize - 1) != 0) {
342             throw new IllegalArgumentException("pageSize: " + pageSize + " (expected: power of 2)");
343         }
344 
345         if (pageSize < alignment) {
346             throw new IllegalArgumentException("Alignment cannot be greater than page size. " +
347                     "Alignment: " + alignment + ", page size: " + pageSize + '.');
348         }
349 
350         // Logarithm base 2. At this point we know that pageSize is a power of two.
351         return Integer.SIZE - 1 - Integer.numberOfLeadingZeros(pageSize);
352     }
353 
354     private static int validateAndCalculateChunkSize(int pageSize, int maxOrder) {
355         if (maxOrder > 14) {
356             throw new IllegalArgumentException("maxOrder: " + maxOrder + " (expected: 0-14)");
357         }
358 
359         // Ensure the resulting chunkSize does not overflow.
360         int chunkSize = pageSize;
361         for (int i = maxOrder; i > 0; i --) {
362             if (chunkSize > MAX_CHUNK_SIZE / 2) {
363                 throw new IllegalArgumentException(String.format(
364                         "pageSize (%d) << maxOrder (%d) must not exceed %d", pageSize, maxOrder, MAX_CHUNK_SIZE));
365             }
366             chunkSize <<= 1;
367         }
368         return chunkSize;
369     }

371     @Override
372     protected ByteBuf newHeapBuffer(int initialCapacity, int maxCapacity) {
373         PoolThreadCache cache = threadCache.get();
374         PoolArena<byte[]> heapArena = cache.heapArena;
375 
376         final ByteBuf buf;
377         if (heapArena != null) {
378             buf = heapArena.allocate(cache, initialCapacity, maxCapacity);
379         } else {
380             buf = PlatformDependent.hasUnsafe() ?
381                     new UnpooledUnsafeHeapByteBuf(this, initialCapacity, maxCapacity) :
382                     new UnpooledHeapByteBuf(this, initialCapacity, maxCapacity);
383         }
384 
385         return toLeakAwareBuffer(buf);
386     }
387 
388     @Override
389     protected ByteBuf newDirectBuffer(int initialCapacity, int maxCapacity) {
390         PoolThreadCache cache = threadCache.get();
391         PoolArena<ByteBuffer> directArena = cache.directArena;
392 
393         final ByteBuf buf;
394         if (directArena != null) {
395             buf = directArena.allocate(cache, initialCapacity, maxCapacity);
396         } else {
397             buf = PlatformDependent.hasUnsafe() ?
398                     UnsafeByteBufUtil.newUnsafeDirectByteBuf(this, initialCapacity, maxCapacity) :
399                     new UnpooledDirectByteBuf(this, initialCapacity, maxCapacity);
400         }
401 
402         return toLeakAwareBuffer(buf);
403     }

500     final class PoolThreadLocalCache extends FastThreadLocal<PoolThreadCache> {
501         private final boolean useCacheForAllThreads;
502 
503         PoolThreadLocalCache(boolean useCacheForAllThreads) {
504             this.useCacheForAllThreads = useCacheForAllThreads;
505         }
506 
507         @Override
508         protected synchronized PoolThreadCache initialValue() {
509             final PoolArena<byte[]> heapArena = leastUsedArena(heapArenas);
510             final PoolArena<ByteBuffer> directArena = leastUsedArena(directArenas);
511 
512             final Thread current = Thread.currentThread();
513             if (useCacheForAllThreads || current instanceof FastThreadLocalThread) {
514                 final PoolThreadCache cache = new PoolThreadCache(
515                         heapArena, directArena, smallCacheSize, normalCacheSize,
516                         DEFAULT_MAX_CACHED_BUFFER_CAPACITY, DEFAULT_CACHE_TRIM_INTERVAL);
517 
518                 if (DEFAULT_CACHE_TRIM_INTERVAL_MILLIS > 0) {
519                     final EventExecutor executor = ThreadExecutorMap.currentExecutor();
520                     if (executor != null) {
521                         executor.scheduleAtFixedRate(trimTask, DEFAULT_CACHE_TRIM_INTERVAL_MILLIS,
522                                 DEFAULT_CACHE_TRIM_INTERVAL_MILLIS, TimeUnit.MILLISECONDS);
523                     }
524                 }
525                 return cache;
526             }
527             // No caching so just use 0 as sizes.
528             return new PoolThreadCache(heapArena, directArena, 0, 0, 0, 0);
529         }
530 
531         @Override
532         protected void onRemoval(PoolThreadCache threadCache) {
533             threadCache.free(false);
534         }
535 
536         private <T> PoolArena<T> leastUsedArena(PoolArena<T>[] arenas) {
537             if (arenas == null || arenas.length == 0) {
538                 return null;
539             }
540 
541             PoolArena<T> minArena = arenas[0];
542             for (int i = 1; i < arenas.length; i++) {
543                 PoolArena<T> arena = arenas[i];
544                 if (arena.numThreadCaches.get() < minArena.numThreadCaches.get()) {
545                     minArena = arena;
546                 }
547             }
548 
549             return minArena;
550         }
551     }

666     private static long usedMemory(PoolArena<?>[] arenas) {
667         if (arenas == null) {

670         long used = 0;
671         for (PoolArena<?> arena : arenas) {
