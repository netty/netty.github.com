417     /**
418      * Set TLS session ticket keys.
419      *
420      * <p> The first key in the list is the primary key. Tickets dervied from the other keys
421      * in the list will be accepted but updated to a new ticket using the primary key. This
422      * is useful for implementing ticket key rotation.
423      * See <a href="https://tools.ietf.org/html/rfc5077">RFC 5077</a>
424      *
425      * @param ctx Server or Client context to use
426      * @param keys the {@link SessionTicketKey}s
427      */
428     public static void setSessionTicketKeys(long ctx, SessionTicketKey[] keys) {
429         if (keys == null || keys.length == 0) {
430             throw new IllegalArgumentException("Length of the keys should be longer than 0.");
431         }
432         byte[] binaryKeys = new byte[keys.length * SessionTicketKey.TICKET_KEY_SIZE];
433         for (int i = 0; i < keys.length; i++) {
434             SessionTicketKey key = keys[i];
435             int dstCurPos = SessionTicketKey.TICKET_KEY_SIZE * i;
436             System.arraycopy(key.name, 0, binaryKeys, dstCurPos, SessionTicketKey.NAME_SIZE);
437             dstCurPos += SessionTicketKey.NAME_SIZE;
438             System.arraycopy(key.hmacKey, 0, binaryKeys, dstCurPos, SessionTicketKey.HMAC_KEY_SIZE);
439             dstCurPos += SessionTicketKey.HMAC_KEY_SIZE;
440             System.arraycopy(key.aesKey, 0, binaryKeys, dstCurPos, SessionTicketKey.AES_KEY_SIZE);
441         }
442         setSessionTicketKeys0(ctx, binaryKeys);
443     }

466     /**
467      * Set Type of Client Certificate verification and Maximum depth of CA Certificates
468      * in Client Certificate verification.
469      * <br>
470      * This directive sets the Certificate verification level for the Client
471      * Authentication. Notice that this directive can be used both in per-server
472      * and per-directory context. In per-server context it applies to the client
473      * authentication process used in the standard SSL handshake when a connection
474      * is established. In per-directory context it forces a SSL renegotiation with
475      * the reconfigured client verification level after the HTTP request was read
476      * but before the HTTP response is sent.
477      * <br>
478      * The following levels are available for level:
479      * <ul>
480      * <li>{@link SSL#SSL_CVERIFY_IGNORED} - The level is ignored. Only depth will change.</li>
481      * <li>{@link SSL#SSL_CVERIFY_NONE} - No client Certificate is required at all</li>
482      * <li>{@link SSL#SSL_CVERIFY_OPTIONAL} - The client may present a valid Certificate</li>
483      * <li>{@link SSL#SSL_CVERIFY_REQUIRED} - The client has to present a valid Certificate</li>
484      * </ul>
485      * The depth actually is the maximum number of intermediate certificate issuers,
486      * i.e. the number of CA certificates which are max allowed to be followed while
487      * verifying the client certificate. A depth of 0 means that self-signed client
488      * certificates are accepted only, the default depth of 1 means the client
489      * certificate can be self-signed or has to be signed by a CA which is directly
490      * known to the server (i.e. the CA's certificate is under
491      * <code>setCACertificatePath</code>), etc.
492      * @param ctx Server or Client context to use.
493      * @param level Type of Client Certificate verification.
494      * @param depth Maximum depth of CA Certificates in Client Certificate
495      *              verification.
496      */
497     public static native void setVerify(long ctx, int level, int depth);

537     private static byte[] protocolsToWireFormat(String[] protocols) {
538         ByteArrayOutputStream out = new ByteArrayOutputStream();
539         try {
540             for (String p : protocols) {
541                 byte[] bytes = p.getBytes(StandardCharsets.US_ASCII);
542                 if (bytes.length <= MAX_ALPN_NPN_PROTO_SIZE) {
543                     out.write(bytes.length);
544                     out.write(bytes);
545                 }
546             }
547            return out.toByteArray();
548         } catch (IOException e) {
549             throw new IllegalStateException(e);
550         } finally {
551             try {
552                 out.close();
553             } catch (IOException ignore) {
554                 // ignore
555             }
556         }
557     }
