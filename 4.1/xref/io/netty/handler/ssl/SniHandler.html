41  /**
42   * <p>Enables <a href="https://tools.ietf.org/html/rfc3546#section-3.1">SNI
43   * (Server Name Indication)</a> extension for server side SSL. For clients
44   * support SNI, the server could have multiple host name bound on a single IP.
45   * The client will send host name in the handshake data so server could decide
46   * which certificate to choose for the host name.</p>
47   */

109     @Override
110     protected void decode(ChannelHandlerContext ctx, ByteBuf in, List<Object> out) throws Exception {
111         if (!suppressRead && !handshakeFailed) {
112             final int writerIndex = in.writerIndex();
113             try {
114                 loop:
115                 for (int i = 0; i < MAX_SSL_RECORDS; i++) {
116                     final int readerIndex = in.readerIndex();
117                     final int readableBytes = writerIndex - readerIndex;
118                     if (readableBytes < SslUtils.SSL_RECORD_HEADER_LENGTH) {
119                         // Not enough data to determine the record type and length.
120                         return;
121                     }
122 
123                     final int command = in.getUnsignedByte(readerIndex);
124 
125                     // tls, but not handshake command
126                     switch (command) {
127                         case SslUtils.SSL_CONTENT_TYPE_CHANGE_CIPHER_SPEC:
128                         case SslUtils.SSL_CONTENT_TYPE_ALERT:
129                             final int len = SslUtils.getEncryptedPacketLength(in, readerIndex);
130 
131                             // Not an SSL/TLS packet
132                             if (len == -1) {
133                                 handshakeFailed = true;
134                                 NotSslRecordException e = new NotSslRecordException(
135                                         "not an SSL/TLS record: " + ByteBufUtil.hexDump(in));
136                                 in.skipBytes(in.readableBytes());
137                                 ctx.fireExceptionCaught(e);
138 
139                                 SslUtils.notifyHandshakeFailure(ctx, e);
140                                 return;
141                             }
142                             if (writerIndex - readerIndex - SslUtils.SSL_RECORD_HEADER_LENGTH < len) {
143                                 // Not enough data
144                                 return;
145                             }
146                             // increase readerIndex and try again.
147                             in.skipBytes(len);
148                             continue;
149                         case SslUtils.SSL_CONTENT_TYPE_HANDSHAKE:
150                             final int majorVersion = in.getUnsignedByte(readerIndex + 1);
151 
152                             // SSLv3 or TLS
153                             if (majorVersion == 3) {
154                                 final int packetLength = in.getUnsignedShort(readerIndex + 3) +
155                                                          SslUtils.SSL_RECORD_HEADER_LENGTH;
156 
157                                 if (readableBytes < packetLength) {
158                                     // client hello incomplete; try again to decode once more data is ready.
159                                     return;
160                                 }
161 
162                                 // See https://tools.ietf.org/html/rfc5246#section-7.4.1.2
163                                 //
164                                 // Decode the ssl client hello packet.
165                                 // We have to skip bytes until SessionID (which sum to 43 bytes).
166                                 //
167                                 // struct {
168                                 //    ProtocolVersion client_version;
169                                 //    Random random;
170                                 //    SessionID session_id;
171                                 //    CipherSuite cipher_suites<2..2^16-2>;
172                                 //    CompressionMethod compression_methods<1..2^8-1>;
173                                 //    select (extensions_present) {
174                                 //        case false:
175                                 //            struct {};
176                                 //        case true:
177                                 //            Extension extensions<0..2^16-1>;
178                                 //    };
179                                 // } ClientHello;
180                                 //
181 
182                                 final int endOffset = readerIndex + packetLength;
183                                 int offset = readerIndex + 43;
184 
185                                 if (endOffset - offset < 6) {
186                                     break loop;
187                                 }
188 
189                                 final int sessionIdLength = in.getUnsignedByte(offset);
190                                 offset += sessionIdLength + 1;
191 
192                                 final int cipherSuitesLength = in.getUnsignedShort(offset);
193                                 offset += cipherSuitesLength + 2;
194 
195                                 final int compressionMethodLength = in.getUnsignedByte(offset);
196                                 offset += compressionMethodLength + 1;
197 
198                                 final int extensionsLength = in.getUnsignedShort(offset);
199                                 offset += 2;
200                                 final int extensionsLimit = offset + extensionsLength;
201 
202                                 if (extensionsLimit > endOffset) {
203                                     // Extensions should never exceed the record boundary.
204                                     break loop;
205                                 }
206 
207                                 for (;;) {
208                                     if (extensionsLimit - offset < 4) {
209                                         break loop;
210                                     }
211 
212                                     final int extensionType = in.getUnsignedShort(offset);
213                                     offset += 2;
214 
215                                     final int extensionLength = in.getUnsignedShort(offset);
216                                     offset += 2;
217 
218                                     if (extensionsLimit - offset < extensionLength) {
219                                         break loop;
220                                     }
221 
222                                     // SNI
223                                     // See https://tools.ietf.org/html/rfc6066#page-6
224                                     if (extensionType == 0) {
225                                         offset += 2;
226                                         if (extensionsLimit - offset < 3) {
227                                             break loop;
228                                         }
229 
230                                         final int serverNameType = in.getUnsignedByte(offset);
231                                         offset++;
232 
233                                         if (serverNameType == 0) {
234                                             final int serverNameLength = in.getUnsignedShort(offset);
235                                             offset += 2;
236 
237                                             if (extensionsLimit - offset < serverNameLength) {
238                                                 break loop;
239                                             }
240 
241                                             final String hostname = in.toString(offset, serverNameLength,
242                                                                                 CharsetUtil.UTF_8);
243 
244                                             select(ctx, IDN.toASCII(hostname,
245                                                                     IDN.ALLOW_UNASSIGNED).toLowerCase(Locale.US));
246                                             return;
247                                         } else {
248                                             // invalid enum value
249                                             break loop;
250                                         }
251                                     }
252 
253                                     offset += extensionLength;
254                                 }
255                             }
256                             // Fall-through
257                         default:
258                             //not tls, ssl or application data, do not try sni
259                             break loop;
260                     }
261                 }
262             } catch (Throwable e) {
263                 // unexpected encoding, ignore sni and use default
264                 if (logger.isDebugEnabled()) {
265                     logger.debug("Unexpected client hello packet: " + ByteBufUtil.hexDump(in), e);
266                 }
267             }
268             // Just select the default SslContext
269             select(ctx, null);
270         }
271     }

273     private void select(final ChannelHandlerContext ctx, final String hostname) {
274         Future<SslContext> future = mapping.map(hostname, ctx.executor().<SslContext>newPromise());
275         if (future.isDone()) {
276             if (future.isSuccess()) {
277                 replaceHandler(ctx, new Selection(future.getNow(), hostname));
278             } else {
279                 throw new DecoderException("failed to get the SslContext for " + hostname, future.cause());
280             }
281         } else {
282             suppressRead = true;
283             future.addListener(new FutureListener<SslContext>() {
284                 @Override
285                 public void operationComplete(Future<SslContext> future) throws Exception {
286                     try {
287                         suppressRead = false;
288                         if (future.isSuccess()) {
289                             replaceHandler(ctx, new Selection(future.getNow(), hostname));
290                         } else {
291                             ctx.fireExceptionCaught(new DecoderException("failed to get the SslContext for "
292                                     + hostname, future.cause()));
293                         }
294                     } finally {
295                         if (readPending) {
296                             readPending = false;
297                             ctx.read();
298                         }
299                     }
300                 }
301             });
302         }
303     }

305     private void replaceHandler(ChannelHandlerContext ctx, Selection selection) {
306         this.selection = selection;
307         SslHandler sslHandler = selection.context.newHandler(ctx.alloc());
308         ctx.pipeline().replace(this, SslHandler.class.getName(), sslHandler);
309     }

356     private static final class AsyncMappingAdapter implements AsyncMapping<String, SslContext> {
357         private final Mapping<? super String, ? extends SslContext> mapping;
358 
359         private AsyncMappingAdapter(Mapping<? super String, ? extends SslContext> mapping) {
360             this.mapping = ObjectUtil.checkNotNull(mapping, "mapping");
361         }
362 
363         @Override
364         public Future<SslContext> map(String input, Promise<SslContext> promise) {
365             final SslContext context;
366             try {
367                 context = mapping.map(input);
368             } catch (Throwable cause) {
369                 return promise.setFailure(cause);
370             }
371             return promise.setSuccess(context);
372         }
373     }
