110     static {
111         Throwable cause = null;
112 
113         if (SystemPropertyUtil.getBoolean("io.netty.handler.ssl.noOpenSsl", false)) {
114             cause = new UnsupportedOperationException(
115                     "OpenSSL was explicit disabled with -Dio.netty.handler.ssl.noOpenSsl=true");
116 
117             logger.debug(
118                     "netty-tcnative explicit disabled; " +
119                             OpenSslEngine.class.getSimpleName() + " will be unavailable.", cause);
120         } else {
121             // Test if netty-tcnative is in the classpath first.
122             try {
123                 Class.forName("io.netty.internal.tcnative.SSLContext", false,
124                         PlatformDependent.getClassLoader(OpenSsl.class));
125             } catch (ClassNotFoundException t) {
126                 cause = t;
127                 logger.debug(
128                         "netty-tcnative not in the classpath; " +
129                                 OpenSslEngine.class.getSimpleName() + " will be unavailable.");
130             }
131 
132             // If in the classpath, try to load the native library and initialize netty-tcnative.
133             if (cause == null) {
134                 try {
135                     // The JNI library was not already loaded. Load it now.
136                     loadTcNative();
137                 } catch (Throwable t) {
138                     cause = t;
139                     logger.debug(
140                             "Failed to load netty-tcnative; " +
141                                     OpenSslEngine.class.getSimpleName() + " will be unavailable, unless the " +
142                                     "application has already loaded the symbols by some other means. " +
143                                     "See https://netty.io/wiki/forked-tomcat-native.html for more information.", t);
144                 }
145 
146                 try {
147                     String engine = SystemPropertyUtil.get("io.netty.handler.ssl.openssl.engine", null);
148                     if (engine == null) {
149                         logger.debug("Initialize netty-tcnative using engine: 'default'");
150                     } else {
151                         logger.debug("Initialize netty-tcnative using engine: '{}'", engine);
152                     }
153                     initializeTcNative(engine);
154 
155                     // The library was initialized successfully. If loading the library failed above,
156                     // reset the cause now since it appears that the library was loaded by some other
157                     // means.
158                     cause = null;
159                 } catch (Throwable t) {
160                     if (cause == null) {
161                         cause = t;
162                     }
163                     logger.debug(
164                             "Failed to initialize netty-tcnative; " +
165                                     OpenSslEngine.class.getSimpleName() + " will be unavailable. " +
166                                     "See https://netty.io/wiki/forked-tomcat-native.html for more information.", t);
167                 }
168             }
169         }
170 
171         UNAVAILABILITY_CAUSE = cause;
172 
173         if (cause == null) {
174             logger.debug("netty-tcnative using native library: {}", SSL.versionString());
175 
176             final List<String> defaultCiphers = new ArrayList<String>();
177             final Set<String> availableOpenSslCipherSuites = new LinkedHashSet<String>(128);
178             boolean supportsKeyManagerFactory = false;
179             boolean useKeyManagerFactory = false;
180             boolean tlsv13Supported = false;
181 
182             IS_BORINGSSL = "BoringSSL".equals(versionString());
183             if (IS_BORINGSSL) {
184                 EXTRA_SUPPORTED_TLS_1_3_CIPHERS = new String [] { "TLS_AES_128_GCM_SHA256",
185                         "TLS_AES_256_GCM_SHA384" ,
186                         "TLS_CHACHA20_POLY1305_SHA256" };
187             }  else {
188                 EXTRA_SUPPORTED_TLS_1_3_CIPHERS = EmptyArrays.EMPTY_STRINGS;
189             }
190 
191             try {
192                 final long sslCtx = SSLContext.make(SSL.SSL_PROTOCOL_ALL, SSL.SSL_MODE_SERVER);
193                 long certBio = 0;
194                 long keyBio = 0;
195                 long cert = 0;
196                 long key = 0;
197                 try {
198                     try {
199                         StringBuilder tlsv13Ciphers = new StringBuilder();
200 
201                         for (String cipher: TLSV13_CIPHERS) {
202                             String converted = CipherSuiteConverter.toOpenSsl(cipher, IS_BORINGSSL);
203                             if (converted != null) {
204                                 tlsv13Ciphers.append(converted).append(':');
205                             }
206                         }
207                         if (tlsv13Ciphers.length() == 0) {
208                             tlsv13Supported = false;
209                         } else {
210                             tlsv13Ciphers.setLength(tlsv13Ciphers.length() - 1);
211                             SSLContext.setCipherSuite(sslCtx, tlsv13Ciphers.toString() , true);
212                             tlsv13Supported = true;
213                         }
214 
215                     } catch (Exception ignore) {
216                         tlsv13Supported = false;
217                     }
218 
219                     SSLContext.setCipherSuite(sslCtx, "ALL", false);
220 
221                     final long ssl = SSL.newSSL(sslCtx, true);
222                     try {
223                         for (String c: SSL.getCiphers(ssl)) {
224                             // Filter out bad input.
225                             if (c == null || c.isEmpty() || availableOpenSslCipherSuites.contains(c) ||
226                                 // Filter out TLSv1.3 ciphers if not supported.
227                                 !tlsv13Supported && isTLSv13Cipher(c)) {
228                                 continue;
229                             }
230                             availableOpenSslCipherSuites.add(c);
231                         }
232                         if (IS_BORINGSSL) {
233                             // Currently BoringSSL does not include these when calling SSL.getCiphers() even when these
234                             // are supported.
235                             Collections.addAll(availableOpenSslCipherSuites, EXTRA_SUPPORTED_TLS_1_3_CIPHERS);
236                             Collections.addAll(availableOpenSslCipherSuites,
237                                                "AEAD-AES128-GCM-SHA256",
238                                                "AEAD-AES256-GCM-SHA384",
239                                                "AEAD-CHACHA20-POLY1305-SHA256");
240                         }
241 
242                         PemEncoded privateKey = PemPrivateKey.valueOf(KEY.getBytes(CharsetUtil.US_ASCII));
243                         try {
244                             // Let's check if we can set a callback, which may not work if the used OpenSSL version
245                             // is to old.
246                             SSLContext.setCertificateCallback(sslCtx, null);
247 
248                             X509Certificate certificate = selfSignedCertificate();
249                             certBio = ReferenceCountedOpenSslContext.toBIO(ByteBufAllocator.DEFAULT, certificate);
250                             cert = SSL.parseX509Chain(certBio);
251 
252                             keyBio = ReferenceCountedOpenSslContext.toBIO(
253                                     UnpooledByteBufAllocator.DEFAULT, privateKey.retain());
254                             key = SSL.parsePrivateKey(keyBio, null);
255 
256                             SSL.setKeyMaterial(ssl, cert, key);
257                             supportsKeyManagerFactory = true;
258                             try {
259                                 boolean propertySet = SystemPropertyUtil.contains(
260                                         "io.netty.handler.ssl.openssl.useKeyManagerFactory");
261                                 if (!IS_BORINGSSL) {
262                                     useKeyManagerFactory = SystemPropertyUtil.getBoolean(
263                                             "io.netty.handler.ssl.openssl.useKeyManagerFactory", true);
264 
265                                     if (propertySet) {
266                                         logger.info("System property " +
267                                                 "'io.netty.handler.ssl.openssl.useKeyManagerFactory'" +
268                                                 " is deprecated and so will be ignored in the future");
269                                     }
270                                 } else {
271                                     useKeyManagerFactory = true;
272                                     if (propertySet) {
273                                         logger.info("System property " +
274                                                 "'io.netty.handler.ssl.openssl.useKeyManagerFactory'" +
275                                                 " is deprecated and will be ignored when using BoringSSL");
276                                     }
277                                 }
278                             } catch (Throwable ignore) {
279                                 logger.debug("Failed to get useKeyManagerFactory system property.");
280                             }
281                         } catch (Error ignore) {
282                             logger.debug("KeyManagerFactory not supported.");
283                         } finally {
284                             privateKey.release();
285                         }
286                     } finally {
287                         SSL.freeSSL(ssl);
288                         if (certBio != 0) {
289                             SSL.freeBIO(certBio);
290                         }
291                         if (keyBio != 0) {
292                             SSL.freeBIO(keyBio);
293                         }
294                         if (cert != 0) {
295                             SSL.freeX509Chain(cert);
296                         }
297                         if (key != 0) {
298                             SSL.freePrivateKey(key);
299                         }
300                     }
301                 } finally {
302                     SSLContext.free(sslCtx);
303                 }
304             } catch (Exception e) {
305                 logger.warn("Failed to get the list of available OpenSSL cipher suites.", e);
306             }
307             AVAILABLE_OPENSSL_CIPHER_SUITES = Collections.unmodifiableSet(availableOpenSslCipherSuites);
308             final Set<String> availableJavaCipherSuites = new LinkedHashSet<String>(
309                     AVAILABLE_OPENSSL_CIPHER_SUITES.size() * 2);
310             for (String cipher: AVAILABLE_OPENSSL_CIPHER_SUITES) {
311                 // Included converted but also openssl cipher name
312                 if (!isTLSv13Cipher(cipher)) {
313                     availableJavaCipherSuites.add(CipherSuiteConverter.toJava(cipher, "TLS"));
314                     availableJavaCipherSuites.add(CipherSuiteConverter.toJava(cipher, "SSL"));
315                 } else {
316                     // TLSv1.3 ciphers have the correct format.
317                     availableJavaCipherSuites.add(cipher);
318                 }
319             }
320 
321             addIfSupported(availableJavaCipherSuites, defaultCiphers, DEFAULT_CIPHER_SUITES);
322             addIfSupported(availableJavaCipherSuites, defaultCiphers, TLSV13_CIPHER_SUITES);
323 
324             useFallbackCiphersIfDefaultIsEmpty(defaultCiphers, availableJavaCipherSuites);
325             DEFAULT_CIPHERS = Collections.unmodifiableList(defaultCiphers);
326 
327             AVAILABLE_JAVA_CIPHER_SUITES = Collections.unmodifiableSet(availableJavaCipherSuites);
328 
329             final Set<String> availableCipherSuites = new LinkedHashSet<String>(
330                     AVAILABLE_OPENSSL_CIPHER_SUITES.size() + AVAILABLE_JAVA_CIPHER_SUITES.size());
331             availableCipherSuites.addAll(AVAILABLE_OPENSSL_CIPHER_SUITES);
332             availableCipherSuites.addAll(AVAILABLE_JAVA_CIPHER_SUITES);
333 
334             AVAILABLE_CIPHER_SUITES = availableCipherSuites;
335             SUPPORTS_KEYMANAGER_FACTORY = supportsKeyManagerFactory;
336             USE_KEYMANAGER_FACTORY = useKeyManagerFactory;
337 
338             Set<String> protocols = new LinkedHashSet<String>(6);
339             // Seems like there is no way to explicitly disable SSLv2Hello in openssl so it is always enabled
340             protocols.add(PROTOCOL_SSL_V2_HELLO);
341             if (doesSupportProtocol(SSL.SSL_PROTOCOL_SSLV2, SSL.SSL_OP_NO_SSLv2)) {
342                 protocols.add(PROTOCOL_SSL_V2);
343             }
344             if (doesSupportProtocol(SSL.SSL_PROTOCOL_SSLV3, SSL.SSL_OP_NO_SSLv3)) {
345                 protocols.add(PROTOCOL_SSL_V3);
346             }
347             if (doesSupportProtocol(SSL.SSL_PROTOCOL_TLSV1, SSL.SSL_OP_NO_TLSv1)) {
348                 protocols.add(PROTOCOL_TLS_V1);
349             }
350             if (doesSupportProtocol(SSL.SSL_PROTOCOL_TLSV1_1, SSL.SSL_OP_NO_TLSv1_1)) {
351                 protocols.add(PROTOCOL_TLS_V1_1);
352             }
353             if (doesSupportProtocol(SSL.SSL_PROTOCOL_TLSV1_2, SSL.SSL_OP_NO_TLSv1_2)) {
354                 protocols.add(PROTOCOL_TLS_V1_2);
355             }
356 
357             // This is only supported by java11 and later.
358             if (tlsv13Supported && doesSupportProtocol(SSL.SSL_PROTOCOL_TLSV1_3, SSL.SSL_OP_NO_TLSv1_3)) {
359                 protocols.add(PROTOCOL_TLS_V1_3);
360                 TLSV13_SUPPORTED = true;
361             } else {
362                 TLSV13_SUPPORTED = false;
363             }
364 
365             SUPPORTED_PROTOCOLS_SET = Collections.unmodifiableSet(protocols);
366             SUPPORTS_OCSP = doesSupportOcsp();
367 
368             if (logger.isDebugEnabled()) {
369                 logger.debug("Supported protocols (OpenSSL): {} ", SUPPORTED_PROTOCOLS_SET);
370                 logger.debug("Default cipher suites (OpenSSL): {}", DEFAULT_CIPHERS);
371             }
372         } else {
373             DEFAULT_CIPHERS = Collections.emptyList();
374             AVAILABLE_OPENSSL_CIPHER_SUITES = Collections.emptySet();
375             AVAILABLE_JAVA_CIPHER_SUITES = Collections.emptySet();
376             AVAILABLE_CIPHER_SUITES = Collections.emptySet();
377             SUPPORTS_KEYMANAGER_FACTORY = false;
378             USE_KEYMANAGER_FACTORY = false;
379             SUPPORTED_PROTOCOLS_SET = Collections.emptySet();
380             SUPPORTS_OCSP = false;
381             TLSV13_SUPPORTED = false;
382             IS_BORINGSSL = false;
383             EXTRA_SUPPORTED_TLS_1_3_CIPHERS = EmptyArrays.EMPTY_STRINGS;
384         }
385     }

391     /**
392      * Returns a self-signed {@link X509Certificate} for {@code netty.io}.
393      */
394     static X509Certificate selfSignedCertificate() throws CertificateException {
395         return (X509Certificate) SslContext.X509_CERT_FACTORY.generateCertificate(
396                 new ByteArrayInputStream(CERT.getBytes(CharsetUtil.US_ASCII))
397         );
398     }

400     private static boolean doesSupportOcsp() {
401         boolean supportsOcsp = false;
402         if (version() >= 0x10002000L) {
403             long sslCtx = -1;
404             try {
405                 sslCtx = SSLContext.make(SSL.SSL_PROTOCOL_TLSV1_2, SSL.SSL_MODE_SERVER);
406                 SSLContext.enableOcsp(sslCtx, false);
407                 supportsOcsp = true;
408             } catch (Exception ignore) {
409                 // ignore
410             } finally {
411                 if (sslCtx != -1) {
412                     SSLContext.free(sslCtx);
413                 }
414             }
415         }
416         return supportsOcsp;
417     }
418     private static boolean doesSupportProtocol(int protocol, int opt) {
419         if (opt == 0) {
420             // If the opt is 0 the protocol is not supported. This is for example the case with BoringSSL and SSLv2.
421             return false;
422         }
423         long sslCtx = -1;
424         try {
425             sslCtx = SSLContext.make(protocol, SSL.SSL_MODE_COMBINED);
426             return true;
427         } catch (Exception ignore) {
428             return false;
429         } finally {
430             if (sslCtx != -1) {
431                 SSLContext.free(sslCtx);
432             }
433         }
434     }

436     /**
437      * Returns {@code true} if and only if
438      * <a href="https://netty.io/wiki/forked-tomcat-native.html">{@code netty-tcnative}</a> and its OpenSSL support
439      * are available.

443     }

569     private static void loadTcNative() throws Exception {
570         String os = PlatformDependent.normalizedOs();
571         String arch = PlatformDependent.normalizedArch();
572 
573         Set<String> libNames = new LinkedHashSet<String>(5);
574         String staticLibName = "netty_tcnative";
575 
576         // First, try loading the platform-specific library. Platform-specific
577         // libraries will be available if using a tcnative uber jar.
578         if ("linux".equals(os)) {
579             Set<String> classifiers = PlatformDependent.normalizedLinuxClassifiers();
580             for (String classifier : classifiers) {
581                 libNames.add(staticLibName + "_" + os + '_' + arch + "_" + classifier);
582             }
583             // generic arch-dependent library
584             libNames.add(staticLibName + "_" + os + '_' + arch);
585 
586             // Fedora SSL lib so naming (libssl.so.10 vs libssl.so.1.0.0).
587             // note: should already be included from the classifiers but if not, we use this as an
588             //       additional fallback option here
589             libNames.add(staticLibName + "_" + os + '_' + arch + "_fedora");
590         } else {
591             libNames.add(staticLibName + "_" + os + '_' + arch);
592         }
593         libNames.add(staticLibName + "_" + arch);
594         libNames.add(staticLibName);
595 
596         NativeLibraryLoader.loadFirstAvailable(PlatformDependent.getClassLoader(SSLContext.class),
597             libNames.toArray(new String[0]));
598     }
