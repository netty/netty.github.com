61  /**
62   * A secure socket protocol implementation which acts as a factory for {@link SSLEngine} and {@link SslHandler}.
63   * Internally, it is implemented via JDK's {@link SSLContext} or OpenSSL's {@code SSL_CTX}.
64   *
65   * <h3>Making your server support SSL/TLS</h3>
66   * <pre>
67   * // In your {@link ChannelInitializer}:
68   * {@link ChannelPipeline} p = channel.pipeline();
69   * {@link SslContext} sslCtx = {@link SslContextBuilder#forServer(File, File) SslContextBuilder.forServer(...)}.build();
70   * p.addLast("ssl", {@link #newEngine(ByteBufAllocator) sslCtx.newEngine(channel.alloc())});
71   * ...
72   * </pre>
73   *
74   * <h3>Making your client support SSL/TLS</h3>
75   * <pre>
76   * // In your {@link ChannelInitializer}:
77   * {@link ChannelPipeline} p = channel.pipeline();
78   * {@link SslContext} sslCtx = {@link SslContextBuilder#forClient() SslContextBuilder.forClient()}.build();
79   * p.addLast("ssl", {@link #newEngine(ByteBufAllocator, String, int) sslCtx.newEngine(channel.alloc(), host, port)});
80   * ...
81   * </pre>
82   */
83  public abstract class SslContext {
84      static final CertificateFactory X509_CERT_FACTORY;
85      static {
86          try {
87              X509_CERT_FACTORY = CertificateFactory.getInstance("X.509");
88          } catch (CertificateException e) {
89              throw new IllegalStateException("unable to instance X.509 CertificateFactory", e);
90          }
91      }

337     /**
338      * Creates a new server-side {@link SslContext}.
339      * @param provider the {@link SslContext} implementation to use.
340      *                 {@code null} to use the current default one.
341      * @param trustCertChainFile an X.509 certificate chain file in PEM format.
342      *                      This provides the certificate chains used for mutual authentication.
343      *                      {@code null} to use the system default
344      * @param trustManagerFactory the {@link TrustManagerFactory} that provides the {@link TrustManager}s
345      *                            that verifies the certificates sent from clients.
346      *                            {@code null} to use the default or the results of parsing {@code trustCertChainFile}.
347      *                            This parameter is ignored if {@code provider} is not {@link SslProvider#JDK}.
348      * @param keyCertChainFile an X.509 certificate chain file in PEM format
349      * @param keyFile a PKCS#8 private key file in PEM format
350      * @param keyPassword the password of the {@code keyFile}.
351      *                    {@code null} if it's not password-protected.
352      * @param keyManagerFactory the {@link KeyManagerFactory} that provides the {@link KeyManager}s
353      *                          that is used to encrypt data being sent to clients.
354      *                          {@code null} to use the default or the results of parsing
355      *                          {@code keyCertChainFile} and {@code keyFile}.
356      *                          This parameter is ignored if {@code provider} is not {@link SslProvider#JDK}.
357      * @param ciphers the cipher suites to enable, in the order of preference.
358      *                {@code null} to use the default cipher suites.
359      * @param cipherFilter a filter to apply over the supplied list of ciphers
360      *                Only required if {@code provider} is {@link SslProvider#JDK}
361      * @param apn Provides a means to configure parameters related to application protocol negotiation.
362      * @param sessionCacheSize the size of the cache used for storing SSL session objects.
363      *                         {@code 0} to use the default value.
364      * @param sessionTimeout the timeout for the cached SSL session objects, in seconds.
365      *                       {@code 0} to use the default value.
366      * @return a new server-side {@link SslContext}
367      * @deprecated Replaced by {@link SslContextBuilder}
368      */
369     @Deprecated
370     public static SslContext newServerContext(
371             SslProvider provider,
372             File trustCertChainFile, TrustManagerFactory trustManagerFactory,
373             File keyCertChainFile, File keyFile, String keyPassword, KeyManagerFactory keyManagerFactory,
374             Iterable<String> ciphers, CipherSuiteFilter cipherFilter, ApplicationProtocolConfig apn,
375             long sessionCacheSize, long sessionTimeout) throws SSLException {
376         try {
377             return newServerContextInternal(provider, toX509Certificates(trustCertChainFile), trustManagerFactory,
378                                             toX509Certificates(keyCertChainFile),
379                                             toPrivateKey(keyFile, keyPassword),
380                                             keyPassword, keyManagerFactory, ciphers, cipherFilter, apn,
381                                             sessionCacheSize, sessionTimeout, ClientAuth.NONE);
382         } catch (Exception e) {
383             if (e instanceof SSLException) {
384                 throw (SSLException) e;
385             }
386             throw new SSLException("failed to initialize the server-side SSL context", e);
387         }
388     }

390     static SslContext newServerContextInternal(
391             SslProvider provider,
392             X509Certificate[] trustCertChain, TrustManagerFactory trustManagerFactory,
393             X509Certificate[] keyCertChain, PrivateKey key, String keyPassword, KeyManagerFactory keyManagerFactory,
394             Iterable<String> ciphers, CipherSuiteFilter cipherFilter, ApplicationProtocolConfig apn,
395             long sessionCacheSize, long sessionTimeout,
396             ClientAuth clientAuth) throws SSLException {
397 
398         if (provider == null) {
399             provider = defaultServerProvider();
400         }
401 
402         switch (provider) {
403         case JDK:
404             return new JdkSslServerContext(
405                     trustCertChain, trustManagerFactory, keyCertChain, key, keyPassword,
406                     keyManagerFactory, ciphers, cipherFilter, apn, sessionCacheSize, sessionTimeout,
407                     clientAuth);
408         case OPENSSL:
409             return new OpenSslServerContext(
410                     trustCertChain, trustManagerFactory, keyCertChain, key, keyPassword,
411                     keyManagerFactory, ciphers, cipherFilter, apn, sessionCacheSize, sessionTimeout,
412                     clientAuth);
413         default:
414             throw new Error(provider.toString());
415         }
416     }

856     /**
857      * Generates a key specification for an (encrypted) private key.
858      *
859      * @param password characters, if {@code null} or empty an unencrypted key is assumed
860      * @param key bytes of the DER encoded private key
861      *
862      * @return a key specification
863      *
864      * @throws IOException if parsing {@code key} fails
865      * @throws NoSuchAlgorithmException if the algorithm used to encrypt {@code key} is unkown
866      * @throws NoSuchPaddingException if the padding scheme specified in the decryption algorithm is unkown
867      * @throws InvalidKeySpecException if the decryption key based on {@code password} cannot be generated
868      * @throws InvalidKeyException if the decryption key based on {@code password} cannot be used to decrypt
869      *                             {@code key}
870      * @throws InvalidAlgorithmParameterException if decryption algorithm parameters are somehow faulty
871      */
872     protected static PKCS8EncodedKeySpec generateKeySpec(char[] password, byte[] key)
873             throws IOException, NoSuchAlgorithmException, NoSuchPaddingException, InvalidKeySpecException,
874             InvalidKeyException, InvalidAlgorithmParameterException {
875 
876         if (password == null || password.length == 0) {
877             return new PKCS8EncodedKeySpec(key);
878         }
879 
880         EncryptedPrivateKeyInfo encryptedPrivateKeyInfo = new EncryptedPrivateKeyInfo(key);
881         SecretKeyFactory keyFactory = SecretKeyFactory.getInstance(encryptedPrivateKeyInfo.getAlgName());
882         PBEKeySpec pbeKeySpec = new PBEKeySpec(password);
883         SecretKey pbeKey = keyFactory.generateSecret(pbeKeySpec);
884 
885         Cipher cipher = Cipher.getInstance(encryptedPrivateKeyInfo.getAlgName());
886         cipher.init(Cipher.DECRYPT_MODE, pbeKey, encryptedPrivateKeyInfo.getAlgParameters());
887 
888         return encryptedPrivateKeyInfo.getKeySpec(cipher);
889     }

929     private static PrivateKey getPrivateKeyFromByteBuffer(ByteBuf encodedKeyBuf, String keyPassword)
930             throws NoSuchAlgorithmException, NoSuchPaddingException, InvalidKeySpecException,
931             InvalidAlgorithmParameterException, KeyException, IOException {
932 
933         byte[] encodedKey = new byte[encodedKeyBuf.readableBytes()];
934         encodedKeyBuf.readBytes(encodedKey).release();
935 
936         PKCS8EncodedKeySpec encodedKeySpec = generateKeySpec(keyPassword == null ? null : keyPassword.toCharArray(),
937                                                              encodedKey);
938 
939         PrivateKey key;
940         try {
941             key = KeyFactory.getInstance("RSA").generatePrivate(encodedKeySpec);
942         } catch (InvalidKeySpecException ignore) {
943             try {
944                 key = KeyFactory.getInstance("DSA").generatePrivate(encodedKeySpec);
945             } catch (InvalidKeySpecException ignore2) {
946                 try {
947                     key = KeyFactory.getInstance("EC").generatePrivate(encodedKeySpec);
948                 } catch (InvalidKeySpecException e) {
949                     throw new InvalidKeySpecException("Neither RSA, DSA nor EC worked", e);
950                 }
951             }
952         }
953         return key;

985     private static X509Certificate[] getCertificatesFromBuffers(ByteBuf[] certs) throws CertificateException {
986         CertificateFactory cf = CertificateFactory.getInstance("X.509");
987         X509Certificate[] x509Certs = new X509Certificate[certs.length];
988 
989         try {
990             for (int i = 0; i < certs.length; i++) {
991                 x509Certs[i] = (X509Certificate) cf.generateCertificate(new ByteBufInputStream(certs[i]));
992             }
993         } finally {
994             for (ByteBuf buf: certs) {
995                 buf.release();
996             }
997         }
998         return x509Certs;
999     }

1001     static TrustManagerFactory buildTrustManagerFactory(
1002             X509Certificate[] certChain, TrustManagerFactory trustManagerFactory)
1003             throws NoSuchAlgorithmException, CertificateException, KeyStoreException, IOException {
1004         KeyStore ks = KeyStore.getInstance("JKS");
1005         ks.load(null, null);
1006 
1007         for (X509Certificate cert: certChain) {
1008             X500Principal principal = cert.getSubjectX500Principal();
1009             ks.setCertificateEntry(principal.getName("RFC2253"), cert);

1013         if (trustManagerFactory == null) {
1014             trustManagerFactory = TrustManagerFactory.getInstance(TrustManagerFactory.getDefaultAlgorithm());
1015         }
1016         trustManagerFactory.init(ks);
