79  /**
80   * Adds <a href="https://en.wikipedia.org/wiki/Transport_Layer_Security">SSL
81   * &middot; TLS</a> and StartTLS support to a {@link Channel}.  Please refer
82   * to the <strong>"SecureChat"</strong> example in the distribution or the web
83   * site for the detailed usage.
84   *
85   * <h3>Beginning the handshake</h3>
86   * <p>
87   * Beside using the handshake {@link ChannelFuture} to get notified about the completion of the handshake it's
88   * also possible to detect it by implement the
89   * {@link ChannelInboundHandler#userEventTriggered(ChannelHandlerContext, Object)}
90   * method and check for a {@link SslHandshakeCompletionEvent}.
91   *
92   * <h3>Handshake</h3>
93   * <p>
94   * The handshake will be automatically issued for you once the {@link Channel} is active and
95   * {@link SSLEngine#getUseClientMode()} returns {@code true}.
96   * So no need to bother with it by your self.
97   *
98   * <h3>Closing the session</h3>
99   * <p>
100  * To close the SSL session, the {@link #closeOutbound()} method should be
101  * called to send the {@code close_notify} message to the remote peer. One
102  * exception is when you close the {@link Channel} - {@link SslHandler}
103  * intercepts the close request and send the {@code close_notify} message
104  * before the channel closure automatically.  Once the SSL session is closed,
105  * it is not reusable, and consequently you should create a new
106  * {@link SslHandler} with a new {@link SSLEngine} as explained in the
107  * following section.
108  *
109  * <h3>Restarting the session</h3>
110  * <p>
111  * To restart the SSL session, you must remove the existing closed
112  * {@link SslHandler} from the {@link ChannelPipeline}, insert a new
113  * {@link SslHandler} with a new {@link SSLEngine} into the pipeline,
114  * and start the handshake process as described in the first section.
115  *
116  * <h3>Implementing StartTLS</h3>
117  * <p>
118  * <a href="https://en.wikipedia.org/wiki/STARTTLS">StartTLS</a> is the
119  * communication pattern that secures the wire in the middle of the plaintext
120  * connection.  Please note that it is different from SSL &middot; TLS, that
121  * secures the wire from the beginning of the connection.  Typically, StartTLS
122  * is composed of three steps:
123  * <ol>
124  * <li>Client sends a StartTLS request to server.</li>
125  * <li>Server sends a StartTLS response to client.</li>
126  * <li>Client begins SSL handshake.</li>
127  * </ol>
128  * If you implement a server, you need to:
129  * <ol>
130  * <li>create a new {@link SslHandler} instance with {@code startTls} flag set
131  *     to {@code true},</li>
132  * <li>insert the {@link SslHandler} to the {@link ChannelPipeline}, and</li>
133  * <li>write a StartTLS response.</li>
134  * </ol>
135  * Please note that you must insert {@link SslHandler} <em>before</em> sending
136  * the StartTLS response.  Otherwise the client can send begin SSL handshake
137  * before {@link SslHandler} is inserted to the {@link ChannelPipeline}, causing
138  * data corruption.
139  * <p>
140  * The client-side implementation is much simpler.
141  * <ol>
142  * <li>Write a StartTLS request,</li>
143  * <li>wait for the StartTLS response,</li>
144  * <li>create a new {@link SslHandler} instance with {@code startTls} flag set
145  *     to {@code false},</li>
146  * <li>insert the {@link SslHandler} to the {@link ChannelPipeline}, and</li>
147  * <li>Initiate SSL handshake.</li>
148  * </ol>
149  *
150  * <h3>Known issues</h3>
151  * <p>
152  * Because of a known issue with the current implementation of the SslEngine that comes
153  * with Java it may be possible that you see blocked IO-Threads while a full GC is done.
154  * <p>
155  * So if you are affected you can workaround this problem by adjust the cache settings
156  * like shown below:
157  *
158  * <pre>
159  *     SslContext context = ...;
160  *     context.getServerSessionContext().setSessionCacheSize(someSaneSize);
161  *     context.getServerSessionContext().setSessionTime(someSameTimeout);
162  * </pre>
163  * <p>
164  * What values to use here depends on the nature of your application and should be set
165  * based on monitoring and debugging of it.
166  * For more details see
167  * <a href="https://github.com/netty/netty/issues/832">#832</a> in our issue tracker.
168  */

185     private enum SslEngineType {
186         TCNATIVE(true, COMPOSITE_CUMULATOR) {
187             @Override
188             SSLEngineResult unwrap(SslHandler handler, ByteBuf in, int readerIndex, int len, ByteBuf out)
189                     throws SSLException {
190                 int nioBufferCount = in.nioBufferCount();
191                 int writerIndex = out.writerIndex();
192                 final SSLEngineResult result;
193                 if (nioBufferCount > 1) {
194                     /*
195                      * If {@link OpenSslEngine} is in use,
196                      * we can use a special {@link OpenSslEngine#unwrap(ByteBuffer[], ByteBuffer[])} method
197                      * that accepts multiple {@link ByteBuffer}s without additional memory copies.
198                      */
199                     ReferenceCountedOpenSslEngine opensslEngine = (ReferenceCountedOpenSslEngine) handler.engine;
200                     try {
201                         handler.singleBuffer[0] = toByteBuffer(out, writerIndex,
202                             out.writableBytes());
203                         result = opensslEngine.unwrap(in.nioBuffers(readerIndex, len), handler.singleBuffer);
204                     } finally {
205                         handler.singleBuffer[0] = null;
206                     }
207                 } else {
208                     result = handler.engine.unwrap(toByteBuffer(in, readerIndex, len),
209                         toByteBuffer(out, writerIndex, out.writableBytes()));
210                 }
211                 out.writerIndex(writerIndex + result.bytesProduced());
212                 return result;
213             }
214 
215             @Override
216             ByteBuf allocateWrapBuffer(SslHandler handler, ByteBufAllocator allocator,
217                                        int pendingBytes, int numComponents) {
218                 return allocator.directBuffer(((ReferenceCountedOpenSslEngine) handler.engine)
219                         .calculateMaxLengthForWrap(pendingBytes, numComponents));
220             }
221 
222             @Override
223             int calculatePendingData(SslHandler handler, int guess) {
224                 int sslPending = ((ReferenceCountedOpenSslEngine) handler.engine).sslPending();
225                 return sslPending > 0 ? sslPending : guess;
226             }
227 
228             @Override
229             boolean jdkCompatibilityMode(SSLEngine engine) {
230                 return ((ReferenceCountedOpenSslEngine) engine).jdkCompatibilityMode;
231             }
232         },
233         CONSCRYPT(true, COMPOSITE_CUMULATOR) {
234             @Override
235             SSLEngineResult unwrap(SslHandler handler, ByteBuf in, int readerIndex, int len, ByteBuf out)
236                     throws SSLException {
237                 int nioBufferCount = in.nioBufferCount();
238                 int writerIndex = out.writerIndex();
239                 final SSLEngineResult result;
240                 if (nioBufferCount > 1) {
241                     /*
242                      * Use a special unwrap method without additional memory copies.
243                      */
244                     try {
245                         handler.singleBuffer[0] = toByteBuffer(out, writerIndex, out.writableBytes());
246                         result = ((ConscryptAlpnSslEngine) handler.engine).unwrap(
247                                 in.nioBuffers(readerIndex, len),
248                                 handler.singleBuffer);
249                     } finally {
250                         handler.singleBuffer[0] = null;
251                     }
252                 } else {
253                     result = handler.engine.unwrap(toByteBuffer(in, readerIndex, len),
254                             toByteBuffer(out, writerIndex, out.writableBytes()));
255                 }
256                 out.writerIndex(writerIndex + result.bytesProduced());
257                 return result;
258             }
259 
260             @Override
261             ByteBuf allocateWrapBuffer(SslHandler handler, ByteBufAllocator allocator,
262                                        int pendingBytes, int numComponents) {
263                 return allocator.directBuffer(
264                         ((ConscryptAlpnSslEngine) handler.engine).calculateOutNetBufSize(pendingBytes, numComponents));
265             }
266 
267             @Override
268             int calculatePendingData(SslHandler handler, int guess) {
269                 return guess;
270             }
271 
272             @Override
273             boolean jdkCompatibilityMode(SSLEngine engine) {
274                 return true;
275             }
276         },
277         JDK(false, MERGE_CUMULATOR) {
278             @Override
279             SSLEngineResult unwrap(SslHandler handler, ByteBuf in, int readerIndex, int len, ByteBuf out)
280                     throws SSLException {
281                 int writerIndex = out.writerIndex();
282                 ByteBuffer inNioBuffer = toByteBuffer(in, readerIndex, len);
283                 int position = inNioBuffer.position();
284                 final SSLEngineResult result = handler.engine.unwrap(inNioBuffer,
285                     toByteBuffer(out, writerIndex, out.writableBytes()));
286                 out.writerIndex(writerIndex + result.bytesProduced());
287 
288                 // This is a workaround for a bug in Android 5.0. Android 5.0 does not correctly update the
289                 // SSLEngineResult.bytesConsumed() in some cases and just return 0.
290                 //
291                 // See:
292                 //     - https://android-review.googlesource.com/c/platform/external/conscrypt/+/122080
293                 //     - https://github.com/netty/netty/issues/7758
294                 if (result.bytesConsumed() == 0) {
295                     int consumed = inNioBuffer.position() - position;
296                     if (consumed != result.bytesConsumed()) {
297                         // Create a new SSLEngineResult with the correct bytesConsumed().
298                         return new SSLEngineResult(
299                                 result.getStatus(), result.getHandshakeStatus(), consumed, result.bytesProduced());
300                     }
301                 }
302                 return result;
303             }
304 
305             @Override
306             ByteBuf allocateWrapBuffer(SslHandler handler, ByteBufAllocator allocator,
307                                        int pendingBytes, int numComponents) {
308                 // As for the JDK SSLEngine we always need to allocate buffers of the size required by the SSLEngine
309                 // (normally ~16KB). This is required even if the amount of data to encrypt is very small. Use heap
310                 // buffers to reduce the native memory usage.
311                 //
312                 // Beside this the JDK SSLEngine also (as of today) will do an extra heap to direct buffer copy
313                 // if a direct buffer is used as its internals operate on byte[].
314                 return allocator.heapBuffer(handler.engine.getSession().getPacketBufferSize());
315             }
316 
317             @Override
318             int calculatePendingData(SslHandler handler, int guess) {
319                 return guess;
320             }
321 
322             @Override
323             boolean jdkCompatibilityMode(SSLEngine engine) {
324                 return true;
325             }
326         };

481     /**
482      * Sets the number of bytes to pass to each {@link SSLEngine#wrap(ByteBuffer[], int, int, ByteBuffer)} call.
483      * <p>
484      * This value will partition data which is passed to write
485      * {@link #write(ChannelHandlerContext, Object, ChannelPromise)}. The partitioning will work as follows:
486      * <ul>
487      * <li>If {@code wrapDataSize <= 0} then we will write each data chunk as is.</li>
488      * <li>If {@code wrapDataSize > data size} then we will attempt to aggregate multiple data chunks together.</li>
489      * <li>If {@code wrapDataSize > data size}  Else if {@code wrapDataSize <= data size} then we will divide the data
490      * into chunks of {@code wrapDataSize} when writing.</li>
491      * </ul>
492      * <p>
493      * If the {@link SSLEngine} doesn't support a gather wrap operation (e.g. {@link SslProvider#OPENSSL}) then
494      * aggregating data before wrapping can help reduce the ratio between TLS overhead vs data payload which will lead
495      * to better goodput. Writing fixed chunks of data can also help target the underlying transport's (e.g. TCP)
496      * frame size. Under lossy/congested network conditions this may help the peer get full TLS packets earlier and
497      * be able to do work sooner, as opposed to waiting for the all the pieces of the TLS packet to arrive.
498      * @param wrapDataSize the number of bytes which will be passed to each
499      *      {@link SSLEngine#wrap(ByteBuffer[], int, int, ByteBuffer)} call.
500      */

503         this.wrapDataSize = wrapDataSize;

778         // created with startTLS flag turned on.
779         if (startTls && !sentFirstMessage) {

790             return;

801     private void wrapAndFlush(ChannelHandlerContext ctx) throws SSLException {
802         if (pendingUnencryptedWrites.isEmpty()) {
803             // It's important to NOT use a voidPromise here as the user
804             // may want to add a ChannelFutureListener to the ChannelPromise later.
805             //
806             // See https://github.com/netty/netty/issues/3364
807             pendingUnencryptedWrites.add(Unpooled.EMPTY_BUFFER, ctx.newPromise());
808         }
809         if (!handshakePromise.isDone()) {
810             flushedBeforeHandshake = true;
811         }
812         try {
813             wrap(ctx, false);
814         } finally {
815             // We may have written some parts of data before an exception was thrown so ensure we always flush.
816             // See https://github.com/netty/netty/issues/3900#issuecomment-172481830
817             forceFlush(ctx);
818         }

821     // This method will not call setHandshakeFailure(...) !
822     private void wrap(ChannelHandlerContext ctx, boolean inUnwrap) throws SSLException {
823         ByteBuf out = null;
824         ByteBufAllocator alloc = ctx.alloc();
825         try {
826             final int wrapDataSize = this.wrapDataSize;
827             // Only continue to loop if the handler was not removed in the meantime.
828             // See https://github.com/netty/netty/issues/5860
829             outer: while (!ctx.isRemoved()) {
830                 ChannelPromise promise = ctx.newPromise();
831                 ByteBuf buf = wrapDataSize > 0 ?
832                         pendingUnencryptedWrites.remove(alloc, wrapDataSize, promise) :
833                         pendingUnencryptedWrites.removeFirst(promise);
834                 if (buf == null) {
835                     break;
836                 }
837 
838                 if (out == null) {
839                     out = allocateOutNetBuf(ctx, buf.readableBytes(), buf.nioBufferCount());
840                 }
841 
842                 SSLEngineResult result = wrap(alloc, engine, buf, out);
843 
844                 if (buf.isReadable()) {
845                     pendingUnencryptedWrites.addFirst(buf, promise);
846                     // When we add the buffer/promise pair back we need to be sure we don't complete the promise
847                     // later. We only complete the promise if the buffer is completely consumed.
848                     promise = null;
849                 } else {
850                     buf.release();
851                 }
852 
853                 // We need to write any data before we invoke any methods which may trigger re-entry, otherwise
854                 // writes may occur out of order and TLS sequencing may be off (e.g. SSLV3_ALERT_BAD_RECORD_MAC).
855                 if (out.isReadable()) {
856                     final ByteBuf b = out;
857                     out = null;
858                     if (promise != null) {
859                         ctx.write(b, promise);
860                     } else {
861                         ctx.write(b);
862                     }
863                 } else if (promise != null) {
864                     ctx.write(Unpooled.EMPTY_BUFFER, promise);
865                 }
866                 // else out is not readable we can re-use it and so save an extra allocation
867 
868                 if (result.getStatus() == Status.CLOSED) {
869                     // Make a best effort to preserve any exception that way previously encountered from the handshake
870                     // or the transport, else fallback to a general error.
871                     Throwable exception = handshakePromise.cause();
872                     if (exception == null) {
873                         exception = sslClosePromise.cause();
874                         if (exception == null) {
875                             exception = new SslClosedEngineException("SSLEngine closed already");
876                         }
877                     }
878                     pendingUnencryptedWrites.releaseAndFailAll(ctx, exception);
879                     return;
880                 } else {
881                     switch (result.getHandshakeStatus()) {
882                         case NEED_TASK:
883                             if (!runDelegatedTasks(inUnwrap)) {
884                                 // We scheduled a task on the delegatingTaskExecutor, so stop processing as we will
885                                 // resume once the task completes.
886                                 break outer;
887                             }
888                             break;
889                         case FINISHED:
890                             setHandshakeSuccess();
891                             break;
892                         case NOT_HANDSHAKING:
893                             setHandshakeSuccessIfStillHandshaking();
894                             break;
895                         case NEED_WRAP:
896                             // If we are expected to wrap again and we produced some data we need to ensure there
897                             // is something in the queue to process as otherwise we will not try again before there
898                             // was more added. Failing to do so may fail to produce an alert that can be
899                             // consumed by the remote peer.
900                             if (result.bytesProduced() > 0 && pendingUnencryptedWrites.isEmpty()) {
901                                 pendingUnencryptedWrites.add(Unpooled.EMPTY_BUFFER);
902                             }
903                             break;
904                         case NEED_UNWRAP:
905                             // The underlying engine is starving so we need to feed it with more data.
906                             // See https://github.com/netty/netty/pull/5039
907                             readIfNeeded(ctx);
908                             return;
909                         default:
910                             throw new IllegalStateException(
911                                     "Unknown handshake status: " + result.getHandshakeStatus());
912                     }
913                 }
914             }
915         } finally {
916             if (out != null) {
917                 out.release();
918             }
919             if (inUnwrap) {
920                 needsFlush = true;
921             }
922         }
923     }

925     /**
926      * This method will not call
927      * {@link #setHandshakeFailure(ChannelHandlerContext, Throwable, boolean, boolean, boolean)} or
928      * {@link #setHandshakeFailure(ChannelHandlerContext, Throwable)}.
929      * @return {@code true} if this method ends on {@link SSLEngineResult.HandshakeStatus#NOT_HANDSHAKING}.
930      */
931     private boolean wrapNonAppData(final ChannelHandlerContext ctx, boolean inUnwrap) throws SSLException {
932         ByteBuf out = null;
933         ByteBufAllocator alloc = ctx.alloc();
934         try {
935             // Only continue to loop if the handler was not removed in the meantime.
936             // See https://github.com/netty/netty/issues/5860
937             outer: while (!ctx.isRemoved()) {
938                 if (out == null) {
939                     // As this is called for the handshake we have no real idea how big the buffer needs to be.
940                     // That said 2048 should give us enough room to include everything like ALPN / NPN data.
941                     // If this is not enough we will increase the buffer in wrap(...).
942                     out = allocateOutNetBuf(ctx, 2048, 1);
943                 }
944                 SSLEngineResult result = wrap(alloc, engine, Unpooled.EMPTY_BUFFER, out);
945                 if (result.bytesProduced() > 0) {
946                     ctx.write(out).addListener(new ChannelFutureListener() {
947                         @Override
948                         public void operationComplete(ChannelFuture future) {
949                             Throwable cause = future.cause();
950                             if (cause != null) {
951                                 setHandshakeFailureTransportFailure(ctx, cause);
952                             }
953                         }
954                     });
955                     if (inUnwrap) {
956                         needsFlush = true;
957                     }
958                     out = null;
959                 }
960 
961                 HandshakeStatus status = result.getHandshakeStatus();
962                 switch (status) {
963                     case FINISHED:
964                         setHandshakeSuccess();
965                         // We may be here because we read data and discovered the remote peer initiated a renegotiation
966                         // and this write is to complete the new handshake. The user may have previously done a
967                         // writeAndFlush which wasn't able to wrap data due to needing the pending handshake, so we
968                         // attempt to wrap application data here if any is pending.
969                         if (inUnwrap && !pendingUnencryptedWrites.isEmpty()) {
970                             wrap(ctx, true);
971                         }
972                         return false;
973                     case NEED_TASK:
974                         if (!runDelegatedTasks(inUnwrap)) {
975                             // We scheduled a task on the delegatingTaskExecutor, so stop processing as we will
976                             // resume once the task completes.
977                             break outer;
978                         }
979                         break;
980                     case NEED_UNWRAP:
981                         if (inUnwrap) {
982                             // If we asked for a wrap, the engine requested an unwrap, and we are in unwrap there is
983                             // no use in trying to call wrap again because we have already attempted (or will after we
984                             // return) to feed more data to the engine.
985                             return false;
986                         }
987 
988                         unwrapNonAppData(ctx);
989                         break;
990                     case NEED_WRAP:
991                         break;
992                     case NOT_HANDSHAKING:
993                         setHandshakeSuccessIfStillHandshaking();
994                         // Workaround for TLS False Start problem reported at:
995                         // https://github.com/netty/netty/issues/1108#issuecomment-14266970
996                         if (!inUnwrap) {
997                             unwrapNonAppData(ctx);
998                         }
999                         return true;
1000                     default:
1001                         throw new IllegalStateException("Unknown handshake status: " + result.getHandshakeStatus());
1002                 }
1003 
1004                 // Check if did not produce any bytes and if so break out of the loop, but only if we did not process
1005                 // a task as last action. It's fine to not produce any data as part of executing a task.
1006                 if (result.bytesProduced() == 0 && status != HandshakeStatus.NEED_TASK) {
1007                     break;
1008                 }
1009 
1010                 // It should not consume empty buffers when it is not handshaking
1011                 // Fix for Android, where it was encrypting empty buffers even when not handshaking
1012                 if (result.bytesConsumed() == 0 && result.getHandshakeStatus() == HandshakeStatus.NOT_HANDSHAKING) {
1013                     break;
1014                 }
1015             }
1016         }  finally {
1017             if (out != null) {
1018                 out.release();
1019             }
1020         }
1021         return false;
1022     }

1024     private SSLEngineResult wrap(ByteBufAllocator alloc, SSLEngine engine, ByteBuf in, ByteBuf out)
1025             throws SSLException {
1026         ByteBuf newDirectIn = null;
1027         try {
1028             int readerIndex = in.readerIndex();
1029             int readableBytes = in.readableBytes();
1030 
1031             // We will call SslEngine.wrap(ByteBuffer[], ByteBuffer) to allow efficient handling of
1032             // CompositeByteBuf without force an extra memory copy when CompositeByteBuffer.nioBuffer() is called.
1033             final ByteBuffer[] in0;
1034             if (in.isDirect() || !engineType.wantsDirectBuffer) {
1035                 // As CompositeByteBuf.nioBufferCount() can be expensive (as it needs to check all composed ByteBuf
1036                 // to calculate the count) we will just assume a CompositeByteBuf contains more then 1 ByteBuf.
1037                 // The worst that can happen is that we allocate an extra ByteBuffer[] in CompositeByteBuf.nioBuffers()
1038                 // which is better then walking the composed ByteBuf in most cases.
1039                 if (!(in instanceof CompositeByteBuf) && in.nioBufferCount() == 1) {
1040                     in0 = singleBuffer;
1041                     // We know its only backed by 1 ByteBuffer so use internalNioBuffer to keep object allocation
1042                     // to a minimum.
1043                     in0[0] = in.internalNioBuffer(readerIndex, readableBytes);
1044                 } else {
1045                     in0 = in.nioBuffers();
1046                 }
1047             } else {
1048                 // We could even go further here and check if its a CompositeByteBuf and if so try to decompose it and
1049                 // only replace the ByteBuffer that are not direct. At the moment we just will replace the whole
1050                 // CompositeByteBuf to keep the complexity to a minimum
1051                 newDirectIn = alloc.directBuffer(readableBytes);
1052                 newDirectIn.writeBytes(in, readerIndex, readableBytes);
1053                 in0 = singleBuffer;
1054                 in0[0] = newDirectIn.internalNioBuffer(newDirectIn.readerIndex(), readableBytes);
1055             }
1056 
1057             for (;;) {
1058                 ByteBuffer out0 = out.nioBuffer(out.writerIndex(), out.writableBytes());
1059                 SSLEngineResult result = engine.wrap(in0, out0);
1060                 in.skipBytes(result.bytesConsumed());
1061                 out.writerIndex(out.writerIndex() + result.bytesProduced());
1062 
1063                 if (result.getStatus() == Status.BUFFER_OVERFLOW) {
1064                     out.ensureWritable(engine.getSession().getPacketBufferSize());
1065                 } else {
1066                     return result;
1067                 }
1068             }
1069         } finally {
1070             // Null out to allow GC of ByteBuffer
1071             singleBuffer[0] = null;
1072 
1073             if (newDirectIn != null) {
1074                 newDirectIn.release();
1075             }
1076         }
1077     }

1126     /**
1127      * Checks if the given {@link Throwable} can be ignore and just "swallowed"
1128      *
1129      * When an ssl connection is closed a close_notify message is sent.
1130      * After that the peer also sends close_notify however, it's not mandatory to receive
1131      * the close_notify. The party who sent the initial close_notify can close the connection immediately
1132      * then the peer will get connection reset error.
1133      *
1134      */
1135     private boolean ignoreException(Throwable t) {
1136         if (!(t instanceof SSLException) && t instanceof IOException && sslClosePromise.isDone()) {
1137             String message = t.getMessage();
1138 
1139             // first try to match connection reset / broke peer based on the regex. This is the fastest way
1140             // but may fail on different jdk impls or OS's
1141             if (message != null && IGNORABLE_ERROR_MESSAGE.matcher(message).matches()) {
1142                 return true;
1143             }
1144 
1145             // Inspect the StackTraceElements to see if it was a connection reset / broken pipe or not
1146             StackTraceElement[] elements = t.getStackTrace();
1147             for (StackTraceElement element: elements) {
1148                 String classname = element.getClassName();
1149                 String methodname = element.getMethodName();
1150 
1151                 // skip all classes that belong to the io.netty package
1152                 if (classname.startsWith("io.netty.")) {
1153                     continue;
1154                 }
1155 
1156                 // check if the method name is read if not skip it
1157                 if (!"read".equals(methodname)) {
1158                     continue;
1159                 }
1160 
1161                 // This will also match against SocketInputStream which is used by openjdk 7 and maybe
1162                 // also others
1163                 if (IGNORABLE_CLASS_IN_STACK.matcher(classname).matches()) {
1164                     return true;
1165                 }
1166 
1167                 try {
1168                     // No match by now.. Try to load the class via classloader and inspect it.
1169                     // This is mainly done as other JDK implementations may differ in name of
1170                     // the impl.
1171                     Class<?> clazz = PlatformDependent.getClassLoader(getClass()).loadClass(classname);
1172 
1173                     if (SocketChannel.class.isAssignableFrom(clazz)
1174                             || DatagramChannel.class.isAssignableFrom(clazz)) {
1175                         return true;
1176                     }
1177 
1178                     // also match against SctpChannel via String matching as it may not present.
1179                     if (PlatformDependent.javaVersion() >= 7
1180                             && "com.sun.nio.sctp.SctpChannel".equals(clazz.getSuperclass().getName())) {
1181                         return true;
1182                     }
1183                 } catch (Throwable cause) {
1184                     if (logger.isDebugEnabled()) {
1185                         logger.debug("Unexpected exception while loading class {} classname {}",
1186                                 getClass(), classname, cause);
1187                     }
1188                 }
1189             }
1190         }
1191 
1192         return false;
1193     }

1233                 in.skipBytes(in.readableBytes());
1234 
1235                 // First fail the handshake promise as we may need to have access to the SSLEngine which may
1236                 // be released because the user will remove the SslHandler in an exceptionCaught(...) implementation.
1237                 setHandshakeFailure(ctx, e);
1238 
1239                 throw e;
1240             }
1241             assert packetLength > 0;
1242             if (packetLength > readableBytes) {
1243                 // wait until the whole packet can be read
1244                 this.packetLength = packetLength;
1245                 return;
1246             }
1247         }
1248 
1249         // Reset the state of this class so we can get the length of the next packet. We assume the entire packet will
1250         // be consumed by the SSLEngine.
1251         this.packetLength = 0;
1252         try {
1253             int bytesConsumed = unwrap(ctx, in, in.readerIndex(), packetLength);
1254             assert bytesConsumed == packetLength || engine.isInboundDone() :
1255                     "we feed the SSLEngine a packets worth of data: " + packetLength + " but it only consumed: " +
1256                             bytesConsumed;
1257             in.skipBytes(bytesConsumed);
1258         } catch (Throwable cause) {
1259             handleUnwrapThrowable(ctx, cause);
1260         }
1261     }

1271     private void handleUnwrapThrowable(ChannelHandlerContext ctx, Throwable cause) {
1272         try {
1273             // We should attempt to notify the handshake failure before writing any pending data. If we are in unwrap
1274             // and failed during the handshake process, and we attempt to wrap, then promises will fail, and if
1275             // listeners immediately close the Channel then we may end up firing the handshake event after the Channel
1276             // has been closed.
1277             if (handshakePromise.tryFailure(cause)) {
1278                 ctx.fireUserEventTriggered(new SslHandshakeCompletionEvent(cause));
1279             }
1280 
1281             // We need to flush one time as there may be an alert that we should send to the remote peer because
1282             // of the SSLException reported here.
1283             wrapAndFlush(ctx);
1284         } catch (SSLException ex) {
1285             logger.debug("SSLException during trying to call SSLEngine.wrap(...)" +
1286                     " because of an previous SSLException, ignoring...", ex);
1287         } finally {
1288             // ensure we always flush and close the channel.
1289             setHandshakeFailure(ctx, cause, true, false, true);
1290         }
1291         PlatformDependent.throwException(cause);
1292     }

1344     /**
1345      * Unwraps inbound SSL records.
1346      */
1347     private int unwrap(
1348             ChannelHandlerContext ctx, ByteBuf packet, int offset, int length) throws SSLException {
1349         final int originalLength = length;
1350         boolean wrapLater = false;
1351         boolean notifyClosure = false;
1352         int overflowReadableBytes = -1;
1353         ByteBuf decodeOut = allocate(ctx, length);
1354         try {
1355             // Only continue to loop if the handler was not removed in the meantime.
1356             // See https://github.com/netty/netty/issues/5860
1357             unwrapLoop: while (!ctx.isRemoved()) {
1358                 final SSLEngineResult result = engineType.unwrap(this, packet, offset, length, decodeOut);
1359                 final Status status = result.getStatus();
1360                 final HandshakeStatus handshakeStatus = result.getHandshakeStatus();
1361                 final int produced = result.bytesProduced();
1362                 final int consumed = result.bytesConsumed();
1363 
1364                 // Update indexes for the next iteration
1365                 offset += consumed;
1366                 length -= consumed;
1367 
1368                 switch (status) {
1369                 case BUFFER_OVERFLOW:
1370                     final int readableBytes = decodeOut.readableBytes();
1371                     final int previousOverflowReadableBytes = overflowReadableBytes;
1372                     overflowReadableBytes = readableBytes;
1373                     int bufferSize = engine.getSession().getApplicationBufferSize() - readableBytes;
1374                     if (readableBytes > 0) {
1375                         firedChannelRead = true;
1376                         ctx.fireChannelRead(decodeOut);
1377 
1378                         // This buffer was handled, null it out.
1379                         decodeOut = null;
1380                         if (bufferSize <= 0) {
1381                             // It may happen that readableBytes >= engine.getSession().getApplicationBufferSize()
1382                             // while there is still more to unwrap, in this case we will just allocate a new buffer
1383                             // with the capacity of engine.getSession().getApplicationBufferSize() and call unwrap
1384                             // again.
1385                             bufferSize = engine.getSession().getApplicationBufferSize();
1386                         }
1387                     } else {
1388                         // This buffer was handled, null it out.
1389                         decodeOut.release();
1390                         decodeOut = null;
1391                     }
1392                     if (readableBytes == 0 && previousOverflowReadableBytes == 0) {
1393                         // If there is two consecutive loops where we overflow and are not able to consume any data,
1394                         // assume the amount of data exceeds the maximum amount for the engine and bail
1395                         throw new IllegalStateException("Two consecutive overflows but no content was consumed. " +
1396                                  SSLSession.class.getSimpleName() + " getApplicationBufferSize: " +
1397                                  engine.getSession().getApplicationBufferSize() + " maybe too small.");
1398                     }
1399                     // Allocate a new buffer which can hold all the rest data and loop again.
1400                     // TODO: We may want to reconsider how we calculate the length here as we may
1401                     // have more then one ssl message to decode.
1402                     decodeOut = allocate(ctx, engineType.calculatePendingData(this, bufferSize));
1403                     continue;
1404                 case CLOSED:
1405                     // notify about the CLOSED state of the SSLEngine. See #137
1406                     notifyClosure = true;
1407                     overflowReadableBytes = -1;
1408                     break;
1409                 default:
1410                     overflowReadableBytes = -1;
1411                     break;
1412                 }
1413 
1414                 switch (handshakeStatus) {
1415                     case NEED_UNWRAP:
1416                         break;
1417                     case NEED_WRAP:
1418                         // If the wrap operation transitions the status to NOT_HANDSHAKING and there is no more data to
1419                         // unwrap then the next call to unwrap will not produce any data. We can avoid the potentially
1420                         // costly unwrap operation and break out of the loop.
1421                         if (wrapNonAppData(ctx, true) && length == 0) {
1422                             break unwrapLoop;
1423                         }
1424                         break;
1425                     case NEED_TASK:
1426                         if (!runDelegatedTasks(true)) {
1427                             // We scheduled a task on the delegatingTaskExecutor, so stop processing as we will
1428                             // resume once the task completes.
1429                             //
1430                             // We break out of the loop only and do NOT return here as we still may need to notify
1431                             // about the closure of the SSLEngine.
1432                             //
1433                             wrapLater = false;
1434                             break unwrapLoop;
1435                         }
1436                         break;
1437                     case FINISHED:
1438                         setHandshakeSuccess();
1439                         wrapLater = true;
1440 
1441                         // We 'break' here and NOT 'continue' as android API version 21 has a bug where they consume
1442                         // data from the buffer but NOT correctly set the SSLEngineResult.bytesConsumed().
1443                         // Because of this it will raise an exception on the next iteration of the for loop on android
1444                         // API version 21. Just doing a break will work here as produced and consumed will both be 0
1445                         // and so we break out of the complete for (;;) loop and so call decode(...) again later on.
1446                         // On other platforms this will have no negative effect as we will just continue with the
1447                         // for (;;) loop if something was either consumed or produced.
1448                         //
1449                         // See:
1450                         //  - https://github.com/netty/netty/issues/4116
1451                         //  - https://code.google.com/p/android/issues/detail?id=198639&thanks=198639&ts=1452501203
1452                         break;
1453                     case NOT_HANDSHAKING:
1454                         if (setHandshakeSuccessIfStillHandshaking()) {
1455                             wrapLater = true;
1456                             continue;
1457                         }
1458 
1459                         // If we are not handshaking and there is no more data to unwrap then the next call to unwrap
1460                         // will not produce any data. We can avoid the potentially costly unwrap operation and break
1461                         // out of the loop.
1462                         if (length == 0) {
1463                             break unwrapLoop;
1464                         }
1465                         break;
1466                     default:
1467                         throw new IllegalStateException("unknown handshake status: " + handshakeStatus);
1468                 }
1469 
1470                 if (status == Status.BUFFER_UNDERFLOW ||
1471                         // If we processed NEED_TASK we should try again even we did not consume or produce anything.
1472                         handshakeStatus != HandshakeStatus.NEED_TASK && consumed == 0 && produced == 0) {
1473                     if (handshakeStatus == HandshakeStatus.NEED_UNWRAP) {
1474                         // The underlying engine is starving so we need to feed it with more data.
1475                         // See https://github.com/netty/netty/pull/5039
1476                         readIfNeeded(ctx);
1477                     }
1478 
1479                     break;
1480                 }
1481             }
1482 
1483             if (flushedBeforeHandshake && handshakePromise.isDone()) {
1484                 // We need to call wrap(...) in case there was a flush done before the handshake completed to ensure
1485                 // we do not stale.
1486                 //
1487                 // See https://github.com/netty/netty/pull/2437
1488                 flushedBeforeHandshake = false;
1489                 wrapLater = true;
1490             }
1491 
1492             if (wrapLater) {
1493                 wrap(ctx, true);
1494             }
1495 
1496             if (notifyClosure) {
1497                 notifyClosePromise(null);
1498             }
1499         } finally {
1500             if (decodeOut != null) {
1501                 if (decodeOut.isReadable()) {
1502                     firedChannelRead = true;
1503 
1504                     ctx.fireChannelRead(decodeOut);
1505                 } else {
1506                     decodeOut.release();
1507                 }
1508             }
1509         }
1510         return originalLength - length;
1511     }

1561      * {@link Runnable} that will be scheduled on the {@code delegatedTaskExecutor} and will take care
1562      * of resume work on the {@link EventExecutor} once the task was executed.

1571         // Handle errors which happened during task processing.
1572         private void taskError(Throwable e) {
1573             if (inUnwrap) {
1574                 // As the error happened while the task was scheduled as part of unwrap(...) we also need to ensure
1575                 // we fire it through the pipeline as inbound error to be consistent with what we do in decode(...).
1576                 //
1577                 // This will also ensure we fail the handshake future and flush all produced data.
1578                 try {
1579                     handleUnwrapThrowable(ctx, e);
1580                 } catch (Throwable cause) {
1581                     safeExceptionCaught(cause);
1582                 }
1583             } else {
1584                 setHandshakeFailure(ctx, e);
1585                 forceFlush(ctx);
1586             }
1587         }

1589         // Try to call exceptionCaught(...)
1590         private void safeExceptionCaught(Throwable cause) {
1591             try {
1592                 exceptionCaught(ctx, wrapIfNeeded(cause));
1593             } catch (Throwable error) {
1594                 ctx.fireExceptionCaught(error);
1595             }
1596         }

1598         private Throwable wrapIfNeeded(Throwable cause) {
1599             if (!inUnwrap) {
1600                 // If we are not in unwrap(...) we can just rethrow without wrapping at all.
1601                 return cause;
1602             }
1603             // As the exception would have been triggered by an inbound operation we will need to wrap it in a
1604             // DecoderException to mimic what a decoder would do when decode(...) throws.
1605             return cause instanceof DecoderException ? cause : new DecoderException(cause);
1606         }

1608         private void tryDecodeAgain() {
1609             try {
1610                 channelRead(ctx, Unpooled.EMPTY_BUFFER);
1611             } catch (Throwable cause) {
1612                 safeExceptionCaught(cause);
1613             } finally {
1614                 // As we called channelRead(...) we also need to call channelReadComplete(...) which
1615                 // will ensure we either call ctx.fireChannelReadComplete() or will trigger a ctx.read() if
1616                 // more data is needed.
1617                 channelReadComplete0(ctx);
1618             }
1619         }

1621         /**
1622          * Executed after the wrapped {@code task} was executed via {@code delegatedTaskExecutor} to resume work
1623          * on the {@link EventExecutor}.
1624          */
1625         private void resumeOnEventExecutor() {
1626             assert ctx.executor().inEventLoop();
1627 
1628             processTask = false;
1629 
1630             try {
1631                 HandshakeStatus status = engine.getHandshakeStatus();
1632                 switch (status) {
1633                     // There is another task that needs to be executed and offloaded to the delegatingTaskExecutor.
1634                     case NEED_TASK:
1635                         executeDelegatedTasks(inUnwrap);
1636 
1637                         break;
1638 
1639                     // The handshake finished, lets notify about the completion of it and resume processing.
1640                     case FINISHED:
1641                         setHandshakeSuccess();
1642 
1643                         // deliberate fall-through
1644 
1645                     // Not handshaking anymore, lets notify about the completion if not done yet and resume processing.
1646                     case NOT_HANDSHAKING:
1647                         setHandshakeSuccessIfStillHandshaking();
1648                         try {
1649                             // Lets call wrap to ensure we produce the alert if there is any pending and also to
1650                             // ensure we flush any queued data..
1651                             wrap(ctx, inUnwrap);
1652                         } catch (Throwable e) {
1653                             taskError(e);
1654                             return;
1655                         }
1656                         if (inUnwrap) {
1657                             // If we were in the unwrap call when the task was processed we should also try to unwrap
1658                             // non app data first as there may not anything left in the inbound buffer to process.
1659                             unwrapNonAppData(ctx);
1660                         }
1661 
1662                         // Flush now as we may have written some data as part of the wrap call.
1663                         forceFlush(ctx);
1664 
1665                         tryDecodeAgain();
1666                         break;
1667 
1668                     // We need more data so lets try to unwrap first and then call decode again which will feed us
1669                     // with buffered data (if there is any).
1670                     case NEED_UNWRAP:
1671                         try {
1672                             unwrapNonAppData(ctx);
1673                         } catch (SSLException e) {
1674                             handleUnwrapThrowable(ctx, e);
1675                             return;
1676                         }
1677                         tryDecodeAgain();
1678                         break;
1679 
1680                     // To make progress we need to call SSLEngine.wrap(...) which may produce more output data
1681                     // that will be written to the Channel.
1682                     case NEED_WRAP:
1683                         try {
1684                             if (!wrapNonAppData(ctx, false) && inUnwrap) {
1685                                 // The handshake finished in wrapNonAppData(...), we need to try call
1686                                 // unwrapNonAppData(...) as we may have some alert that we should read.
1687                                 //
1688                                 // This mimics what we would do when we are calling this method while in unwrap(...).
1689                                 unwrapNonAppData(ctx);
1690                             }
1691 
1692                             // Flush now as we may have written some data as part of the wrap call.
1693                             forceFlush(ctx);
1694                         } catch (Throwable e) {
1695                             taskError(e);
1696                             return;
1697                         }
1698 
1699                         // Now try to feed in more data that we have buffered.
1700                         tryDecodeAgain();
1701                         break;
1702 
1703                     default:
1704                         // Should never reach here as we handle all cases.
1705                         throw new AssertionError();
1706                 }
1707             } catch (Throwable cause) {
1708                 safeExceptionCaught(cause);
1709             }
1710         }

1714             try {
1715                 runAllDelegatedTasks(engine);
1716 
1717                 // All tasks were processed.
1718                 assert engine.getHandshakeStatus() != HandshakeStatus.NEED_TASK;

1768      * @return {@code true} if {@link #handshakePromise} was set successfully and a {@link SslHandshakeCompletionEvent}

1773             readDuringHandshake = false;
1774             ctx.read();
1775         }
1776         // Our control flow may invoke this method multiple times for a single FINISHED event. For example
1777         // wrapNonAppData may drain pendingUnencryptedWrites in wrap which transitions to handshake from FINISHED to
1778         // NOT_HANDSHAKING which invokes setHandshakeSuccessIfStillHandshaking, and then wrapNonAppData also directly
1779         // invokes this method.
1780         if (handshakePromise.trySuccess(ctx.channel())) {

1784                         "{} HANDSHAKEN: protocol:{} cipher suite:{}",
1785                         ctx.channel(),
1786                         session.getProtocol(),
1787                         session.getCipherSuite());
1788             }
1789             ctx.fireUserEventTriggered(SslHandshakeCompletionEvent.SUCCESS);
1790             return true;
1791         }
1792         return false;
1793     }

1803      * Notify all the handshake futures about the failure during the handshake.
1804      */
1805     private void setHandshakeFailure(ChannelHandlerContext ctx, Throwable cause, boolean closeInbound,
1806                                      boolean notify, boolean alwaysFlushAndClose) {
1807         try {
1808             // Release all resources such as internal buffers that SSLEngine
1809             // is managing.
1810             outboundClosed = true;
1811             engine.closeOutbound();
1812 
1813             if (closeInbound) {
1814                 try {
1815                     engine.closeInbound();
1816                 } catch (SSLException e) {
1817                     if (logger.isDebugEnabled()) {
1818                         // only log in debug mode as it most likely harmless and latest chrome still trigger
1819                         // this all the time.
1820                         //
1821                         // See https://github.com/netty/netty/issues/1340
1822                         String msg = e.getMessage();
1823                         if (msg == null || !(msg.contains("possible truncation attack") ||
1824                                 msg.contains("closing inbound before receiving peer's close_notify"))) {
1825                             logger.debug("{} SSLEngine.closeInbound() raised an exception.", ctx.channel(), e);
1826                         }
1827                     }
1828                 }
1829             }
1830             if (handshakePromise.tryFailure(cause) || alwaysFlushAndClose) {
1831                 SslUtils.handleHandshakeFailure(ctx, cause, notify);
1832             }
1833         } finally {
1834             // Ensure we remove and fail all pending writes in all cases and so release memory quickly.
1835             releaseAndFailAll(ctx, cause);
1836         }
1837     }

1839     private void setHandshakeFailureTransportFailure(ChannelHandlerContext ctx, Throwable cause) {
1840         // If TLS control frames fail to write we are in an unknown state and may become out of
1841         // sync with our peer. We give up and close the channel. This will also take care of
1842         // cleaning up any outstanding state (e.g. handshake promise, queued unencrypted data).
1843         try {
1844             SSLException transportFailure = new SSLException("failure when writing TLS control frames", cause);
1845             releaseAndFailAll(ctx, transportFailure);
1846             if (handshakePromise.tryFailure(transportFailure)) {
1847                 ctx.fireUserEventTriggered(new SslHandshakeCompletionEvent(transportFailure));
1848             }
1849         } finally {
1850             ctx.close();
1851         }
1852     }
