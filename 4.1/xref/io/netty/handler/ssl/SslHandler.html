71  /**
72   * Adds <a href="http://en.wikipedia.org/wiki/Transport_Layer_Security">SSL
73   * &middot; TLS</a> and StartTLS support to a {@link Channel}.  Please refer
74   * to the <strong>"SecureChat"</strong> example in the distribution or the web
75   * site for the detailed usage.
76   *
77   * <h3>Beginning the handshake</h3>
78   * <p>
79   * You must make sure not to write a message while the handshake is in progress unless you are
80   * renegotiating.  You will be notified by the {@link Future} which is
81   * returned by the {@link #handshakeFuture()} method when the handshake
82   * process succeeds or fails.
83   * <p>
84   * Beside using the handshake {@link ChannelFuture} to get notified about the completation of the handshake it's
85   * also possible to detect it by implement the
86   * {@link ChannelInboundHandler#userEventTriggered(ChannelHandlerContext, Object)}
87   * method and check for a {@link SslHandshakeCompletionEvent}.
88   *
89   * <h3>Handshake</h3>
90   * <p>
91   * The handshake will be automaticly issued for you once the {@link Channel} is active and
92   * {@link SSLEngine#getUseClientMode()} returns {@code true}.
93   * So no need to bother with it by your self.
94   *
95   * <h3>Closing the session</h3>
96   * <p>
97   * To close the SSL session, the {@link #close()} method should be
98   * called to send the {@code close_notify} message to the remote peer.  One
99   * exception is when you close the {@link Channel} - {@link SslHandler}
100  * intercepts the close request and send the {@code close_notify} message
101  * before the channel closure automatically.  Once the SSL session is closed,
102  * it is not reusable, and consequently you should create a new
103  * {@link SslHandler} with a new {@link SSLEngine} as explained in the
104  * following section.
105  *
106  * <h3>Restarting the session</h3>
107  * <p>
108  * To restart the SSL session, you must remove the existing closed
109  * {@link SslHandler} from the {@link ChannelPipeline}, insert a new
110  * {@link SslHandler} with a new {@link SSLEngine} into the pipeline,
111  * and start the handshake process as described in the first section.
112  *
113  * <h3>Implementing StartTLS</h3>
114  * <p>
115  * <a href="http://en.wikipedia.org/wiki/STARTTLS">StartTLS</a> is the
116  * communication pattern that secures the wire in the middle of the plaintext
117  * connection.  Please note that it is different from SSL &middot; TLS, that
118  * secures the wire from the beginning of the connection.  Typically, StartTLS
119  * is composed of three steps:
120  * <ol>
121  * <li>Client sends a StartTLS request to server.</li>
122  * <li>Server sends a StartTLS response to client.</li>
123  * <li>Client begins SSL handshake.</li>
124  * </ol>
125  * If you implement a server, you need to:
126  * <ol>
127  * <li>create a new {@link SslHandler} instance with {@code startTls} flag set
128  *     to {@code true},</li>
129  * <li>insert the {@link SslHandler} to the {@link ChannelPipeline}, and</li>
130  * <li>write a StartTLS response.</li>
131  * </ol>
132  * Please note that you must insert {@link SslHandler} <em>before</em> sending
133  * the StartTLS response.  Otherwise the client can send begin SSL handshake
134  * before {@link SslHandler} is inserted to the {@link ChannelPipeline}, causing
135  * data corruption.
136  * <p>
137  * The client-side implementation is much simpler.
138  * <ol>
139  * <li>Write a StartTLS request,</li>
140  * <li>wait for the StartTLS response,</li>
141  * <li>create a new {@link SslHandler} instance with {@code startTls} flag set
142  *     to {@code false},</li>
143  * <li>insert the {@link SslHandler} to the {@link ChannelPipeline}, and</li>
144  * <li>Initiate SSL handshake.</li>
145  * </ol>
146  *
147  * <h3>Known issues</h3>
148  * <p>
149  * Because of a known issue with the current implementation of the SslEngine that comes
150  * with Java it may be possible that you see blocked IO-Threads while a full GC is done.
151  * <p>
152  * So if you are affected you can workaround this problem by adjust the cache settings
153  * like shown below:
154  *
155  * <pre>
156  *     SslContext context = ...;
157  *     context.getServerSessionContext().setSessionCacheSize(someSaneSize);
158  *     context.getServerSessionContext().setSessionTime(someSameTimeout);
159  * </pre>
160  * <p>
161  * What values to use here depends on the nature of your application and should be set
162  * based on monitoring and debugging of it.
163  * For more details see
164  * <a href="https://github.com/netty/netty/issues/832">#832</a> in our issue tracker.
165  */

394      * See {@link #close()}

396     public ChannelFuture close(final ChannelPromise future) {
397         final ChannelHandlerContext ctx = this.ctx;
398         ctx.executor().execute(new OneTimeTask() {
399             @Override
400             public void run() {
401                 outboundClosed = true;
402                 engine.closeOutbound();
403                 try {
404                     write(ctx, Unpooled.EMPTY_BUFFER, future);
405                     flush(ctx);
406                 } catch (Exception e) {
407                     if (!future.tryFailure(e)) {
408                         logger.warn("{} flush() raised a masked exception.", ctx.channel(), e);
409                     }
410                 }
411             }
412         });
413 
414         return future;
415     }

477     @Override
478     public void write(final ChannelHandlerContext ctx, Object msg, ChannelPromise promise) throws Exception {
479         if (!(msg instanceof ByteBuf)) {
480             promise.setFailure(new UnsupportedMessageTypeException(msg, ByteBuf.class));
481             return;
482         }
483         pendingUnencryptedWrites.add(msg, promise);
484     }

486     @Override
487     public void flush(ChannelHandlerContext ctx) throws Exception {
488         // Do not encrypt the first write request if this handler is
489         // created with startTLS flag turned on.
490         if (startTls && !sentFirstMessage) {
491             sentFirstMessage = true;
492             pendingUnencryptedWrites.removeAndWriteAll();
493             ctx.flush();
494             return;
495         }
496         if (pendingUnencryptedWrites.isEmpty()) {
497             // It's important to NOT use a voidPromise here as the user
498             // may want to add a ChannelFutureListener to the ChannelPromise later.
499             //
500             // See https://github.com/netty/netty/issues/3364
501             pendingUnencryptedWrites.add(Unpooled.EMPTY_BUFFER, ctx.newPromise());
502         }
503         if (!handshakePromise.isDone()) {
504             flushedBeforeHandshake = true;
505         }
506         try {
507             wrap(ctx, false);
508         } finally {
509             // We may have written some parts of data before an exception was thrown so ensure we always flush.
510             // See https://github.com/netty/netty/issues/3900#issuecomment-172481830
511             ctx.flush();
512         }
513     }

515     private void wrap(ChannelHandlerContext ctx, boolean inUnwrap) throws SSLException {
516         ByteBuf out = null;
517         ChannelPromise promise = null;
518         ByteBufAllocator alloc = ctx.alloc();
519         try {
520             for (;;) {
521                 Object msg = pendingUnencryptedWrites.current();
522                 if (msg == null) {
523                     break;
524                 }
525 
526                 ByteBuf buf = (ByteBuf) msg;
527                 if (out == null) {
528                     out = allocateOutNetBuf(ctx, buf.readableBytes());
529                 }
530 
531                 SSLEngineResult result = wrap(alloc, engine, buf, out);
532                 if (!buf.isReadable()) {
533                     promise = pendingUnencryptedWrites.remove();
534                 } else {
535                     promise = null;
536                 }
537 
538                 if (result.getStatus() == Status.CLOSED) {
539                     // SSLEngine has been closed already.
540                     // Any further write attempts should be denied.
541                     pendingUnencryptedWrites.removeAndFailAll(SSLENGINE_CLOSED);
542                     return;
543                 } else {
544                     switch (result.getHandshakeStatus()) {
545                         case NEED_TASK:
546                             runDelegatedTasks();
547                             break;
548                         case FINISHED:
549                             setHandshakeSuccess();
550                             // deliberate fall-through
551                         case NOT_HANDSHAKING:
552                             setHandshakeSuccessIfStillHandshaking();
553                             // deliberate fall-through
554                         case NEED_WRAP:
555                             finishWrap(ctx, out, promise, inUnwrap);
556                             promise = null;
557                             out = null;
558                             break;
559                         case NEED_UNWRAP:
560                             return;
561                         default:
562                             throw new IllegalStateException(
563                                     "Unknown handshake status: " + result.getHandshakeStatus());
564                     }
565                 }
566             }
567         } catch (SSLException e) {
568             setHandshakeFailure(ctx, e);
569             throw e;
570         } finally {
571             finishWrap(ctx, out, promise, inUnwrap);
572         }
573     }
574 
575     private void finishWrap(ChannelHandlerContext ctx, ByteBuf out, ChannelPromise promise, boolean inUnwrap) {
576         if (out == null) {
577             out = Unpooled.EMPTY_BUFFER;
578         } else if (!out.isReadable()) {
579             out.release();
580             out = Unpooled.EMPTY_BUFFER;
581         }
582 
583         if (promise != null) {
584             ctx.write(out, promise);
585         } else {
586             ctx.write(out);
587         }
588 
589         if (inUnwrap) {
590             needsFlush = true;
591         }
592     }

594     private void wrapNonAppData(ChannelHandlerContext ctx, boolean inUnwrap) throws SSLException {
595         ByteBuf out = null;
596         ByteBufAllocator alloc = ctx.alloc();
597         try {
598             for (;;) {
599                 if (out == null) {
600                     out = allocateOutNetBuf(ctx, 0);
601                 }
602                 SSLEngineResult result = wrap(alloc, engine, Unpooled.EMPTY_BUFFER, out);
603 
604                 if (result.bytesProduced() > 0) {
605                     ctx.write(out);
606                     if (inUnwrap) {
607                         needsFlush = true;
608                     }
609                     out = null;
610                 }
611 
612                 switch (result.getHandshakeStatus()) {
613                     case FINISHED:
614                         setHandshakeSuccess();
615                         break;
616                     case NEED_TASK:
617                         runDelegatedTasks();
618                         break;
619                     case NEED_UNWRAP:
620                         if (!inUnwrap) {
621                             unwrapNonAppData(ctx);
622                         }
623                         break;
624                     case NEED_WRAP:
625                         break;
626                     case NOT_HANDSHAKING:
627                         setHandshakeSuccessIfStillHandshaking();
628                         // Workaround for TLS False Start problem reported at:
629                         // https://github.com/netty/netty/issues/1108#issuecomment-14266970
630                         if (!inUnwrap) {
631                             unwrapNonAppData(ctx);
632                         }
633                         break;
634                     default:
635                         throw new IllegalStateException("Unknown handshake status: " + result.getHandshakeStatus());
636                 }
637 
638                 if (result.bytesProduced() == 0) {
639                     break;
640                 }
641 
642                 // It should not consume empty buffers when it is not handshaking
643                 // Fix for Android, where it was encrypting empty buffers even when not handshaking
644                 if (result.bytesConsumed() == 0 && result.getHandshakeStatus() == HandshakeStatus.NOT_HANDSHAKING) {
645                     break;
646                 }
647             }
648         } catch (SSLException e) {
649             setHandshakeFailure(ctx, e);
650 
651             // We may have written some parts of data before an exception was thrown so ensure we always flush.
652             // See https://github.com/netty/netty/issues/3900#issuecomment-172481830
653             flushIfNeeded(ctx);
654             throw e;
655         }  finally {
656             if (out != null) {
657                 out.release();
658             }
659         }
660     }

662     private SSLEngineResult wrap(ByteBufAllocator alloc, SSLEngine engine, ByteBuf in, ByteBuf out)
663             throws SSLException {
664         ByteBuf newDirectIn = null;
665         try {
666             int readerIndex = in.readerIndex();
667             int readableBytes = in.readableBytes();
668 
669             // We will call SslEngine.wrap(ByteBuffer[], ByteBuffer) to allow efficient handling of
670             // CompositeByteBuf without force an extra memory copy when CompositeByteBuffer.nioBuffer() is called.
671             final ByteBuffer[] in0;
672             if (in.isDirect() || !wantsDirectBuffer) {
673                 // As CompositeByteBuf.nioBufferCount() can be expensive (as it needs to check all composed ByteBuf
674                 // to calculate the count) we will just assume a CompositeByteBuf contains more then 1 ByteBuf.
675                 // The worst that can happen is that we allocate an extra ByteBuffer[] in CompositeByteBuf.nioBuffers()
676                 // which is better then walking the composed ByteBuf in most cases.
677                 if (!(in instanceof CompositeByteBuf) && in.nioBufferCount() == 1) {
678                     in0 = singleBuffer;
679                     // We know its only backed by 1 ByteBuffer so use internalNioBuffer to keep object allocation
680                     // to a minimum.
681                     in0[0] = in.internalNioBuffer(readerIndex, readableBytes);
682                 } else {
683                     in0 = in.nioBuffers();
684                 }
685             } else {
686                 // We could even go further here and check if its a CompositeByteBuf and if so try to decompose it and
687                 // only replace the ByteBuffer that are not direct. At the moment we just will replace the whole
688                 // CompositeByteBuf to keep the complexity to a minimum
689                 newDirectIn = alloc.directBuffer(readableBytes);
690                 newDirectIn.writeBytes(in, readerIndex, readableBytes);
691                 in0 = singleBuffer;
692                 in0[0] = newDirectIn.internalNioBuffer(0, readableBytes);
693             }
694 
695             for (;;) {
696                 ByteBuffer out0 = out.nioBuffer(out.writerIndex(), out.writableBytes());
697                 SSLEngineResult result = engine.wrap(in0, out0);
698                 in.skipBytes(result.bytesConsumed());
699                 out.writerIndex(out.writerIndex() + result.bytesProduced());
700 
701                 switch (result.getStatus()) {
702                 case BUFFER_OVERFLOW:
703                     out.ensureWritable(maxPacketBufferSize);
704                     break;
705                 default:
706                     return result;
707                 }
708             }
709         } finally {
710             // Null out to allow GC of ByteBuffer
711             singleBuffer[0] = null;
712 
713             if (newDirectIn != null) {
714                 newDirectIn.release();
715             }
716         }
717     }

729         if (ignoreException(cause)) {
730             // It is safe to ignore the 'connection reset by peer' or
731             // 'broken pipe' error after sending close_notify.

735                         "while writing close_notify in response to the peer's close_notify", ctx.channel(), cause);
736             }
737

739             // did not close the connection automatically.

745         }

748     /**
749      * Checks if the given {@link Throwable} can be ignore and just "swallowed"
750      *
751      * When an ssl connection is closed a close_notify message is sent.
752      * After that the peer also sends close_notify however, it's not mandatory to receive
753      * the close_notify. The party who sent the initial close_notify can close the connection immediately
754      * then the peer will get connection reset error.
755      *
756      */
757     private boolean ignoreException(Throwable t) {
758         if (!(t instanceof SSLException) && t instanceof IOException && sslCloseFuture.isDone()) {
759             String message = String.valueOf(t.getMessage()).toLowerCase();
760 
761             // first try to match connection reset / broke peer based on the regex. This is the fastest way
762             // but may fail on different jdk impls or OS's
763             if (IGNORABLE_ERROR_MESSAGE.matcher(message).matches()) {
764                 return true;
765             }
766 
767             // Inspect the StackTraceElements to see if it was a connection reset / broken pipe or not
768             StackTraceElement[] elements = t.getStackTrace();
769             for (StackTraceElement element: elements) {
770                 String classname = element.getClassName();
771                 String methodname = element.getMethodName();
772 
773                 // skip all classes that belong to the io.netty package
774                 if (classname.startsWith("io.netty.")) {
775                     continue;
776                 }
777 
778                 // check if the method name is read if not skip it
779                 if (!"read".equals(methodname)) {
780                     continue;
781                 }
782 
783                 // This will also match against SocketInputStream which is used by openjdk 7 and maybe
784                 // also others
785                 if (IGNORABLE_CLASS_IN_STACK.matcher(classname).matches()) {
786                     return true;
787                 }
788 
789                 try {
790                     // No match by now.. Try to load the class via classloader and inspect it.
791                     // This is mainly done as other JDK implementations may differ in name of
792                     // the impl.
793                     Class<?> clazz = PlatformDependent.getClassLoader(getClass()).loadClass(classname);
794 
795                     if (SocketChannel.class.isAssignableFrom(clazz)
796                             || DatagramChannel.class.isAssignableFrom(clazz)) {
797                         return true;
798                     }
799 
800                     // also match against SctpChannel via String matching as it may not present.
801                     if (PlatformDependent.javaVersion() >= 7
802                             && "com.sun.nio.sctp.SctpChannel".equals(clazz.getSuperclass().getName())) {
803                         return true;
804                     }
805                 } catch (ClassNotFoundException e) {
806                     // This should not happen just ignore
807                 }
808             }
809         }
810 
811         return false;
812     }

814     /**
815      * Returns {@code true} if the given {@link ByteBuf} is encrypted. Be aware that this method
816      * will not increase the readerIndex of the given {@link ByteBuf}.
817      *
818      * @param   buffer
819      *                  The {@link ByteBuf} to read from. Be aware that it must have at least 5 bytes to read,
820      *                  otherwise it will throw an {@link IllegalArgumentException}.
821      * @return encrypted
822      *                  {@code true} if the {@link ByteBuf} is encrypted, {@code false} otherwise.
823      * @throws IllegalArgumentException
824      *                  Is thrown if the given {@link ByteBuf} has not at least 5 bytes to read.
825      */
826     public static boolean isEncrypted(ByteBuf buffer) {
827         if (buffer.readableBytes() < SslUtils.SSL_RECORD_HEADER_LENGTH) {
828             throw new IllegalArgumentException(
829                     "buffer must have at least " + SslUtils.SSL_RECORD_HEADER_LENGTH + " readable bytes");
830         }
831         return getEncryptedPacketLength(buffer, buffer.readerIndex()) != -1;
832     }

834     @Override
835     protected void decode(ChannelHandlerContext ctx, ByteBuf in, List<Object> out) throws SSLException {
836         final int startOffset = in.readerIndex();
837         final int endOffset = in.writerIndex();
838         int offset = startOffset;
839         int totalLength = 0;
840 
841         // If we calculated the length of the current SSL record before, use that information.
842         if (packetLength > 0) {
843             if (endOffset - startOffset < packetLength) {
844                 return;
845             } else {
846                 offset += packetLength;
847                 totalLength = packetLength;
848                 packetLength = 0;
849             }
850         }
851 
852         boolean nonSslRecord = false;
853 
854         while (totalLength < OpenSslEngine.MAX_ENCRYPTED_PACKET_LENGTH) {
855             final int readableBytes = endOffset - offset;
856             if (readableBytes < SslUtils.SSL_RECORD_HEADER_LENGTH) {
857                 break;
858             }
859 
860             final int packetLength = getEncryptedPacketLength(in, offset);
861             if (packetLength == -1) {
862                 nonSslRecord = true;
863                 break;
864             }
865 
866             assert packetLength > 0;
867 
868             if (packetLength > readableBytes) {
869                 // wait until the whole packet can be read
870                 this.packetLength = packetLength;
871                 break;
872             }
873 
874             int newTotalLength = totalLength + packetLength;
875             if (newTotalLength > OpenSslEngine.MAX_ENCRYPTED_PACKET_LENGTH) {
876                 // Don't read too much.
877                 break;
878             }
879 
880             // We have a whole packet.
881             // Increment the offset to handle the next packet.
882             offset += packetLength;
883             totalLength = newTotalLength;
884         }
885 
886         if (totalLength > 0) {
887             boolean decoded = false;
888 
889             // The buffer contains one or more full SSL records.
890             // Slice out the whole packet so unwrap will only be called with complete packets.
891             // Also directly reset the packetLength. This is needed as unwrap(..) may trigger
892             // decode(...) again via:
893             // 1) unwrap(..) is called
894             // 2) wrap(...) is called from within unwrap(...)
895             // 3) wrap(...) calls unwrapLater(...)
896             // 4) unwrapLater(...) calls decode(...)
897             //
898             // See https://github.com/netty/netty/issues/1534
899 
900             in.skipBytes(totalLength);
901 
902             // If SSLEngine expects a heap buffer for unwrapping, do the conversion.
903             if (in.isDirect() && wantsInboundHeapBuffer) {
904                 ByteBuf copy = ctx.alloc().heapBuffer(totalLength);
905                 try {
906                     copy.writeBytes(in, startOffset, totalLength);
907                     decoded = unwrap(ctx, copy, 0, totalLength);
908                 } finally {
909                     copy.release();
910                 }
911             } else {
912                 decoded = unwrap(ctx, in, startOffset, totalLength);
913             }
914 
915             if (!firedChannelRead) {
916                 // Check first if firedChannelRead is not set yet as it may have been set in a
917                 // previous decode(...) call.
918                 firedChannelRead = decoded;
919             }
920         }
921 
922         if (nonSslRecord) {
923             // Not an SSL/TLS packet
924             NotSslRecordException e = new NotSslRecordException(
925                     "not an SSL/TLS record: " + ByteBufUtil.hexDump(in));
926             in.skipBytes(in.readableBytes());
927             ctx.fireExceptionCaught(e);
928             setHandshakeFailure(ctx, e);
929         }
930     }

934         // Discard bytes of the cumulation buffer if needed.
935         discardSomeReadBytes();
936 
937         flushIfNeeded(ctx);

942             // yet, which means we need to trigger the read to ensure we not encounter any stalls.
943             ctx.read();
944         }
945

947         ctx.fireChannelReadComplete();
948     }

964     /**
965      * Unwraps inbound SSL records.
966      */
967     private boolean unwrap(
968             ChannelHandlerContext ctx, ByteBuf packet, int offset, int length) throws SSLException {
969 
970         boolean decoded = false;
971         boolean wrapLater = false;
972         boolean notifyClosure = false;
973         ByteBuf decodeOut = allocate(ctx, length);
974         try {
975             for (;;) {
976                 final SSLEngineResult result = unwrap(engine, packet, offset, length, decodeOut);
977                 final Status status = result.getStatus();
978                 final HandshakeStatus handshakeStatus = result.getHandshakeStatus();
979                 final int produced = result.bytesProduced();
980                 final int consumed = result.bytesConsumed();
981 
982                 // Update indexes for the next iteration
983                 offset += consumed;
984                 length -= consumed;
985 
986                 switch (status) {
987                 case BUFFER_OVERFLOW:
988                     int readableBytes = decodeOut.readableBytes();
989                     if (readableBytes > 0) {
990                         decoded = true;
991                         ctx.fireChannelRead(decodeOut);
992                     } else {
993                         decodeOut.release();
994                     }
995                     // Allocate a new buffer which can hold all the rest data and loop again.
996                     // TODO: We may want to reconsider how we calculate the length here as we may
997                     // have more then one ssl message to decode.
998                     decodeOut = allocate(ctx, engine.getSession().getApplicationBufferSize() - readableBytes);
999                     continue;
1000                 case CLOSED:
1001                     // notify about the CLOSED state of the SSLEngine. See #137
1002                     notifyClosure = true;
1003                     break;
1004                 default:
1005                     break;
1006                 }
1007 
1008                 switch (handshakeStatus) {
1009                     case NEED_UNWRAP:
1010                         break;
1011                     case NEED_WRAP:
1012                         wrapNonAppData(ctx, true);
1013                         break;
1014                     case NEED_TASK:
1015                         runDelegatedTasks();
1016                         break;
1017                     case FINISHED:
1018                         setHandshakeSuccess();
1019                         wrapLater = true;
1020 
1021                         // We 'break' here and NOT 'continue' as android API version 21 has a bug where they consume
1022                         // data from the buffer but NOT correctly set the SSLEngineResult.bytesConsumed().
1023                         // Because of this it will raise an exception on the next iteration of the for loop on android
1024                         // API version 21. Just doing a break will work here as produced and consumed will both be 0
1025                         // and so we break out of the complete for (;;) loop and so call decode(...) again later on.
1026                         // On other platforms this will have no negative effect as we will just continue with the
1027                         // for (;;) loop if something was either consumed or produced.
1028                         //
1029                         // See:
1030                         //  - https://github.com/netty/netty/issues/4116
1031                         //  - https://code.google.com/p/android/issues/detail?id=198639&thanks=198639&ts=1452501203
1032                         break;
1033                     case NOT_HANDSHAKING:
1034                         if (setHandshakeSuccessIfStillHandshaking()) {
1035                             wrapLater = true;
1036                             continue;
1037                         }
1038                         if (flushedBeforeHandshake) {
1039                             // We need to call wrap(...) in case there was a flush done before the handshake completed.
1040                             //
1041                             // See https://github.com/netty/netty/pull/2437
1042                             flushedBeforeHandshake = false;
1043                             wrapLater = true;
1044                         }
1045 
1046                         break;
1047                     default:
1048                         throw new IllegalStateException("unknown handshake status: " + handshakeStatus);
1049                 }
1050 
1051                 if (status == Status.BUFFER_UNDERFLOW || consumed == 0 && produced == 0) {
1052                     break;
1053                 }
1054             }
1055 
1056             if (wrapLater) {
1057                 wrap(ctx, true);
1058             }
1059 
1060             if (notifyClosure) {
1061                 sslCloseFuture.trySuccess(ctx.channel());
1062             }
1063         } catch (SSLException e) {
1064             setHandshakeFailure(ctx, e);
1065             throw e;
1066         } finally {
1067             if (decodeOut.isReadable()) {
1068                 decoded = true;
1069 
1070                 ctx.fireChannelRead(decodeOut);
1071             } else {
1072                 decodeOut.release();
1073             }
1074         }
1075         return decoded;
1076     }

1078     private SSLEngineResult unwrap(
1079             SSLEngine engine, ByteBuf in, int readerIndex, int len, ByteBuf out) throws SSLException {
1080         int nioBufferCount = in.nioBufferCount();
1081         int writerIndex = out.writerIndex();
1082         final SSLEngineResult result;
1083         if (engine instanceof OpenSslEngine && nioBufferCount > 1) {
1084             /**
1085              * If {@link OpenSslEngine} is in use,
1086              * we can use a special {@link OpenSslEngine#unwrap(ByteBuffer[], ByteBuffer[])} method
1087              * that accepts multiple {@link ByteBuffer}s without additional memory copies.
1088              */
1089             OpenSslEngine opensslEngine = (OpenSslEngine) engine;
1090             try {
1091                 singleBuffer[0] = toByteBuffer(out, writerIndex, out.writableBytes());
1092                 result = opensslEngine.unwrap(in.nioBuffers(readerIndex, len), singleBuffer);
1093                 out.writerIndex(writerIndex + result.bytesProduced());
1094             } finally {
1095                 singleBuffer[0] = null;
1096             }
1097         } else {
1098             result = engine.unwrap(toByteBuffer(in, readerIndex, len),
1099                                                    toByteBuffer(out, writerIndex, out.writableBytes()));
1100         }
1101         out.writerIndex(writerIndex + result.bytesProduced());
1102         return result;
1103     }

1105     private static ByteBuffer toByteBuffer(ByteBuf out, int index, int len) {
1106         return out.nioBufferCount() == 1 ? out.internalNioBuffer(index, len) :
1107                 out.nioBuffer(index, len);

1110     /**
1111      * Fetches all delegated tasks from the {@link SSLEngine} and runs them via the {@link #delegatedTaskExecutor}.
1112      * If the {@link #delegatedTaskExecutor} is {@link ImmediateExecutor}, just call {@link Runnable#run()} directly
1113      * instead of using {@link Executor#execute(Runnable)}.  Otherwise, run the tasks via
1114      * the {@link #delegatedTaskExecutor} and wait until the tasks are finished.
1115      */
1116     private void runDelegatedTasks() {
1117         if (delegatedTaskExecutor == ImmediateExecutor.INSTANCE) {
1118             for (;;) {
1119                 Runnable task = engine.getDelegatedTask();
1120                 if (task == null) {
1121                     break;
1122                 }
1123 
1124                 task.run();
1125             }
1126         } else {
1127             final List<Runnable> tasks = new ArrayList<Runnable>(2);
1128             for (;;) {
1129                 final Runnable task = engine.getDelegatedTask();
1130                 if (task == null) {
1131                     break;
1132                 }
1133 
1134                 tasks.add(task);
1135             }
1136 
1137             if (tasks.isEmpty()) {
1138                 return;
1139             }
1140 
1141             final CountDownLatch latch = new CountDownLatch(1);
1142             delegatedTaskExecutor.execute(new OneTimeTask() {
1143                 @Override
1144                 public void run() {
1145                     try {
1146                         for (Runnable task: tasks) {
1147                             task.run();
1148                         }
1149                     } catch (Exception e) {
1150                         ctx.fireExceptionCaught(e);
1151                     } finally {
1152                         latch.countDown();
1153                     }
1154                 }
1155             });
1156 
1157             boolean interrupted = false;
1158             while (latch.getCount() != 0) {
1159                 try {
1160                     latch.await();
1161                 } catch (InterruptedException e) {
1162                     // Interrupt later.
1163                     interrupted = true;
1164                 }
1165             }
1166 
1167             if (interrupted) {
1168                 Thread.currentThread().interrupt();
1169             }
1170         }
1171     }

1326     /**
1327      * Performs TLS (re)negotiation.
1328      *
1329      * @param newHandshakePromise if {@code null}, use the existing {@link #handshakePromise},
1330      *                            assuming that the current negotiation has not been finished.
1331      *                            Currently, {@code null} is expected only for the initial handshake.
1332      */
1333     private void handshake(final Promise<Channel> newHandshakePromise) {
1334         final Promise<Channel> p;
1335         if (newHandshakePromise != null) {
1336             final Promise<Channel> oldHandshakePromise = handshakePromise;
1337             if (!oldHandshakePromise.isDone()) {
1338                 // There's no need to handshake because handshake is in progress already.
1339                 // Merge the new promise into the old one.
1340                 oldHandshakePromise.addListener(new FutureListener<Channel>() {
1341                     @Override
1342                     public void operationComplete(Future<Channel> future) throws Exception {
1343                         if (future.isSuccess()) {
1344                             newHandshakePromise.setSuccess(future.getNow());
1345                         } else {
1346                             newHandshakePromise.setFailure(future.cause());
1347                         }
1348                     }
1349                 });
1350                 return;
1351             }
1352 
1353             handshakePromise = p = newHandshakePromise;
1354         } else {
1355             // Forced to reuse the old handshake.
1356             p = handshakePromise;
1357             assert !p.isDone();
1358         }
1359 
1360         // Begin handshake.
1361         final ChannelHandlerContext ctx = this.ctx;
1362         try {
1363             engine.beginHandshake();
1364             wrapNonAppData(ctx, false);
1365         } catch (Exception e) {
1366             notifyHandshakeFailure(e);
1367         } finally {
1368             // We have may haven written some parts of data before an exception was thrown so ensure we always flush.
1369             // See https://github.com/netty/netty/issues/3900#issuecomment-172481830
1370             ctx.flush();
1371         }
1372 
1373         // Set timeout if necessary.
1374         final long handshakeTimeoutMillis = this.handshakeTimeoutMillis;
1375         if (handshakeTimeoutMillis <= 0 || p.isDone()) {
1376             return;
1377         }
1378 
1379         final ScheduledFuture<?> timeoutFuture = ctx.executor().schedule(new OneTimeTask() {
1380             @Override
1381             public void run() {
1382                 if (p.isDone()) {
1383                     return;
1384                 }
1385                 notifyHandshakeFailure(HANDSHAKE_TIMED_OUT);
1386             }
1387         }, handshakeTimeoutMillis, TimeUnit.MILLISECONDS);
1388 
1389         // Cancel the handshake timeout when handshake is finished.
1390         p.addListener(new FutureListener<Channel>() {
1391             @Override
1392             public void operationComplete(Future<Channel> f) throws Exception {
1393                 timeoutFuture.cancel(false);
1394             }
1395         });
1396     }

1444                 }
