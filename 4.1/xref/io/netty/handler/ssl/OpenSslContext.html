141     OpenSslContext(Iterable<String> ciphers, CipherSuiteFilter cipherFilter,
142                    OpenSslApplicationProtocolNegotiator apn, long sessionCacheSize,
143                    long sessionTimeout, int mode, Certificate[] keyCertChain,
144                    ClientAuth clientAuth) throws SSLException {
145         OpenSsl.ensureAvailability();
146 
147         if (mode != SSL.SSL_MODE_SERVER && mode != SSL.SSL_MODE_CLIENT) {
148             throw new IllegalArgumentException("mode most be either SSL.SSL_MODE_SERVER or SSL.SSL_MODE_CLIENT");
149         }
150         this.mode = mode;
151         this.clientAuth = isServer() ? checkNotNull(clientAuth, "clientAuth") : ClientAuth.NONE;
152 
153         if (mode == SSL.SSL_MODE_SERVER) {
154             rejectRemoteInitiatedRenegotiation =
155                     JDK_REJECT_CLIENT_INITIATED_RENEGOTIATION;
156         }
157         this.keyCertChain = keyCertChain == null ? null : keyCertChain.clone();
158         final List<String> convertedCiphers;
159         if (ciphers == null) {
160             convertedCiphers = null;
161         } else {
162             convertedCiphers = new ArrayList<String>();
163             for (String c: ciphers) {
164                 if (c == null) {
165                     break;
166                 }
167 
168                 String converted = CipherSuiteConverter.toOpenSsl(c);
169                 if (converted != null) {
170                     c = converted;
171                 }
172                 convertedCiphers.add(c);
173             }
174         }
175 
176         unmodifiableCiphers = Arrays.asList(checkNotNull(cipherFilter, "cipherFilter").filterCipherSuites(
177                 convertedCiphers, DEFAULT_CIPHERS, OpenSsl.availableCipherSuites()));
178 
179         this.apn = checkNotNull(apn, "apn");
180 
181         // Allocate a new APR pool.
182         aprPool = Pool.create(0);
183 
184         // Create a new SSL_CTX and configure it.
185         boolean success = false;
186         try {
187             synchronized (OpenSslContext.class) {
188                 try {
189                     ctx = SSLContext.make(aprPool, SSL.SSL_PROTOCOL_ALL, mode);
190                 } catch (Exception e) {
191                     throw new SSLException("failed to create an SSL_CTX", e);
192                 }
193 
194                 SSLContext.setOptions(ctx, SSL.SSL_OP_ALL);
195                 SSLContext.setOptions(ctx, SSL.SSL_OP_NO_SSLv2);
196                 SSLContext.setOptions(ctx, SSL.SSL_OP_NO_SSLv3);
197                 SSLContext.setOptions(ctx, SSL.SSL_OP_CIPHER_SERVER_PREFERENCE);
198                 SSLContext.setOptions(ctx, SSL.SSL_OP_SINGLE_ECDH_USE);
199                 SSLContext.setOptions(ctx, SSL.SSL_OP_SINGLE_DH_USE);
200                 SSLContext.setOptions(ctx, SSL.SSL_OP_NO_SESSION_RESUMPTION_ON_RENEGOTIATION);
201 
202                 // We need to enable SSL_MODE_ACCEPT_MOVING_WRITE_BUFFER as the memory address may change between
203                 // calling OpenSSLEngine.wrap(...).
204                 // See https://github.com/netty/netty-tcnative/issues/100
205                 SSLContext.setMode(ctx, SSLContext.getMode(ctx) | SSL.SSL_MODE_ACCEPT_MOVING_WRITE_BUFFER);
206 
207                 /* List the ciphers that are permitted to negotiate. */
208                 try {
209                     SSLContext.setCipherSuite(ctx, CipherSuiteConverter.toOpenSsl(unmodifiableCiphers));
210                 } catch (SSLException e) {
211                     throw e;
212                 } catch (Exception e) {
213                     throw new SSLException("failed to set cipher suite: " + unmodifiableCiphers, e);
214                 }
215 
216                 List<String> nextProtoList = apn.protocols();
217                 /* Set next protocols for next protocol negotiation extension, if specified */
218                 if (!nextProtoList.isEmpty()) {
219                     String[] protocols = nextProtoList.toArray(new String[nextProtoList.size()]);
220                     int selectorBehavior = opensslSelectorFailureBehavior(apn.selectorFailureBehavior());
221 
222                     switch (apn.protocol()) {
223                     case NPN:
224                         SSLContext.setNpnProtos(ctx, protocols, selectorBehavior);
225                         break;
226                     case ALPN:
227                         SSLContext.setAlpnProtos(ctx, protocols, selectorBehavior);
228                         break;
229                     case NPN_AND_ALPN:
230                         SSLContext.setNpnProtos(ctx, protocols, selectorBehavior);
231                         SSLContext.setAlpnProtos(ctx, protocols, selectorBehavior);
232                         break;
233                     default:
234                         throw new Error();
235                     }
236                 }
237 
238                 /* Set session cache size, if specified */
239                 if (sessionCacheSize > 0) {
240                     this.sessionCacheSize = sessionCacheSize;
241                     SSLContext.setSessionCacheSize(ctx, sessionCacheSize);
242                 } else {
243                     // Get the default session cache size using SSLContext.setSessionCacheSize()
244                     this.sessionCacheSize = sessionCacheSize = SSLContext.setSessionCacheSize(ctx, 20480);
245                     // Revert the session cache size to the default value.
246                     SSLContext.setSessionCacheSize(ctx, sessionCacheSize);
247                 }
248 
249                 /* Set session timeout, if specified */
250                 if (sessionTimeout > 0) {
251                     this.sessionTimeout = sessionTimeout;
252                     SSLContext.setSessionCacheTimeout(ctx, sessionTimeout);
253                 } else {
254                     // Get the default session timeout using SSLContext.setSessionCacheTimeout()
255                     this.sessionTimeout = sessionTimeout = SSLContext.setSessionCacheTimeout(ctx, 300);
256                     // Revert the session timeout to the default value.
257                     SSLContext.setSessionCacheTimeout(ctx, sessionTimeout);
258                 }
259             }
260             success = true;
261         } finally {
262             if (!success) {
263                 destroy();
264             }
265         }
266     }

268     private static int opensslSelectorFailureBehavior(SelectorFailureBehavior behavior) {
269         switch (behavior) {

272         case CHOOSE_MY_LAST_PROTOCOL:
273             return SSL.SSL_SELECTOR_FAILURE_CHOOSE_MY_LAST_PROTOCOL;
274         default:
275             throw new Error();

375     protected final void destroy() {
376         synchronized (OpenSslContext.class) {
377             if (ctx != 0) {
378                 SSLContext.free(ctx);
379                 ctx = 0;
380             }

382             // Guard against multiple destroyPools() calls triggered by construction exception and finalize() later
383             if (aprPool != 0) {

385                 aprPool = 0;

407     /**
408      * Translate a {@link ApplicationProtocolConfig} object to a
409      * {@link OpenSslApplicationProtocolNegotiator} object.
410      * @param config The configuration which defines the translation
411      * @return The results of the translation
412      */
413     static OpenSslApplicationProtocolNegotiator toNegotiator(ApplicationProtocolConfig config) {
414         if (config == null) {
415             return NONE_PROTOCOL_NEGOTIATOR;
416         }
417 
418         switch (config.protocol()) {
419         case NONE:
420             return NONE_PROTOCOL_NEGOTIATOR;
421         case ALPN:
422         case NPN:
423         case NPN_AND_ALPN:
424             switch (config.selectedListenerFailureBehavior()) {
425             case CHOOSE_MY_LAST_PROTOCOL:
426             case ACCEPT:
427                 switch (config.selectorFailureBehavior()) {
428                 case CHOOSE_MY_LAST_PROTOCOL:
429                 case NO_ADVERTISE:
430                     return new OpenSslDefaultApplicationProtocolNegotiator(
431                             config);
432                 default:
433                     throw new UnsupportedOperationException(
434                             new StringBuilder("OpenSSL provider does not support ")
435                                     .append(config.selectorFailureBehavior())
436                                     .append(" behavior").toString());
437                 }
438             default:
439                 throw new UnsupportedOperationException(
440                         new StringBuilder("OpenSSL provider does not support ")
441                                 .append(config.selectedListenerFailureBehavior())
442                                 .append(" behavior").toString());
443             }
444         default:
445             throw new Error();
446         }
447     }

453     abstract class AbstractCertificateVerifier implements CertificateVerifier {
454         @Override
455         public final int verify(long ssl, byte[][] chain, String auth) {
456             X509Certificate[] peerCerts = certificates(chain);
457             final OpenSslEngine engine = engineMap.remove(ssl);
458             try {
459                 verify(engine, peerCerts, auth);
460                 return CertificateVerifier.X509_V_OK;
461             } catch (Throwable cause) {
462                 logger.debug("verification of certificate failed", cause);
463                 SSLHandshakeException e = new SSLHandshakeException("General OpenSslEngine problem");
464                 e.initCause(cause);
465                 engine.handshakeException = e;
466 
467                 if (cause instanceof OpenSslCertificateException) {
468                     return ((OpenSslCertificateException) cause).errorCode();
469                 }
470                 if (cause instanceof CertificateExpiredException) {
471                     return CertificateVerifier.X509_V_ERR_CERT_HAS_EXPIRED;
472                 }
473                 if (cause instanceof CertificateNotYetValidException) {
474                     return CertificateVerifier.X509_V_ERR_CERT_NOT_YET_VALID;
475                 }
476                 if (PlatformDependent.javaVersion() >= 7 && cause instanceof CertificateRevokedException) {
477                     return CertificateVerifier.X509_V_ERR_CERT_REVOKED;
478                 }
479                 return CertificateVerifier.X509_V_ERR_UNSPECIFIED;
480             }
481         }

483         abstract void verify(OpenSslEngine engine, X509Certificate[] peerCerts, String auth) throws Exception;

499     /**
500      * Return the pointer to a <a href="https://www.openssl.org/docs/crypto/BIO_get_mem_ptr.html">in-memory BIO</a>
501      * or {@code 0} if the {@code key} is {@code null}. The BIO contains the content of the {@code key}.
502      */
503     static long toBIO(PrivateKey key) throws Exception {
504         if (key == null) {
505             return 0;
506         }
507         ByteBuf buffer = Unpooled.directBuffer();
508         try {
509             buffer.writeBytes(BEGIN_PRIVATE_KEY);
510             ByteBuf wrappedBuf = Unpooled.wrappedBuffer(key.getEncoded());
511             final ByteBuf encodedBuf;
512             try {
513                 encodedBuf = Base64.encode(wrappedBuf, true);
514                 try {
515                     buffer.writeBytes(encodedBuf);
516                 } finally {
517                     encodedBuf.release();
518                 }
519             } finally {
520                 wrappedBuf.release();
521             }
522             buffer.writeBytes(END_PRIVATE_KEY);
523             return newBIO(buffer);
524         } finally {
525             buffer.release();
526         }
527     }

529     /**
530      * Return the pointer to a <a href="https://www.openssl.org/docs/crypto/BIO_get_mem_ptr.html">in-memory BIO</a>
531      * or {@code 0} if the {@code certChain} is {@code null}. The BIO contains the content of the {@code certChain}.
532      */
533     static long toBIO(X509Certificate[] certChain) throws Exception {
534         if (certChain == null) {
535             return 0;
536         }
537         ByteBuf buffer = Unpooled.directBuffer();
538         try {
539             for (X509Certificate cert: certChain) {
540                 buffer.writeBytes(BEGIN_CERT);
541                 ByteBuf wrappedBuf = Unpooled.wrappedBuffer(cert.getEncoded());
542                 try {
543                     ByteBuf encodedBuf = Base64.encode(wrappedBuf, true);
544                     try {
545                         buffer.writeBytes(encodedBuf);
546                     } finally {
547                         encodedBuf.release();
548                     }
549                 } finally {
550                     wrappedBuf.release();
551                 }
552                 buffer.writeBytes(END_CERT);
553             }
554             return newBIO(buffer);
555         }  finally {
556             buffer.release();
557         }
558     }

560     private static long newBIO(ByteBuf buffer) throws Exception {
561         long bio = SSL.newMemBIO();
562         int readable = buffer.readableBytes();
563         if (SSL.writeToBIO(bio, OpenSsl.memoryAddress(buffer), readable) != readable) {
564             SSL.freeBIO(bio);
565             throw new IllegalStateException("Could not write data to memory BIO");
566         }
567         return bio;
568     }
