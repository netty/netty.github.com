66      /**
67       * Returns {@code true} if and only if the connection can remain open and
68       * thus 'kept alive'.  This methods respects the value of the.
69       * {@code "Connection"} header first and then the return value of
70       * {@link HttpVersion#isKeepAliveDefault()}.
71       */
72      public static boolean isKeepAlive(HttpMessage message) {
73          CharSequence connection = message.headers().get(HttpHeaderNames.CONNECTION);
74          if (connection != null && HttpHeaderValues.CLOSE.contentEqualsIgnoreCase(connection)) {
75              return false;
76          }
77  
78          if (message.protocolVersion().isKeepAliveDefault()) {
79              return !HttpHeaderValues.CLOSE.contentEqualsIgnoreCase(connection);
80          } else {
81              return HttpHeaderValues.KEEP_ALIVE.contentEqualsIgnoreCase(connection);
82          }
83      }

103      * @see #setKeepAlive(HttpHeaders, HttpVersion, boolean)

109     /**
110      * Sets the value of the {@code "Connection"} header depending on the
111      * protocol version of the specified message. This getMethod sets or removes
112      * the {@code "Connection"} header depending on what the default keep alive
113      * mode of the message's protocol version is, as specified by
114      * {@link HttpVersion#isKeepAliveDefault()}.
115      * <ul>
116      * <li>If the connection is kept alive by default:
117      *     <ul>
118      *     <li>set to {@code "close"} if {@code keepAlive} is {@code false}.</li>
119      *     <li>remove otherwise.</li>
120      *     </ul></li>
121      * <li>If the connection is closed by default:
122      *     <ul>
123      *     <li>set to {@code "keep-alive"} if {@code keepAlive} is {@code true}.</li>
124      *     <li>remove otherwise.</li>
125      *     </ul></li>
126      * </ul>
127      */
128     public static void setKeepAlive(HttpHeaders h, HttpVersion httpVersion, boolean keepAlive) {
129         if (httpVersion.isKeepAliveDefault()) {
130             if (keepAlive) {
131                 h.remove(HttpHeaderNames.CONNECTION);
132             } else {
133                 h.set(HttpHeaderNames.CONNECTION, HttpHeaderValues.CLOSE);
134             }
135         } else {
136             if (keepAlive) {
137                 h.set(HttpHeaderNames.CONNECTION, HttpHeaderValues.KEEP_ALIVE);
138             } else {
139                 h.remove(HttpHeaderNames.CONNECTION);
140             }
141         }
142     }

144     /**
145      * Returns the length of the content. Please note that this value is
146      * not retrieved from {@link HttpContent#content()} but from the
147      * {@code "Content-Length"} header, and thus they are independent from each
148      * other.
149      *
150      * @return the content length
151      *
152      * @throws NumberFormatException
153      *         if the message does not have the {@code "Content-Length"} header
154      *         or its value is not a number
155      */
156     public static long getContentLength(HttpMessage message) {
157         String value = message.headers().get(HttpHeaderNames.CONTENT_LENGTH);
158         if (value != null) {
159             return Long.parseLong(value);
160         }
161 
162         // We know the content length if it's a Web Socket message even if
163         // Content-Length header is missing.
164         long webSocketContentLength = getWebSocketContentLength(message);
165         if (webSocketContentLength >= 0) {
166             return webSocketContentLength;
167         }
168 
169         // Otherwise we don't.
170         throw new NumberFormatException("header not found: " + HttpHeaderNames.CONTENT_LENGTH);
171     }

210     /**
211      * Returns the content length of the specified web socket message. If the
212      * specified message is not a web socket message, {@code -1} is returned.
213      */
214     private static int getWebSocketContentLength(HttpMessage message) {
215         // WebSockset messages have constant content-lengths.
216         HttpHeaders h = message.headers();
217         if (message instanceof HttpRequest) {
218             HttpRequest req = (HttpRequest) message;
219             if (HttpMethod.GET.equals(req.method()) &&
220                     h.contains(HttpHeaderNames.SEC_WEBSOCKET_KEY1) &&
221                     h.contains(HttpHeaderNames.SEC_WEBSOCKET_KEY2)) {
222                 return 8;
223             }
224         } else if (message instanceof HttpResponse) {
225             HttpResponse res = (HttpResponse) message;
226             if (res.status().code() == 101 &&
227                     h.contains(HttpHeaderNames.SEC_WEBSOCKET_ORIGIN) &&
228                     h.contains(HttpHeaderNames.SEC_WEBSOCKET_LOCATION)) {
229                 return 16;
230             }
231         }
232 
233         // Not a web socket message
234         return -1;
235     }

248     /**
249      * Returns {@code true} if and only if the specified message contains the
250      * {@code "Expect: 100-continue"} header.
251      */
252     public static boolean is100ContinueExpected(HttpMessage message) {
253         // Expect: 100-continue is for requests only.
254         if (!(message instanceof HttpRequest)) {
255             return false;
256         }
257 
258         // It works only on HTTP/1.1 or later.
259         if (message.protocolVersion().compareTo(HttpVersion.HTTP_1_1) < 0) {
260             return false;
261         }
262 
263         // In most cases, there will be one or zero 'Expect' header.
264         CharSequence value = message.headers().get(HttpHeaderNames.EXPECT);
265         if (value == null) {
266             return false;
267         }
268         if (HttpHeaderValues.CONTINUE.contentEqualsIgnoreCase(value)) {
269             return true;
270         }
271 
272         // Multiple 'Expect' headers.  Search through them.
273         return message.headers().contains(HttpHeaderNames.EXPECT, HttpHeaderValues.CONTINUE, true);
274     }

301     /**
302      * Set the {@link HttpHeaderNames#TRANSFER_ENCODING} to either include {@link HttpHeaderValues#CHUNKED} if
303      * {@code chunked} is {@code true}, or remove {@link HttpHeaderValues#CHUNKED} if {@code chunked} is {@code false}.
304      * @param m The message which contains the headers to modify.
305      * @param chunked if {@code true} then include {@link HttpHeaderValues#CHUNKED} in the headers. otherwise remove
306      * {@link HttpHeaderValues#CHUNKED} from the headers.
307      */
308     public static void setTransferEncodingChunked(HttpMessage m, boolean chunked) {
309         if (chunked) {
310             m.headers().add(HttpHeaderNames.TRANSFER_ENCODING, HttpHeaderValues.CHUNKED);
311             m.headers().remove(HttpHeaderNames.CONTENT_LENGTH);
312         } else {
313             List<String> encodings = m.headers().getAll(HttpHeaderNames.TRANSFER_ENCODING);
314             if (encodings.isEmpty()) {
315                 return;
316             }
317             List<CharSequence> values = new ArrayList<CharSequence>(encodings);
318             Iterator<CharSequence> valuesIt = values.iterator();
319             while (valuesIt.hasNext()) {
320                 CharSequence value = valuesIt.next();
321                 if (HttpHeaderValues.CHUNKED.contentEqualsIgnoreCase(value)) {
322                     valuesIt.remove();
323                 }
324             }
325             if (values.isEmpty()) {
326                 m.headers().remove(HttpHeaderNames.TRANSFER_ENCODING);
327             } else {
328                 m.headers().set(HttpHeaderNames.TRANSFER_ENCODING, values);
329             }
330         }
331     }

349     public static Charset getCharset(HttpMessage message, Charset defaultCharset) {

351         if (charsetCharSequence != null) {
352             try {
353                 return Charset.forName(charsetCharSequence.toString());
354             } catch (UnsupportedCharsetException unsupportedException) {

356             }
357         } else {
358             return defaultCharset;

362     /**
363      * Fetch charset string from message's Content-Type header.
364      *
365      * A lot of sites/possibly clients have charset="CHARSET", for example charset="utf-8". Or "utf8" instead of "utf-8"
366      * This is not according to standard, but this method provide an ability to catch desired mistakes manually in code
367      *
368      * @return the charset string from message's Content-Type header or {@code null} if charset is not presented
369      */
370     public static CharSequence getCharsetAsString(HttpMessage message) {
371         CharSequence contentTypeValue = message.headers().get(HttpHeaderNames.CONTENT_TYPE);
372         if (contentTypeValue != null) {
373             int indexOfCharset = AsciiString.indexOfIgnoreCaseAscii(contentTypeValue, CHARSET_EQUALS, 0);
374             if (indexOfCharset != AsciiString.INDEX_NOT_FOUND) {
375                 int indexOfEncoding = indexOfCharset + CHARSET_EQUALS.length();
376                 if (indexOfEncoding < contentTypeValue.length()) {
377                     return contentTypeValue.subSequence(indexOfEncoding, contentTypeValue.length());
378                 }
379             }
380         }
381         return null;
382     }
