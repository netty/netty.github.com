33  /**
34   * A <a href="http://tools.ietf.org/html/rfc6265">RFC6265</a> compliant cookie encoder to be used client side, so
35   * only name=value pairs are sent.
36   *
37   * User-Agents are not supposed to interpret cookies, so, if present, {@link Cookie#rawValue()} will be used.
38   * Otherwise, {@link Cookie#value()} will be used unquoted.
39   *
40   * Note that multiple cookies are supposed to be sent at once in a single "Cookie" header.
41   *
42   * <pre>
43   * // Example
44   * {@link HttpRequest} req = ...;
45   * res.setHeader("Cookie", {@link ClientCookieEncoder}.encode("JSESSIONID", "1234"));
46   * </pre>
47   *
48   * @see ClientCookieDecoder
49   */
50  public final class ClientCookieEncoder extends CookieEncoder {
51  
52      /**
53       * Strict encoder that validates that name and value chars are in the valid scope and (for methods that accept
54       * multiple cookies) sorts cookies into order of decreasing path length, as specified in RFC6265.
55       */
56      public static final ClientCookieEncoder STRICT = new ClientCookieEncoder(true);
57  
58      /**
59       * Lax instance that doesn't validate name and value, and (for methods that accept multiple cookies) keeps
60       * cookies in the order in which they were given.
61       */
62      public static final ClientCookieEncoder LAX = new ClientCookieEncoder(false);
63  
64      private ClientCookieEncoder(boolean strict) {
65          super(strict);
66      }

68      /**
69       * Encodes the specified cookie into a Cookie header value.
70       *
71       * @param name
72       *            the cookie name
73       * @param value
74       *            the cookie value
75       * @return a Rfc6265 style Cookie header value
76       */
77      public String encode(String name, String value) {
78          return encode(new DefaultCookie(name, value));
79      }

81      /**
82       * Encodes the specified cookie into a Cookie header value.
83       *
84       * @param cookie the specified cookie
85       * @return a Rfc6265 style Cookie header value
86       */
87      public String encode(Cookie cookie) {
88          StringBuilder buf = stringBuilder();
89          encode(buf, checkNotNull(cookie, "cookie"));
90          return stripTrailingSeparator(buf);
91      }

93      /**
94       * Sort cookies into decreasing order of path length, breaking ties by sorting into increasing chronological
95       * order of creation time, as recommended by RFC 6265.
96       */
97      private static final Comparator<Cookie> COOKIE_COMPARATOR = new Comparator<Cookie>() {
98          @Override
99          public int compare(Cookie c1, Cookie c2) {
100             String path1 = c1.path();
101             String path2 = c2.path();
102             // Cookies with unspecified path default to the path of the request. We don't
103             // know the request path here, but we assume that the length of an unspecified
104             // path is longer than any specified path (i.e. pathless cookies come first),
105             // because setting cookies with a path longer than the request path is of
106             // limited use.
107             int len1 = path1 == null ? Integer.MAX_VALUE : path1.length();
108             int len2 = path2 == null ? Integer.MAX_VALUE : path2.length();
109             int diff = len2 - len1;
110             if (diff != 0) {
111                 return diff;
112             }
113             // Rely on Java's sort stability to retain creation order in cases where
114             // cookies have same path length.
115             return -1;
116         }
117     };

119     /**
120      * Encodes the specified cookies into a single Cookie header value.
121      *
122      * @param cookies
123      *            some cookies
124      * @return a Rfc6265 style Cookie header value, null if no cookies are passed.
125      */
126     public String encode(Cookie... cookies) {
127         if (checkNotNull(cookies, "cookies").length == 0) {
128             return null;
129         }
130 
131         StringBuilder buf = stringBuilder();
132         if (strict) {
133             if (cookies.length == 1) {
134                 encode(buf, cookies[0]);
135             } else {
136                 Cookie[] cookiesSorted = Arrays.copyOf(cookies, cookies.length);
137                 Arrays.sort(cookiesSorted, COOKIE_COMPARATOR);
138                 for (Cookie c : cookiesSorted) {
139                     encode(buf, c);
140                 }
141             }
142         } else {
143             for (Cookie c : cookies) {
144                 encode(buf, c);
145             }
146         }
147         return stripTrailingSeparatorOrNull(buf);
148     }

150     /**
151      * Encodes the specified cookies into a single Cookie header value.
152      *
153      * @param cookies
154      *            some cookies
155      * @return a Rfc6265 style Cookie header value, null if no cookies are passed.
156      */
157     public String encode(Collection<? extends Cookie> cookies) {
158         if (checkNotNull(cookies, "cookies").isEmpty()) {
159             return null;
160         }
161 
162         StringBuilder buf = stringBuilder();
163         if (strict) {
164             if (cookies.size() == 1) {
165                 encode(buf, cookies.iterator().next());
166             } else {
167                 Cookie[] cookiesSorted = cookies.toArray(new Cookie[cookies.size()]);
168                 Arrays.sort(cookiesSorted, COOKIE_COMPARATOR);
169                 for (Cookie c : cookiesSorted) {
170                     encode(buf, c);
171                 }
172             }
173         } else {
174             for (Cookie c : cookies) {
175                 encode(buf, c);
176             }
177         }
178         return stripTrailingSeparatorOrNull(buf);
179     }
180 
181     /**
182      * Encodes the specified cookies into a single Cookie header value.
183      *
184      * @param cookies some cookies
185      * @return a Rfc6265 style Cookie header value, null if no cookies are passed.
186      */
187     public String encode(Iterable<? extends Cookie> cookies) {
188         Iterator<? extends Cookie> cookiesIt = checkNotNull(cookies, "cookies").iterator();
189         if (!cookiesIt.hasNext()) {
190             return null;
191         }
192 
193         StringBuilder buf = stringBuilder();
194         if (strict) {
195             Cookie firstCookie = cookiesIt.next();
196             if (!cookiesIt.hasNext()) {
197                 encode(buf, firstCookie);
198             } else {
199                 List<Cookie> cookiesList = InternalThreadLocalMap.get().arrayList();
200                 cookiesList.add(firstCookie);
201                 while (cookiesIt.hasNext()) {
202                     cookiesList.add(cookiesIt.next());
203                 }
204                 Cookie[] cookiesSorted = cookiesList.toArray(new Cookie[cookiesList.size()]);
205                 Arrays.sort(cookiesSorted, COOKIE_COMPARATOR);
206                 for (Cookie c : cookiesSorted) {
207                     encode(buf, c);
208                 }
209             }
210         } else {
211             while (cookiesIt.hasNext()) {
212                 encode(buf, cookiesIt.next());
213             }
214         }
215         return stripTrailingSeparatorOrNull(buf);
216     }
217 
218     private void encode(StringBuilder buf, Cookie c) {
219         final String name = c.name();
220         final String value = c.value() != null ? c.value() : "";
221 
222         validateCookie(name, value);
223 
224         if (c.wrap()) {
225             addQuoted(buf, name, value);
226         } else {
227             add(buf, name, value);
228         }
229     }
