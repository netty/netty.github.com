30  import static io.netty.handler.codec.http.HttpMethod.*;

32  import static io.netty.util.ReferenceCountUtil.*;
33  import static io.netty.util.internal.ObjectUtil.checkNotNull;

35  /**
36   * Handles <a href="http://www.w3.org/TR/cors/">Cross Origin Resource Sharing</a> (CORS) requests.
37   * <p>
38   * This handler can be configured using a {@link CorsConfig}, please
39   * refer to this class for details about the configuration options available.
40   */
41  public class CorsHandler extends ChannelDuplexHandler {
42  
43      private static final InternalLogger logger = InternalLoggerFactory.getInstance(CorsHandler.class);
44      private static final String ANY_ORIGIN = "*";
45      private final CorsConfig config;
46  
47      private HttpRequest request;
48  
49      /**
50       * Creates a new instance with the specified {@link CorsConfig}.
51       */
52      public CorsHandler(final CorsConfig config) {
53          this.config = checkNotNull(config, "config");
54      }
55  
56      @Override
57      public void channelRead(final ChannelHandlerContext ctx, final Object msg) throws Exception {
58          if (config.isCorsSupportEnabled() && msg instanceof HttpRequest) {
59              request = (HttpRequest) msg;
60              if (isPreflightRequest(request)) {
61                  handlePreflight(ctx, request);
62                  return;
63              }
64              if (config.isShortCircuit() && !validateOrigin()) {
65                  forbidden(ctx, request);
66                  return;
67              }
68          }
69          ctx.fireChannelRead(msg);
70      }
71  
72      private void handlePreflight(final ChannelHandlerContext ctx, final HttpRequest request) {
73          final HttpResponse response = new DefaultFullHttpResponse(request.protocolVersion(), OK, true, true);
74          if (setOrigin(response)) {
75              setAllowMethods(response);
76              setAllowHeaders(response);
77              setAllowCredentials(response);
78              setMaxAge(response);
79              setPreflightHeaders(response);
80          }
81          release(request);
82          ctx.writeAndFlush(response).addListener(ChannelFutureListener.CLOSE);
83      }
84  
85      /**
86       * This is a non CORS specification feature which enables the setting of preflight
87       * response headers that might be required by intermediaries.
88       *
89       * @param response the HttpResponse to which the preflight response headers should be added.
90       */
91      private void setPreflightHeaders(final HttpResponse response) {
92          response.headers().add(config.preflightResponseHeaders());
93      }
94  
95      private boolean setOrigin(final HttpResponse response) {
96          final String origin = request.headers().get(HttpHeaderNames.ORIGIN);
97          if (origin != null) {
98              if ("null".equals(origin) && config.isNullOriginAllowed()) {
99                  setAnyOrigin(response);
100                 return true;
101             }
102             if (config.isAnyOriginSupported()) {
103                 if (config.isCredentialsAllowed()) {
104                     echoRequestOrigin(response);
105                     setVaryHeader(response);
106                 } else {
107                     setAnyOrigin(response);
108                 }
109                 return true;
110             }
111             if (config.origins().contains(origin)) {
112                 setOrigin(response, origin);
113                 setVaryHeader(response);
114                 return true;
115             }
116             logger.debug("Request origin [{}]] was not among the configured origins [{}]", origin, config.origins());
117         }
118         return false;
119     }
120 
121     private boolean validateOrigin() {
122         if (config.isAnyOriginSupported()) {
123             return true;
124         }
125 
126         final String origin = request.headers().get(HttpHeaderNames.ORIGIN);
127         if (origin == null) {
128             // Not a CORS request so we cannot validate it. It may be a non CORS request.
129             return true;
130         }
131 
132         if ("null".equals(origin) && config.isNullOriginAllowed()) {
133             return true;
134         }
135 
136         return config.origins().contains(origin);
137     }
138 
139     private void echoRequestOrigin(final HttpResponse response) {
140         setOrigin(response, request.headers().get(HttpHeaderNames.ORIGIN));
141     }
142 
143     private static void setVaryHeader(final HttpResponse response) {
144         response.headers().set(HttpHeaderNames.VARY, HttpHeaderNames.ORIGIN);
145     }
146 
147     private static void setAnyOrigin(final HttpResponse response) {
148         setOrigin(response, ANY_ORIGIN);
149     }
150 
151     private static void setOrigin(final HttpResponse response, final String origin) {
152         response.headers().set(HttpHeaderNames.ACCESS_CONTROL_ALLOW_ORIGIN, origin);
153     }
154 
155     private void setAllowCredentials(final HttpResponse response) {
156         if (config.isCredentialsAllowed()
157                 && !response.headers().get(HttpHeaderNames.ACCESS_CONTROL_ALLOW_ORIGIN).equals(ANY_ORIGIN)) {
158             response.headers().set(HttpHeaderNames.ACCESS_CONTROL_ALLOW_CREDENTIALS, "true");
159         }
160     }
161 
162     private static boolean isPreflightRequest(final HttpRequest request) {
163         final HttpHeaders headers = request.headers();
164         return request.method().equals(OPTIONS) &&
165                 headers.contains(HttpHeaderNames.ORIGIN) &&
166                 headers.contains(HttpHeaderNames.ACCESS_CONTROL_REQUEST_METHOD);
167     }
168 
169     private void setExposeHeaders(final HttpResponse response) {
170         if (!config.exposedHeaders().isEmpty()) {
171             response.headers().set(HttpHeaderNames.ACCESS_CONTROL_EXPOSE_HEADERS, config.exposedHeaders());
172         }
173     }
174 
175     private void setAllowMethods(final HttpResponse response) {
176         response.headers().set(HttpHeaderNames.ACCESS_CONTROL_ALLOW_METHODS, config.allowedRequestMethods());
177     }
178 
179     private void setAllowHeaders(final HttpResponse response) {
180         response.headers().set(HttpHeaderNames.ACCESS_CONTROL_ALLOW_HEADERS, config.allowedRequestHeaders());
181     }
182 
183     private void setMaxAge(final HttpResponse response) {
184         response.headers().set(HttpHeaderNames.ACCESS_CONTROL_MAX_AGE, config.maxAge());
185     }
186 
187     @Override
188     public void write(final ChannelHandlerContext ctx, final Object msg, final ChannelPromise promise)
189             throws Exception {
190         if (config.isCorsSupportEnabled() && msg instanceof HttpResponse) {
191             final HttpResponse response = (HttpResponse) msg;
192             if (setOrigin(response)) {
193                 setAllowCredentials(response);
194                 setAllowHeaders(response);
195                 setExposeHeaders(response);
196             }
197         }
198         ctx.writeAndFlush(msg, promise);
199     }
200 
201     private static void forbidden(final ChannelHandlerContext ctx, final HttpRequest request) {
202         ctx.writeAndFlush(new DefaultFullHttpResponse(request.protocolVersion(), FORBIDDEN))
203                 .addListener(ChannelFutureListener.CLOSE);
204         release(request);
205     }
