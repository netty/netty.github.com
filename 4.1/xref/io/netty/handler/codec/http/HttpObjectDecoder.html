34  /**
35   * Decodes {@link ByteBuf}s into {@link HttpMessage}s and
36   * {@link HttpContent}s.
37   *
38   * <h3>Parameters that prevents excessive memory consumption</h3>
39   * <table border="1">
40   * <tr>
41   * <th>Name</th><th>Default value</th><th>Meaning</th>
42   * </tr>
43   * <tr>
44   * <td>{@code maxInitialLineLength}</td>
45   * <td>{@value #DEFAULT_MAX_INITIAL_LINE_LENGTH}</td>
46   * <td>The maximum length of the initial line
47   *     (e.g. {@code "GET / HTTP/1.0"} or {@code "HTTP/1.0 200 OK"})
48   *     If the length of the initial line exceeds this value, a
49   *     {@link TooLongFrameException} will be raised.</td>
50   * </tr>
51   * <tr>
52   * <td>{@code maxHeaderSize}</td>
53   * <td>{@value #DEFAULT_MAX_HEADER_SIZE}</td>
54   * <td>The maximum length of all headers.  If the sum of the length of each
55   *     header exceeds this value, a {@link TooLongFrameException} will be raised.</td>
56   * </tr>
57   * <tr>
58   * <td>{@code maxChunkSize}</td>
59   * <td>{@value #DEFAULT_MAX_CHUNK_SIZE}</td>
60   * <td>The maximum length of the content or each chunk.  If the content length
61   *     (or the length of each chunk) exceeds this value, the content or chunk
62   *     will be split into multiple {@link HttpContent}s whose length is
63   *     {@code maxChunkSize} at maximum.</td>
64   * </tr>
65   * </table>
66   *
67   * <h3>Parameters that control parsing behavior</h3>
68   * <table border="1">
69   * <tr>
70   * <th>Name</th><th>Default value</th><th>Meaning</th>
71   * </tr>
72   * <tr>
73   * <td>{@code allowDuplicateContentLengths}</td>
74   * <td>{@value #DEFAULT_ALLOW_DUPLICATE_CONTENT_LENGTHS}</td>
75   * <td>When set to {@code false}, will reject any messages that contain multiple Content-Length header fields.
76   *     When set to {@code true}, will allow multiple Content-Length headers only if they are all the same decimal value.
77   *     The duplicated field-values will be replaced with a single valid Content-Length field.
78   *     See <a href="https://tools.ietf.org/html/rfc7230#section-3.3.2">RFC 7230, Section 3.3.2</a>.</td>
79   * </tr>
80   * </table>
81   *
82   * <h3>Chunked Content</h3>
83   *
84   * If the content of an HTTP message is greater than {@code maxChunkSize} or
85   * the transfer encoding of the HTTP message is 'chunked', this decoder
86   * generates one {@link HttpMessage} instance and its following
87   * {@link HttpContent}s per single HTTP message to avoid excessive memory
88   * consumption. For example, the following HTTP message:
89   * <pre>
90   * GET / HTTP/1.1
91   * Transfer-Encoding: chunked
92   *
93   * 1a
94   * abcdefghijklmnopqrstuvwxyz
95   * 10
96   * 1234567890abcdef
97   * 0
98   * Content-MD5: ...
99   * <i>[blank line]</i>
100  * </pre>
101  * triggers {@link HttpRequestDecoder} to generate 3 objects:
102  * <ol>
103  * <li>An {@link HttpRequest},</li>
104  * <li>The first {@link HttpContent} whose content is {@code 'abcdefghijklmnopqrstuvwxyz'},</li>
105  * <li>The second {@link LastHttpContent} whose content is {@code '1234567890abcdef'}, which marks
106  * the end of the content.</li>
107  * </ol>
108  *
109  * If you prefer not to handle {@link HttpContent}s by yourself for your
110  * convenience, insert {@link HttpObjectAggregator} after this decoder in the
111  * {@link ChannelPipeline}.  However, please note that your server might not
112  * be as memory efficient as without the aggregator.
113  *
114  * <h3>Extensibility</h3>
115  *
116  * Please note that this decoder is designed to be extended to implement
117  * a protocol derived from HTTP, such as
118  * <a href="https://en.wikipedia.org/wiki/Real_Time_Streaming_Protocol">RTSP</a> and
119  * <a href="https://en.wikipedia.org/wiki/Internet_Content_Adaptation_Protocol">ICAP</a>.
120  * To implement the decoder of such a derived protocol, extend this class and
121  * implement all abstract methods properly.
122  */
123 public abstract class HttpObjectDecoder extends ByteToMessageDecoder {

128     public static final boolean DEFAULT_VALIDATE_HEADERS = true;
129     public static final int DEFAULT_INITIAL_BUFFER_SIZE = 128;
130     public static final boolean DEFAULT_ALLOW_DUPLICATE_CONTENT_LENGTHS = false;

211     protected HttpObjectDecoder(
212             int maxInitialLineLength, int maxHeaderSize, int maxChunkSize,
213             boolean chunkedSupported, boolean validateHeaders, int initialBufferSize,
214             boolean allowDuplicateContentLengths) {
215         checkPositive(maxInitialLineLength, "maxInitialLineLength");
216         checkPositive(maxHeaderSize, "maxHeaderSize");
217         checkPositive(maxChunkSize, "maxChunkSize");
218 
219         AppendableCharSequence seq = new AppendableCharSequence(initialBufferSize);
220         lineParser = new LineParser(seq, maxInitialLineLength);
221         headerParser = new HeaderParser(seq, maxHeaderSize);
222         this.maxChunkSize = maxChunkSize;
223         this.chunkedSupported = chunkedSupported;
224         this.validateHeaders = validateHeaders;
225         this.allowDuplicateContentLengths = allowDuplicateContentLengths;
226     }

228     @Override
229     protected void decode(ChannelHandlerContext ctx, ByteBuf buffer, List<Object> out) throws Exception {
230         if (resetRequested) {
231             resetNow();
232         }
233 
234         switch (currentState) {
235         case SKIP_CONTROL_CHARS:
236             // Fall-through
237         case READ_INITIAL: try {
238             AppendableCharSequence line = lineParser.parse(buffer);
239             if (line == null) {
240                 return;
241             }
242             String[] initialLine = splitInitialLine(line);
243             if (initialLine.length < 3) {
244                 // Invalid initial line - ignore.
245                 currentState = State.SKIP_CONTROL_CHARS;
246                 return;
247             }
248 
249             message = createMessage(initialLine);
250             currentState = State.READ_HEADER;
251             // fall-through
252         } catch (Exception e) {
253             out.add(invalidMessage(buffer, e));
254             return;
255         }
256         case READ_HEADER: try {
257             State nextState = readHeaders(buffer);
258             if (nextState == null) {
259                 return;
260             }
261             currentState = nextState;
262             switch (nextState) {
263             case SKIP_CONTROL_CHARS:
264                 // fast-path
265                 // No content is expected.
266                 out.add(message);
267                 out.add(LastHttpContent.EMPTY_LAST_CONTENT);
268                 resetNow();
269                 return;
270             case READ_CHUNK_SIZE:
271                 if (!chunkedSupported) {
272                     throw new IllegalArgumentException("Chunked messages not supported");
273                 }
274                 // Chunked encoding - generate HttpMessage first.  HttpChunks will follow.
275                 out.add(message);
276                 return;
277             default:
278                 /**
279                  * <a href="https://tools.ietf.org/html/rfc7230#section-3.3.3">RFC 7230, 3.3.3</a> states that if a
280                  * request does not have either a transfer-encoding or a content-length header then the message body
281                  * length is 0. However for a response the body length is the number of octets received prior to the
282                  * server closing the connection. So we treat this as variable length chunked encoding.
283                  */
284                 long contentLength = contentLength();
285                 if (contentLength == 0 || contentLength == -1 && isDecodingRequest()) {
286                     out.add(message);
287                     out.add(LastHttpContent.EMPTY_LAST_CONTENT);
288                     resetNow();
289                     return;
290                 }
291 
292                 assert nextState == State.READ_FIXED_LENGTH_CONTENT ||
293                         nextState == State.READ_VARIABLE_LENGTH_CONTENT;
294 
295                 out.add(message);
296 
297                 if (nextState == State.READ_FIXED_LENGTH_CONTENT) {
298                     // chunkSize will be decreased as the READ_FIXED_LENGTH_CONTENT state reads data chunk by chunk.
299                     chunkSize = contentLength;
300                 }
301 
302                 // We return here, this forces decode to be called again where we will decode the content
303                 return;
304             }
305         } catch (Exception e) {
306             out.add(invalidMessage(buffer, e));
307             return;
308         }
309         case READ_VARIABLE_LENGTH_CONTENT: {
310             // Keep reading data as a chunk until the end of connection is reached.
311             int toRead = Math.min(buffer.readableBytes(), maxChunkSize);
312             if (toRead > 0) {
313                 ByteBuf content = buffer.readRetainedSlice(toRead);
314                 out.add(new DefaultHttpContent(content));
315             }
316             return;
317         }
318         case READ_FIXED_LENGTH_CONTENT: {
319             int readLimit = buffer.readableBytes();
320 
321             // Check if the buffer is readable first as we use the readable byte count
322             // to create the HttpChunk. This is needed as otherwise we may end up with
323             // create an HttpChunk instance that contains an empty buffer and so is
324             // handled like it is the last HttpChunk.
325             //
326             // See https://github.com/netty/netty/issues/433
327             if (readLimit == 0) {
328                 return;
329             }
330 
331             int toRead = Math.min(readLimit, maxChunkSize);
332             if (toRead > chunkSize) {
333                 toRead = (int) chunkSize;
334             }
335             ByteBuf content = buffer.readRetainedSlice(toRead);
336             chunkSize -= toRead;
337 
338             if (chunkSize == 0) {
339                 // Read all content.
340                 out.add(new DefaultLastHttpContent(content, validateHeaders));
341                 resetNow();
342             } else {
343                 out.add(new DefaultHttpContent(content));
344             }
345             return;
346         }
347         /**
348          * everything else after this point takes care of reading chunked content. basically, read chunk size,
349          * read chunk, read and ignore the CRLF and repeat until 0
350          */
351         case READ_CHUNK_SIZE: try {
352             AppendableCharSequence line = lineParser.parse(buffer);
353             if (line == null) {
354                 return;
355             }
356             int chunkSize = getChunkSize(line.toString());
357             this.chunkSize = chunkSize;
358             if (chunkSize == 0) {
359                 currentState = State.READ_CHUNK_FOOTER;
360                 return;
361             }
362             currentState = State.READ_CHUNKED_CONTENT;
363             // fall-through
364         } catch (Exception e) {
365             out.add(invalidChunk(buffer, e));
366             return;
367         }
368         case READ_CHUNKED_CONTENT: {
369             assert chunkSize <= Integer.MAX_VALUE;
370             int toRead = Math.min((int) chunkSize, maxChunkSize);
371             toRead = Math.min(toRead, buffer.readableBytes());
372             if (toRead == 0) {
373                 return;
374             }
375             HttpContent chunk = new DefaultHttpContent(buffer.readRetainedSlice(toRead));
376             chunkSize -= toRead;
377 
378             out.add(chunk);
379 
380             if (chunkSize != 0) {
381                 return;
382             }
383             currentState = State.READ_CHUNK_DELIMITER;
384             // fall-through
385         }
386         case READ_CHUNK_DELIMITER: {
387             final int wIdx = buffer.writerIndex();
388             int rIdx = buffer.readerIndex();
389             while (wIdx > rIdx) {
390                 byte next = buffer.getByte(rIdx++);
391                 if (next == HttpConstants.LF) {
392                     currentState = State.READ_CHUNK_SIZE;
393                     break;
394                 }
395             }
396             buffer.readerIndex(rIdx);
397             return;
398         }
399         case READ_CHUNK_FOOTER: try {
400             LastHttpContent trailer = readTrailingHeaders(buffer);
401             if (trailer == null) {
402                 return;
403             }
404             out.add(trailer);
405             resetNow();
406             return;
407         } catch (Exception e) {
408             out.add(invalidChunk(buffer, e));
409             return;
410         }
411         case BAD_MESSAGE: {
412             // Keep discarding until disconnection.
413             buffer.skipBytes(buffer.readableBytes());
414             break;
415         }
416         case UPGRADED: {
417             int readableBytes = buffer.readableBytes();
418             if (readableBytes > 0) {
419                 // Keep on consuming as otherwise we may trigger an DecoderException,
420                 // other handler will replace this codec with the upgraded protocol codec to
421                 // take the traffic over at some point then.
422                 // See https://github.com/netty/netty/issues/2173
423                 out.add(buffer.readBytes(readableBytes));
424             }
425             break;
426         }
427         }
428     }

430     @Override
431     protected void decodeLast(ChannelHandlerContext ctx, ByteBuf in, List<Object> out) throws Exception {
432         super.decodeLast(ctx, in, out);
433 
434         if (resetRequested) {
435             // If a reset was requested by decodeLast() we need to do it now otherwise we may produce a
436             // LastHttpContent while there was already one.
437             resetNow();
438         }
439         // Handle the last unfinished message.
440         if (message != null) {
441             boolean chunked = HttpUtil.isTransferEncodingChunked(message);
442             if (currentState == State.READ_VARIABLE_LENGTH_CONTENT && !in.isReadable() && !chunked) {
443                 // End of connection.
444                 out.add(LastHttpContent.EMPTY_LAST_CONTENT);
445                 resetNow();
446                 return;
447             }
448 
449             if (currentState == State.READ_HEADER) {
450                 // If we are still in the state of reading headers we need to create a new invalid message that
451                 // signals that the connection was closed before we received the headers.
452                 out.add(invalidMessage(Unpooled.EMPTY_BUFFER,
453                         new PrematureChannelClosureException("Connection closed before received headers")));
454                 resetNow();
455                 return;
456             }
457 
458             // Check if the closure of the connection signifies the end of the content.
459             boolean prematureClosure;
460             if (isDecodingRequest() || chunked) {
461                 // The last request did not wait for a response.
462                 prematureClosure = true;
463             } else {
464                 // Compare the length of the received content and the 'Content-Length' header.
465                 // If the 'Content-Length' header is absent, the length of the content is determined by the end of the
466                 // connection, so it is perfectly fine.
467                 prematureClosure = contentLength() > 0;
468             }
469 
470             if (!prematureClosure) {
471                 out.add(LastHttpContent.EMPTY_LAST_CONTENT);
472             }
473             resetNow();
474         }
475     }

477     @Override
478     public void userEventTriggered(ChannelHandlerContext ctx, Object evt) throws Exception {
479         if (evt instanceof HttpExpectationFailedEvent) {
480             switch (currentState) {
481             case READ_FIXED_LENGTH_CONTENT:
482             case READ_VARIABLE_LENGTH_CONTENT:
483             case READ_CHUNK_SIZE:
484                 reset();
485                 break;
486             default:
487                 break;
488             }
489         }
490         super.userEventTriggered(ctx, evt);
491     }

493     protected boolean isContentAlwaysEmpty(HttpMessage msg) {
494         if (msg instanceof HttpResponse) {
495             HttpResponse res = (HttpResponse) msg;
496             int code = res.status().code();
497 
498             // Correctly handle return codes of 1xx.
499             //
500             // See:
501             //     - https://www.w3.org/Protocols/rfc2616/rfc2616-sec4.html Section 4.4
502             //     - https://github.com/netty/netty/issues/222
503             if (code >= 100 && code < 200) {
504                 // One exception: Hixie 76 websocket handshake response
505                 return !(code == 101 && !res.headers().contains(HttpHeaderNames.SEC_WEBSOCKET_ACCEPT)
506                          && res.headers().contains(HttpHeaderNames.UPGRADE, HttpHeaderValues.WEBSOCKET, true));
507             }
508 
509             switch (code) {
510             case 204: case 304:
511                 return true;
512             }
513         }
514         return false;
515     }

517     /**
518      * Returns true if the server switched to a different protocol than HTTP/1.0 or HTTP/1.1, e.g. HTTP/2 or Websocket.
519      * Returns false if the upgrade happened in a different layer, e.g. upgrade from HTTP/1.1 to HTTP/1.1 over TLS.
520      */
521     protected boolean isSwitchingToNonHttp1Protocol(HttpResponse msg) {
522         if (msg.status().code() != HttpResponseStatus.SWITCHING_PROTOCOLS.code()) {
523             return false;
524         }
525         String newProtocol = msg.headers().get(HttpHeaderNames.UPGRADE);
526         return newProtocol == null ||
527                 !newProtocol.contains(HttpVersion.HTTP_1_0.text()) &&
528                 !newProtocol.contains(HttpVersion.HTTP_1_1.text());
529     }

531     /**
532      * Resets the state of the decoder so that it is ready to decode a new message.
533      * This method is useful for handling a rejected request with {@code Expect: 100-continue} header.
534      */
535     public void reset() {
536         resetRequested = true;
537     }

539     private void resetNow() {
540         HttpMessage message = this.message;
541         this.message = null;
542         name = null;
543         value = null;
544         contentLength = Long.MIN_VALUE;
545         lineParser.reset();
546         headerParser.reset();
547         trailer = null;
548         if (!isDecodingRequest()) {
549             HttpResponse res = (HttpResponse) message;
550             if (res != null && isSwitchingToNonHttp1Protocol(res)) {
551                 currentState = State.UPGRADED;
552                 return;
553             }
554         }
555 
556         resetRequested = false;
557         currentState = State.SKIP_CONTROL_CHARS;
558     }

560     private HttpMessage invalidMessage(ByteBuf in, Exception cause) {
561         currentState = State.BAD_MESSAGE;
562 
563         // Advance the readerIndex so that ByteToMessageDecoder does not complain
564         // when we produced an invalid message without consuming anything.
565         in.skipBytes(in.readableBytes());
566 
567         if (message == null) {
568             message = createInvalidMessage();
569         }
570         message.setDecoderResult(DecoderResult.failure(cause));
571 
572         HttpMessage ret = message;
573         message = null;
574         return ret;
575     }
576 
577     private HttpContent invalidChunk(ByteBuf in, Exception cause) {
578         currentState = State.BAD_MESSAGE;
579 
580         // Advance the readerIndex so that ByteToMessageDecoder does not complain
581         // when we produced an invalid message without consuming anything.
582         in.skipBytes(in.readableBytes());
583 
584         HttpContent chunk = new DefaultLastHttpContent(Unpooled.EMPTY_BUFFER);
585         chunk.setDecoderResult(DecoderResult.failure(cause));
586         message = null;
587         trailer = null;
588         return chunk;
589     }

591     private State readHeaders(ByteBuf buffer) {
592         final HttpMessage message = this.message;
593         final HttpHeaders headers = message.headers();
594 
595         AppendableCharSequence line = headerParser.parse(buffer);
596         if (line == null) {
597             return null;
598         }
599         if (line.length() > 0) {
600             do {
601                 char firstChar = line.charAtUnsafe(0);
602                 if (name != null && (firstChar == ' ' || firstChar == '\t')) {
603                     //please do not make one line from below code
604                     //as it breaks +XX:OptimizeStringConcat optimization
605                     String trimmedLine = line.toString().trim();
606                     String valueStr = String.valueOf(value);
607                     value = valueStr + ' ' + trimmedLine;
608                 } else {
609                     if (name != null) {
610                         headers.add(name, value);
611                     }
612                     splitHeader(line);
613                 }
614 
615                 line = headerParser.parse(buffer);
616                 if (line == null) {
617                     return null;
618                 }
619             } while (line.length() > 0);
620         }
621 
622         // Add the last header.
623         if (name != null) {
624             headers.add(name, value);
625         }
626 
627         // reset name and value fields
628         name = null;
629         value = null;
630 
631         // Done parsing initial line and headers. Set decoder result.
632         HttpMessageDecoderResult decoderResult = new HttpMessageDecoderResult(lineParser.size, headerParser.size);
633         message.setDecoderResult(decoderResult);
634 
635         List<String> contentLengthFields = headers.getAll(HttpHeaderNames.CONTENT_LENGTH);
636         if (!contentLengthFields.isEmpty()) {
637             HttpVersion version = message.protocolVersion();
638             boolean isHttp10OrEarlier = version.majorVersion() < 1 || (version.majorVersion() == 1
639                     && version.minorVersion() == 0);
640             // Guard against multiple Content-Length headers as stated in
641             // https://tools.ietf.org/html/rfc7230#section-3.3.2:
642             contentLength = HttpUtil.normalizeAndGetContentLength(contentLengthFields,
643                     isHttp10OrEarlier, allowDuplicateContentLengths);
644             if (contentLength != -1) {
645                 headers.set(HttpHeaderNames.CONTENT_LENGTH, contentLength);
646             }
647         }
648 
649         if (isContentAlwaysEmpty(message)) {
650             HttpUtil.setTransferEncodingChunked(message, false);
651             return State.SKIP_CONTROL_CHARS;
652         } else if (HttpUtil.isTransferEncodingChunked(message)) {
653             if (!contentLengthFields.isEmpty() && message.protocolVersion() == HttpVersion.HTTP_1_1) {
654                 handleTransferEncodingChunkedWithContentLength(message);
655             }
656             return State.READ_CHUNK_SIZE;
657         } else if (contentLength() >= 0) {
658             return State.READ_FIXED_LENGTH_CONTENT;
659         } else {
660             return State.READ_VARIABLE_LENGTH_CONTENT;
661         }
662     }

664     /**
665      * Invoked when a message with both a "Transfer-Encoding: chunked" and a "Content-Length" header field is detected.

697     private LastHttpContent readTrailingHeaders(ByteBuf buffer) {
698         AppendableCharSequence line = headerParser.parse(buffer);
699         if (line == null) {
700             return null;
701         }
702         LastHttpContent trailer = this.trailer;
703         if (line.length() == 0 && trailer == null) {
704             // We have received the empty line which signals the trailer is complete and did not parse any trailers
705             // before. Just return an empty last content to reduce allocations.
706             return LastHttpContent.EMPTY_LAST_CONTENT;
707         }
708 
709         CharSequence lastHeader = null;
710         if (trailer == null) {
711             trailer = this.trailer = new DefaultLastHttpContent(Unpooled.EMPTY_BUFFER, validateHeaders);
712         }
713         while (line.length() > 0) {
714             char firstChar = line.charAtUnsafe(0);
715             if (lastHeader != null && (firstChar == ' ' || firstChar == '\t')) {
716                 List<String> current = trailer.trailingHeaders().getAll(lastHeader);
717                 if (!current.isEmpty()) {
718                     int lastPos = current.size() - 1;
719                     //please do not make one line from below code
720                     //as it breaks +XX:OptimizeStringConcat optimization
721                     String lineTrimmed = line.toString().trim();
722                     String currentLastPos = current.get(lastPos);
723                     current.set(lastPos, currentLastPos + lineTrimmed);
724                 }
725             } else {
726                 splitHeader(line);
727                 CharSequence headerName = name;
728                 if (!HttpHeaderNames.CONTENT_LENGTH.contentEqualsIgnoreCase(headerName) &&
729                         !HttpHeaderNames.TRANSFER_ENCODING.contentEqualsIgnoreCase(headerName) &&
730                         !HttpHeaderNames.TRAILER.contentEqualsIgnoreCase(headerName)) {
731                     trailer.trailingHeaders().add(headerName, value);
732                 }
733                 lastHeader = name;
734                 // reset name and value fields
735                 name = null;
736                 value = null;
737             }
738             line = headerParser.parse(buffer);
739             if (line == null) {
740                 return null;
741             }
742         }
743 
744         this.trailer = null;
745         return trailer;
746     }

788     private void splitHeader(AppendableCharSequence sb) {
789         final int length = sb.length();
790         int nameStart;
791         int nameEnd;
792         int colonEnd;
793         int valueStart;
794         int valueEnd;
795 
796         nameStart = findNonWhitespace(sb, 0, false);
797         for (nameEnd = nameStart; nameEnd < length; nameEnd ++) {
798             char ch = sb.charAtUnsafe(nameEnd);
799             // https://tools.ietf.org/html/rfc7230#section-3.2.4
800             //
801             // No whitespace is allowed between the header field-name and colon. In
802             // the past, differences in the handling of such whitespace have led to
803             // security vulnerabilities in request routing and response handling. A
804             // server MUST reject any received request message that contains
805             // whitespace between a header field-name and colon with a response code
806             // of 400 (Bad Request). A proxy MUST remove any such whitespace from a
807             // response message before forwarding the message downstream.
808             if (ch == ':' ||
809                     // In case of decoding a request we will just continue processing and header validation
810                     // is done in the DefaultHttpHeaders implementation.
811                     //
812                     // In the case of decoding a response we will "skip" the whitespace.
813                     (!isDecodingRequest() && isOWS(ch))) {
814                 break;
815             }
816         }
817 
818         if (nameEnd == length) {
819             // There was no colon present at all.
820             throw new IllegalArgumentException("No colon found");
821         }
822 
823         for (colonEnd = nameEnd; colonEnd < length; colonEnd ++) {
824             if (sb.charAtUnsafe(colonEnd) == ':') {
825                 colonEnd ++;
826                 break;
827             }
828         }
829 
830         name = sb.subStringUnsafe(nameStart, nameEnd);
831         valueStart = findNonWhitespace(sb, colonEnd, true);
832         if (valueStart == length) {
833             value = EMPTY_VALUE;
834         } else {
835             valueEnd = findEndOfString(sb);
836             value = sb.subStringUnsafe(valueStart, valueEnd);
837         }
838     }

840     private static int findNonSPLenient(AppendableCharSequence sb, int offset) {
841         for (int result = offset; result < sb.length(); ++result) {
842             char c = sb.charAtUnsafe(result);
843             // See https://tools.ietf.org/html/rfc7230#section-3.5
844             if (isSPLenient(c)) {
845                 continue;
846             }
847             if (Character.isWhitespace(c)) {
848                 // Any other whitespace delimiter is invalid
849                 throw new IllegalArgumentException("Invalid separator");
850             }
851             return result;
852         }
853         return sb.length();
854     }

856     private static int findSPLenient(AppendableCharSequence sb, int offset) {
857         for (int result = offset; result < sb.length(); ++result) {
858             if (isSPLenient(sb.charAtUnsafe(result))) {
859                 return result;
860             }
861         }
862         return sb.length();
863     }

865     private static boolean isSPLenient(char c) {
866         // See https://tools.ietf.org/html/rfc7230#section-3.5
867         return c == ' ' || c == (char) 0x09 || c == (char) 0x0B || c == (char) 0x0C || c == (char) 0x0D;
868     }

870     private static int findNonWhitespace(AppendableCharSequence sb, int offset, boolean validateOWS) {
871         for (int result = offset; result < sb.length(); ++result) {
872             char c = sb.charAtUnsafe(result);
873             if (!Character.isWhitespace(c)) {
874                 return result;
875             } else if (validateOWS && !isOWS(c)) {
876                 // Only OWS is supported for whitespace
877                 throw new IllegalArgumentException("Invalid separator, only a single space or horizontal tab allowed," +
878                         " but received a '" + c + "' (0x" + Integer.toHexString(c) + ")");
879             }
880         }
881         return sb.length();
882     }

884     private static int findEndOfString(AppendableCharSequence sb) {
885         for (int result = sb.length() - 1; result > 0; --result) {
886             if (!Character.isWhitespace(sb.charAtUnsafe(result))) {
887                 return result + 1;
888             }
889         }
890         return 0;
891     }

893     private static boolean isOWS(char ch) {
894         return ch == ' ' || ch == (char) 0x09;

897     private static class HeaderParser implements ByteProcessor {
898         private final AppendableCharSequence seq;
899         private final int maxLength;
900         int size;
901 
902         HeaderParser(AppendableCharSequence seq, int maxLength) {
903             this.seq = seq;
904             this.maxLength = maxLength;
905         }
906 
907         public AppendableCharSequence parse(ByteBuf buffer) {
908             final int oldSize = size;
909             seq.reset();
910             int i = buffer.forEachByte(this);
911             if (i == -1) {
912                 size = oldSize;
913                 return null;
914             }
915             buffer.readerIndex(i + 1);
916             return seq;
917         }
918 
919         public void reset() {
920             size = 0;
921         }
922 
923         @Override
924         public boolean process(byte value) throws Exception {
925             char nextByte = (char) (value & 0xFF);
926             if (nextByte == HttpConstants.LF) {
927                 int len = seq.length();
928                 // Drop CR if we had a CRLF pair
929                 if (len >= 1 && seq.charAtUnsafe(len - 1) == HttpConstants.CR) {
930                     -- size;
931                     seq.setLength(len - 1);
932                 }
933                 return false;
934             }
935 
936             increaseCount();
937 
938             seq.append(nextByte);
939             return true;
940         }
941 
942         protected final void increaseCount() {
943             if (++ size > maxLength) {
944                 // TODO: Respond with Bad Request and discard the traffic
945                 //    or close the connection.
946                 //       No need to notify the upstream handlers - just log.
947                 //       If decoding a response, just throw an exception.
948                 throw newException(maxLength);
949             }
950         }
951 
952         protected TooLongFrameException newException(int maxLength) {
953             return new TooLongFrameException("HTTP header is larger than " + maxLength + " bytes.");
954         }
955     }
