31  /**
32   * Client-side handler for handling an HTTP upgrade handshake to another protocol. When the first
33   * HTTP request is sent, this handler will add all appropriate headers to perform an upgrade to the
34   * new protocol. If the upgrade fails (i.e. response is not 101 Switching Protocols), this handler
35   * simply removes itself from the pipeline. If the upgrade is successful, upgrades the pipeline to
36   * the new protocol.
37   */
38  public class HttpClientUpgradeHandler extends HttpObjectAggregator implements ChannelOutboundHandler {
39  
40      /**
41       * User events that are fired to notify about upgrade status.
42       */
43      public enum UpgradeEvent {
44          /**
45           * The Upgrade request was sent to the server.
46           */
47          UPGRADE_ISSUED,
48  
49          /**
50           * The Upgrade to the new protocol was successful.
51           */
52          UPGRADE_SUCCESSFUL,
53  
54          /**
55           * The Upgrade was unsuccessful due to the server not issuing
56           * with a 101 Switching Protocols response.
57           */
58          UPGRADE_REJECTED
59      }
60  
61      /**
62       * The source codec that is used in the pipeline initially.
63       */
64      public interface SourceCodec {
65          /**
66           * Removes this codec (i.e. all associated handlers) from the pipeline.
67           */
68          void upgradeFrom(ChannelHandlerContext ctx);
69      }
70  
71      /**
72       * A codec that the source can be upgraded to.
73       */
74      public interface UpgradeCodec {
75          /**
76           * Returns the name of the protocol supported by this codec, as indicated by the {@code 'UPGRADE'} header.
77           */
78          CharSequence protocol();
79  
80          /**
81           * Sets any protocol-specific headers required to the upgrade request. Returns the names of
82           * all headers that were added. These headers will be used to populate the CONNECTION header.
83           */
84          Collection<CharSequence> setUpgradeHeaders(ChannelHandlerContext ctx, HttpRequest upgradeRequest);
85  
86          /**
87           * Performs an HTTP protocol upgrade from the source codec. This method is responsible for
88           * adding all handlers required for the new protocol.
89           *
90           * @param ctx the context for the current handler.
91           * @param upgradeResponse the 101 Switching Protocols response that indicates that the server
92           *            has switched to this protocol.
93           */
94          void upgradeTo(ChannelHandlerContext ctx, FullHttpResponse upgradeResponse) throws Exception;
95      }
96  
97      private final SourceCodec sourceCodec;
98      private final UpgradeCodec upgradeCodec;
99      private boolean upgradeRequested;
100 
101     /**
102      * Constructs the client upgrade handler.
103      *
104      * @param sourceCodec the codec that is being used initially.
105      * @param upgradeCodec the codec that the client would like to upgrade to.
106      * @param maxContentLength the maximum length of the aggregated content.
107      */
108     public HttpClientUpgradeHandler(SourceCodec sourceCodec, UpgradeCodec upgradeCodec,
109                                     int maxContentLength) {
110         super(maxContentLength);
111         if (sourceCodec == null) {
112             throw new NullPointerException("sourceCodec");
113         }
114         if (upgradeCodec == null) {
115             throw new NullPointerException("upgradeCodec");
116         }
117         this.sourceCodec = sourceCodec;
118         this.upgradeCodec = upgradeCodec;
119     }

152     @Override
153     public void write(ChannelHandlerContext ctx, Object msg, ChannelPromise promise)
154             throws Exception {
155         if (!(msg instanceof HttpRequest)) {
156             ctx.write(msg, promise);
157             return;
158         }
159 
160         if (upgradeRequested) {
161             promise.setFailure(new IllegalStateException(
162                     "Attempting to write HTTP request with upgrade in progress"));
163             return;
164         }
165 
166         upgradeRequested = true;
167         setUpgradeRequestHeaders(ctx, (HttpRequest) msg);
168 
169         // Continue writing the request.
170         ctx.write(msg, promise);
171 
172         // Notify that the upgrade request was issued.
173         ctx.fireUserEventTriggered(UpgradeEvent.UPGRADE_ISSUED);
174         // Now we wait for the next HTTP response to see if we switch protocols.
175     }

178     public void flush(ChannelHandlerContext ctx) throws Exception {
179         ctx.flush();
180     }

182     @Override
183     protected void decode(ChannelHandlerContext ctx, HttpObject msg, List<Object> out)
184             throws Exception {
185         FullHttpResponse response = null;
186         try {
187             if (!upgradeRequested) {
188                 throw new IllegalStateException("Read HTTP response without requesting protocol switch");
189             }
190 
191             if (msg instanceof FullHttpResponse) {
192                 response = (FullHttpResponse) msg;
193                 // Need to retain since the base class will release after returning from this method.
194                 response.retain();
195                 out.add(response);
196             } else {
197                 // Call the base class to handle the aggregation of the full request.
198                 super.decode(ctx, msg, out);
199                 if (out.isEmpty()) {
200                     // The full request hasn't been created yet, still awaiting more data.
201                     return;
202                 }
203 
204                 assert out.size() == 1;
205                 response = (FullHttpResponse) out.get(0);
206             }
207 
208             if (!SWITCHING_PROTOCOLS.equals(response.status())) {
209                 // The server does not support the requested protocol, just remove this handler
210                 // and continue processing HTTP.
211                 // NOTE: not releasing the response since we're letting it propagate to the
212                 // next handler.
213                 ctx.fireUserEventTriggered(UpgradeEvent.UPGRADE_REJECTED);
214                 removeThisHandler(ctx);
215                 return;
216             }
217 
218             CharSequence upgradeHeader = response.headers().get(HttpHeaderNames.UPGRADE);
219             if (upgradeHeader != null && !AsciiString.contentEqualsIgnoreCase(upgradeCodec.protocol(), upgradeHeader)) {
220                 throw new IllegalStateException(
221                         "Switching Protocols response with unexpected UPGRADE protocol: " + upgradeHeader);
222             }
223 
224             // Upgrade to the new protocol.
225             sourceCodec.upgradeFrom(ctx);
226             upgradeCodec.upgradeTo(ctx, response);
227 
228             // Notify that the upgrade to the new protocol completed successfully.
229             ctx.fireUserEventTriggered(UpgradeEvent.UPGRADE_SUCCESSFUL);
230 
231             // We switched protocols, so we're done with the upgrade response.
232             // Release it and clear it from the output.
233             response.release();
234             out.clear();
235             removeThisHandler(ctx);
236         } catch (Throwable t) {
237             release(response);
238             ctx.fireExceptionCaught(t);
239             removeThisHandler(ctx);
240         }
241     }
242 
243     private static void removeThisHandler(ChannelHandlerContext ctx) {
244         ctx.pipeline().remove(ctx.name());
245     }
246 
247     /**
248      * Adds all upgrade request headers necessary for an upgrade to the supported protocols.
249      */
250     private void setUpgradeRequestHeaders(ChannelHandlerContext ctx, HttpRequest request) {
251         // Set the UPGRADE header on the request.
252         request.headers().set(HttpHeaderNames.UPGRADE, upgradeCodec.protocol());
253 
254         // Add all protocol-specific headers to the request.
255         Set<CharSequence> connectionParts = new LinkedHashSet<CharSequence>(2);
256         connectionParts.addAll(upgradeCodec.setUpgradeHeaders(ctx, request));
257 
258         // Set the CONNECTION header from the set of all protocol-specific headers that were added.
259         StringBuilder builder = new StringBuilder();
260         for (CharSequence part : connectionParts) {
261             builder.append(part);
262             builder.append(',');
263         }
264         builder.append(HttpHeaderValues.UPGRADE);
265         request.headers().set(HttpHeaderNames.CONNECTION, builder.toString());
266     }
