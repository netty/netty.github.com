173     public HttpPostMultipartRequestDecoder(HttpDataFactory factory, HttpRequest request, Charset charset) {
174         if (factory == null) {
175             throw new NullPointerException("factory");
176         }
177         if (request == null) {
178             throw new NullPointerException("request");
179         }
180         if (charset == null) {
181             throw new NullPointerException("charset");
182         }
183         this.request = request;
184         this.charset = charset;
185         this.factory = factory;
186         // Fill default values
187 
188         setMultipart(this.request.headers().get(HttpHeaderNames.CONTENT_TYPE));
189         if (request instanceof HttpContent) {
190             // Offer automatically if the given request is als type of HttpContent
191             // See #1089
192             offer((HttpContent) request);
193         } else {
194             undecodedChunk = buffer();
195             parseBody();
196         }
197     }

321     /**
322      * Initialized the internals from a new chunk
323      *
324      * @param content
325      *            the new received chunk
326      * @throws ErrorDataDecoderException
327      *             if there is a problem with the charset decoding or other
328      *             errors
329      */
330     @Override
331     public HttpPostMultipartRequestDecoder offer(HttpContent content) {
332         checkDestroyed();
333 
334         // Maybe we should better not copy here for performance reasons but this will need
335         // more care by the caller to release the content in a correct manner later
336         // So maybe something to optimize on a later stage
337         ByteBuf buf = content.content();
338         if (undecodedChunk == null) {
339             undecodedChunk = buf.copy();
340         } else {
341             undecodedChunk.writeBytes(buf);
342         }
343         if (content instanceof LastHttpContent) {
344             isLastChunk = true;
345         }
346         parseBody();
347         if (undecodedChunk != null && undecodedChunk.writerIndex() > discardThreshold) {
348             undecodedChunk.discardReadBytes();
349         }
350         return this;
351     }

354      * True if at current getStatus, there is an available decoded

356      *

359      * @return True if at current getStatus, there is a decoded InterfaceHttpData

365         checkDestroyed();
366 
367         if (currentStatus == MultiPartStatus.EPILOGUE) {
368             // OK except if end of list
369             if (bodyListHttpDataRank >= bodyListHttpData.size()) {
370                 throw new EndOfDataDecoderException();
371             }
372         }
373         return !bodyListHttpData.isEmpty() && bodyListHttpDataRank < bodyListHttpData.size();

410      * @throws ErrorDataDecoderException
411      *             if there is a problem with the charset decoding or other
412      *             errors

414     private void parseBody() {
415         if (currentStatus == MultiPartStatus.PREEPILOGUE || currentStatus == MultiPartStatus.EPILOGUE) {
416             if (isLastChunk) {
417                 currentStatus = MultiPartStatus.EPILOGUE;
418             }
419             return;
420         }
421         parseBodyMultipart();
422     }

424     /**
425      * Utility function to add a new decoded data
426      */
427     protected void addHttpData(InterfaceHttpData data) {
428         if (data == null) {
429             return;
430         }
431         List<InterfaceHttpData> datas = bodyMapHttpData.get(data.getName());
432         if (datas == null) {
433             datas = new ArrayList<InterfaceHttpData>(1);
434             bodyMapHttpData.put(data.getName(), datas);
435         }
436         datas.add(data);
437         bodyListHttpData.add(data);
438     }

440     /**
441      * Parse the Body for multipart
442      *
443      * @throws ErrorDataDecoderException
444      *             if there is a problem with the charset decoding or other
445      *             errors
446      */
447     private void parseBodyMultipart() {
448         if (undecodedChunk == null || undecodedChunk.readableBytes() == 0) {
449             // nothing to decode
450             return;
451         }
452         InterfaceHttpData data = decodeMultipart(currentStatus);
453         while (data != null) {
454             addHttpData(data);
455             if (currentStatus == MultiPartStatus.PREEPILOGUE || currentStatus == MultiPartStatus.EPILOGUE) {
456                 break;
457             }
458             data = decodeMultipart(currentStatus);
459         }
460     }

462     /**
463      * Decode a multipart request by pieces<br>
464      * <br>
465      * NOTSTARTED PREAMBLE (<br>
466      * (HEADERDELIMITER DISPOSITION (FIELD | FILEUPLOAD))*<br>
467      * (HEADERDELIMITER DISPOSITION MIXEDPREAMBLE<br>
468      * (MIXEDDELIMITER MIXEDDISPOSITION MIXEDFILEUPLOAD)+<br>
469      * MIXEDCLOSEDELIMITER)*<br>
470      * CLOSEDELIMITER)+ EPILOGUE<br>
471      *
472      * Inspired from HttpMessageDecoder
473      *
474      * @return the next decoded InterfaceHttpData or null if none until now.
475      * @throws ErrorDataDecoderException
476      *             if an error occurs
477      */
478     private InterfaceHttpData decodeMultipart(MultiPartStatus state) {
479         switch (state) {
480         case NOTSTARTED:
481             throw new ErrorDataDecoderException("Should not be called with the current getStatus");
482         case PREAMBLE:
483             // Content-type: multipart/form-data, boundary=AaB03x
484             throw new ErrorDataDecoderException("Should not be called with the current getStatus");
485         case HEADERDELIMITER: {
486             // --AaB03x or --AaB03x--
487             return findMultipartDelimiter(multipartDataBoundary, MultiPartStatus.DISPOSITION,
488                     MultiPartStatus.PREEPILOGUE);
489         }
490         case DISPOSITION: {
491             // content-disposition: form-data; name="field1"
492             // content-disposition: form-data; name="pics"; filename="file1.txt"
493             // and other immediate values like
494             // Content-type: image/gif
495             // Content-Type: text/plain
496             // Content-Type: text/plain; charset=ISO-8859-1
497             // Content-Transfer-Encoding: binary
498             // The following line implies a change of mode (mixed mode)
499             // Content-type: multipart/mixed, boundary=BbC04y
500             return findMultipartDisposition();
501         }
502         case FIELD: {
503             // Now get value according to Content-Type and Charset
504             Charset localCharset = null;
505             Attribute charsetAttribute = currentFieldAttributes.get(HttpHeaderValues.CHARSET);
506             if (charsetAttribute != null) {
507                 try {
508                     localCharset = Charset.forName(charsetAttribute.getValue());
509                 } catch (IOException e) {
510                     throw new ErrorDataDecoderException(e);
511                 }
512             }
513             Attribute nameAttribute = currentFieldAttributes.get(HttpHeaderValues.NAME);
514             if (currentAttribute == null) {
515                 Attribute lengthAttribute = currentFieldAttributes
516                         .get(HttpHeaderNames.CONTENT_LENGTH);
517                 long size;
518                 try {
519                     size = lengthAttribute != null? Long.parseLong(lengthAttribute
520                             .getValue()) : 0L;
521                 } catch (IOException e) {
522                     throw new ErrorDataDecoderException(e);
523                 } catch (NumberFormatException e) {
524                     size = 0;
525                 }
526                 try {
527                     if (size > 0) {
528                         currentAttribute = factory.createAttribute(request,
529                                 cleanString(nameAttribute.getValue()), size);
530                     } else {
531                         currentAttribute = factory.createAttribute(request,
532                                 cleanString(nameAttribute.getValue()));
533                     }
534                 } catch (NullPointerException e) {
535                     throw new ErrorDataDecoderException(e);
536                 } catch (IllegalArgumentException e) {
537                     throw new ErrorDataDecoderException(e);
538                 } catch (IOException e) {
539                     throw new ErrorDataDecoderException(e);
540                 }
541                 if (localCharset != null) {
542                     currentAttribute.setCharset(localCharset);
543                 }
544             }
545             // load data
546             try {
547                 loadFieldMultipart(multipartDataBoundary);
548             } catch (NotEnoughDataDecoderException ignored) {
549                 return null;
550             }
551             Attribute finalAttribute = currentAttribute;
552             currentAttribute = null;
553             currentFieldAttributes = null;
554             // ready to load the next one
555             currentStatus = MultiPartStatus.HEADERDELIMITER;
556             return finalAttribute;
557         }
558         case FILEUPLOAD: {
559             // eventually restart from existing FileUpload
560             return getFileUpload(multipartDataBoundary);
561         }
562         case MIXEDDELIMITER: {
563             // --AaB03x or --AaB03x--
564             // Note that currentFieldAttributes exists
565             return findMultipartDelimiter(multipartMixedBoundary, MultiPartStatus.MIXEDDISPOSITION,
566                     MultiPartStatus.HEADERDELIMITER);
567         }
568         case MIXEDDISPOSITION: {
569             return findMultipartDisposition();
570         }
571         case MIXEDFILEUPLOAD: {
572             // eventually restart from existing FileUpload
573             return getFileUpload(multipartMixedBoundary);
574         }
575         case PREEPILOGUE:
576             return null;
577         case EPILOGUE:
578             return null;
579         default:
580             throw new ErrorDataDecoderException("Shouldn't reach here.");
581         }
582     }

584     /**
585      * Skip control Characters
586      *
587      * @throws NotEnoughDataDecoderException
588      */
589     void skipControlCharacters() {
590         SeekAheadOptimize sao;
591         try {
592             sao = new SeekAheadOptimize(undecodedChunk);
593         } catch (SeekAheadNoBackArrayException ignored) {
594             try {
595                 skipControlCharactersStandard();
596             } catch (IndexOutOfBoundsException e1) {
597                 throw new NotEnoughDataDecoderException(e1);
598             }
599             return;
600         }
601 
602         while (sao.pos < sao.limit) {
603             char c = (char) (sao.bytes[sao.pos++] & 0xFF);
604             if (!Character.isISOControl(c) && !Character.isWhitespace(c)) {
605                 sao.setReadPosition(1);
606                 return;
607             }
608         }
609         throw new NotEnoughDataDecoderException("Access out of bounds");
610     }
611 
612     void skipControlCharactersStandard() {
613         for (;;) {
614             char c = (char) undecodedChunk.readUnsignedByte();
615             if (!Character.isISOControl(c) && !Character.isWhitespace(c)) {
616                 undecodedChunk.readerIndex(undecodedChunk.readerIndex() - 1);
617                 break;
618             }
619         }
620     }

622     /**
623      * Find the next Multipart Delimiter
624      *
625      * @param delimiter
626      *            delimiter to find
627      * @param dispositionStatus
628      *            the next getStatus if the delimiter is a start
629      * @param closeDelimiterStatus
630      *            the next getStatus if the delimiter is a close delimiter
631      * @return the next InterfaceHttpData if any
632      * @throws ErrorDataDecoderException
633      */
634     private InterfaceHttpData findMultipartDelimiter(String delimiter, MultiPartStatus dispositionStatus,
635             MultiPartStatus closeDelimiterStatus) {
636         // --AaB03x or --AaB03x--
637         int readerIndex = undecodedChunk.readerIndex();
638         try {
639             skipControlCharacters();
640         } catch (NotEnoughDataDecoderException ignored) {
641             undecodedChunk.readerIndex(readerIndex);
642             return null;
643         }
644         skipOneLine();
645         String newline;
646         try {
647             newline = readDelimiter(delimiter);
648         } catch (NotEnoughDataDecoderException ignored) {
649             undecodedChunk.readerIndex(readerIndex);
650             return null;
651         }
652         if (newline.equals(delimiter)) {
653             currentStatus = dispositionStatus;
654             return decodeMultipart(dispositionStatus);
655         }
656         if (newline.equals(delimiter + "--")) {
657             // CLOSEDELIMITER or MIXED CLOSEDELIMITER found
658             currentStatus = closeDelimiterStatus;
659             if (currentStatus == MultiPartStatus.HEADERDELIMITER) {
660                 // MIXEDCLOSEDELIMITER
661                 // end of the Mixed part
662                 currentFieldAttributes = null;
663                 return decodeMultipart(MultiPartStatus.HEADERDELIMITER);
664             }
665             return null;
666         }
667         undecodedChunk.readerIndex(readerIndex);
668         throw new ErrorDataDecoderException("No Multipart delimiter found");
669     }

671     /**
672      * Find the next Disposition
673      *
674      * @return the next InterfaceHttpData if any
675      * @throws ErrorDataDecoderException
676      */
677     private InterfaceHttpData findMultipartDisposition() {
678         int readerIndex = undecodedChunk.readerIndex();
679         if (currentStatus == MultiPartStatus.DISPOSITION) {
680             currentFieldAttributes = new TreeMap<CharSequence, Attribute>(CaseIgnoringComparator.INSTANCE);
681         }
682         // read many lines until empty line with newline found! Store all data
683         while (!skipOneLine()) {
684             String newline;
685             try {
686                 skipControlCharacters();
687                 newline = readLine();
688             } catch (NotEnoughDataDecoderException ignored) {
689                 undecodedChunk.readerIndex(readerIndex);
690                 return null;
691             }
692             String[] contents = splitMultipartHeader(newline);
693             if (HttpHeaderNames.CONTENT_DISPOSITION.contentEqualsIgnoreCase(contents[0])) {
694                 boolean checkSecondArg;
695                 if (currentStatus == MultiPartStatus.DISPOSITION) {
696                     checkSecondArg = HttpHeaderValues.FORM_DATA.contentEqualsIgnoreCase(contents[1]);
697                 } else {
698                     checkSecondArg = HttpHeaderValues.ATTACHMENT.contentEqualsIgnoreCase(contents[1])
699                             || HttpHeaderValues.FILE.contentEqualsIgnoreCase(contents[1]);
700                 }
701                 if (checkSecondArg) {
702                     // read next values and store them in the map as Attribute
703                     for (int i = 2; i < contents.length; i++) {
704                         String[] values = StringUtil.split(contents[i], '=', 2);
705                         Attribute attribute;
706                         try {
707                             String name = cleanString(values[0]);
708                             String value = values[1];
709 
710                             // See http://www.w3.org/Protocols/rfc2616/rfc2616-sec19.html
711                             if (HttpHeaderValues.FILENAME.contentEquals(name)) {
712                                 // filename value is quoted string so strip them
713                                 value = value.substring(1, value.length() - 1);
714                             } else {
715                                 // otherwise we need to clean the value
716                                 value = cleanString(value);
717                             }
718                             attribute = factory.createAttribute(request, name, value);
719                         } catch (NullPointerException e) {
720                             throw new ErrorDataDecoderException(e);
721                         } catch (IllegalArgumentException e) {
722                             throw new ErrorDataDecoderException(e);
723                         }
724                         currentFieldAttributes.put(attribute.getName(), attribute);
725                     }
726                 }
727             } else if (HttpHeaderNames.CONTENT_TRANSFER_ENCODING.contentEqualsIgnoreCase(contents[0])) {
728                 Attribute attribute;
729                 try {
730                     attribute = factory.createAttribute(request, HttpHeaderNames.CONTENT_TRANSFER_ENCODING.toString(),
731                             cleanString(contents[1]));
732                 } catch (NullPointerException e) {
733                     throw new ErrorDataDecoderException(e);
734                 } catch (IllegalArgumentException e) {
735                     throw new ErrorDataDecoderException(e);
736                 }
737 
738                 currentFieldAttributes.put(HttpHeaderNames.CONTENT_TRANSFER_ENCODING, attribute);

743                             cleanString(contents[1]));

752                 // Take care of possible "multipart/mixed"

754                     if (currentStatus == MultiPartStatus.DISPOSITION) {
755                         String values = StringUtil.substringAfter(contents[2], '=');

761                     }

769                                         cleanString(values));

782                                 throw new ErrorDataDecoderException(e);
783                             } catch (IllegalArgumentException e) {
784                                 throw new ErrorDataDecoderException(e);
785                             }

789                 }
790             } else {
791                 throw new ErrorDataDecoderException("Unknown Params: " + newline);
792             }
793         }

795         Attribute filenameAttribute = currentFieldAttributes.get(HttpHeaderValues.FILENAME);

811                 currentStatus = MultiPartStatus.MIXEDFILEUPLOAD;

829     protected InterfaceHttpData getFileUpload(String delimiter) {

832         Attribute encoding = currentFieldAttributes.get(HttpHeaderNames.CONTENT_TRANSFER_ENCODING);
833         Charset localCharset = charset;
834         // Default
835         TransferEncodingMechanism mechanism = TransferEncodingMechanism.BIT7;

839                 code = encoding.getValue().toLowerCase();
840             } catch (IOException e) {
841                 throw new ErrorDataDecoderException(e);
842             }
843             if (code.equals(HttpPostBodyUtil.TransferEncodingMechanism.BIT7.value())) {

848             } else if (code.equals(HttpPostBodyUtil.TransferEncodingMechanism.BINARY.value())) {
849                 // no real charset, so let the default
850                 mechanism = TransferEncodingMechanism.BINARY;
851             } else {
852                 throw new ErrorDataDecoderException("TransferEncoding Unknown: " + code);
853             }
854         }
855         Attribute charsetAttribute = currentFieldAttributes.get(HttpHeaderValues.CHARSET);
856         if (charsetAttribute != null) {

859             } catch (IOException e) {
860                 throw new ErrorDataDecoderException(e);
861             }
862         }
863         if (currentFileUpload == null) {
864             Attribute filenameAttribute = currentFieldAttributes.get(HttpHeaderValues.FILENAME);
865             Attribute nameAttribute = currentFieldAttributes.get(HttpHeaderValues.NAME);
866             Attribute contentTypeAttribute = currentFieldAttributes.get(HttpHeaderNames.CONTENT_TYPE);
867             Attribute lengthAttribute = currentFieldAttributes.get(HttpHeaderNames.CONTENT_LENGTH);
868             long size;
869             try {
870                 size = lengthAttribute != null ? Long.parseLong(lengthAttribute.getValue()) : 0L;
871             } catch (IOException e) {
872                 throw new ErrorDataDecoderException(e);
873             } catch (NumberFormatException ignored) {
874                 size = 0;
875             }
876             try {
877                 String contentType;
878                 if (contentTypeAttribute != null) {
879                     contentType = contentTypeAttribute.getValue();
880                 } else {
881                     contentType = HttpPostBodyUtil.DEFAULT_BINARY_CONTENT_TYPE;
882                 }
883                 currentFileUpload = factory.createFileUpload(request,
884                         cleanString(nameAttribute.getValue()), cleanString(filenameAttribute.getValue()),
885                         contentType, mechanism.value(), localCharset,
886                         size);
887             } catch (NullPointerException e) {
888                 throw new ErrorDataDecoderException(e);
889             } catch (IllegalArgumentException e) {
890                 throw new ErrorDataDecoderException(e);
891             } catch (IOException e) {
892                 throw new ErrorDataDecoderException(e);
893             }
894         }
895         // load data as much as possible
896         try {
897             readFileUploadByteMultipart(delimiter);
898         } catch (NotEnoughDataDecoderException e) {
899             // do not change the buffer position
900             // since some can be already saved into FileUpload
901             // So do not change the currentStatus
902             return null;
903         }
904         if (currentFileUpload.isCompleted()) {
905             // ready to load the next one
906             if (currentStatus == MultiPartStatus.FILEUPLOAD) {
907                 currentStatus = MultiPartStatus.HEADERDELIMITER;
908                 currentFieldAttributes = null;
909             } else {
910                 currentStatus = MultiPartStatus.MIXEDDELIMITER;
911                 cleanMixedAttributes();
912             }
913             FileUpload fileUpload = currentFileUpload;
914             currentFileUpload = null;
915             return fileUpload;
916         }
917         // do not change the buffer position
918         // since some can be already saved into FileUpload
919         // So do not change the currentStatus
920         return null;
921     }

1288      * Read a FileUpload data as Byte (Binary) and add the bytes directly to the
1289      * FileUpload. If the delimiter is found, the FileUpload is completed.
1290      *
1291      * @throws NotEnoughDataDecoderException
1292      *             Need more chunks but do not reset the readerInder since some
1293      *             values will be already added to the FileOutput
1294      * @throws ErrorDataDecoderException
1295      *             write IO error occurs with the FileUpload
1296      */
1297     private void readFileUploadByteMultipartStandard(String delimiter) {
1298         int readerIndex = undecodedChunk.readerIndex();
1299         // found the decoder limit
1300         boolean newLine = true;
1301         int index = 0;
1302         int lastPosition = undecodedChunk.readerIndex();
1303         boolean found = false;
1304         while (undecodedChunk.isReadable()) {
1305             byte nextByte = undecodedChunk.readByte();
1306             if (newLine) {
1307                 // Check the delimiter
1308                 if (nextByte == delimiter.codePointAt(index)) {
1309                     index++;
1310                     if (delimiter.length() == index) {
1311                         found = true;
1312                         break;
1313                     }
1314                     continue;
1315                 } else {
1316                     newLine = false;
1317                     index = 0;
1318                     // continue until end of line
1319                     if (nextByte == HttpConstants.CR) {
1320                         if (undecodedChunk.isReadable()) {
1321                             nextByte = undecodedChunk.readByte();
1322                             if (nextByte == HttpConstants.LF) {
1323                                 newLine = true;
1324                                 index = 0;
1325                                 lastPosition = undecodedChunk.readerIndex() - 2;
1326                             } else {
1327                                 // save last valid position
1328                                 lastPosition = undecodedChunk.readerIndex() - 1;
1329 
1330                                 // Unread next byte.
1331                                 undecodedChunk.readerIndex(lastPosition);
1332                             }
1333                         }
1334                     } else if (nextByte == HttpConstants.LF) {
1335                         newLine = true;
1336                         index = 0;
1337                         lastPosition = undecodedChunk.readerIndex() - 1;
1338                     } else {
1339                         // save last valid position
1340                         lastPosition = undecodedChunk.readerIndex();
1341                     }
1342                 }
1343             } else {
1344                 // continue until end of line
1345                 if (nextByte == HttpConstants.CR) {
1346                     if (undecodedChunk.isReadable()) {
1347                         nextByte = undecodedChunk.readByte();
1348                         if (nextByte == HttpConstants.LF) {
1349                             newLine = true;
1350                             index = 0;
1351                             lastPosition = undecodedChunk.readerIndex() - 2;
1352                         } else {
1353                             // save last valid position
1354                             lastPosition = undecodedChunk.readerIndex() - 1;
1355 
1356                             // Unread next byte.
1357                             undecodedChunk.readerIndex(lastPosition);
1358                         }
1359                     }
1360                 } else if (nextByte == HttpConstants.LF) {
1361                     newLine = true;
1362                     index = 0;
1363                     lastPosition = undecodedChunk.readerIndex() - 1;
1364                 } else {
1365                     // save last valid position
1366                     lastPosition = undecodedChunk.readerIndex();
1367                 }
1368             }
1369         }
1370         ByteBuf buffer = undecodedChunk.copy(readerIndex, lastPosition - readerIndex);
1371         if (found) {
1372             // found so lastPosition is correct and final
1373             try {
1374                 currentFileUpload.addContent(buffer, true);
1375                 // just before the CRLF and delimiter
1376                 undecodedChunk.readerIndex(lastPosition);
1377             } catch (IOException e) {
1378                 throw new ErrorDataDecoderException(e);
1379             }
1380         } else {
1381             // possibly the delimiter is partially found but still the last
1382             // position is OK
