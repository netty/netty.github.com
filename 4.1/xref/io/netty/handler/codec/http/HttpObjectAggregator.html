35  /**
36   * A {@link ChannelHandler} that aggregates an {@link HttpMessage}
37   * and its following {@link HttpContent}s into a single {@link FullHttpRequest}
38   * or {@link FullHttpResponse} (depending on if it used to handle requests or responses)
39   * with no following {@link HttpContent}s.  It is useful when you don't want to take
40   * care of HTTP messages whose transfer encoding is 'chunked'.  Insert this
41   * handler after {@link HttpObjectDecoder} in the {@link ChannelPipeline}:
42   * <pre>
43   * {@link ChannelPipeline} p = ...;
44   * ...
45   * p.addLast("encoder", new {@link HttpResponseEncoder}());
46   * p.addLast("decoder", new {@link HttpRequestDecoder}());
47   * p.addLast("aggregator", <b>new {@link HttpObjectAggregator}(1048576)</b>);
48   * ...
49   * p.addLast("handler", new HttpRequestHandler());
50   * </pre>
51   * Be aware that you need to have the {@link HttpResponseEncoder} or {@link HttpRequestEncoder}
52   * before the {@link HttpObjectAggregator} in the {@link ChannelPipeline}.
53   */
54  public class HttpObjectAggregator
55          extends MessageAggregator<HttpObject, HttpMessage, HttpContent, FullHttpMessage> {
56      private static final InternalLogger logger = InternalLoggerFactory.getInstance(HttpObjectAggregator.class);
57      private static final FullHttpResponse CONTINUE =
58              new DefaultFullHttpResponse(HttpVersion.HTTP_1_1, HttpResponseStatus.CONTINUE, Unpooled.EMPTY_BUFFER);
59      private static final FullHttpResponse EXPECTATION_FAILED = new DefaultFullHttpResponse(
60              HttpVersion.HTTP_1_1, HttpResponseStatus.EXPECTATION_FAILED, Unpooled.EMPTY_BUFFER);
61      private static final FullHttpResponse TOO_LARGE = new DefaultFullHttpResponse(
62              HttpVersion.HTTP_1_1, HttpResponseStatus.REQUEST_ENTITY_TOO_LARGE, Unpooled.EMPTY_BUFFER);
63  
64      static {
65          EXPECTATION_FAILED.headers().set(CONTENT_LENGTH, 0);
66          TOO_LARGE.headers().set(CONTENT_LENGTH, 0);
67      }

83       * @param maxContentLength the maximum length of the aggregated content in bytes.
84       * If the length of the aggregated content exceeds this value,

91          super(maxContentLength);

120     @Override
121     protected Object newContinueResponse(HttpMessage start, int maxContentLength, ChannelPipeline pipeline) {
122         if (HttpUtil.is100ContinueExpected(start)) {
123             if (getContentLength(start, -1L) <= maxContentLength) {
124                 return CONTINUE.duplicate().retain();
125             }
126 
127             pipeline.fireUserEventTriggered(HttpExpectationFailedEvent.INSTANCE);
128             return EXPECTATION_FAILED.duplicate().retain();
129         }
130         return null;
131     }
132 
133     @Override
134     protected boolean closeAfterContinueResponse(Object msg) {
135         return closeOnExpectationFailed && ignoreContentAfterContinueResponse(msg);
136     }

138     @Override
139     protected boolean ignoreContentAfterContinueResponse(Object msg) {
140         return msg instanceof HttpResponse &&
141                ((HttpResponse) msg).status().code() == HttpResponseStatus.EXPECTATION_FAILED.code();
142     }

144     @Override
145     protected FullHttpMessage beginAggregation(HttpMessage start, ByteBuf content) throws Exception {
146         assert !(start instanceof FullHttpMessage);
147 
148         HttpUtil.setTransferEncodingChunked(start, false);
149 
150         AggregatedFullHttpMessage ret;
151         if (start instanceof HttpRequest) {
152             ret = new AggregatedFullHttpRequest((HttpRequest) start, content, null);
153         } else if (start instanceof HttpResponse) {
154             ret = new AggregatedFullHttpResponse((HttpResponse) start, content, null);
155         } else {
156             throw new Error();
157         }
158         return ret;
159     }
160 
161     @Override
162     protected void aggregate(FullHttpMessage aggregated, HttpContent content) throws Exception {
163         if (content instanceof LastHttpContent) {
164             // Merge trailing headers into the message.
165             ((AggregatedFullHttpMessage) aggregated).setTrailingHeaders(((LastHttpContent) content).trailingHeaders());
166         }
167     }

169     @Override
170     protected void finishAggregation(FullHttpMessage aggregated) throws Exception {
171         // Set the 'Content-Length' header. If one isn't already set.
172         // This is important as HEAD responses will use a 'Content-Length' header which
173         // does not match the actual body, but the number of bytes that would be
174         // transmitted if a GET would have been used.
175         //
176         // See rfc2616 14.13 Content-Length
177         if (!HttpUtil.isContentLengthSet(aggregated)) {
178             aggregated.headers().set(
179                     HttpHeaderNames.CONTENT_LENGTH,
180                     String.valueOf(aggregated.content().readableBytes()));
181         }
182     }

184     @Override
185     protected void handleOversizedMessage(final ChannelHandlerContext ctx, HttpMessage oversized) throws Exception {
186         if (oversized instanceof HttpRequest) {
187             // send back a 413 and close the connection
188             ChannelFuture future = ctx.writeAndFlush(TOO_LARGE.duplicate().retain()).addListener(
189                     new ChannelFutureListener() {
190                 @Override
191                 public void operationComplete(ChannelFuture future) throws Exception {
192                     if (!future.isSuccess()) {
193                         logger.debug("Failed to send a 413 Request Entity Too Large.", future.cause());
194                         ctx.close();
195                     }
196                 }
197             });
198 
199             // If the client started to send data already, close because it's impossible to recover.
200             // If keep-alive is off and 'Expect: 100-continue' is missing, no need to leave the connection open.
201             if (oversized instanceof FullHttpMessage ||
202                 !HttpUtil.is100ContinueExpected(oversized) && !HttpUtil.isKeepAlive(oversized)) {
203                 future.addListener(ChannelFutureListener.CLOSE);
204             }
205 
206             // If an oversized request was handled properly and the connection is still alive
207             // (i.e. rejected 100-continue). the decoder should prepare to handle a new message.
208             HttpObjectDecoder decoder = ctx.pipeline().get(HttpObjectDecoder.class);
209             if (decoder != null) {
210                 decoder.reset();
211             }
212         } else if (oversized instanceof HttpResponse) {
213             ctx.close();
214             throw new TooLongFrameException("Response entity too large: " + oversized);
215         } else {
216             throw new IllegalStateException();
217         }
218     }
