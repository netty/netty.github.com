155     /**
156      * Begins the opening handshake
157      *
158      * @param channel
159      *            Channel
160      * @param promise
161      *            the {@link ChannelPromise} to be notified when the opening handshake is sent
162      */
163     public final ChannelFuture handshake(Channel channel, final ChannelPromise promise) {
164         FullHttpRequest request =  newHandshakeRequest();
165 
166         HttpResponseDecoder decoder = channel.pipeline().get(HttpResponseDecoder.class);
167         if (decoder == null) {
168             HttpClientCodec codec = channel.pipeline().get(HttpClientCodec.class);
169             if (codec == null) {
170                promise.setFailure(new IllegalStateException("ChannelPipeline does not contain " +
171                        "a HttpResponseDecoder or HttpClientCodec"));
172                return promise;
173             }
174         }
175 
176         channel.writeAndFlush(request).addListener(new ChannelFutureListener() {
177             @Override
178             public void operationComplete(ChannelFuture future) {
179                 if (future.isSuccess()) {
180                     ChannelPipeline p = future.channel().pipeline();
181                     ChannelHandlerContext ctx = p.context(HttpRequestEncoder.class);
182                     if (ctx == null) {
183                         ctx = p.context(HttpClientCodec.class);
184                     }
185                     if (ctx == null) {
186                         promise.setFailure(new IllegalStateException("ChannelPipeline does not contain " +
187                                 "a HttpRequestEncoder or HttpClientCodec"));
188                         return;
189                     }
190                     p.addAfter(ctx.name(), "ws-encoder", newWebSocketEncoder());
191 
192                     promise.setSuccess();
193                 } else {
194                     promise.setFailure(future.cause());
195                 }
196             }
197         });
198         return promise;
199     }

206     /**
207      * Validates and finishes the opening handshake initiated by {@link #handshake}}.
208      *
209      * @param channel
210      *            Channel
211      * @param response
212      *            HTTP response containing the closing handshake details
213      */
214     public final void finishHandshake(Channel channel, FullHttpResponse response) {
215         verify(response);
216 
217         // Verify the subprotocol that we received from the server.
218         // This must be one of our expected subprotocols - or null/empty if we didn't want to speak a subprotocol
219         String receivedProtocol = response.headers().get(HttpHeaderNames.SEC_WEBSOCKET_PROTOCOL);
220         receivedProtocol = receivedProtocol != null ? receivedProtocol.trim() : null;
221         String expectedProtocol = expectedSubprotocol != null ? expectedSubprotocol : "";
222         boolean protocolValid = false;
223 
224         if (expectedProtocol.isEmpty() && receivedProtocol == null) {
225             // No subprotocol required and none received
226             protocolValid = true;
227             setActualSubprotocol(expectedSubprotocol); // null or "" - we echo what the user requested
228         } else if (!expectedProtocol.isEmpty() && receivedProtocol != null && !receivedProtocol.isEmpty()) {
229             // We require a subprotocol and received one -> verify it
230             for (String protocol : StringUtil.split(expectedSubprotocol, ',')) {
231                 if (protocol.trim().equals(receivedProtocol)) {
232                     protocolValid = true;
233                     setActualSubprotocol(receivedProtocol);
234                     break;
235                 }
236             }
237         } // else mixed cases - which are all errors
238 
239         if (!protocolValid) {
240             throw new WebSocketHandshakeException(String.format(
241                     "Invalid subprotocol. Actual: %s. Expected one of: %s",
242                     receivedProtocol, expectedSubprotocol));
243         }
244 
245         setHandshakeComplete();
246 
247         final ChannelPipeline p = channel.pipeline();
248         // Remove decompressor from pipeline if its in use
249         HttpContentDecompressor decompressor = p.get(HttpContentDecompressor.class);
250         if (decompressor != null) {
251             p.remove(decompressor);
252         }
253 
254         // Remove aggregator if present before
255         HttpObjectAggregator aggregator = p.get(HttpObjectAggregator.class);
256         if (aggregator != null) {
257             p.remove(aggregator);
258         }
259 
260         ChannelHandlerContext ctx = p.context(HttpResponseDecoder.class);
261         if (ctx == null) {
262             ctx = p.context(HttpClientCodec.class);
263             if (ctx == null) {
264                 throw new IllegalStateException("ChannelPipeline does not contain " +
265                         "a HttpRequestEncoder or HttpClientCodec");
266             }
267             final HttpClientCodec codec =  (HttpClientCodec) ctx.handler();
268             // Remove the encoder part of the codec as the user may start writing frames after this method returns.
269             codec.removeOutboundHandler();
270 
271             p.addAfter(ctx.name(), "ws-decoder", newWebsocketDecoder());
272 
273             // Delay the removal of the decoder so the user can setup the pipeline if needed to handle
274             // WebSocketFrame messages.
275             // See https://github.com/netty/netty/issues/4533
276             channel.eventLoop().execute(new OneTimeTask() {
277                 @Override
278                 public void run() {
279                     p.remove(codec);
280                 }
281             });
282         } else {
283             if (p.get(HttpRequestEncoder.class) != null) {
284                 // Remove the encoder part of the codec as the user may start writing frames after this method returns.
285                 p.remove(HttpRequestEncoder.class);
286             }
287             final ChannelHandlerContext context = ctx;
288             p.addAfter(context.name(), "ws-decoder", newWebsocketDecoder());
289 
290             // Delay the removal of the decoder so the user can setup the pipeline if needed to handle
291             // WebSocketFrame messages.
292             // See https://github.com/netty/netty/issues/4533
293             channel.eventLoop().execute(new OneTimeTask() {
294                 @Override
295                 public void run() {
296                     p.remove(context.handler());
297                 }
298             });
299         }
300     }

316     /**
317      * Process the opening handshake initiated by {@link #handshake}}.
318      *
319      * @param channel
320      *            Channel
321      * @param response
322      *            HTTP response containing the closing handshake details
323      * @param promise
324      *            the {@link ChannelPromise} to notify once the handshake completes.
325      * @return future
326      *            the {@link ChannelFuture} which is notified once the handshake completes.
327      */
328     public final ChannelFuture processHandshake(final Channel channel, HttpResponse response,
329                                                 final ChannelPromise promise) {
330         if (response instanceof FullHttpResponse) {
331             try {
332                 finishHandshake(channel, (FullHttpResponse) response);
333                 promise.setSuccess();
334             } catch (Throwable cause) {
335                 promise.setFailure(cause);
336             }
337         } else {
338             ChannelPipeline p = channel.pipeline();
339             ChannelHandlerContext ctx = p.context(HttpResponseDecoder.class);
340             if (ctx == null) {
341                 ctx = p.context(HttpClientCodec.class);
342                 if (ctx == null) {
343                     return promise.setFailure(new IllegalStateException("ChannelPipeline does not contain " +
344                             "a HttpResponseDecoder or HttpClientCodec"));
345                 }
346             }
347             // Add aggregator and ensure we feed the HttpResponse so it is aggregated. A limit of 8192 should be more
348             // then enough for the websockets handshake payload.
349             //
350             // TODO: Make handshake work without HttpObjectAggregator at all.
351             String aggregatorName = "httpAggregator";
352             p.addAfter(ctx.name(), aggregatorName, new HttpObjectAggregator(8192));
353             p.addAfter(aggregatorName, "handshaker", new SimpleChannelInboundHandler<FullHttpResponse>() {
354                 @Override
355                 protected void channelRead0(ChannelHandlerContext ctx, FullHttpResponse msg) throws Exception {
356                     // Remove ourself and do the actual handshake
357                     ctx.pipeline().remove(this);
358                     try {
359                         finishHandshake(channel, msg);
360                         promise.setSuccess();
361                     } catch (Throwable cause) {
362                         promise.setFailure(cause);
363                     }
364                 }
365 
366                 @Override
367                 public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) throws Exception {
368                     // Remove ourself and fail the handshake promise.
369                     ctx.pipeline().remove(this);
370                     promise.setFailure(cause);
371                 }
372 
373                 @Override
374                 public void channelInactive(ChannelHandlerContext ctx) throws Exception {
375                     // Fail promise if Channel was closed
376                     promise.tryFailure(CLOSED_CHANNEL_EXCEPTION);
377                     ctx.fireChannelInactive();
378                 }
379             });
380             try {
381                 ctx.fireChannelRead(ReferenceCountUtil.retain(response));
382             } catch (Throwable cause) {
383                 promise.setFailure(cause);
384             }
385         }
386         return promise;
387     }

436     /**
437      * Return the constructed raw path for the give {@link URI}.
438      */
439     static String rawPath(URI wsURL) {
440         String path = wsURL.getRawPath();
441         String query = wsURL.getQuery();
442         if (query != null && !query.isEmpty()) {
443             path = path + '?' + query;
444         }
445 
446         return path == null || path.isEmpty() ? "/" : path;
447     }
