33  /**
34   * <p>
35   * Performs client side opening and closing handshakes for web socket specification version <a
36   * href="http://tools.ietf.org/html/draft-ietf-hybi-thewebsocketprotocol-17" >draft-ietf-hybi-thewebsocketprotocol-
37   * 17</a>
38   * </p>
39   */
40  public class WebSocketClientHandshaker13 extends WebSocketClientHandshaker {
41  
42      private static final InternalLogger logger = InternalLoggerFactory.getInstance(WebSocketClientHandshaker13.class);
43  
44      public static final String MAGIC_GUID = "258EAFA5-E914-47DA-95CA-C5AB0DC85B11";

101         super(webSocketURL, version, subprotocol, customHeaders, maxFramePayloadLength);

103         this.performMasking = performMasking;
104         this.allowMaskMismatch = allowMaskMismatch;

107     /**
108      * /**
109      * <p>
110      * Sends the opening request to the server:
111      * </p>
112      *
113      * <pre>
114      * GET /chat HTTP/1.1
115      * Host: server.example.com
116      * Upgrade: websocket
117      * Connection: Upgrade
118      * Sec-WebSocket-Key: dGhlIHNhbXBsZSBub25jZQ==
119      * Sec-WebSocket-Origin: http://example.com
120      * Sec-WebSocket-Protocol: chat, superchat
121      * Sec-WebSocket-Version: 13
122      * </pre>
123      *
124      */
125     @Override
126     protected FullHttpRequest newHandshakeRequest() {
127         // Get path
128         URI wsURL = uri();
129         String path = rawPath(wsURL);
130 
131         // Get 16 bit nonce and base 64 encode it
132         byte[] nonce = WebSocketUtil.randomBytes(16);
133         String key = WebSocketUtil.base64(nonce);
134 
135         String acceptSeed = key + MAGIC_GUID;
136         byte[] sha1 = WebSocketUtil.sha1(acceptSeed.getBytes(CharsetUtil.US_ASCII));
137         expectedChallengeResponseString = WebSocketUtil.base64(sha1);
138 
139         if (logger.isDebugEnabled()) {
140             logger.debug(
141                     "WebSocket version 13 client handshake key: {}, expected response: {}",
142                     key, expectedChallengeResponseString);
143         }
144 
145         // Format request
146         int wsPort = wsURL.getPort();
147         // check if the URI contained a port if not set the correct one depending on the schema.
148         // See https://github.com/netty/netty/pull/1558
149         if (wsPort == -1) {
150             if ("wss".equals(wsURL.getScheme())) {
151                 wsPort = 443;
152             } else {
153                 wsPort = 80;
154             }
155         }
156 
157         FullHttpRequest request = new DefaultFullHttpRequest(HttpVersion.HTTP_1_1, HttpMethod.GET, path);
158         HttpHeaders headers = request.headers();
159 
160         headers.add(HttpHeaderNames.UPGRADE, HttpHeaderValues.WEBSOCKET)
161                .add(HttpHeaderNames.CONNECTION, HttpHeaderValues.UPGRADE)
162                .add(HttpHeaderNames.SEC_WEBSOCKET_KEY, key)
163                .add(HttpHeaderNames.HOST, wsURL.getHost() + ':' + wsPort);
164 
165         String originValue = "http://" + wsURL.getHost();
166         if (wsPort != 80 && wsPort != 443) {
167             // if the port is not standard (80/443) its needed to add the port to the header.
168             // See http://tools.ietf.org/html/rfc6454#section-6.2
169             originValue = originValue + ':' + wsPort;
170         }
171         headers.add(HttpHeaderNames.SEC_WEBSOCKET_ORIGIN, originValue);
172 
173         String expectedSubprotocol = expectedSubprotocol();
174         if (expectedSubprotocol != null && !expectedSubprotocol.isEmpty()) {
175             headers.add(HttpHeaderNames.SEC_WEBSOCKET_PROTOCOL, expectedSubprotocol);
176         }
177 
178         headers.add(HttpHeaderNames.SEC_WEBSOCKET_VERSION, "13");
179 
180         if (customHeaders != null) {
181             headers.add(customHeaders);
182         }
183         return request;
184     }

186     /**
187      * <p>
188      * Process server response:
189      * </p>
190      *
191      * <pre>
192      * HTTP/1.1 101 Switching Protocols
193      * Upgrade: websocket
194      * Connection: Upgrade
195      * Sec-WebSocket-Accept: s3pPLMBiTxaQ9kYGzzhZRbK+xOo=
196      * Sec-WebSocket-Protocol: chat
197      * </pre>
198      *
199      * @param response
200      *            HTTP response returned from the server for the request sent by beginOpeningHandshake00().
201      * @throws WebSocketHandshakeException
202      */
203     @Override
204     protected void verify(FullHttpResponse response) {
205         final HttpResponseStatus status = HttpResponseStatus.SWITCHING_PROTOCOLS;
206         final HttpHeaders headers = response.headers();
207 
208         if (!response.status().equals(status)) {
209             throw new WebSocketHandshakeException("Invalid handshake response getStatus: " + response.status());
210         }
211 
212         CharSequence upgrade = headers.get(HttpHeaderNames.UPGRADE);
213         if (!HttpHeaderValues.WEBSOCKET.contentEqualsIgnoreCase(upgrade)) {
214             throw new WebSocketHandshakeException("Invalid handshake response upgrade: " + upgrade);
215         }
216 
217         if (!headers.containsValue(HttpHeaderNames.CONNECTION, HttpHeaderValues.UPGRADE, true)) {
218             throw new WebSocketHandshakeException("Invalid handshake response connection: "
219                     + headers.get(HttpHeaderNames.CONNECTION));
220         }
221 
222         CharSequence accept = headers.get(HttpHeaderNames.SEC_WEBSOCKET_ACCEPT);
223         if (accept == null || !accept.equals(expectedChallengeResponseString)) {
224             throw new WebSocketHandshakeException(String.format(
225                     "Invalid challenge. Actual: %s. Expected: %s", accept, expectedChallengeResponseString));
226         }
227     }
