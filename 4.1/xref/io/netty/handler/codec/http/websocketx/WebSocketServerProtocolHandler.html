36  /**
37   * This handler does all the heavy lifting for you to run a websocket server.
38   *
39   * It takes care of websocket handshaking as well as processing of control frames (Close, Ping, Pong). Text and Binary
40   * data frames are passed to the next handler in the pipeline (implemented by you) for processing.
41   *
42   * See <tt>io.netty.example.http.websocketx.html5.WebSocketServer</tt> for usage.
43   *
44   * The implementation of this handler assumes that you just want to run  a websocket server and not process other types
45   * HTTP requests (like GET and POST). If you wish to support both HTTP requests and websockets in the one server, refer
46   * to the <tt>io.netty.example.http.websocketx.server.WebSocketServer</tt> example.
47   *
48   * To know once a handshake was done you can intercept the
49   * {@link ChannelInboundHandler#userEventTriggered(ChannelHandlerContext, Object)} and check if the event was of type
50   * {@link ServerHandshakeStateEvent#HANDSHAKE_COMPLETE}.
51   */
52  public class WebSocketServerProtocolHandler extends WebSocketProtocolHandler {
53  
54      /**
55       * Events that are fired to notify about handshake status
56       */
57      public enum ServerHandshakeStateEvent {
58          /**
59           * The Handshake was complete succesful and so the channel was upgraded to websockets
60           */
61          HANDSHAKE_COMPLETE
62      }
63  
64      private static final AttributeKey<WebSocketServerHandshaker> HANDSHAKER_ATTR_KEY =
65              AttributeKey.valueOf(WebSocketServerHandshaker.class, "HANDSHAKER");

67      private final String websocketPath;
68      private final String subprotocols;
69      private final boolean allowExtensions;
70      private final int maxFramePayloadLength;
71      private final boolean allowMaskMismatch;

73      public WebSocketServerProtocolHandler(String websocketPath) {
74          this(websocketPath, null, false);
75      }
76  
77      public WebSocketServerProtocolHandler(String websocketPath, String subprotocols) {
78          this(websocketPath, subprotocols, false);
79      }

82          this(websocketPath, subprotocols, allowExtensions, 65536);

85      public WebSocketServerProtocolHandler(String websocketPath, String subprotocols,
86                                            boolean allowExtensions, int maxFrameSize) {
87          this(websocketPath, subprotocols, allowExtensions, maxFrameSize, false);
88      }

90      public WebSocketServerProtocolHandler(String websocketPath, String subprotocols,
91              boolean allowExtensions, int maxFrameSize, boolean allowMaskMismatch) {
92          this.websocketPath = websocketPath;
93          this.subprotocols = subprotocols;
94          this.allowExtensions = allowExtensions;
95          maxFramePayloadLength = maxFrameSize;
96          this.allowMaskMismatch = allowMaskMismatch;
97      }
98  
99      @Override
100     public void handlerAdded(ChannelHandlerContext ctx) {
101         ChannelPipeline cp = ctx.pipeline();
102         if (cp.get(WebSocketServerProtocolHandshakeHandler.class) == null) {
103             // Add the WebSocketHandshakeHandler before this one.
104             ctx.pipeline().addBefore(ctx.name(), WebSocketServerProtocolHandshakeHandler.class.getName(),
105                     new WebSocketServerProtocolHandshakeHandler(websocketPath, subprotocols,
106                             allowExtensions, maxFramePayloadLength, allowMaskMismatch));
107         }
108         if (cp.get(Utf8FrameValidator.class) == null) {
109             // Add the UFT8 checking before this one.
110             ctx.pipeline().addBefore(ctx.name(), Utf8FrameValidator.class.getName(),
111                     new Utf8FrameValidator());
112         }
113     }
114 
115     @Override
116     protected void decode(ChannelHandlerContext ctx, WebSocketFrame frame, List<Object> out) throws Exception {
117         if (frame instanceof CloseWebSocketFrame) {
118             WebSocketServerHandshaker handshaker = getHandshaker(ctx.channel());
119             if (handshaker != null) {
120                 frame.retain();
121                 handshaker.close(ctx.channel(), (CloseWebSocketFrame) frame);
122             } else {
123                 ctx.writeAndFlush(Unpooled.EMPTY_BUFFER).addListener(ChannelFutureListener.CLOSE);
124             }
125             return;
126         }
127         super.decode(ctx, frame, out);
128     }
129 
130     @Override
131     public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) throws Exception {
132         if (cause instanceof WebSocketHandshakeException) {
133             FullHttpResponse response = new DefaultFullHttpResponse(
134                     HTTP_1_1, HttpResponseStatus.BAD_REQUEST, Unpooled.wrappedBuffer(cause.getMessage().getBytes()));
135             ctx.channel().writeAndFlush(response).addListener(ChannelFutureListener.CLOSE);
136         } else {
137             ctx.close();
138         }
139     }
140 
141     static WebSocketServerHandshaker getHandshaker(Channel channel) {
142         return channel.attr(HANDSHAKER_ATTR_KEY).get();
143     }
144 
145     static void setHandshaker(Channel channel, WebSocketServerHandshaker handshaker) {
146         channel.attr(HANDSHAKER_ATTR_KEY).set(handshaker);
147     }
148 
149     static ChannelHandler forbiddenHttpRequestResponder() {
150         return new ChannelInboundHandlerAdapter() {
151             @Override
152             public void channelRead(ChannelHandlerContext ctx, Object msg) throws Exception {
153                 if (msg instanceof FullHttpRequest) {
154                     ((FullHttpRequest) msg).release();
155                     FullHttpResponse response =
156                             new DefaultFullHttpResponse(HTTP_1_1, HttpResponseStatus.FORBIDDEN);
157                     ctx.channel().writeAndFlush(response);
158                 } else {
159                     ctx.fireChannelRead(msg);
160                 }
161             }
162         };
163     }
