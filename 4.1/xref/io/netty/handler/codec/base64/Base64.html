110     public static ByteBuf encode(
111             ByteBuf src, int off, int len, boolean breakLines, Base64Dialect dialect, ByteBufAllocator allocator) {
112 
113         if (src == null) {
114             throw new NullPointerException("src");
115         }
116         if (dialect == null) {
117             throw new NullPointerException("dialect");
118         }
119 
120         int len43 = len * 4 / 3;
121         ByteBuf dest = allocator.buffer(
122                 len43 +
123                         (len % 3 > 0 ? 4 : 0) + // Account for padding
124                         (breakLines ? len43 / MAX_LINE_LENGTH : 0)).order(src.order()); // New lines
125         int d = 0;
126         int e = 0;
127         int len2 = len - 2;
128         int lineLength = 0;
129         for (; d < len2; d += 3, e += 4) {
130             encode3to4(src, d + off, 3, dest, e, dialect);
131 
132             lineLength += 4;
133 
134             if (breakLines && lineLength == MAX_LINE_LENGTH) {
135                 dest.setByte(e + 4, NEW_LINE);
136                 e ++;
137                 lineLength = 0;
138             } // end if: end of line
139         } // end for: each piece of array
140 
141         if (d < len) {
142             encode3to4(src, d + off, len - d, dest, e, dialect);
143             e += 4;
144         } // end if: some padding needed
145 
146         // Remove last byte if it's a newline
147         if (e > 1 && dest.getByte(e - 1) == NEW_LINE) {
148             e--;
149         }
150 
151         return dest.slice(0, e);
152     }

154     private static void encode3to4(
155             ByteBuf src, int srcOffset, int numSigBytes,
156             ByteBuf dest, int destOffset, Base64Dialect dialect) {
157 
158         byte[] ALPHABET = alphabet(dialect);
159 
160         //           1         2         3
161         // 01234567890123456789012345678901 Bit position
162         // --------000000001111111122222222 Array position from threeBytes
163         // --------|    ||    ||    ||    | Six bit groups to index ALPHABET
164         //          >>18  >>12  >> 6  >> 0  Right shift necessary
165         //                0x3f  0x3f  0x3f  Additional AND
166 
167         // Create buffer with zero-padding if there are only one or two
168         // significant bytes passed in the array.
169         // We have to shift left 24 in order to flush out the 1's that appear
170         // when Java treats a value as negative that is cast from a byte to an int.
171         int inBuff =
172                 (numSigBytes > 0? src.getByte(srcOffset)     << 24 >>>  8 : 0) |
173                 (numSigBytes > 1? src.getByte(srcOffset + 1) << 24 >>> 16 : 0) |
174                 (numSigBytes > 2? src.getByte(srcOffset + 2) << 24 >>> 24 : 0);
175 
176         switch (numSigBytes) {
177         case 3:
178             dest.setByte(destOffset    , ALPHABET[inBuff >>> 18       ]);
179             dest.setByte(destOffset + 1, ALPHABET[inBuff >>> 12 & 0x3f]);
180             dest.setByte(destOffset + 2, ALPHABET[inBuff >>>  6 & 0x3f]);
181             dest.setByte(destOffset + 3, ALPHABET[inBuff        & 0x3f]);
182             break;
183         case 2:
184             dest.setByte(destOffset    , ALPHABET[inBuff >>> 18       ]);
185             dest.setByte(destOffset + 1, ALPHABET[inBuff >>> 12 & 0x3f]);
186             dest.setByte(destOffset + 2, ALPHABET[inBuff >>> 6  & 0x3f]);
187             dest.setByte(destOffset + 3, EQUALS_SIGN);
188             break;
189         case 1:
190             dest.setByte(destOffset    , ALPHABET[inBuff >>> 18       ]);
191             dest.setByte(destOffset + 1, ALPHABET[inBuff >>> 12 & 0x3f]);
192             dest.setByte(destOffset + 2, EQUALS_SIGN);
193             dest.setByte(destOffset + 3, EQUALS_SIGN);
194             break;
195         }
196     }

223     public static ByteBuf decode(
224             ByteBuf src, int off, int len, Base64Dialect dialect, ByteBufAllocator allocator) {
225 
226         if (src == null) {
227             throw new NullPointerException("src");
228         }
229         if (dialect == null) {
230             throw new NullPointerException("dialect");
231         }
232 
233         byte[] DECODABET = decodabet(dialect);
234 
235         int len34 = len * 3 / 4;
236         ByteBuf dest = allocator.buffer(len34).order(src.order()); // Upper limit on size of output
237         int outBuffPosn = 0;
238 
239         byte[] b4 = new byte[4];
240         int b4Posn = 0;
241         int i;
242         byte sbiCrop;
243         byte sbiDecode;
244         for (i = off; i < off + len; i ++) {
245             sbiCrop = (byte) (src.getByte(i) & 0x7f); // Only the low seven bits
246             sbiDecode = DECODABET[sbiCrop];
247 
248             if (sbiDecode >= WHITE_SPACE_ENC) { // White space, Equals sign or better
249                 if (sbiDecode >= EQUALS_SIGN_ENC) { // Equals sign or better
250                     b4[b4Posn ++] = sbiCrop;
251                     if (b4Posn > 3) { // Quartet built
252                         outBuffPosn += decode4to3(
253                                 b4, 0, dest, outBuffPosn, dialect);
254                         b4Posn = 0;
255 
256                         // If that was the equals sign, break out of 'for' loop
257                         if (sbiCrop == EQUALS_SIGN) {
258                             break;
259                         }
260                     }
261                 }
262             } else {
263                 throw new IllegalArgumentException(
264                         "bad Base64 input character at " + i + ": " +
265                         src.getUnsignedByte(i) + " (decimal)");
266             }
267         }
268 
269         return dest.slice(0, outBuffPosn);
270     }

272     private static int decode4to3(
273             byte[] src, int srcOffset,
274             ByteBuf dest, int destOffset, Base64Dialect dialect) {
275 
276         byte[] DECODABET = decodabet(dialect);
277 
278         if (src[srcOffset + 2] == EQUALS_SIGN) {
279             // Example: Dk==
280             int outBuff =
281                     (DECODABET[src[srcOffset    ]] & 0xFF) << 18 |
282                     (DECODABET[src[srcOffset + 1]] & 0xFF) << 12;
283 
284             dest.setByte(destOffset, (byte) (outBuff >>> 16));
285             return 1;
286         } else if (src[srcOffset + 3] == EQUALS_SIGN) {
287             // Example: DkL=
288             int outBuff =
289                     (DECODABET[src[srcOffset    ]] & 0xFF) << 18 |
290                     (DECODABET[src[srcOffset + 1]] & 0xFF) << 12 |
291                     (DECODABET[src[srcOffset + 2]] & 0xFF) <<  6;
292 
293             dest.setByte(destOffset    , (byte) (outBuff >>> 16));
294             dest.setByte(destOffset + 1, (byte) (outBuff >>>  8));
295             return 2;
296         } else {
297             // Example: DkLE
298             int outBuff;
299             try {
300                 outBuff =
301                         (DECODABET[src[srcOffset    ]] & 0xFF) << 18 |
302                         (DECODABET[src[srcOffset + 1]] & 0xFF) << 12 |
303                         (DECODABET[src[srcOffset + 2]] & 0xFF) <<  6 |
304                          DECODABET[src[srcOffset + 3]] & 0xFF;
305             } catch (IndexOutOfBoundsException ignored) {
306                 throw new IllegalArgumentException("not encoded in Base64");
307             }
308 
309             dest.setByte(destOffset    , (byte) (outBuff >> 16));
310             dest.setByte(destOffset + 1, (byte) (outBuff >>  8));
311             dest.setByte(destOffset + 2, (byte)  outBuff);
312             return 3;
313         }
