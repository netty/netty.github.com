16  package io.netty.handler.codec.memcache.binary;
17  
18  import io.netty.buffer.ByteBuf;
19  import io.netty.handler.codec.memcache.AbstractMemcacheObjectAggregator;
20  import io.netty.handler.codec.memcache.FullMemcacheMessage;
21  import io.netty.handler.codec.memcache.MemcacheContent;
22  import io.netty.handler.codec.memcache.MemcacheObject;
23  
24  /**
25   * An object aggregator for the memcache binary protocol.
26   *
27   * It aggregates {@link BinaryMemcacheMessage}s and {@link MemcacheContent} into {@link FullBinaryMemcacheRequest}s
28   * or {@link FullBinaryMemcacheResponse}s.
29   */
30  public class BinaryMemcacheObjectAggregator extends AbstractMemcacheObjectAggregator<BinaryMemcacheMessage> {
31  
32      public BinaryMemcacheObjectAggregator(int maxContentLength) {
33          super(maxContentLength);
34      }
35  
36      @Override
37      protected boolean isStartMessage(MemcacheObject msg) throws Exception {
38          return msg instanceof BinaryMemcacheMessage;
39      }
40  
41      @Override
42      protected FullMemcacheMessage beginAggregation(BinaryMemcacheMessage start, ByteBuf content) throws Exception {
43          if (start instanceof BinaryMemcacheRequest) {
44              return toFullRequest((BinaryMemcacheRequest) start, content);
45          }
46  
47          if (start instanceof BinaryMemcacheResponse) {
48              return toFullResponse((BinaryMemcacheResponse) start, content);
49          }
50  
51          // Should not reach here.
52          throw new Error();
53      }
54  
55      private static FullBinaryMemcacheRequest toFullRequest(BinaryMemcacheRequest request, ByteBuf content) {
56          ByteBuf key = request.key() == null ? null : request.key().retain();
57          ByteBuf extras = request.extras() == null ? null : request.extras().retain();
58          FullBinaryMemcacheRequest fullRequest =
59                  new DefaultFullBinaryMemcacheRequest(key, extras, content);
60  
61          fullRequest.setMagic(request.magic());
62          fullRequest.setOpcode(request.opcode());
63          fullRequest.setKeyLength(request.keyLength());
64          fullRequest.setExtrasLength(request.extrasLength());
65          fullRequest.setDataType(request.dataType());
66          fullRequest.setTotalBodyLength(request.totalBodyLength());
67          fullRequest.setOpaque(request.opaque());
68          fullRequest.setCas(request.cas());
69          fullRequest.setReserved(request.reserved());
70  
71          return fullRequest;
72      }
73  
74      private static FullBinaryMemcacheResponse toFullResponse(BinaryMemcacheResponse response, ByteBuf content) {
75          ByteBuf key = response.key() == null ? null : response.key().retain();
76          ByteBuf extras = response.extras() == null ? null : response.extras().retain();
77          FullBinaryMemcacheResponse fullResponse =
78                  new DefaultFullBinaryMemcacheResponse(key, extras, content);
79  
80          fullResponse.setMagic(response.magic());
81          fullResponse.setOpcode(response.opcode());
82          fullResponse.setKeyLength(response.keyLength());
83          fullResponse.setExtrasLength(response.extrasLength());
84          fullResponse.setDataType(response.dataType());
85          fullResponse.setTotalBodyLength(response.totalBodyLength());
86          fullResponse.setOpaque(response.opaque());
87          fullResponse.setCas(response.cas());
88          fullResponse.setStatus(response.status());
89  
90          return fullResponse;
91      }
92  }
