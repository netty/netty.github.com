127     @Override
128     public ChannelFuture writeData(ChannelHandlerContext ctx, int streamId, ByteBuf data,
129             int padding, boolean endStream, ChannelPromise promise) {
130         final SimpleChannelPromiseAggregator promiseAggregator =
131                 new SimpleChannelPromiseAggregator(promise, ctx.channel(), ctx.executor());
132         final DataFrameHeader header = new DataFrameHeader(ctx, streamId);
133         boolean needToReleaseHeaders = true;
134         boolean needToReleaseData = true;
135         try {
136             verifyStreamId(streamId, STREAM_ID);
137             verifyPadding(padding);
138 
139             boolean lastFrame;
140             int remainingData = data.readableBytes();
141             do {
142                 // Determine how much data and padding to write in this frame. Put all padding at the end.
143                 int frameDataBytes = min(remainingData, maxFrameSize);
144                 int framePaddingBytes = min(padding, max(0, (maxFrameSize - 1) - frameDataBytes));
145 
146                 // Decrement the remaining counters.
147                 padding -= framePaddingBytes;
148                 remainingData -= frameDataBytes;
149 
150                 // Determine whether or not this is the last frame to be sent.
151                 lastFrame = remainingData == 0 && padding == 0;
152 
153                 // Only the last frame is not retained. Until then, the outer finally must release.
154                 ByteBuf frameHeader = header.slice(frameDataBytes, framePaddingBytes, lastFrame && endStream);
155                 needToReleaseHeaders = !lastFrame;
156                 ctx.write(lastFrame ? frameHeader : frameHeader.retain(), promiseAggregator.newPromise());
157 
158                 // Write the frame data.
159                 ByteBuf frameData = data.readSlice(frameDataBytes);
160                 // Only the last frame is not retained. Until then, the outer finally must release.
161                 needToReleaseData = !lastFrame;
162                 ctx.write(lastFrame ? frameData : frameData.retain(), promiseAggregator.newPromise());
163 
164                 // Write the frame padding.
165                 if (framePaddingBytes > 0) {
166                     ctx.write(ZERO_BUFFER.slice(0, framePaddingBytes), promiseAggregator.newPromise());
167                 }
168             } while (!lastFrame);
169 
170             return promiseAggregator.doneAllocatingPromises();
171         } catch (Throwable t) {
172             if (needToReleaseHeaders) {
173                 header.release();
174             }
175             if (needToReleaseData) {
176                 data.release();
177             }
178             return promiseAggregator.setFailure(t);
179         }
180     }

288     @Override
289     public ChannelFuture writePushPromise(ChannelHandlerContext ctx, int streamId,
290             int promisedStreamId, Http2Headers headers, int padding, ChannelPromise promise) {
291         ByteBuf headerBlock = null;
292         SimpleChannelPromiseAggregator promiseAggregator =
293                 new SimpleChannelPromiseAggregator(promise, ctx.channel(), ctx.executor());
294         try {
295             verifyStreamId(streamId, STREAM_ID);
296             verifyStreamId(promisedStreamId, "Promised Stream ID");
297             verifyPadding(padding);
298 
299             // Encode the entire header block into an intermediate buffer.
300             headerBlock = ctx.alloc().buffer();
301             headersEncoder.encodeHeaders(headers, headerBlock);
302 
303             // Read the first fragment (possibly everything).
304             Http2Flags flags = new Http2Flags().paddingPresent(padding > 0);
305             // INT_FIELD_LENGTH is for the length of the promisedStreamId
306             int nonFragmentLength = INT_FIELD_LENGTH + padding + flags.getPaddingPresenceFieldLength();
307             int maxFragmentLength = maxFrameSize - nonFragmentLength;
308             ByteBuf fragment =
309                     headerBlock.readSlice(min(headerBlock.readableBytes(), maxFragmentLength)).retain();
310 
311             flags.endOfHeaders(!headerBlock.isReadable());
312 
313             int payloadLength = fragment.readableBytes() + nonFragmentLength;
314             ByteBuf buf = ctx.alloc().buffer(PUSH_PROMISE_FRAME_HEADER_LENGTH);
315             writeFrameHeaderInternal(buf, payloadLength, PUSH_PROMISE, flags, streamId);
316             writePaddingLength(buf, padding);
317 
318             // Write out the promised stream ID.
319             buf.writeInt(promisedStreamId);
320             ctx.write(buf, promiseAggregator.newPromise());
321 
322             // Write the first fragment.
323             ctx.write(fragment, promiseAggregator.newPromise());
324 
325             if (padding > 0) { // Write out the padding, if any.
326                 ctx.write(ZERO_BUFFER.slice(0, padding), promiseAggregator.newPromise());
327             }
328 
329             if (!flags.endOfHeaders()) {
330                 writeContinuationFrames(ctx, streamId, headerBlock, padding, promiseAggregator);
331             }
332 
333             return promiseAggregator.doneAllocatingPromises();
334         } catch (Throwable t) {
335             return promiseAggregator.setFailure(t);
336         } finally {
337             if (headerBlock != null) {
338                 headerBlock.release();
339             }
340         }
341     }

410     private ChannelFuture writeHeadersInternal(ChannelHandlerContext ctx,
411             int streamId, Http2Headers headers, int padding, boolean endStream,
412             boolean hasPriority, int streamDependency, short weight, boolean exclusive, ChannelPromise promise) {
413         ByteBuf headerBlock = null;
414         SimpleChannelPromiseAggregator promiseAggregator =
415                 new SimpleChannelPromiseAggregator(promise, ctx.channel(), ctx.executor());
416         try {
417             verifyStreamId(streamId, STREAM_ID);
418             if (hasPriority) {
419                 verifyStreamOrConnectionId(streamDependency, STREAM_DEPENDENCY);
420                 verifyPadding(padding);
421                 verifyWeight(weight);
422             }
423 
424             // Encode the entire header block.
425             headerBlock = ctx.alloc().buffer();
426             headersEncoder.encodeHeaders(headers, headerBlock);
427 
428             Http2Flags flags =
429                     new Http2Flags().endOfStream(endStream).priorityPresent(hasPriority).paddingPresent(padding > 0);
430 
431             // Read the first fragment (possibly everything).
432             int nonFragmentBytes = padding + flags.getNumPriorityBytes() + flags.getPaddingPresenceFieldLength();
433             int maxFragmentLength = maxFrameSize - nonFragmentBytes;
434             ByteBuf fragment =
435                     headerBlock.readSlice(min(headerBlock.readableBytes(), maxFragmentLength)).retain();
436 
437             // Set the end of headers flag for the first frame.
438             flags.endOfHeaders(!headerBlock.isReadable());
439 
440             int payloadLength = fragment.readableBytes() + nonFragmentBytes;
441             ByteBuf buf = ctx.alloc().buffer(HEADERS_FRAME_HEADER_LENGTH);
442             writeFrameHeaderInternal(buf, payloadLength, HEADERS, flags, streamId);
443             writePaddingLength(buf, padding);
444 
445             if (hasPriority) {
446                 long word1 = exclusive ? 0x80000000L | streamDependency : streamDependency;
447                 writeUnsignedInt(word1, buf);
448 
449                 // Adjust the weight so that it fits into a single byte on the wire.
450                 buf.writeByte(weight - 1);
451             }
452             ctx.write(buf, promiseAggregator.newPromise());
453 
454             // Write the first fragment.
455             ctx.write(fragment, promiseAggregator.newPromise());
456 
457             if (padding > 0) { // Write out the padding, if any.
458                 ctx.write(ZERO_BUFFER.slice(0, padding), promiseAggregator.newPromise());
459             }
460 
461             if (!flags.endOfHeaders()) {
462                 writeContinuationFrames(ctx, streamId, headerBlock, padding, promiseAggregator);
463             }
464 
465             return promiseAggregator.doneAllocatingPromises();
466         } catch (Throwable t) {
467             return promiseAggregator.setFailure(t);
468         } finally {
469             if (headerBlock != null) {
470                 headerBlock.release();
471             }
472         }
473     }

475     /**
476      * Writes as many continuation frames as needed until {@code padding} and {@code headerBlock} are consumed.
477      */
478     private ChannelFuture writeContinuationFrames(ChannelHandlerContext ctx, int streamId,
479             ByteBuf headerBlock, int padding, SimpleChannelPromiseAggregator promiseAggregator) {
480         Http2Flags flags = new Http2Flags().paddingPresent(padding > 0);
481         int nonFragmentLength = padding + flags.getPaddingPresenceFieldLength();
482         int maxFragmentLength = maxFrameSize - nonFragmentLength;
483         // TODO: same padding is applied to all frames, is this desired?
484         if (maxFragmentLength <= 0) {
485             return promiseAggregator.setFailure(new IllegalArgumentException(
486                     "Padding [" + padding + "] is too large for max frame size [" + maxFrameSize + "]"));
487         }
488 
489         if (headerBlock.isReadable()) {
490             // The frame header (and padding) only changes on the last frame, so allocate it once and re-use
491             int fragmentReadableBytes = min(headerBlock.readableBytes(), maxFragmentLength);
492             int payloadLength = fragmentReadableBytes + nonFragmentLength;
493             ByteBuf buf = ctx.alloc().buffer(CONTINUATION_FRAME_HEADER_LENGTH);
494             writeFrameHeaderInternal(buf, payloadLength, CONTINUATION, flags, streamId);
495             writePaddingLength(buf, padding);
496 
497             do {
498                 fragmentReadableBytes = min(headerBlock.readableBytes(), maxFragmentLength);
499                 ByteBuf fragment = headerBlock.readSlice(fragmentReadableBytes).retain();
500 
501                 payloadLength = fragmentReadableBytes + nonFragmentLength;
502                 if (headerBlock.isReadable()) {
503                     ctx.write(buf.retain(), promiseAggregator.newPromise());
504                 } else {
505                     // The frame header is different for the last frame, so re-allocate and release the old buffer
506                     flags = flags.endOfHeaders(true);
507                     buf.release();
508                     buf = ctx.alloc().buffer(CONTINUATION_FRAME_HEADER_LENGTH);
509                     writeFrameHeaderInternal(buf, payloadLength, CONTINUATION, flags, streamId);
510                     writePaddingLength(buf, padding);
511                     ctx.write(buf, promiseAggregator.newPromise());
512                 }
513 
514                 ctx.write(fragment, promiseAggregator.newPromise());
515 
516                 // Write out the padding, if any.
517                 if (padding > 0) {
518                     ctx.write(ZERO_BUFFER.slice(0, padding), promiseAggregator.newPromise());
519                 }
520             } while(headerBlock.isReadable());
521         }
522         return promiseAggregator;
523     }

526         if (paddingLength > 0) {
527             // It is assumed that the padding length has been bounds checked before this
528             buf.writeByte(paddingLength);
529         }
530     }

574     /**
575      * Utility class that manages the creation of frame header buffers for {@code DATA} frames. Attempts
576      * to reuse the same buffer repeatedly when splitting data into multiple frames.
577      */
578     private static final class DataFrameHeader {
579         private final int streamId;
580         private final ByteBuf buffer;
581         private final Http2Flags flags = new Http2Flags();
582         private int prevData;
583         private int prevPadding;
584         private ByteBuf frameHeader;
585 
586         DataFrameHeader(ChannelHandlerContext ctx, int streamId) {
587             // All padding will be put at the end, so in the worst case we need 3 headers:
588             // a repeated no-padding frame of maxFrameSize, a frame that has part data and part
589             // padding, and a frame that has the remainder of the padding.
590             buffer = ctx.alloc().buffer(3 * DATA_FRAME_HEADER_LENGTH);
591             this.streamId = streamId;
592         }
593 
594         /**
595          * Gets the frame header buffer configured for the current frame.
596          */
597         ByteBuf slice(int data, int padding, boolean endOfStream) {
598             // Since we're reusing the current frame header whenever possible, check if anything changed
599             // that requires a new header.
600             if (data != prevData || padding != prevPadding
