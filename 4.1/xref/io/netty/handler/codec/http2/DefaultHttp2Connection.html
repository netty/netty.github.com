114     @Override
115     public Future<Void> close(final Promise<Void> promise) {
116         checkNotNull(promise, "promise");
117         // Since we allow this method to be called multiple times, we must make sure that all the promises are notified
118         // when all streams are removed and the close operation completes.
119         if (closePromise != null) {
120             if (closePromise == promise) {
121                 // Do nothing
122             } else if ((promise instanceof ChannelPromise) && ((ChannelPromise) closePromise).isVoid()) {
123                 closePromise = promise;
124             } else {
125                 closePromise.addListener(new FutureListener<Void>() {
126                     @Override
127                     public void operationComplete(Future<Void> future) throws Exception {
128                         if (future.isSuccess()) {
129                             promise.trySuccess(null);
130                         } else if (future.isCancelled()) {
131                             promise.cancel(false);
132                         } else {
133                             promise.tryFailure(future.cause());
134                         }
135                     }
136                 });
137             }
138         } else {
139             closePromise = promise;
140         }
141         if (isStreamMapEmpty()) {
142             promise.trySuccess(null);
143             return promise;
144         }
145         Iterator<PrimitiveEntry<Http2Stream>> itr = streamMap.entries().iterator();
146         // We must take care while iterating the streamMap as to not modify while iterating in case there are other code
147         // paths iterating over the active streams.
148         if (activeStreams.allowModifications()) {
149             while (itr.hasNext()) {
150                 Http2Stream stream = itr.next().value();
151                 if (stream.id() != CONNECTION_STREAM_ID) {
152                     // If modifications of the activeStream map is allowed, then a stream close operation will also
153                     // modify the streamMap. We must prevent concurrent modifications to the streamMap, so use the
154                     // iterator to remove the current stream.
155                     itr.remove();
156                     stream.close();
157                 }
158             }
159         } else {
160             while (itr.hasNext()) {
161                 Http2Stream stream = itr.next().value();
162                 if (stream.id() != CONNECTION_STREAM_ID) {
163                     // We are not allowed to make modifications, so the close calls will be executed after this
164                     // iteration completes.
165                     stream.close();
166                 }
167             }
168         }
169         return closePromise;
170     }

227     @Override
228     public void goAwayReceived(final int lastKnownStream, long errorCode, ByteBuf debugData) {
229         localEndpoint.lastStreamKnownByPeer(lastKnownStream);
230         for (int i = 0; i < listeners.size(); ++i) {
231             try {
232                 listeners.get(i).onGoAwayReceived(lastKnownStream, errorCode, debugData);
233             } catch (RuntimeException e) {
234                 logger.error("Caught RuntimeException from listener onGoAwayReceived.", e);
235             }
236         }
237 
238         try {
239             forEachActiveStream(new Http2StreamVisitor() {
240                 @Override
241                 public boolean visit(Http2Stream stream) {
242                     if (stream.id() > lastKnownStream && localEndpoint.isValidStreamId(stream.id())) {
243                         stream.close();
244                     }
245                     return true;
246                 }
247             });
248         } catch (Http2Exception e) {
249             PlatformDependent.throwException(e);
250         }
251     }

291     /**
292      * Closed streams may stay in the priority tree if they have dependents that are in prioritizable states.
293      * When a stream is requested to be removed we can only actually remove that stream when there are no more
294      * prioritizable children.
295      * (see [1] {@link Http2Stream#prioritizableForTree()} and [2] {@link DefaultStream#removeChild(DefaultStream)}).
296      * When a priority tree edge changes we also have to re-evaluate viable nodes
297      * (see [3] {@link DefaultStream#takeChild(DefaultStream, boolean, List)}).
298      * @param stream The stream to remove.
299      */
300     void removeStream(DefaultStream stream) {
301         // [1] Check if this stream can be removed because it has no prioritizable descendants.
302         if (stream.parent().removeChild(stream)) {
303             streamMap.remove(stream.id());
304 
305             for (int i = 0; i < listeners.size(); i++) {
306                 try {
307                     listeners.get(i).onStreamRemoved(stream);
308                 } catch (RuntimeException e) {
309                     logger.error("Caught RuntimeException from listener onStreamRemoved.", e);
310                 }
311             }
312 
313             if (closePromise != null && isStreamMapEmpty()) {
314                 closePromise.trySuccess(null);
315             }
316         }
317     }

319     static State activeState(int streamId, State initialState, boolean isLocal, boolean halfClosed)
320             throws Http2Exception {
321         switch (initialState) {
322         case IDLE:
323             return halfClosed ? isLocal ? HALF_CLOSED_LOCAL : HALF_CLOSED_REMOTE : OPEN;
324         case RESERVED_LOCAL:
325             return HALF_CLOSED_REMOTE;
326         case RESERVED_REMOTE:
327             return HALF_CLOSED_LOCAL;
328         default:
329             throw streamError(streamId, PROTOCOL_ERROR, "Attempting to open a stream in an invalid state: "
330                     + initialState);
331         }
332     }

476         @Override
477         public Http2Stream setPriority(int parentStreamId, short weight, boolean exclusive) throws Http2Exception {
478             if (weight < MIN_WEIGHT || weight > MAX_WEIGHT) {
479                 throw new IllegalArgumentException(String.format(
480                         "Invalid weight: %d.  Must be between %d and %d (inclusive).", weight, MIN_WEIGHT, MAX_WEIGHT));
481             }
482 
483             DefaultStream newParent = (DefaultStream) stream(parentStreamId);
484             if (newParent == null) {
485                 // Streams can depend on other streams in the IDLE state. We must ensure
486                 // the stream has been "created" in order to use it in the priority tree.
487                 newParent = createdBy().createIdleStream(parentStreamId);
488             } else if (this == newParent) {
489                 throw new IllegalArgumentException("A stream cannot depend on itself");
490             }
491 
492             // Already have a priority. Re-prioritize the stream.
493             weight(weight);
494 
495             if (newParent != parent() || (exclusive && newParent.numChildren() != 1)) {
496                 final List<ParentChangedEvent> events;
497                 if (newParent.isDescendantOf(this)) {
498                     events = new ArrayList<ParentChangedEvent>(2 + (exclusive ? newParent.numChildren() : 0));
499                     parent.takeChild(newParent, false, events);
500                 } else {
501                     events = new ArrayList<ParentChangedEvent>(1 + (exclusive ? newParent.numChildren() : 0));
502                 }
503                 newParent.takeChild(this, exclusive, events);
504                 notifyParentChanged(events);
505             }
506 
507             return this;
508         }

569         /**
570          * Recursively increment the {@link #prioritizableForTree} for this object up the parent links until
571          * either we go past the root or {@code oldParent} is encountered.
572          * @param amt The amount to increment by. This must be positive.
573          * @param oldParent The previous parent for this stream.
574          */
575         private void incrementPrioritizableForTree(int amt, Http2Stream oldParent) {
576             if (amt != 0) {
577                 incrementPrioritizableForTree0(amt, oldParent);
578             }
579         }

581         /**
582          * Direct calls to this method are discouraged.
583          * Instead use {@link #incrementPrioritizableForTree(int, Http2Stream)}.
584          */
585         private void incrementPrioritizableForTree0(int amt, Http2Stream oldParent) {
586             assert amt > 0 && Integer.MAX_VALUE - amt >= prioritizableForTree;
587             prioritizableForTree += amt;
588             if (parent != null && parent != oldParent) {
589                 parent.incrementPrioritizableForTree0(amt, oldParent);
590             }
591         }

593         /**
594          * Recursively increment the {@link #prioritizableForTree} for this object up the parent links until
595          * either we go past the root.
596          * @param amt The amount to decrement by. This must be positive.
597          */
598         private void decrementPrioritizableForTree(int amt) {
599             if (amt != 0) {
600                 decrementPrioritizableForTree0(amt);
601             }
602         }
603 
604         /**
605          * Direct calls to this method are discouraged. Instead use {@link #decrementPrioritizableForTree(int)}.
606          */
607         private void decrementPrioritizableForTree0(int amt) {
608             assert amt > 0 && prioritizableForTree >= amt;
609             prioritizableForTree -= amt;
610             if (parent != null) {
611                 parent.decrementPrioritizableForTree0(amt);
612             }
613         }

640         final void weight(short weight) {
641             if (weight != this.weight) {
642                 final short oldWeight = this.weight;
643                 this.weight = weight;
644                 for (int i = 0; i < listeners.size(); i++) {
645                     try {
646                         listeners.get(i).onWeightChanged(this, oldWeight);
647                     } catch (RuntimeException e) {
648                         logger.error("Caught RuntimeException from listener onWeightChanged.", e);
649                     }
650                 }
651             }
652         }

654         /**
655          * Remove all children with the exception of {@code streamToRetain}.
656          * This method is intended to be used to support an exclusive priority dependency operation.
657          * @return The map of children prior to this operation, excluding {@code streamToRetain} if present.
658          */
659         private IntObjectMap<DefaultStream> retain(DefaultStream streamToRetain) {
660             streamToRetain = children.remove(streamToRetain.id());
661             IntObjectMap<DefaultStream> prevChildren = children;
662             // This map should be re-initialized in anticipation for the 1 exclusive child which will be added.
663             // It will either be added directly in this method, or after this method is called...but it will be added.
664             initChildren();
665             if (streamToRetain == null) {
666                 prioritizableForTree = isPrioritizable() ? 1 : 0;
667             } else {
668                 // prioritizableForTree does not change because it is assumed all children node will still be
669                 // descendants through an exclusive priority tree operation.
670                 children.put(streamToRetain.id(), streamToRetain);
671             }
672             return prevChildren;
673         }

675         /**
676          * Adds a child to this priority. If exclusive is set, any children of this node are moved to being dependent on
677          * the child.
678          */
679         final void takeChild(DefaultStream child, boolean exclusive, List<ParentChangedEvent> events) {
680             DefaultStream oldParent = child.parent();
681 
682             if (oldParent != this) {
683                 events.add(new ParentChangedEvent(child, oldParent));
684                 notifyParentChanging(child, this);
685                 child.parent = this;
686                 // We need the removal operation to happen first so the prioritizableForTree for the old parent to root
687                 // path is updated with the correct child.prioritizableForTree() value. Note that the removal operation
688                 // may not be successful and may return null. This is because when an exclusive dependency is processed
689                 // the children are removed in a previous recursive call but the child's parent link is updated here.
690                 if (oldParent != null && oldParent.children.remove(child.id()) != null &&
691                         !child.isDescendantOf(oldParent)) {
692                     oldParent.decrementPrioritizableForTree(child.prioritizableForTree());
693                     if (oldParent.prioritizableForTree() == 0) {
694                         // There are a few risks with immediately removing nodes from the priority tree:
695                         // 1. We are removing nodes while we are potentially shifting the tree. There are no
696                         // concrete cases known but is risky because it could invalidate the data structure.
697                         // 2. We are notifying listeners of the removal while the tree is in flux. Currently the
698                         // codec listeners make no assumptions about priority tree structure when being notified.
699                         removeStream(oldParent);
700                     }
701                 }
702 
703                 // Lazily initialize the children to save object allocations.
704                 initChildrenIfEmpty();
705 
706                 final Http2Stream oldChild = children.put(child.id(), child);
707                 assert oldChild == null : "A stream with the same stream ID was already in the child map.";
708                 incrementPrioritizableForTree(child.prioritizableForTree(), oldParent);
709             }
710 
711             if (exclusive && !children.isEmpty()) {
712                 // If it was requested that this child be the exclusive dependency of this node,
713                 // move any previous children to the child node, becoming grand children of this node.
714                 for (DefaultStream grandchild : retain(child).values()) {
715                     child.takeChild(grandchild, false, events);
716                 }
717             }
718         }
719 
720         /**
721          * Removes the child priority and moves any of its dependencies to being direct dependencies on this node.
722          */
723         final boolean removeChild(DefaultStream child) {
724             if (child.prioritizableForTree() == 0 && children.remove(child.id()) != null) {
725                 List<ParentChangedEvent> events = new ArrayList<ParentChangedEvent>(1 + child.numChildren());
726                 events.add(new ParentChangedEvent(child, child.parent()));
727                 notifyParentChanging(child, null);
728                 child.parent = null;
729                 decrementPrioritizableForTree(child.prioritizableForTree());
730 
731                 // Move up any grand children to be directly dependent on this node.
732                 for (DefaultStream grandchild : child.children.values()) {
733                     takeChild(grandchild, false, events);
734                 }
735 
736                 if (prioritizableForTree() == 0) {
737                     // There are a few risks with immediately removing nodes from the priority tree:
738                     // 1. We are removing nodes while we are potentially shifting the tree. There are no
739                     // concrete cases known but is risky because it could invalidate the data structure.
740                     // 2. We are notifying listeners of the removal while the tree is in flux. Currently the
741                     // codec listeners make no assumptions about priority tree structure when being notified.
742                     removeStream(this);
743                 }
744                 notifyParentChanged(events);
745                 return true;
746             }
747             return false;
748         }

1100         private void checkNewStreamAllowed(int streamId, State state) throws Http2Exception {
1101             if (goAwayReceived() && streamId > localEndpoint.lastStreamKnownByPeer()) {
1102                 throw connectionError(PROTOCOL_ERROR, "Cannot create stream %d since this endpoint has received a " +
1103                                                       "GOAWAY frame with last stream id %d.", streamId,
1104                                                       localEndpoint.lastStreamKnownByPeer());
1105             }
1106             if (streamId < 0) {
1107                 throw new Http2NoMoreStreamIdsException();
1108             }
1109             if (!isValidStreamId(streamId)) {
1110                 throw connectionError(PROTOCOL_ERROR, "Request stream %d is not correct for %s connection", streamId,
1111                         server ? "server" : "client");
1112             }
1113             // This check must be after all id validated checks, but before the max streams check because it may be
1114             // recoverable to some degree for handling frames which can be sent on closed streams.
1115             if (streamId < nextStreamIdToCreate) {
1116                 throw closedStreamError(PROTOCOL_ERROR, "Request stream %d is behind the next expected stream %d",
1117                         streamId, nextStreamIdToCreate);
1118             }
1119             if (nextStreamIdToCreate <= 0) {
1120                 throw connectionError(REFUSED_STREAM, "Stream IDs are exhausted for this endpoint.");
1121             }
1122             if ((state.localSideOpen() || state.remoteSideOpen()) && !canOpenStream()) {
1123                 throw connectionError(REFUSED_STREAM, "Maximum active streams violated for this endpoint.");
1124             }
1125             if (isClosed()) {
1126                 throw connectionError(INTERNAL_ERROR, "Attempted to create stream id %d after connection was closed",
1127                                       streamId);
1128             }
1129         }
