153     @Override
154     public void goAwayReceived(final int lastKnownStream, long errorCode, ByteBuf debugData) {
155         localEndpoint.lastStreamKnownByPeer(lastKnownStream);
156         for (int i = 0; i < listeners.size(); ++i) {
157             try {
158                 listeners.get(i).onGoAwayReceived(lastKnownStream, errorCode, debugData);
159             } catch (RuntimeException e) {
160                 logger.error("Caught RuntimeException from listener onGoAwayReceived.", e);
161             }
162         }
163 
164         try {
165             forEachActiveStream(new Http2StreamVisitor() {
166                 @Override
167                 public boolean visit(Http2Stream stream) {
168                     if (stream.id() > lastKnownStream && localEndpoint.isValidStreamId(stream.id())) {
169                         stream.close();
170                     }
171                     return true;
172                 }
173             });
174         } catch (Http2Exception e) {
175             PlatformDependent.throwException(e);
176         }
177     }

210     /**
211      * Closed streams may stay in the priority tree if they have dependents that are in prioritizable states.
212      * When a stream is requested to be removed we can only actually remove that stream when there are no more
213      * prioritizable children.
214      * (see [1] {@link Http2Stream#prioritizableForTree()} and [2] {@link DefaultStream#removeChild(DefaultStream)}).
215      * When a priority tree edge changes we also have to re-evaluate viable nodes
216      * (see [3] {@link DefaultStream#takeChild(DefaultStream, boolean, List)}).
217      * @param stream The stream to remove.
218      */
219     void removeStream(DefaultStream stream) {
220         // [1] Check if this stream can be removed because it has no prioritizable descendants.
221         if (stream.parent().removeChild(stream)) {
222             // Remove it from the map and priority tree.
223             streamMap.remove(stream.id());
224 
225             for (int i = 0; i < listeners.size(); i++) {
226                 try {
227                     listeners.get(i).onStreamRemoved(stream);
228                 } catch (RuntimeException e) {
229                     logger.error("Caught RuntimeException from listener onStreamRemoved.", e);
230                 }
231             }
232         }
233     }

235     static State activeState(int streamId, State initialState, boolean isLocal, boolean halfClosed)
236             throws Http2Exception {
237         switch (initialState) {
238         case IDLE:
239             return halfClosed ? isLocal ? HALF_CLOSED_LOCAL : HALF_CLOSED_REMOTE : OPEN;
240         case RESERVED_LOCAL:
241             return HALF_CLOSED_REMOTE;
242         case RESERVED_REMOTE:
243             return HALF_CLOSED_LOCAL;
244         default:
245             throw streamError(streamId, PROTOCOL_ERROR, "Attempting to open a stream in an invalid state: "
246                     + initialState);
247         }
248     }

404         @Override
405         public Http2Stream setPriority(int parentStreamId, short weight, boolean exclusive) throws Http2Exception {
406             if (weight < MIN_WEIGHT || weight > MAX_WEIGHT) {
407                 throw new IllegalArgumentException(String.format(
408                         "Invalid weight: %d.  Must be between %d and %d (inclusive).", weight, MIN_WEIGHT, MAX_WEIGHT));
409             }
410 
411             DefaultStream newParent = (DefaultStream) stream(parentStreamId);
412             if (newParent == null) {
413                 // Streams can depend on other streams in the IDLE state. We must ensure
414                 // the stream has been "created" in order to use it in the priority tree.
415                 newParent = createdBy().createIdleStream(parentStreamId);
416             } else if (this == newParent) {
417                 throw new IllegalArgumentException("A stream cannot depend on itself");
418             }
419 
420             // Already have a priority. Re-prioritize the stream.
421             weight(weight);
422 
423             if (newParent != parent() || (exclusive && newParent.numChildren() != 1)) {
424                 final List<ParentChangedEvent> events;
425                 if (newParent.isDescendantOf(this)) {
426                     events = new ArrayList<ParentChangedEvent>(2 + (exclusive ? newParent.numChildren() : 0));
427                     parent.takeChild(newParent, false, events);
428                 } else {
429                     events = new ArrayList<ParentChangedEvent>(1 + (exclusive ? newParent.numChildren() : 0));
430                 }
431                 newParent.takeChild(this, exclusive, events);
432                 notifyParentChanged(events);
433             }
434 
435             return this;
436         }

497         /**
498          * Recursively increment the {@link #prioritizableForTree} for this object up the parent links until
499          * either we go past the root or {@code oldParent} is encountered.
500          * @param amt The amount to increment by. This must be positive.
501          * @param oldParent The previous parent for this stream.
502          */
503         private void incrementPrioritizableForTree(int amt, Http2Stream oldParent) {
504             if (amt != 0) {
505                 incrementPrioritizableForTree0(amt, oldParent);
506             }
507         }
508 
509         /**
510          * Direct calls to this method are discouraged.
511          * Instead use {@link #incrementPrioritizableForTree(int, Http2Stream)}.
512          */
513         private void incrementPrioritizableForTree0(int amt, Http2Stream oldParent) {
514             assert amt > 0 && Integer.MAX_VALUE - amt >= prioritizableForTree;
515             prioritizableForTree += amt;
516             if (parent != null && parent != oldParent) {
517                 parent.incrementPrioritizableForTree0(amt, oldParent);
518             }
519         }
520 
521         /**
522          * Recursively increment the {@link #prioritizableForTree} for this object up the parent links until
523          * either we go past the root.
524          * @param amt The amount to decrement by. This must be positive.
525          */
526         private void decrementPrioritizableForTree(int amt) {
527             if (amt != 0) {
528                 decrementPrioritizableForTree0(amt);
529             }
530         }
531 
532         /**
533          * Direct calls to this method are discouraged. Instead use {@link #decrementPrioritizableForTree(int)}.
534          */
535         private void decrementPrioritizableForTree0(int amt) {
536             assert amt > 0 && prioritizableForTree >= amt;
537             prioritizableForTree -= amt;
538             if (parent != null) {
539                 parent.decrementPrioritizableForTree0(amt);
540             }

568         final void weight(short weight) {
569             if (weight != this.weight) {
570                 final short oldWeight = this.weight;
571                 this.weight = weight;
572                 for (int i = 0; i < listeners.size(); i++) {
573                     try {
574                         listeners.get(i).onWeightChanged(this, oldWeight);
575                     } catch (RuntimeException e) {
576                         logger.error("Caught RuntimeException from listener onWeightChanged.", e);
577                     }
578                 }
579             }
580         }

582         /**
583          * Remove all children with the exception of {@code streamToRetain}.
584          * This method is intended to be used to support an exclusive priority dependency operation.
585          * @return The map of children prior to this operation, excluding {@code streamToRetain} if present.
586          */
587         private IntObjectMap<DefaultStream> retain(DefaultStream streamToRetain) {
588             streamToRetain = children.remove(streamToRetain.id());
589             IntObjectMap<DefaultStream> prevChildren = children;
590             // This map should be re-initialized in anticipation for the 1 exclusive child which will be added.
591             // It will either be added directly in this method, or after this method is called...but it will be added.
592             initChildren();
593             if (streamToRetain == null) {
594                 prioritizableForTree = isPrioritizable() ? 1 : 0;
595             } else {
596                 // prioritizableForTree does not change because it is assumed all children node will still be
597                 // descendants through an exclusive priority tree operation.
598                 children.put(streamToRetain.id(), streamToRetain);
599             }
600             return prevChildren;
601         }

603         /**
604          * Adds a child to this priority. If exclusive is set, any children of this node are moved to being dependent on
605          * the child.
606          */
607         final void takeChild(DefaultStream child, boolean exclusive, List<ParentChangedEvent> events) {
608             DefaultStream oldParent = child.parent();
609 
610             if (oldParent != this) {
611                 events.add(new ParentChangedEvent(child, oldParent));
612                 notifyParentChanging(child, this);
613                 child.parent = this;
614                 // We need the removal operation to happen first so the prioritizableForTree for the old parent to root
615                 // path is updated with the correct child.prioritizableForTree() value. Note that the removal operation
616                 // may not be successful and may return null. This is because when an exclusive dependency is processed
617                 // the children are removed in a previous recursive call but the child's parent link is updated here.
618                 if (oldParent != null && oldParent.children.remove(child.id()) != null) {
619                     if (!child.isDescendantOf(oldParent)) {
620                         oldParent.decrementPrioritizableForTree(child.prioritizableForTree());
621                         if (oldParent.prioritizableForTree() == 0) {
622                             // There are a few risks with immediately removing nodes from the priority tree:
623                             // 1. We are removing nodes while we are potentially shifting the tree. There are no
624                             // concrete cases known but is risky because it could invalidate the data structure.
625                             // 2. We are notifying listeners of the removal while the tree is in flux. Currently the
626                             // codec listeners make no assumptions about priority tree structure when being notified.
627                             removeStream(oldParent);
628                         }
629                     }
630                 }
631 
632                 // Lazily initialize the children to save object allocations.
633                 initChildrenIfEmpty();
634 
635                 final Http2Stream oldChild = children.put(child.id(), child);
636                 assert oldChild == null : "A stream with the same stream ID was already in the child map.";
637                 incrementPrioritizableForTree(child.prioritizableForTree(), oldParent);
638             }
639 
640             if (exclusive && !children.isEmpty()) {
641                 // If it was requested that this child be the exclusive dependency of this node,
642                 // move any previous children to the child node, becoming grand children of this node.
643                 for (DefaultStream grandchild : retain(child).values()) {
644                     child.takeChild(grandchild, false, events);
645                 }
646             }
647         }
648 
649         /**
650          * Removes the child priority and moves any of its dependencies to being direct dependencies on this node.
651          */
652         final boolean removeChild(DefaultStream child) {
653             if (child.prioritizableForTree() == 0 && children.remove(child.id()) != null) {
654                 List<ParentChangedEvent> events = new ArrayList<ParentChangedEvent>(1 + child.numChildren());
655                 events.add(new ParentChangedEvent(child, child.parent()));
656                 notifyParentChanging(child, null);
657                 child.parent = null;
658                 decrementPrioritizableForTree(child.prioritizableForTree());
659 
660                 // Move up any grand children to be directly dependent on this node.
661                 for (DefaultStream grandchild : child.children.values()) {
662                     takeChild(grandchild, false, events);
663                 }
664 
665                 if (prioritizableForTree() == 0) {
666                     // There are a few risks with immediately removing nodes from the priority tree:
667                     // 1. We are removing nodes while we are potentially shifting the tree. There are no
668                     // concrete cases known but is risky because it could invalidate the data structure.
669                     // 2. We are notifying listeners of the removal while the tree is in flux. Currently the
670                     // codec listeners make no assumptions about priority tree structure when being notified.
671                     removeStream(this);
672                 }
673                 notifyParentChanged(events);
674                 return true;
675             }
676             return false;
677         }

679         /**
680          * Provides the lazy initialization for the {@link DefaultStream} data map.
681          */
682         private class PropertyMap {
683             Object[] values = EmptyArrays.EMPTY_OBJECTS;
684 
685             <V> V add(DefaultPropertyKey key, V value) {
686                 resizeIfNecessary(key.index);
687                 @SuppressWarnings("unchecked")
688                 V prevValue = (V) values[key.index];
689                 values[key.index] = value;
690                 return prevValue;
691             }
692 
693             @SuppressWarnings("unchecked")
694             <V> V get(DefaultPropertyKey key) {
695                 if (key.index >= values.length) {
696                     return null;
697                 }
698                 return (V) values[key.index];
699             }
700 
701             @SuppressWarnings("unchecked")
702             <V> V remove(DefaultPropertyKey key) {
703                 V prevValue = null;
704                 if (key.index < values.length) {
705                     prevValue = (V) values[key.index];
706                     values[key.index] = null;
707                 }
708                 return prevValue;

1024         private void checkNewStreamAllowed(int streamId, State state) throws Http2Exception {
1025             if (goAwayReceived() && streamId > localEndpoint.lastStreamKnownByPeer()) {
1026                 throw connectionError(PROTOCOL_ERROR, "Cannot create stream %d since this endpoint has received a " +
1027                                                       "GOAWAY frame with last stream id %d.", streamId,
1028                                                       localEndpoint.lastStreamKnownByPeer());
1029             }
1030             if (streamId < 0) {
1031                 throw new Http2NoMoreStreamIdsException();
1032             }
1033             if (!isValidStreamId(streamId)) {
1034                 throw connectionError(PROTOCOL_ERROR, "Request stream %d is not correct for %s connection", streamId,
1035                         server ? "server" : "client");
1036             }
1037             // This check must be after all id validated checks, but before the max streams check because it may be
1038             // recoverable to some degree for handling frames which can be sent on closed streams.
1039             if (streamId < nextStreamId) {
1040                 throw closedStreamError(PROTOCOL_ERROR, "Request stream %d is behind the next expected stream %d",
1041                         streamId, nextStreamId);
1042             }
1043             if (isExhausted()) {
1044                 throw connectionError(REFUSED_STREAM, "Stream IDs are exhausted for this endpoint.");
1045             }
1046             if ((state.localSideOpen() || state.remoteSideOpen()) && !canOpenStream()) {
1047                 throw connectionError(REFUSED_STREAM, "Maximum active streams violated for this endpoint.");
1048             }
1049         }

1072      * active streams in order to prevent modification while iterating.
1073      */
1074     private final class ActiveStreams {
1075 
1076         private final List<Listener> listeners;
1077         private final Queue<Event> pendingEvents = new ArrayDeque<Event>(4);
1078         private final Set<Http2Stream> streams = new LinkedHashSet<Http2Stream>();
1079         private int pendingIterations;

1082             this.listeners = listeners;

1086             return streams.size();
1087         }

1089         public void activate(final DefaultStream stream) {
1090             if (allowModifications()) {
1091                 addToActiveStreams(stream);
1092             } else {
1093                 pendingEvents.add(new Event() {
1094                     @Override
1095                     public void process() {
1096                         addToActiveStreams(stream);
1097                     }
1098                 });
1099             }
1100         }
1101 
1102         public void deactivate(final DefaultStream stream) {
1103             if (allowModifications()) {
1104                 removeFromActiveStreams(stream);
1105             } else {
1106                 pendingEvents.add(new Event() {
1107                     @Override
1108                     public void process() {

1110                     }
1111                 });

1115         public Http2Stream forEachActiveStream(Http2StreamVisitor visitor) throws Http2Exception {
1116             ++pendingIterations;
1117             try {
1118                 for (Http2Stream stream : streams) {
1119                     if (!visitor.visit(stream)) {
1120                         return stream;
1121                     }
1122                 }
1123                 return null;
1124             } finally {
1125                 --pendingIterations;
1126                 if (allowModifications()) {
1127                     for (;;) {
1128                         Event event = pendingEvents.poll();
1129                         if (event == null) {
1130                             break;
1131                         }
1132                         try {
1133                             event.process();
1134                         } catch (RuntimeException e) {
1135                             logger.error("Caught RuntimeException while processing pending ActiveStreams$Event.", e);
1136                         }
1137                     }
1138                 }
1139             }
1140         }

1143             if (streams.add(stream)) {
1144                 // Update the number of active streams initiated by the endpoint.
1145                 stream.createdBy().numActiveStreams++;

1147                 for (int i = 0; i < listeners.size(); i++) {
1148                     try {
1149                         listeners.get(i).onStreamActive(stream);

1152                     }
1153                 }
1154             }
1155         }

1159                 // Update the number of active streams initiated by the endpoint.
1160                 stream.createdBy().numActiveStreams--;
1161             }
1162             notifyClosed(stream);
1163             removeStream(stream);
1164         }
