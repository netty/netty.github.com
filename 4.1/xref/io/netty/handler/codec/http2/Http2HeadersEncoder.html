16  package io.netty.handler.codec.http2;
17  
18  import io.netty.buffer.ByteBuf;
19  
20  /**
21   * Encodes {@link Http2Headers} into HPACK-encoded headers blocks.
22   */
23  public interface Http2HeadersEncoder {
24      /**
25       * Configuration related elements for the {@link Http2HeadersEncoder} interface
26       */
27      interface Configuration {
28          /**
29           * Access the Http2HeaderTable for this {@link Http2HeadersEncoder}
30           */
31          Http2HeaderTable headerTable();
32      }
33  
34      /**
35       * Determine if a header name/value pair is treated as
36       * <a href="http://tools.ietf.org/html/draft-ietf-httpbis-header-compression-12#section-7.1.3">sensitive</a>.
37       * If the object can be dynamically modified and shared across multiple connections it may need to be thread safe.
38       */
39      interface SensitivityDetector {
40          /**
41           * Determine if a header {@code name}/{@code value} pair should be treated as
42           * <a href="http://tools.ietf.org/html/draft-ietf-httpbis-header-compression-12#section-7.1.3">sensitive</a>.
43           * @param name The name for the header.
44           * @param value The value of the header.
45           * @return {@code true} if a header {@code name}/{@code value} pair should be treated as
46           * <a href="http://tools.ietf.org/html/draft-ietf-httpbis-header-compression-12#section-7.1.3">sensitive</a>.
47           * {@code false} otherwise.
48           */
49          boolean isSensitive(CharSequence name, CharSequence value);
50      }
51  
52      /**
53       * Encodes the given headers and writes the output headers block to the given output buffer.
54       *
55       * @param headers the headers to be encoded.
56       * @param buffer the buffer to receive the encoded headers.
57       */
58      void encodeHeaders(Http2Headers headers, ByteBuf buffer) throws Http2Exception;
59  
60      /**
61       * Get the {@link Configuration} for this {@link Http2HeadersEncoder}
62       */
63      Configuration configuration();
64  
65      /**
66       * Always return {@code false} for {@link SensitivityDetector#isSensitive(CharSequence, CharSequence)}.
67       */
68      SensitivityDetector NEVER_SENSITIVE = new SensitivityDetector() {
69          @Override
70          public boolean isSensitive(CharSequence name, CharSequence value) {
71              return false;
72          }
73      };
74  
75      /**
76       * Always return {@code true} for {@link SensitivityDetector#isSensitive(CharSequence, CharSequence)}.
77       */
78      SensitivityDetector ALWAYS_SENSITIVE = new SensitivityDetector() {
79          @Override
80          public boolean isSensitive(CharSequence name, CharSequence value) {
81              return true;
82          }
83      };
84  }
