128     @Override
129     public void readFrame(ChannelHandlerContext ctx, ByteBuf input, Http2FrameListener listener)
130             throws Http2Exception {
131         try {
132             while (input.isReadable()) {
133                 switch (state) {
134                     case FRAME_HEADER:
135                         processHeaderState(input);
136                         if (state == State.FRAME_HEADER) {
137                             // Wait until the entire header has arrived.
138                             return;
139                         }
140 
141                         // The header is complete, fall into the next case to process the payload.
142                         // This is to ensure the proper handling of zero-length payloads. In this
143                         // case, we don't want to loop around because there may be no more data
144                         // available, causing us to exit the loop. Instead, we just want to perform
145                         // the first pass at payload processing now.
146                     case FRAME_PAYLOAD:
147                         processPayloadState(ctx, input, listener);
148                         if (state == State.FRAME_PAYLOAD) {
149                             // Wait until the entire payload has arrived.
150                             return;
151                         }
152                         break;
153                     case ERROR:
154                         input.skipBytes(input.readableBytes());
155                         return;
156                     default:
157                         throw new IllegalStateException("Should never get here");
158                 }
159             }
160         } catch (Http2Exception e) {
161             state = State.ERROR;
162             throw e;
163         } catch (RuntimeException e) {
164             state = State.ERROR;
165             throw e;
166         } catch (Error e) {
167             state = State.ERROR;
168             throw e;
169         }
170     }
171 
172     private void processHeaderState(ByteBuf in) throws Http2Exception {
173         if (in.readableBytes() < FRAME_HEADER_LENGTH) {
174             // Wait until the entire frame header has been read.
175             return;
176         }
177 
178         // Read the header and prepare the unmarshaller to read the frame.
179         payloadLength = in.readUnsignedMedium();
180         if (payloadLength > maxFrameSize) {
181             throw connectionError(PROTOCOL_ERROR, "Frame length: %d exceeds maximum: %d", payloadLength, maxFrameSize);
182         }
183         frameType = in.readByte();
184         flags = new Http2Flags(in.readUnsignedByte());
185         streamId = readUnsignedInt(in);
186 
187         switch (frameType) {
188             case DATA:
189                 verifyDataFrame();
190                 break;
191             case HEADERS:
192                 verifyHeadersFrame();
193                 break;
194             case PRIORITY:
195                 verifyPriorityFrame();
196                 break;
197             case RST_STREAM:
198                 verifyRstStreamFrame();
199                 break;
200             case SETTINGS:
201                 verifySettingsFrame();
202                 break;
203             case PUSH_PROMISE:
204                 verifyPushPromiseFrame();
205                 break;
206             case PING:
207                 verifyPingFrame();
208                 break;
209             case GO_AWAY:
210                 verifyGoAwayFrame();
211                 break;
212             case WINDOW_UPDATE:
213                 verifyWindowUpdateFrame();
214                 break;
215             case CONTINUATION:
216                 verifyContinuationFrame();
217                 break;
218             default:
219                 // Unknown frame type, could be an extension.
220                 break;
221         }
222 
223         // Start reading the payload for the frame.
224         state = State.FRAME_PAYLOAD;
225     }
226 
227     private void processPayloadState(ChannelHandlerContext ctx, ByteBuf in, Http2FrameListener listener)
228                     throws Http2Exception {
229         if (in.readableBytes() < payloadLength) {
230             // Wait until the entire payload has been read.
231             return;
232         }
233 
234         // Get a view of the buffer for the size of the payload.
235         ByteBuf payload = in.readSlice(payloadLength);
236 
237         // Read the payload and fire the frame event to the listener.
238         switch (frameType) {
239             case DATA:
240                 readDataFrame(ctx, payload, listener);
241                 break;
242             case HEADERS:
243                 readHeadersFrame(ctx, payload, listener);
244                 break;
245             case PRIORITY:
246                 readPriorityFrame(ctx, payload, listener);
247                 break;
248             case RST_STREAM:
249                 readRstStreamFrame(ctx, payload, listener);
250                 break;
251             case SETTINGS:
252                 readSettingsFrame(ctx, payload, listener);
253                 break;
254             case PUSH_PROMISE:
255                 readPushPromiseFrame(ctx, payload, listener);
256                 break;
257             case PING:
258                 readPingFrame(ctx, payload, listener);
259                 break;
260             case GO_AWAY:
261                 readGoAwayFrame(ctx, payload, listener);
262                 break;
263             case WINDOW_UPDATE:
264                 readWindowUpdateFrame(ctx, payload, listener);
265                 break;
266             case CONTINUATION:
267                 readContinuationFrame(payload, listener);
268                 break;
269             default:
270                 readUnknownFrame(ctx, payload, listener);
271                 break;
272         }
273 
274         // Go back to reading the next frame header.
275         state = State.FRAME_HEADER;
276     }

278     private void verifyDataFrame() throws Http2Exception {
279         verifyNotProcessingHeaders();
280         verifyPayloadLength(payloadLength);
281 
282         if (payloadLength < flags.getPaddingPresenceFieldLength()) {
283             throw streamError(streamId, FRAME_SIZE_ERROR,
284                     "Frame length %d too small.", payloadLength);
285         }
286     }
287 
288     private void verifyHeadersFrame() throws Http2Exception {
289         verifyNotProcessingHeaders();
290         verifyPayloadLength(payloadLength);
291 
292         int requiredLength = flags.getPaddingPresenceFieldLength() + flags.getNumPriorityBytes();
293         if (payloadLength < requiredLength) {
294             throw streamError(streamId, FRAME_SIZE_ERROR,
295                     "Frame length too small." + payloadLength);
296         }
297     }
298 
299     private void verifyPriorityFrame() throws Http2Exception {
300         verifyNotProcessingHeaders();
301 
302         if (payloadLength != PRIORITY_ENTRY_LENGTH) {
303             throw streamError(streamId, FRAME_SIZE_ERROR,
304                     "Invalid frame length %d.", payloadLength);
305         }
306     }
307 
308     private void verifyRstStreamFrame() throws Http2Exception {
309         verifyNotProcessingHeaders();
310 
311         if (payloadLength != INT_FIELD_LENGTH) {
312             throw streamError(streamId, FRAME_SIZE_ERROR,
313                     "Invalid frame length %d.", payloadLength);
314         }
315     }
316 
317     private void verifySettingsFrame() throws Http2Exception {
318         verifyNotProcessingHeaders();
319         verifyPayloadLength(payloadLength);
320         if (streamId != 0) {
321             throw connectionError(PROTOCOL_ERROR, "A stream ID must be zero.");
322         }
323         if (flags.ack() && payloadLength > 0) {
324             throw connectionError(PROTOCOL_ERROR, "Ack settings frame must have an empty payload.");
325         }
326         if (payloadLength % SETTING_ENTRY_LENGTH > 0) {
327             throw connectionError(FRAME_SIZE_ERROR, "Frame length %d invalid.", payloadLength);
328         }
329     }
330 
331     private void verifyPushPromiseFrame() throws Http2Exception {
332         verifyNotProcessingHeaders();
333         verifyPayloadLength(payloadLength);
334 
335         // Subtract the length of the promised stream ID field, to determine the length of the
336         // rest of the payload (header block fragment + payload).
337         int minLength = flags.getPaddingPresenceFieldLength() + INT_FIELD_LENGTH;
338         if (payloadLength < minLength) {
339             throw streamError(streamId, FRAME_SIZE_ERROR,
340                     "Frame length %d too small.", payloadLength);
341         }
342     }
343 
344     private void verifyPingFrame() throws Http2Exception {
345         verifyNotProcessingHeaders();
346         if (streamId != 0) {
347             throw connectionError(PROTOCOL_ERROR, "A stream ID must be zero.");
348         }
349         if (payloadLength != PING_FRAME_PAYLOAD_LENGTH) {
350             throw connectionError(FRAME_SIZE_ERROR,
351                     "Frame length %d incorrect size for ping.", payloadLength);
352         }
353     }
354 
355     private void verifyGoAwayFrame() throws Http2Exception {
356         verifyNotProcessingHeaders();
357         verifyPayloadLength(payloadLength);
358 
359         if (streamId != 0) {
360             throw connectionError(PROTOCOL_ERROR, "A stream ID must be zero.");
361         }
362         if (payloadLength < 8) {
363             throw connectionError(FRAME_SIZE_ERROR, "Frame length %d too small.", payloadLength);
364         }
365     }
366 
367     private void verifyWindowUpdateFrame() throws Http2Exception {
368         verifyNotProcessingHeaders();
369         verifyStreamOrConnectionId(streamId, "Stream ID");
370 
371         if (payloadLength != INT_FIELD_LENGTH) {
372             throw streamError(streamId, FRAME_SIZE_ERROR,
373                     "Invalid frame length %d.", payloadLength);
374         }
375     }

377     private void verifyContinuationFrame() throws Http2Exception {
378         verifyPayloadLength(payloadLength);
379 
380         if (headersContinuation == null) {
381             throw connectionError(PROTOCOL_ERROR, "Received %s frame but not currently processing headers.",
382                     frameType);
383         }
384 
385         if (streamId != headersContinuation.getStreamId()) {
386             throw connectionError(PROTOCOL_ERROR, "Continuation stream ID does not match pending headers. "
387                     + "Expected %d, but received %d.", headersContinuation.getStreamId(), streamId);
388         }
389 
390         if (payloadLength < flags.getPaddingPresenceFieldLength()) {
391             throw streamError(streamId, FRAME_SIZE_ERROR,
392                     "Frame length %d too small for padding.", payloadLength);
393         }
394     }

396     private void readDataFrame(ChannelHandlerContext ctx, ByteBuf payload,
397             Http2FrameListener listener) throws Http2Exception {
398         short padding = readPadding(payload);
399 
400         // Determine how much data there is to read by removing the trailing
401         // padding.
402         int dataLength = payload.readableBytes() - padding;
403         if (dataLength < 0) {
404             throw streamError(streamId, FRAME_SIZE_ERROR,
405                     "Frame payload too small for padding.");
406         }
407 
408         ByteBuf data = payload.readSlice(dataLength);
409         listener.onDataRead(ctx, streamId, data, padding, flags.endOfStream());
410         payload.skipBytes(payload.readableBytes());
411     }

413     private void readHeadersFrame(final ChannelHandlerContext ctx, ByteBuf payload,
414             Http2FrameListener listener) throws Http2Exception {
415         final int headersStreamId = streamId;
416         final Http2Flags headersFlags = flags;
417         final int padding = readPadding(payload);
418 
419         // The callback that is invoked is different depending on whether priority information
420         // is present in the headers frame.
421         if (flags.priorityPresent()) {
422             long word1 = payload.readUnsignedInt();
423             final boolean exclusive = (word1 & 0x80000000L) != 0;
424             final int streamDependency = (int) (word1 & 0x7FFFFFFFL);
425             final short weight = (short) (payload.readUnsignedByte() + 1);
426             final ByteBuf fragment = payload.readSlice(payload.readableBytes() - padding);
427 
428             // Create a handler that invokes the listener when the header block is complete.
429             headersContinuation = new HeadersContinuation() {
430                 @Override
431                 public int getStreamId() {
432                     return headersStreamId;
433                 }
434 
435                 @Override
436                 public void processFragment(boolean endOfHeaders, ByteBuf fragment,
437                         Http2FrameListener listener) throws Http2Exception {
438                     final HeadersBlockBuilder hdrBlockBuilder = headersBlockBuilder();
439                     hdrBlockBuilder.addFragment(fragment, ctx.alloc(), endOfHeaders);
440                     if (endOfHeaders) {
441                         listener.onHeadersRead(ctx, headersStreamId, hdrBlockBuilder.headers(), streamDependency,
442                                 weight, exclusive, padding, headersFlags.endOfStream());
443                     }
444                 }
445             };
446 
447             // Process the initial fragment, invoking the listener's callback if end of headers.
448             headersContinuation.processFragment(flags.endOfHeaders(), fragment, listener);
449             return;
450         }
451 
452         // The priority fields are not present in the frame. Prepare a continuation that invokes
453         // the listener callback without priority information.
454         headersContinuation = new HeadersContinuation() {
455             @Override
456             public int getStreamId() {
457                 return headersStreamId;
458             }
459 
460             @Override
461             public void processFragment(boolean endOfHeaders, ByteBuf fragment,
462                     Http2FrameListener listener) throws Http2Exception {
463                 final HeadersBlockBuilder hdrBlockBuilder = headersBlockBuilder();
464                 hdrBlockBuilder.addFragment(fragment, ctx.alloc(), endOfHeaders);
465                 if (endOfHeaders) {
466                     listener.onHeadersRead(ctx, headersStreamId, hdrBlockBuilder.headers(), padding,
467                                     headersFlags.endOfStream());
468                 }
469             }
470         };
471 
472         // Process the initial fragment, invoking the listener's callback if end of headers.
473         final ByteBuf fragment = payload.readSlice(payload.readableBytes() - padding);
474         headersContinuation.processFragment(flags.endOfHeaders(), fragment, listener);
475     }

477     private void readPriorityFrame(ChannelHandlerContext ctx, ByteBuf payload,
478             Http2FrameListener listener) throws Http2Exception {
479         long word1 = payload.readUnsignedInt();
480         boolean exclusive = (word1 & 0x80000000L) != 0;
481         int streamDependency = (int) (word1 & 0x7FFFFFFFL);
482         short weight = (short) (payload.readUnsignedByte() + 1);
483         listener.onPriorityRead(ctx, streamId, streamDependency, weight, exclusive);
484     }

486     private void readRstStreamFrame(ChannelHandlerContext ctx, ByteBuf payload,
487             Http2FrameListener listener) throws Http2Exception {
488         long errorCode = payload.readUnsignedInt();
489         listener.onRstStreamRead(ctx, streamId, errorCode);
490     }

492     private void readSettingsFrame(ChannelHandlerContext ctx, ByteBuf payload,
493             Http2FrameListener listener) throws Http2Exception {
494         if (flags.ack()) {
495             listener.onSettingsAckRead(ctx);
496         } else {
497             int numSettings = payloadLength / SETTING_ENTRY_LENGTH;
498             Http2Settings settings = new Http2Settings();
499             for (int index = 0; index < numSettings; ++index) {
500                 char id = (char) payload.readUnsignedShort();
501                 long value = payload.readUnsignedInt();
502                 try {
503                     settings.put(id, Long.valueOf(value));
504                 } catch (IllegalArgumentException e) {
505                     switch(id) {
506                     case SETTINGS_MAX_FRAME_SIZE:
507                         throw connectionError(FRAME_SIZE_ERROR, e, e.getMessage());
508                     case SETTINGS_INITIAL_WINDOW_SIZE:
509                         throw connectionError(FLOW_CONTROL_ERROR, e, e.getMessage());
510                     default:
511                         throw connectionError(PROTOCOL_ERROR, e, e.getMessage());
512                     }
513                 }
514             }
515             listener.onSettingsRead(ctx, settings);
516         }
517     }

519     private void readPushPromiseFrame(final ChannelHandlerContext ctx, ByteBuf payload,
520             Http2FrameListener listener) throws Http2Exception {
521         final int pushPromiseStreamId = streamId;
522         final int padding = readPadding(payload);
523         final int promisedStreamId = readUnsignedInt(payload);
524 
525         // Create a handler that invokes the listener when the header block is complete.
526         headersContinuation = new HeadersContinuation() {
527             @Override
528             public int getStreamId() {
529                 return pushPromiseStreamId;
530             }
531 
532             @Override
533             public void processFragment(boolean endOfHeaders, ByteBuf fragment,
534                     Http2FrameListener listener) throws Http2Exception {
535                 headersBlockBuilder().addFragment(fragment, ctx.alloc(), endOfHeaders);
536                 if (endOfHeaders) {
537                     listener.onPushPromiseRead(ctx, pushPromiseStreamId, promisedStreamId,
538                             headersBlockBuilder().headers(), padding);
539                 }
540             }
541         };
542 
543         // Process the initial fragment, invoking the listener's callback if end of headers.
544         final ByteBuf fragment = payload.readSlice(payload.readableBytes() - padding);
545         headersContinuation.processFragment(flags.endOfHeaders(), fragment, listener);
546     }

548     private void readPingFrame(ChannelHandlerContext ctx, ByteBuf payload,
549             Http2FrameListener listener) throws Http2Exception {
550         ByteBuf data = payload.readSlice(payload.readableBytes());
551         if (flags.ack()) {
552             listener.onPingAckRead(ctx, data);
553         } else {
554             listener.onPingRead(ctx, data);
555         }
556     }

560         int lastStreamId = readUnsignedInt(payload);
561         long errorCode = payload.readUnsignedInt();
562         ByteBuf debugData = payload.readSlice(payload.readableBytes());
563         listener.onGoAwayRead(ctx, lastStreamId, errorCode, debugData);

566     private void readWindowUpdateFrame(ChannelHandlerContext ctx, ByteBuf payload,
567             Http2FrameListener listener) throws Http2Exception {
568         int windowSizeIncrement = readUnsignedInt(payload);
569         if (windowSizeIncrement == 0) {
570             throw streamError(streamId, PROTOCOL_ERROR,
571                     "Received WINDOW_UPDATE with delta 0 for stream: %d", streamId);
572         }
573         listener.onWindowUpdateRead(ctx, streamId, windowSizeIncrement);
574     }

576     private void readContinuationFrame(ByteBuf payload, Http2FrameListener listener)
577             throws Http2Exception {
578         // Process the initial fragment, invoking the listener's callback if end of headers.
579         final ByteBuf continuationFragment = payload.readSlice(payload.readableBytes());
580         headersContinuation.processFragment(flags.endOfHeaders(), continuationFragment,
581                 listener);

584     private void readUnknownFrame(ChannelHandlerContext ctx, ByteBuf payload, Http2FrameListener listener)

586         payload = payload.readSlice(payload.readableBytes());
587         listener.onUnknownFrame(ctx, frameType, streamId, flags, payload);

590     /**
591      * If padding is present in the payload, reads the next byte as padding. Otherwise, returns zero.
592      */
593     private short readPadding(ByteBuf payload) {
594         if (!flags.paddingPresent()) {
595             return 0;
596         }
597         return payload.readUnsignedByte();
598     }

600     /**
601      * Base class for processing of HEADERS and PUSH_PROMISE header blocks that potentially span
602      * multiple frames. The implementation of this interface will perform the final callback to the
603      * {@link Http2FrameListener} once the end of headers is reached.
604      */
605     private abstract class HeadersContinuation {
606         private final HeadersBlockBuilder builder = new HeadersBlockBuilder();
607 
608         /**
609          * Returns the stream for which headers are currently being processed.
610          */
611         abstract int getStreamId();
612 
613         /**
614          * Processes the next fragment for the current header block.
615          *
616          * @param endOfHeaders whether the fragment is the last in the header block.
617          * @param fragment the fragment of the header block to be added.
618          * @param listener the listener to be notified if the header block is completed.
619          */
620         abstract void processFragment(boolean endOfHeaders, ByteBuf fragment,
621                 Http2FrameListener listener) throws Http2Exception;
622 
623         final HeadersBlockBuilder headersBlockBuilder() {
624             return builder;
625         }
626 
627         /**
628          * Free any allocated resources.
629          */
630         final void close() {
631             builder.close();
632         }
633     }
634 
635     /**
636      * Utility class to help with construction of the headers block that may potentially span
637      * multiple frames.
638      */
639     protected class HeadersBlockBuilder {
640         private ByteBuf headerBlock;
641 
642         /**
643          * The local header size maximum has been exceeded while accumulating bytes.
644          * @throws Http2Exception A connection error indicating too much data has been received.
645          */
646         private void headerSizeExceeded() throws Http2Exception {
647             close();
648             throw connectionError(ENHANCE_YOUR_CALM, "Header size exceeded max allowed size (%d)",
649                     headersDecoder.configuration().maxHeaderSize());
650         }
651 
652         /**
653          * Adds a fragment to the block.
654          *
655          * @param fragment the fragment of the headers block to be added.
656          * @param alloc allocator for new blocks if needed.
657          * @param endOfHeaders flag indicating whether the current frame is the end of the headers.
658          *            This is used for an optimization for when the first fragment is the full
659          *            block. In that case, the buffer is used directly without copying.
660          */
661         final void addFragment(ByteBuf fragment, ByteBufAllocator alloc, boolean endOfHeaders) throws Http2Exception {
662             if (headerBlock == null) {
663                 if (fragment.readableBytes() > headersDecoder.configuration().maxHeaderSize()) {
664                     headerSizeExceeded();
665                 }
666                 if (endOfHeaders) {
667                     // Optimization - don't bother copying, just use the buffer as-is. Need
668                     // to retain since we release when the header block is built.
669                     headerBlock = fragment.retain();
670                 } else {
671                     headerBlock = alloc.buffer(fragment.readableBytes());
672                     headerBlock.writeBytes(fragment);
673                 }
674                 return;
675             }
676             if (headersDecoder.configuration().maxHeaderSize() - fragment.readableBytes() <
677                     headerBlock.readableBytes()) {
678                 headerSizeExceeded();
679             }
680             if (headerBlock.isWritable(fragment.readableBytes())) {
681                 // The buffer can hold the requested bytes, just write it directly.
682                 headerBlock.writeBytes(fragment);
683             } else {
684                 // Allocate a new buffer that is big enough to hold the entire header block so far.
685                 ByteBuf buf = alloc.buffer(headerBlock.readableBytes() + fragment.readableBytes());
686                 buf.writeBytes(headerBlock);
687                 buf.writeBytes(fragment);
