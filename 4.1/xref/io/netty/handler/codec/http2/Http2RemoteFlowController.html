15  package io.netty.handler.codec.http2;
16  
17  import io.netty.channel.ChannelHandlerContext;
18  
19  /**
20   * A {@link Http2FlowController} for controlling the flow of outbound {@code DATA} frames to the remote
21   * endpoint.
22   */
23  public interface Http2RemoteFlowController extends Http2FlowController {
24      /**
25       * Get the {@link ChannelHandlerContext} for which to apply flow control on.
26       * <p>
27       * This is intended for us by {@link FlowControlled} implementations only. Use with caution.
28       * @return The {@link ChannelHandlerContext} for which to apply flow control on.
29       */
30      ChannelHandlerContext channelHandlerContext();
31  
32      /**
33       * Queues a payload for transmission to the remote endpoint. There is no guarantee as to when the data
34       * will be written or how it will be assigned to frames.
35       * before sending.
36       * <p>
37       * Writes do not actually occur until {@link #writePendingBytes()} is called.
38       *
39       * @param stream the subject stream. Must not be the connection stream object.
40       * @param payload payload to write subject to flow-control accounting and ordering rules.
41       */
42      void addFlowControlled(Http2Stream stream, FlowControlled payload);
43  
44      /**
45       * Determine if {@code stream} has any {@link FlowControlled} frames currently queued.
46       * @param stream the stream to check if it has flow controlled frames.
47       * @return {@code true} if {@code stream} has any {@link FlowControlled} frames currently queued.
48       */
49      boolean hasFlowControlled(Http2Stream stream);
50  
51      /**
52       * Write all data pending in the flow controller up to the flow-control limits.
53       *
54       * @throws Http2Exception throws if a protocol-related error occurred.
55       */
56      void writePendingBytes() throws Http2Exception;
57  
58      /**
59       * Set the active listener on the flow-controller.
60       *
61       * @param listener to notify when the a write occurs, can be {@code null}.
62       */
63      void listener(Listener listener);
64  
65      /**
66       * Determine if the {@code stream} has bytes remaining for use in the flow control window.
67       * <p>
68       * Note that this only takes into account HTTP/2 flow control. It does <strong>not</strong> take into account
69       * the underlying {@link io.netty.channel.Channel#isWritable()}.
70       * @param stream The stream to test.
71       * @return {@code true} if if the {@code stream} has bytes remaining for use in the flow control window.
72       * {@code false} otherwise.
73       */
74      boolean isWritable(Http2Stream stream);
75  
76      /**
77       * Notification that the writability of {@link #channelHandlerContext()} has changed.
78       * @throws Http2Exception If any writes occur as a result of this call and encounter errors.
79       */
80      void channelWritabilityChanged() throws Http2Exception;
81  
82      /**
83       * Implementations of this interface are used to progressively write chunks of the underlying
84       * payload to the stream. A payload is considered to be fully written if {@link #write} has
85       * been called at least once and it's {@link #size} is now zero.
86       */
87      interface FlowControlled {
88          /**
89           * The size of the payload in terms of bytes applied to the flow-control window.
90           * Some payloads like {@code HEADER} frames have no cost against flow control and would
91           * return 0 for this value even though they produce a non-zero number of bytes on
92           * the wire. Other frames like {@code DATA} frames have both their payload and padding count
93           * against flow-control.
94           */
95          int size();
96  
97          /**
98           * Called to indicate that an error occurred before this object could be completely written.
99           * <p>
100          * The {@link Http2RemoteFlowController} will make exactly one call to either
101          * this method or {@link #writeComplete()}.
102          * </p>
103          *
104          * @param ctx The context to use if any communication needs to occur as a result of the error.
105          * This may be {@code null} if an exception occurs when the connection has not been established yet.
106          * @param cause of the error.
107          */
108         void error(ChannelHandlerContext ctx, Throwable cause);
109 
110         /**
111          * Called after this object has been successfully written.
112          * <p>
113          * The {@link Http2RemoteFlowController} will make exactly one call to either
114          * this method or {@link #error(ChannelHandlerContext, Throwable)}.
115          * </p>
116          */
117         void writeComplete();
118 
119         /**
120          * Writes up to {@code allowedBytes} of the encapsulated payload to the stream. Note that
121          * a value of 0 may be passed which will allow payloads with flow-control size == 0 to be
122          * written. The flow-controller may call this method multiple times with different values until
123          * the payload is fully written, i.e it's size after the write is 0.
124          * <p>
125          * When an exception is thrown the {@link Http2RemoteFlowController} will make a call to
126          * {@link #error(ChannelHandlerContext, Throwable)}.
127          * </p>
128          *
129          * @param ctx The context to use for writing.
130          * @param allowedBytes an upper bound on the number of bytes the payload can write at this time.
131          */
132         void write(ChannelHandlerContext ctx, int allowedBytes);
133 
134         /**
135          * Merge the contents of the {@code next} message into this message so they can be written out as one unit.
136          * This allows many small messages to be written as a single DATA frame.
137          *
138          * @return {@code true} if {@code next} was successfully merged and does not need to be enqueued,
139          *     {@code false} otherwise.
140          */
141         boolean merge(ChannelHandlerContext ctx, FlowControlled next);
142     }
143 
144     /**
145      * Listener to the number of flow-controlled bytes written per stream.
146      */
147     interface Listener {
148         /**
149          * Notification that {@link Http2RemoteFlowController#isWritable(Http2Stream)} has changed for {@code stream}.
150          * <p>
151          * This method should not throw. Any thrown exceptions are considered a programming error and are ignored.
152          * @param stream The stream which writability has changed for.
153          */
154         void writabilityChanged(Http2Stream stream);
155     }
156 }
