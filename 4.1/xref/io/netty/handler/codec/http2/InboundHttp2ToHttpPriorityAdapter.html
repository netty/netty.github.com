15  package io.netty.handler.codec.http2;
16  
17  import io.netty.channel.ChannelHandlerContext;
18  import io.netty.handler.codec.http.DefaultHttpHeaders;
19  import io.netty.handler.codec.http.FullHttpMessage;
20  import io.netty.handler.codec.http.HttpHeaders;
21  import io.netty.util.AsciiString;
22  
23  import java.util.Iterator;
24  import java.util.Map.Entry;
25  
26  import static io.netty.handler.codec.http2.Http2Error.PROTOCOL_ERROR;
27  import static io.netty.handler.codec.http2.Http2Exception.connectionError;
28  
29  /**
30   * Translate header/data/priority HTTP/2 frame events into HTTP events.  Just as {@link InboundHttp2ToHttpAdapter}
31   * may generate multiple {@link FullHttpMessage} objects per stream, this class is more likely to
32   * generate multiple messages per stream because the chances of an HTTP/2 event happening outside
33   * the header/data message flow is more likely.
34   */
35  public final class InboundHttp2ToHttpPriorityAdapter extends InboundHttp2ToHttpAdapter {
36      private static final AsciiString OUT_OF_MESSAGE_SEQUENCE_METHOD = new AsciiString(
37              HttpConversionUtil.OUT_OF_MESSAGE_SEQUENCE_METHOD.toString());
38      private static final AsciiString OUT_OF_MESSAGE_SEQUENCE_PATH = new AsciiString(
39              HttpConversionUtil.OUT_OF_MESSAGE_SEQUENCE_PATH);
40      private static final AsciiString OUT_OF_MESSAGE_SEQUENCE_RETURN_CODE = new AsciiString(
41              HttpConversionUtil.OUT_OF_MESSAGE_SEQUENCE_RETURN_CODE.toString());
42      private final Http2Connection.PropertyKey outOfMessageFlowHeadersKey;
43  
44      InboundHttp2ToHttpPriorityAdapter(Http2Connection connection, int maxContentLength,
45                                        boolean validateHttpHeaders,
46                                        boolean propagateSettings) {
47          super(connection, maxContentLength, validateHttpHeaders, propagateSettings);
48          outOfMessageFlowHeadersKey = connection.newKey();
49      }
50  
51      private HttpHeaders getOutOfMessageFlowHeaders(Http2Stream stream) {
52          return stream.getProperty(outOfMessageFlowHeadersKey);
53      }
54  
55      private void putOutOfMessageFlowHeaders(Http2Stream stream, HttpHeaders headers) {
56          stream.setProperty(outOfMessageFlowHeadersKey, headers);
57      }
58  
59      private HttpHeaders removeOutOfMessageFlowHeaders(Http2Stream stream) {
60          return stream.removeProperty(outOfMessageFlowHeadersKey);
61      }
62  
63      /**
64       * Get either the header or the trailing headers depending on which is valid to add to
65       * @param msg The message containing the headers and trailing headers
66       * @return The headers object which can be appended to or modified
67       */
68      private static HttpHeaders getActiveHeaders(FullHttpMessage msg) {
69          return msg.content().isReadable() ? msg.trailingHeaders() : msg.headers();
70      }
71  
72      /**
73       * This method will add the {@code headers} to the out of order headers map
74       * @param stream The stream associated with {@code headers}
75       * @param headers Newly encountered out of order headers which must be stored for future use
76       */
77      private void importOutOfMessageFlowHeaders(Http2Stream stream, HttpHeaders headers) {
78          final HttpHeaders outOfMessageFlowHeader = getOutOfMessageFlowHeaders(stream);
79          if (outOfMessageFlowHeader == null) {
80              putOutOfMessageFlowHeaders(stream, headers);
81          } else {
82              outOfMessageFlowHeader.setAll(headers);
83          }
84      }
85  
86      /**
87       * Take any saved out of order headers and export them to {@code headers}
88       * @param stream The stream to search for out of order headers for
89       * @param headers If any out of order headers exist for {@code stream} they will be added to this object
90       */
91      private void exportOutOfMessageFlowHeaders(Http2Stream stream, final HttpHeaders headers) {
92          final HttpHeaders outOfMessageFlowHeader = getOutOfMessageFlowHeaders(stream);
93          if (outOfMessageFlowHeader != null) {
94              headers.setAll(outOfMessageFlowHeader);
95          }
96      }
97  
98      /**
99       * This will remove all headers which are related to priority tree events
100      * @param headers The headers to remove the priority tree elements from
101      */
102     private static void removePriorityRelatedHeaders(HttpHeaders headers) {
103         headers.remove(HttpConversionUtil.ExtensionHeaderNames.STREAM_DEPENDENCY_ID.text());
104         headers.remove(HttpConversionUtil.ExtensionHeaderNames.STREAM_WEIGHT.text());
105     }
106 
107     /**
108      * Initializes the pseudo header fields for out of message flow HTTP/2 headers
109      * @param headers The headers to be initialized with pseudo header values
110      */
111     private void initializePseudoHeaders(Http2Headers headers) {
112         if (connection.isServer()) {
113             headers.method(OUT_OF_MESSAGE_SEQUENCE_METHOD).path(OUT_OF_MESSAGE_SEQUENCE_PATH);
114         } else {
115             headers.status(OUT_OF_MESSAGE_SEQUENCE_RETURN_CODE);
116         }
117     }
118 
119     /**
120      * Add all the HTTP headers into the HTTP/2 headers object
121      * @param httpHeaders The HTTP headers to translate to HTTP/2
122      * @param http2Headers The target HTTP/2 headers
123      */
124     private static void addHttpHeadersToHttp2Headers(HttpHeaders httpHeaders, final Http2Headers http2Headers) {
125         Iterator<Entry<CharSequence, CharSequence>> iter = httpHeaders.iteratorCharSequence();
126         while (iter.hasNext()) {
127             Entry<CharSequence, CharSequence> entry = iter.next();
128             http2Headers.add(AsciiString.of(entry.getKey()), AsciiString.of(entry.getValue()));
129         }
130     }
131 
132     @Override
133     protected void fireChannelRead(ChannelHandlerContext ctx, FullHttpMessage msg, boolean release,
134                                    Http2Stream stream) {
135         exportOutOfMessageFlowHeaders(stream, getActiveHeaders(msg));
136         super.fireChannelRead(ctx, msg, release, stream);
137     }
138 
139     @Override
140     protected FullHttpMessage processHeadersBegin(ChannelHandlerContext ctx, Http2Stream stream, Http2Headers headers,
141             boolean endOfStream, boolean allowAppend, boolean appendToTrailer) throws Http2Exception {
142         FullHttpMessage msg = super.processHeadersBegin(ctx, stream, headers,
143                 endOfStream, allowAppend, appendToTrailer);
144         if (msg != null) {
145             exportOutOfMessageFlowHeaders(stream, getActiveHeaders(msg));
146         }
147         return msg;
148     }
149 
150     @Override
151     public void onPriorityTreeParentChanged(Http2Stream stream, Http2Stream oldParent) {
152         Http2Stream parent = stream.parent();
153         FullHttpMessage msg = getMessage(stream);
154         if (msg == null) {
155             // msg may be null if a HTTP/2 frame event is received outside the HTTP message flow
156             // For example a PRIORITY frame can be received in any state but the HTTP message flow
157             // takes place while the stream is OPEN.
158             if (parent != null && !parent.equals(connection.connectionStream())) {
159                 HttpHeaders headers = new DefaultHttpHeaders();
160                 headers.setInt(HttpConversionUtil.ExtensionHeaderNames.STREAM_DEPENDENCY_ID.text(), parent.id());
161                 importOutOfMessageFlowHeaders(stream, headers);
162             }
163         } else {
164             if (parent == null) {
165                 removePriorityRelatedHeaders(msg.headers());
166                 removePriorityRelatedHeaders(msg.trailingHeaders());
167             } else if (!parent.equals(connection.connectionStream())) {
168                 HttpHeaders headers = getActiveHeaders(msg);
169                 headers.setInt(HttpConversionUtil.ExtensionHeaderNames.STREAM_DEPENDENCY_ID.text(), parent.id());
170             }
171         }
172     }
173 
174     @Override
175     public void onWeightChanged(Http2Stream stream, short oldWeight) {
176         FullHttpMessage msg = getMessage(stream);
177         final HttpHeaders headers;
178         if (msg == null) {
179             // msg may be null if a HTTP/2 frame event in received outside the HTTP message flow
180             // For example a PRIORITY frame can be received in any state
181             // and the HTTP message flow exists in OPEN.
182             headers = new DefaultHttpHeaders();
183             importOutOfMessageFlowHeaders(stream, headers);
184         } else {
185             headers = getActiveHeaders(msg);
186         }
187         headers.setShort(HttpConversionUtil.ExtensionHeaderNames.STREAM_WEIGHT.text(), stream.weight());
188     }
189 
190     @Override
191     public void onPriorityRead(ChannelHandlerContext ctx, int streamId, int streamDependency, short weight,
192                     boolean exclusive) throws Http2Exception {
193         Http2Stream stream = connection.stream(streamId);
194         if (stream == null) {
195             return;
196         }
197         FullHttpMessage msg = getMessage(stream);
198         if (msg == null) {
199             HttpHeaders httpHeaders = removeOutOfMessageFlowHeaders(stream);
200             if (httpHeaders == null) {
201                 throw connectionError(PROTOCOL_ERROR, "Priority Frame recieved for unknown stream id %d", streamId);
202             }
203 
204             Http2Headers http2Headers = new DefaultHttp2Headers(validateHttpHeaders, httpHeaders.size());
205             initializePseudoHeaders(http2Headers);
206             addHttpHeadersToHttp2Headers(httpHeaders, http2Headers);
207             msg = newMessage(stream, http2Headers, validateHttpHeaders, ctx.alloc());
208             fireChannelRead(ctx, msg, false, stream);
209         }
210     }
