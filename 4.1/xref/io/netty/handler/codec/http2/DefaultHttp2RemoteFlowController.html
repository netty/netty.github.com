36   * Basic implementation of {@link Http2RemoteFlowController}.
37   * <p>
38   * This class is <strong>NOT</strong> thread safe. The assumption is all methods must be invoked from a single thread.

66      public DefaultHttp2RemoteFlowController(Http2Connection connection,
67                                              StreamByteDistributor streamByteDistributor,
68                                              final Listener listener) {
69          this.connection = checkNotNull(connection, "connection");
70          this.streamByteDistributor = checkNotNull(streamByteDistributor, "streamWriteDistributor");
71  
72          // Add a flow state for the connection.
73          stateKey = connection.newKey();
74          connectionState = new DefaultState(connection.connectionStream(), initialWindowSize,
75                  initialWindowSize > 0 && isChannelWritable());
76          connection.connectionStream().setProperty(stateKey, connectionState);
77  
78          // Monitor may depend upon connectionState, and so initialize after connectionState
79          listener(listener);
80  
81          // Register for notification of new streams.
82          connection.addListener(new Http2ConnectionAdapter() {
83              @Override
84              public void onStreamAdded(Http2Stream stream) {
85                  // If the stream state is not open then the stream is not yet eligible for flow controlled frames and
86                  // only requires the ReducedFlowState. Otherwise the full amount of memory is required.
87                  stream.setProperty(stateKey, stream.state() == IDLE ?
88                          new ReducedState(stream) :
89                          new DefaultState(stream, 0,
90                                  isWritable(DefaultHttp2RemoteFlowController.this.connection.connectionStream())));
91              }
92  
93              @Override
94              public void onStreamActive(Http2Stream stream) {
95                  // If the object was previously created, but later activated then we have to ensure
96                  // the full state is allocated and the proper initialWindowSize is used.
97                  AbstractState state = state(stream);
98                  if (state.getClass() == DefaultState.class) {
99                      state.window(initialWindowSize);
100                 } else {
101                     stream.setProperty(stateKey, new DefaultState(state, initialWindowSize));
102                 }
103             }
104 
105             @Override
106             public void onStreamClosed(Http2Stream stream) {
107                 // Any pending frames can never be written, cancel and
108                 // write errors for any pending frames.
109                 AbstractState state = state(stream);
110                 state.cancel();
111 
112                 // If the stream is now eligible for removal, but will persist in the priority tree then we can
113                 // decrease the amount of memory required for this stream because no flow controlled frames can
114                 // be exchanged on this stream
115                 if (stream.prioritizableForTree() != 0) {
116                     state = new ReducedState(state);
117                     stream.setProperty(stateKey, state);
118                 }
119                 // Tell the monitor after cancel has been called and after the new state is used.
120                 monitor.stateCancelled(state);
121             }
122 
123             @Override
124             public void onStreamHalfClosed(Http2Stream stream) {
125                 if (HALF_CLOSED_LOCAL.equals(stream.state())) {
126                     /**
127                      * When this method is called there should not be any
128                      * pending frames left if the API is used correctly. However,
129                      * it is possible that a erroneous application can sneak
130                      * in a frame even after having already written a frame with the
131                      * END_STREAM flag set, as the stream state might not transition
132                      * immediately to HALF_CLOSED_LOCAL / CLOSED due to flow control
133                      * delaying the write.
134                      *
135                      * This is to cancel any such illegal writes.
136                      */
137                     AbstractState state = state(stream);
138                     state.cancel();
139                     monitor.stateCancelled(state);
140                 }
141             }
142         });
143     }

250     private int minUsableChannelBytes() {
251         // The current allocation algorithm values "fairness" and doesn't give any consideration to "goodput". It
252         // is possible that 1 byte will be allocated to many streams. In an effort to try to make "goodput"
253         // reasonable with the current allocation algorithm we have this "cheap" check up front to ensure there is
254         // an "adequate" amount of connection window before allocation is attempted. This is not foolproof as if the
255         // number of streams is >= this minimal number then we may still have the issue, but the idea is to narrow the
256         // circumstances in which this can happen without rewriting the allocation algorithm.
257         return max(ctx.channel().config().getWriteBufferLowWaterMark(), MIN_WRITABLE_CHUNK);
258     }
259 
260     private int maxUsableChannelBytes() {
261         // If the channel isWritable, allow at least minUseableChannelBytes.
262         int channelWritableBytes = (int) min(Integer.MAX_VALUE, ctx.channel().bytesBeforeUnwritable());
263         int useableBytes = channelWritableBytes > 0 ? max(channelWritableBytes, minUsableChannelBytes()) : 0;
264 
265         // Clip the usable bytes by the connection window.
266         return min(connectionState.windowSize(), useableBytes);
267     }

282     /**
283      * The remote flow control state for a single stream.
284      */
285     private final class DefaultState extends AbstractState {
286         private final Deque<FlowControlled> pendingWriteQueue;
287         private int window;
288         private int pendingBytes;
289         // Set to true while a frame is being written, false otherwise.
290         private boolean writing;
291         // Set to true if cancel() was called.
292         private boolean cancelled;
293 
294         DefaultState(Http2Stream stream, int initialWindowSize, boolean markedWritable) {
295             super(stream, markedWritable);
296             window(initialWindowSize);
297             pendingWriteQueue = new ArrayDeque<FlowControlled>(2);
298         }
299 
300         DefaultState(AbstractState existingState, int initialWindowSize) {
301             super(existingState);
302             window(initialWindowSize);
303             pendingWriteQueue = new ArrayDeque<FlowControlled>(2);
304         }
305 
306         @Override
307         public int windowSize() {
308             return window;
309         }
310 
311         @Override
312         int initialWindowSize() {
313             return initialWindowSize;
314         }
315 
316         @Override
317         void window(int initialWindowSize) {
318             window = initialWindowSize;
319         }
320 
321         @Override
322         int writeAllocatedBytes(int allocated) {
323             final int initialAllocated = allocated;
324             int writtenBytes;
325             // In case an exception is thrown we want to remember it and pass it to cancel(Throwable).
326             Throwable cause = null;
327             FlowControlled frame;
328             try {
329                 assert !writing;
330                 writing = true;
331 
332                 // Write the remainder of frames that we are allowed to
333                 boolean writeOccurred = false;
334                 while (!cancelled && (frame = peek()) != null) {
335                     int maxBytes = min(allocated, writableWindow());
336                     if (maxBytes <= 0 && frame.size() > 0) {
337                         // The frame still has data, but the amount of allocated bytes has been exhausted.
338                         // Don't write needless empty frames.
339                         break;
340                     }
341                     writeOccurred = true;
342                     int initialFrameSize = frame.size();
343                     try {
344                         frame.write(ctx, max(0, maxBytes));
345                         if (frame.size() == 0) {
346                             // This frame has been fully written, remove this frame and notify it.
347                             // Since we remove this frame first, we're guaranteed that its error
348                             // method will not be called when we call cancel.
349                             pendingWriteQueue.remove();
350                             frame.writeComplete();
351                         }
352                     } finally {
353                         // Decrement allocated by how much was actually written.
354                         allocated -= initialFrameSize - frame.size();
355                     }
356                 }
357 
358                 if (!writeOccurred) {
359                     // Either there was no frame, or the amount of allocated bytes has been exhausted.
360                     return -1;
361                 }
362 
363             } catch (Throwable t) {
364                 // Mark the state as cancelled, we'll clear the pending queue via cancel() below.
365                 cancelled = true;
366                 cause = t;
367             } finally {
368                 writing = false;
369                 // Make sure we always decrement the flow control windows
370                 // by the bytes written.
371                 writtenBytes = initialAllocated - allocated;
372 
373                 decrementPendingBytes(writtenBytes, false);
374                 decrementFlowControlWindow(writtenBytes);
375 
376                 // If a cancellation occurred while writing, call cancel again to
377                 // clear and error all of the pending writes.
378                 if (cancelled) {
379                     cancel(cause);
380                 }
381             }
382             return writtenBytes;
383         }
384 
385         @Override
386         int incrementStreamWindow(int delta) throws Http2Exception {
387             if (delta > 0 && Integer.MAX_VALUE - delta < window) {
388                 throw streamError(stream.id(), FLOW_CONTROL_ERROR,
389                         "Window size overflow for stream: %d", stream.id());
390             }
391             window += delta;
392 
393             streamByteDistributor.updateStreamableBytes(this);
394             return window;
395         }
396 
397         /**
398          * Returns the maximum writable window (minimum of the stream and connection windows).
399          */
400         private int writableWindow() {
401             return min(window, connectionWindowSize());
402         }
403 
404         @Override
405         public int pendingBytes() {
406             return pendingBytes;
407         }
408 
409         @Override
410         void enqueueFrame(FlowControlled frame) {
411             FlowControlled last = pendingWriteQueue.peekLast();
412             if (last == null) {
413                 enqueueFrameWithoutMerge(frame);
414                 return;
415             }
416 
417             int lastSize = last.size();
418             if (last.merge(ctx, frame)) {
419                 incrementPendingBytes(last.size() - lastSize, true);
420                 return;
421             }
422             enqueueFrameWithoutMerge(frame);
423         }
424 
425         private void enqueueFrameWithoutMerge(FlowControlled frame) {
426             pendingWriteQueue.offer(frame);
427             // This must be called after adding to the queue in order so that hasFrame() is
428             // updated before updating the stream state.
429             incrementPendingBytes(frame.size(), true);
430         }
431 
432         @Override
433         public boolean hasFrame() {
434             return !pendingWriteQueue.isEmpty();
435         }

437         /**
438          * Returns the the head of the pending queue, or {@code null} if empty.
439          */
440         private FlowControlled peek() {
441             return pendingWriteQueue.peek();
442         }
443 
444         @Override
445         void cancel() {
446             cancel(null);
447         }

449         /**
450          * Clears the pending queue and writes errors for each remaining frame.
451          * @param cause the {@link Throwable} that caused this method to be invoked.
452          */
453         private void cancel(Throwable cause) {
454             cancelled = true;
455             // Ensure that the queue can't be modified while we are writing.
456             if (writing) {
457                 return;
458             }
459             for (;;) {
460                 FlowControlled frame = pendingWriteQueue.poll();
461                 if (frame == null) {
462                     break;
463                 }
464                 writeError(frame, streamError(stream.id(), INTERNAL_ERROR, cause,
465                                               "Stream closed before write could take place"));
466             }
467 
468             streamByteDistributor.updateStreamableBytes(this);
469         }
470 
471         /**
472          * Increments the number of pending bytes for this node and optionally updates the
473          * {@link StreamByteDistributor}.
474          */
475         private void incrementPendingBytes(int numBytes, boolean updateStreamableBytes) {
476             pendingBytes += numBytes;
477             monitor.incrementPendingBytes(numBytes);
478             if (updateStreamableBytes) {
479                 streamByteDistributor.updateStreamableBytes(this);
480             }
481         }
482 
483         /**
484          * If this frame is in the pending queue, decrements the number of pending bytes for the stream.
485          */
486         private void decrementPendingBytes(int bytes, boolean updateStreamableBytes) {
487             incrementPendingBytes(-bytes, updateStreamableBytes);
488         }
489 
490         /**
491          * Decrement the per stream and connection flow control window by {@code bytes}.
492          */
493         private void decrementFlowControlWindow(int bytes) {
494             try {
495                 int negativeBytes = -bytes;
496                 connectionState.incrementStreamWindow(negativeBytes);
497                 incrementStreamWindow(negativeBytes);
498             } catch (Http2Exception e) {
499                 // Should never get here since we're decrementing.
500                 throw new IllegalStateException("Invalid window state when writing frame: " + e.getMessage(), e);
501             }
502         }
503 
504         /**
505          * Discards this {@link FlowControlled}, writing an error. If this frame is in the pending queue,
506          * the unwritten bytes are removed from this branch of the priority tree.
507          */
508         private void writeError(FlowControlled frame, Http2Exception cause) {
509             assert ctx != null;
510             decrementPendingBytes(frame.size(), true);
511             frame.error(ctx, cause);
512         }
513     }

669          * Increment the window size for a particular stream.

674         public void incrementWindowSize(AbstractState state, int delta) throws Http2Exception {
675             state.incrementStreamWindow(delta);

678         /**
679          * Add a frame to be sent via flow control.

681          * @param frame the frame to enqueue.
682          * @throws Http2Exception If a writability error occurs.
683          */
684         public void enqueueFrame(AbstractState state, FlowControlled frame) throws Http2Exception {
685             state.enqueueFrame(frame);
686         }
687 
688         /**
689          * Increment the total amount of pending bytes for all streams. When any stream's pending bytes changes
690          * method should be called.
691          * @param delta The amount to increment by.
692          */
693         public final void incrementPendingBytes(int delta) {
694             totalPendingBytes += delta;
695 
696             // Notification of writibilty change should be delayed until the end of the top level event.
697             // This is to ensure the flow controller is more consistent state before calling external listener methods.
698         }
699 
700         /**
701          * Determine if the stream associated with {@code state} is writable.
702          * @param state The state which is associated with the stream to test writability for.
703          * @return {@code true} if {@link AbstractState#stream()} is writable. {@code false} otherwise.
704          */
705         public final boolean isWritable(AbstractState state) {
706             return isWritableConnection() && state.windowSize() - state.pendingBytes() > 0;
707         }
708 
709         protected final void writePendingBytes() throws Http2Exception {
710             int bytesToWrite = writableBytes();
711 
712             // Make sure we always write at least once, regardless if we have bytesToWrite or not.
713             // This ensures that zero-length frames will always be written.
714             for (;;) {
715                 if (!streamByteDistributor.distribute(bytesToWrite, writer) ||
716                     (bytesToWrite = writableBytes()) <= 0 ||
717                     !isChannelWritable0()) {
718                     break;
719                 }
720             }
721         }
722 
723         protected void initialWindowSize(int newWindowSize) throws Http2Exception {
724             if (newWindowSize < 0) {
725                 throw new IllegalArgumentException("Invalid initial window size: " + newWindowSize);
726             }
727 
728             final int delta = newWindowSize - initialWindowSize;
729             initialWindowSize = newWindowSize;
730             connection.forEachActiveStream(new Http2StreamVisitor() {
731                 @Override
732                 public boolean visit(Http2Stream stream) throws Http2Exception {
733                     state(stream).incrementStreamWindow(delta);
734                     return true;
735                 }
736             });
737 
738             if (delta > 0) {
739                 // The window size increased, send any pending frames for all streams.
740                 writePendingBytes();
741             }
742         }
743 
744         protected final boolean isWritableConnection() {
745             return connectionState.windowSize() - totalPendingBytes > 0 && isChannelWritable();
746         }

749     /**
750      * Writability of a {@code stream} is calculated using the following:
751      * <pre>
752      * Connection Window - Total Queued Bytes > 0 &&
753      * Stream Window - Bytes Queued for Stream > 0 &&
754      * isChannelWritable()
755      * </pre>
756      */
757     private final class ListenerWritabilityMonitor extends WritabilityMonitor {
758         private final Listener listener;
759         private final Http2StreamVisitor checkStreamWritabilityVisitor = new Http2StreamVisitor() {
760             @Override
761             public boolean visit(Http2Stream stream) throws Http2Exception {
762                 AbstractState state = state(stream);
763                 if (isWritable(state) != state.markWritability()) {
764                     notifyWritabilityChanged(state);
765                 }
766                 return true;
767             }
768         };
769 
770         ListenerWritabilityMonitor(Listener listener) {
771             this.listener = listener;
772         }
773 
774         @Override
775         public void incrementWindowSize(AbstractState state, int delta) throws Http2Exception {
776             super.incrementWindowSize(state, delta);
777             if (isWritable(state) != state.markWritability()) {
778                 if (state == connectionState) {
779                     checkAllWritabilityChanged();
780                 } else {
781                     notifyWritabilityChanged(state);
782                 }
783             }
784         }
785 
786         @Override
787         protected void initialWindowSize(int newWindowSize) throws Http2Exception {
788             super.initialWindowSize(newWindowSize);
789             if (isWritableConnection()) {
790                 // If the write operation does not occur we still need to check all streams because they
791                 // may have transitioned from writable to not writable.
792                 checkAllWritabilityChanged();
793             }
794         }
795 
796         @Override
797         public void enqueueFrame(AbstractState state, FlowControlled frame) throws Http2Exception {
798             super.enqueueFrame(state, frame);
799             checkConnectionThenStreamWritabilityChanged(state);
800         }
801 
802         @Override
803         public void stateCancelled(AbstractState state) {
804             try {
805                 checkConnectionThenStreamWritabilityChanged(state);
806             } catch (Http2Exception e) {
807                 logger.error("Caught unexpected exception from checkAllWritabilityChanged", e);
808             }
809         }
810 
811         @Override
812         public void channelWritabilityChange() throws Http2Exception {
813             if (connectionState.markWritability() != isChannelWritable()) {
814                 checkAllWritabilityChanged();
815             }
816         }
817 
818         private void notifyWritabilityChanged(AbstractState state) {
819             state.markWritability(!state.markWritability());
820             try {
821                 listener.writabilityChanged(state.stream);
822             } catch (RuntimeException e) {
823                 logger.error("Caught unexpected exception from listener.writabilityChanged", e);
824             }
825         }
826 
827         private void checkConnectionThenStreamWritabilityChanged(AbstractState state) throws Http2Exception {
828             // It is possible that the connection window and/or the individual stream writability could change.
829             if (isWritableConnection() != connectionState.markWritability()) {
830                 checkAllWritabilityChanged();
831             } else if (isWritable(state) != state.markWritability()) {
832                 notifyWritabilityChanged(state);
833             }
834         }
835 
836         private void checkAllWritabilityChanged() throws Http2Exception {
837             // Make sure we mark that we have notified as a result of this change.
838             connectionState.markWritability(isWritableConnection());
839             connection.forEachActiveStream(checkStreamWritabilityVisitor);
840         }
