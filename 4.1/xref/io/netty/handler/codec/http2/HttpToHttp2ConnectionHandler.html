16  package io.netty.handler.codec.http2;
17  
18  import io.netty.buffer.ByteBuf;
19  import io.netty.channel.ChannelHandlerContext;
20  import io.netty.channel.ChannelPromise;
21  import io.netty.handler.codec.http.FullHttpMessage;
22  import io.netty.handler.codec.http.HttpContent;
23  import io.netty.handler.codec.http.HttpHeaders;
24  import io.netty.handler.codec.http.HttpMessage;
25  import io.netty.handler.codec.http.LastHttpContent;
26  import io.netty.handler.codec.http2.Http2CodecUtil.SimpleChannelPromiseAggregator;
27  import io.netty.util.ReferenceCountUtil;
28  
29  /**
30   * Translates HTTP/1.x object writes into HTTP/2 frames.
31   * <p>
32   * See {@link InboundHttp2ToHttpAdapter} to get translation from HTTP/2 frames to HTTP/1.x objects.
33   */
34  public class HttpToHttp2ConnectionHandler extends Http2ConnectionHandler {
35  
36      private final boolean validateHeaders;
37      private int currentStreamId;
38  
39      protected HttpToHttp2ConnectionHandler(Http2ConnectionDecoder decoder, Http2ConnectionEncoder encoder,
40                                             Http2Settings initialSettings, boolean validateHeaders) {
41          super(decoder, encoder, initialSettings);
42          this.validateHeaders = validateHeaders;
43      }
44  
45      /**
46       * Get the next stream id either from the {@link HttpHeaders} object or HTTP/2 codec
47       *
48       * @param httpHeaders The HTTP/1.x headers object to look for the stream id
49       * @return The stream id to use with this {@link HttpHeaders} object
50       * @throws Exception If the {@code httpHeaders} object specifies an invalid stream id
51       */
52      private int getStreamId(HttpHeaders httpHeaders) throws Exception {
53          return httpHeaders.getInt(HttpConversionUtil.ExtensionHeaderNames.STREAM_ID.text(),
54                                    connection().local().nextStreamId());
55      }
56  
57      /**
58       * Handles conversion of {@link HttpMessage} and {@link HttpContent} to HTTP/2 frames.
59       */
60      @Override
61      public void write(ChannelHandlerContext ctx, Object msg, ChannelPromise promise) {
62  
63          if (!(msg instanceof HttpMessage || msg instanceof HttpContent)) {
64              ctx.write(msg, promise);
65              return;
66          }
67  
68          boolean release = true;
69          SimpleChannelPromiseAggregator promiseAggregator =
70                  new SimpleChannelPromiseAggregator(promise, ctx.channel(), ctx.executor());
71          try {
72              Http2ConnectionEncoder encoder = encoder();
73              boolean endStream = false;
74              if (msg instanceof HttpMessage) {
75                  final HttpMessage httpMsg = (HttpMessage) msg;
76  
77                  // Provide the user the opportunity to specify the streamId
78                  currentStreamId = getStreamId(httpMsg.headers());
79  
80                  // Convert and write the headers.
81                  Http2Headers http2Headers = HttpConversionUtil.toHttp2Headers(httpMsg, validateHeaders);
82                  endStream = msg instanceof FullHttpMessage && !((FullHttpMessage) msg).content().isReadable();
83                  encoder.writeHeaders(ctx, currentStreamId, http2Headers, 0, endStream, promiseAggregator.newPromise());
84              }
85  
86              if (!endStream && msg instanceof HttpContent) {
87                  boolean isLastContent = false;
88                  Http2Headers trailers = EmptyHttp2Headers.INSTANCE;
89                  if (msg instanceof LastHttpContent) {
90                      isLastContent = true;
91  
92                      // Convert any trailing headers.
93                      final LastHttpContent lastContent = (LastHttpContent) msg;
94                      trailers = HttpConversionUtil.toHttp2Headers(lastContent.trailingHeaders(), validateHeaders);
95                  }
96  
97                  // Write the data
98                  final ByteBuf content = ((HttpContent) msg).content();
99                  endStream = isLastContent && trailers.isEmpty();
100                 release = false;
101                 encoder.writeData(ctx, currentStreamId, content, 0, endStream, promiseAggregator.newPromise());
102 
103                 if (!trailers.isEmpty()) {
104                     // Write trailing headers.
105                     encoder.writeHeaders(ctx, currentStreamId, trailers, 0, true, promiseAggregator.newPromise());
106                 }
107             }
108 
109             promiseAggregator.doneAllocatingPromises();
110         } catch (Throwable t) {
111             promiseAggregator.setFailure(t);
112         } finally {
113             if (release) {
114                 ReferenceCountUtil.release(msg);
115             }
116         }
117     }
118 }
