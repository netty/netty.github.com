32  package io.netty.handler.codec.http2.hpack;
33  
34  import static io.netty.handler.codec.http2.hpack.HpackUtil.ISO_8859_1;
35  import static io.netty.handler.codec.http2.hpack.HpackUtil.requireNonNull;
36  
37  class HeaderField implements Comparable<HeaderField> {
38  
39      // Section 4.1. Calculating Table Size
40      // The additional 32 octets account for an estimated
41      // overhead associated with the structure.
42      static final int HEADER_ENTRY_OVERHEAD = 32;
43  
44      static int sizeOf(byte[] name, byte[] value) {
45          return name.length + value.length + HEADER_ENTRY_OVERHEAD;
46      }
47  
48      final byte[] name;
49      final byte[] value;
50  
51      // This constructor can only be used if name and value are ISO-8859-1 encoded.
52      HeaderField(String name, String value) {
53          this(name.getBytes(ISO_8859_1), value.getBytes(ISO_8859_1));
54      }
55  
56      HeaderField(byte[] name, byte[] value) {
57          this.name = requireNonNull(name);
58          this.value = requireNonNull(value);
59      }
60  
61      int size() {
62          return name.length + value.length + HEADER_ENTRY_OVERHEAD;
63      }
64  
65      @Override
66      public int hashCode() {
67          // TODO(nmittler): Netty's build rules require this. Probably need a better implementation.
68          return super.hashCode();
69      }
70  
71      @Override
72      public int compareTo(HeaderField anotherHeaderField) {
73          int ret = compareTo(name, anotherHeaderField.name);
74          if (ret == 0) {
75              ret = compareTo(value, anotherHeaderField.value);
76          }
77          return ret;
78      }
79  
80      private int compareTo(byte[] s1, byte[] s2) {
81          int len1 = s1.length;
82          int len2 = s2.length;
83          int lim = Math.min(len1, len2);
84  
85          int k = 0;
86          while (k < lim) {
87              byte b1 = s1[k];
88              byte b2 = s2[k];
89              if (b1 != b2) {
90                  return b1 - b2;
91              }
92              k++;
93          }
94          return len1 - len2;
95      }
96  
97      @Override
98      public boolean equals(Object obj) {
99          if (obj == this) {
100             return true;
101         }
102         if (!(obj instanceof HeaderField)) {
103             return false;
104         }
105         HeaderField other = (HeaderField) obj;
106         boolean nameEquals = HpackUtil.equals(name, other.name);
107         boolean valueEquals = HpackUtil.equals(value, other.value);
108         return nameEquals && valueEquals;
109     }
110 
111     @Override
112     public String toString() {
113         String nameString = new String(name);
114         String valueString = new String(value);
115         return nameString + ": " + valueString;
116     }
117 }
