109     private static final Map<String, Integer> STATIC_INDEX_BY_NAME = createMap();
110 
111     /**
112      * The number of header fields in the static table.
113      */
114     static final int length = STATIC_TABLE.size();
115 
116     /**
117      * Return the header field at the given index value.
118      */
119     static HeaderField getEntry(int index) {
120         return STATIC_TABLE.get(index - 1);
121     }
122 
123     /**
124      * Returns the lowest index value for the given header field name in the static table. Returns
125      * -1 if the header field name is not in the static table.
126      */
127     static int getIndex(byte[] name) {
128         String nameString = new String(name, 0, name.length, HpackUtil.ISO_8859_1);
129         Integer index = STATIC_INDEX_BY_NAME.get(nameString);
130         if (index == null) {
131             return -1;
132         }
133         return index;
134     }
135 
136     /**
137      * Returns the index value for the given header field in the static table. Returns -1 if the
138      * header field is not in the static table.
139      */
140     static int getIndex(byte[] name, byte[] value) {
141         int index = getIndex(name);
142         if (index == -1) {
143             return -1;
144         }
145 
146         // Note this assumes all entries for a given header field are sequential.
147         while (index <= length) {
148             HeaderField entry = getEntry(index);
149             if (!HpackUtil.equals(name, entry.name)) {
150                 break;
151             }
152             if (HpackUtil.equals(value, entry.value)) {
153                 return index;
154             }
155             index++;
156         }
157 
158         return -1;
159     }
160 
161     // create a map of header name to index value to allow quick lookup
162     private static Map<String, Integer> createMap() {
163         int length = STATIC_TABLE.size();
164         HashMap<String, Integer> ret = new HashMap<String, Integer>(length);
165         // Iterate through the static table in reverse order to
166         // save the smallest index for a given name in the map.
167         for (int index = length; index > 0; index--) {
168             HeaderField entry = getEntry(index);
169             String name = new String(entry.name, 0, entry.name.length, HpackUtil.ISO_8859_1);
170             ret.put(name, index);
171         }
172         return ret;
173     }
