40      /**
41       * A string compare that doesn't leak timing information.
42       */
43      static boolean equals(byte[] s1, byte[] s2) {
44          if (s1.length != s2.length) {
45              return false;
46          }
47          char c = 0;
48          for (int i = 0; i < s1.length; i++) {
49              c |= s1[i] ^ s2[i];
50          }
51          return c == 0;
52      }
