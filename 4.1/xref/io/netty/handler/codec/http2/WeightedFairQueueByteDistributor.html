30  /**
31   * A {@link StreamByteDistributor} that is sensitive to stream priority and uses
32   * <a href="https://en.wikipedia.org/wiki/Weighted_fair_queueing">Weighted Fair Queueing</a> approach for distributing
33   * bytes.
34   * <p>
35   * Inspiration for this distributor was taken from Linux's
36   * <a href="https://git.kernel.org/cgit/linux/kernel/git/stable/linux-stable.git/tree/Documentation/scheduler
37   * /sched-design-CFS.txt">Completely Fair Scheduler</a>
38   * to model the distribution of bytes to simulate an "ideal multi-tasking CPU", but in this case we are simulating
39   * an "ideal multi-tasking NIC".
40   * <p>
41   * Each write operation will use the {@link #allocationQuantum(int)} to know how many more bytes should be allocated
42   * relative to the next stream which wants to write. This is to balance fairness while also considering goodput.
43   */
44  public final class WeightedFairQueueByteDistributor implements StreamByteDistributor {
45      private final Http2Connection.PropertyKey stateKey;
46      private final State connectionState;
47      /**
48       * The minimum number of bytes that we will attempt to allocate to a stream. This is to
49       * help improve goodput on a per-stream basis.
50       */
51      private int allocationQuantum = 1024;
52  
53      public WeightedFairQueueByteDistributor(Http2Connection connection) {
54          stateKey = connection.newKey();
55          Http2Stream connectionStream = connection.connectionStream();
56          connectionStream.setProperty(stateKey, connectionState = new State(connectionStream, 16));
57  
58          // Register for notification of new streams.
59          connection.addListener(new Http2ConnectionAdapter() {
60              @Override
61              public void onStreamAdded(Http2Stream stream) {
62                  stream.setProperty(stateKey, new State(stream));
63              }
64  
65              @Override
66              public void onWeightChanged(Http2Stream stream, short oldWeight) {
67                  Http2Stream parent;
68                  if (state(stream).activeCountForTree != 0 && (parent = stream.parent()) != null) {
69                      state(parent).totalQueuedWeights += stream.weight() - oldWeight;
70                  }
71              }
72  
73              @Override
74              public void onStreamClosed(Http2Stream stream) {
75                  state(stream).close();
76              }
77  
78              @Override
79              public void onPriorityTreeParentChanged(Http2Stream stream, Http2Stream oldParent) {
80                  Http2Stream parent = stream.parent();
81                  if (parent != null) {
82                      State state = state(stream);
83                      if (state.activeCountForTree != 0) {
84                          State pState = state(parent);
85                          pState.offerAndInitializePseudoTime(state);
86                          pState.isActiveCountChangeForTree(state.activeCountForTree);
87                      }
88                  }
89              }
90  
91              @Override
92              public void onPriorityTreeParentChanging(Http2Stream stream, Http2Stream newParent) {
93                  Http2Stream parent = stream.parent();
94                  if (parent != null) {
95                      State state = state(stream);
96                      if (state.activeCountForTree != 0) {
97                          State pState = state(parent);
98                          pState.remove(state);
99                          pState.isActiveCountChangeForTree(-state.activeCountForTree);
100                     }
101                 }
102             }
103         });
104     }

107     public void updateStreamableBytes(StreamState state) {
108         state(state.stream()).updateStreamableBytes(streamableBytes(state),
109                                                     state.hasFrame() && state.windowSize() >= 0);

112     @Override
113     public boolean distribute(int maxBytes, Writer writer) throws Http2Exception {
114         checkNotNull(writer, "writer");
115 
116         // As long as there is some active frame we should write at least 1 time.
117         if (connectionState.activeCountForTree == 0) {
118             return false;
119         }
120 
121         // The goal is to write until we write all the allocated bytes or are no longer making progress.
122         // We still attempt to write even after the number of allocated bytes has been exhausted to allow empty frames
123         // to be sent. Making progress means the active streams rooted at the connection stream has changed.
124         int oldIsActiveCountForTree;
125         do {
126             oldIsActiveCountForTree = connectionState.activeCountForTree;
127             // connectionState will never be active, so go right to its children.
128             maxBytes -= distributeToChildren(maxBytes, writer, connectionState);
129         } while (connectionState.activeCountForTree != 0 &&
130                 (maxBytes > 0 || oldIsActiveCountForTree != connectionState.activeCountForTree));
131 
132         return connectionState.activeCountForTree != 0;
133     }

135     /**
136      * Sets the amount of bytes that will be allocated to each stream. Defaults to 1KiB.
137      * @param allocationQuantum the amount of bytes that will be allocated to each stream. Must be &gt; 0.
138      */
139     public void allocationQuantum(int allocationQuantum) {
140         if (allocationQuantum <= 0) {
141             throw new IllegalArgumentException("allocationQuantum must be > 0");
142         }
143         this.allocationQuantum = allocationQuantum;
144     }

146     private int distribute(int maxBytes, Writer writer, State state) throws Http2Exception {
147         if (state.active) {
148             int nsent = min(maxBytes, state.streamableBytes);
149             state.write(nsent, writer);
150             if (nsent == 0 && maxBytes != 0) {
151                 // If a stream sends zero bytes, then we gave it a chance to write empty frames and it is now
152                 // considered inactive until the next call to updateStreamableBytes. This allows descendant streams to
153                 // be allocated bytes when the parent stream can't utilize them. This may be as a result of the
154                 // stream's flow control window being 0.
155                 state.updateStreamableBytes(state.streamableBytes, false);
156             }
157             return nsent;
158         }
159 
160         return distributeToChildren(maxBytes, writer, state);
161     }
162 
163     /**
164      * It is a pre-condition that {@code state.poll()} returns a non-{@code null} value. This is a result of the way
165      * the allocation algorithm is structured and can be explained in the following cases:
166      * <h3>For the recursive case</h3>
167      * If a stream has no children (in the allocation tree) than that node must be active or it will not be in the
168      * allocation tree. If a node is active then it will not delegate to children and recursion ends.
169      * <h3>For the initial case</h3>
170      * We check connectionState.activeCountForTree == 0 before any allocation is done. So if the connection stream
171      * has no active children we don't get into this method.
172      */
173     private int distributeToChildren(int maxBytes, Writer writer, State state) throws Http2Exception {
174         long oldTotalQueuedWeights = state.totalQueuedWeights;
175         State childState = state.poll();
176         State nextChildState = state.peek();
177         try {
178             assert nextChildState == null || nextChildState.pseudoTimeToWrite >= childState.pseudoTimeToWrite :
179                 "nextChildState.pseudoTime(" + nextChildState.pseudoTimeToWrite + ") < " + " childState.pseudoTime(" +
180                     childState.pseudoTimeToWrite + ")";
181             int nsent = distribute(nextChildState == null ? maxBytes :
182                             min(maxBytes, (int) min((nextChildState.pseudoTimeToWrite - childState.pseudoTimeToWrite) *
183                                                 childState.stream.weight() / oldTotalQueuedWeights + allocationQuantum,
184                                                 Integer.MAX_VALUE)
185                                ),
186                                writer,
187                                childState);
188             state.pseudoTime += nsent;
189             childState.updatePseudoTime(state, nsent, oldTotalQueuedWeights);
190             return nsent;
191         } finally {
192             // Do in finally to ensure the internal state is not corrupted if an exception is thrown.
193             // The offer operation is delayed until we unroll up the recursive stack, so we don't have to remove from
194             // the priority queue due to a write operation.
195             if (childState.activeCountForTree != 0) {
196                 state.offer(childState);
197             }
198         }
199     }

212     /**
213      * The remote flow control state for a single stream.
214      */
215     private final class State implements PriorityQueueNode<State> {
216         final Http2Stream stream;
217         private final Queue<State> queue;
218         int streamableBytes;
219         /**
220          * Count of nodes rooted at this sub tree with {@link #active} equal to {@code true}.
221          */
222         int activeCountForTree;
223         private int priorityQueueIndex = INDEX_NOT_IN_QUEUE;
224         /**
225          * An estimate of when this node should be given the opportunity to write data.
226          */
227         long pseudoTimeToWrite;
228         /**
229          * A pseudo time maintained for immediate children to base their {@link pseudoTimeToSend} off of.
230          */
231         long pseudoTime;
232         long totalQueuedWeights;
233         boolean active;
234 
235         State(Http2Stream stream) {
236             this(stream, 0);
237         }
238 
239         State(Http2Stream stream, int initialSize) {
240             this.stream = stream;
241             queue = new PriorityQueue<State>(initialSize);
242         }
243 
244         void write(int numBytes, Writer writer) throws Http2Exception {
245             try {
246                 writer.write(stream, numBytes);
247             } catch (Throwable t) {
248                 throw connectionError(INTERNAL_ERROR, t, "byte distribution write error");
249             }
250         }
251 
252         void isActiveCountChangeForTree(int increment) {
253             assert activeCountForTree + increment >= 0;
254             activeCountForTree += increment;
255             if (!stream.isRoot()) {
256                 State pState = state(stream.parent());
257                 if (activeCountForTree == 0) {
258                     pState.remove(this);
259                 } else if (activeCountForTree - increment == 0) { // if frame count was 0 but is now not, then queue.
260                     pState.offerAndInitializePseudoTime(this);
261                 }
262                 pState.isActiveCountChangeForTree(increment);
263             }
264         }
265 
266         void updateStreamableBytes(int newStreamableBytes, boolean isActive) {
267             if (this.active != isActive) {
268                 isActiveCountChangeForTree(isActive ? 1 : -1);
269                 this.active = isActive;
270             }
271 
272             streamableBytes = newStreamableBytes;
273         }
274 
275         /**
276          * Assumes the parents {@link #totalQueuedWeights} includes this node's weight.
277          */
278         void updatePseudoTime(State parentState, int nsent, long totalQueuedWeights) {
279             assert stream.id() != CONNECTION_STREAM_ID && nsent >= 0;
280             // If the current pseudoTimeToSend is greater than parentState.pseudoTime then we previously over accounted
281             // and should use parentState.pseudoTime.
282             pseudoTimeToWrite = min(pseudoTimeToWrite, parentState.pseudoTime) +
283                                 nsent * totalQueuedWeights / stream.weight();
284         }
285 
286         /**
287          * The concept of pseudoTime can be influenced by priority tree manipulations or if a stream goes from "active"
288          * to "non-active". This method accounts for that by initializing the {@link #pseudoTimeToWrite} for
289          * {@code state} to {@link #pseudoTime} of this node and then calls {@link #offer(State)}.
290          */
291         void offerAndInitializePseudoTime(State state) {
292             state.pseudoTimeToWrite = pseudoTime;
293             offer(state);
294         }
295 
296         void offer(State state) {
297             queue.offer(state);
298             totalQueuedWeights += state.stream.weight();
299         }
300 
301         /**
302          * Must only be called if the queue is non-empty!
303          */
304         State poll() {
305             State state = queue.poll();
306             // This method is only ever called if the queue is non-empty.
307             totalQueuedWeights -= state.stream.weight();
308             return state;
309         }
310 
311         void remove(State state) {
312             if (queue.remove(state)) {
313                 totalQueuedWeights -= state.stream.weight();
314             }
315         }
316 
317         State peek() {
318             return queue.peek();
319         }
320 
321         void close() {
322             updateStreamableBytes(0, false);
323         }
324 
325         @Override
326         public int compareTo(State o) {
327             return MathUtil.compare(pseudoTimeToWrite, o.pseudoTimeToWrite);
328         }
329 
330         @Override
331         public int priorityQueueIndex() {
332             return priorityQueueIndex;
333         }
334 
335         @Override
336         public void priorityQueueIndex(int i) {
337             priorityQueueIndex = i;
338         }
339     }
