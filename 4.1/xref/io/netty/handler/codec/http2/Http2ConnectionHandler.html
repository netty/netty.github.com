122     /**
123      * Handles the client-side (cleartext) upgrade from HTTP to HTTP/2.
124      * Reserves local stream 1 for the HTTP/2 response.
125      */
126     public void onHttpClientUpgrade() throws Http2Exception {
127         if (connection().isServer()) {
128             throw connectionError(PROTOCOL_ERROR, "Client-side HTTP upgrade requested for a server");
129         }
130         if (prefaceSent() || decoder.prefaceReceived()) {
131             throw connectionError(PROTOCOL_ERROR, "HTTP upgrade must occur before HTTP/2 preface is sent or received");
132         }
133 
134         // Create a local stream used for the HTTP cleartext upgrade.
135         connection().local().createStream(HTTP_UPGRADE_STREAM_ID, true);
136     }
137 
138     /**
139      * Handles the server-side (cleartext) upgrade from HTTP to HTTP/2.
140      * @param settings the settings for the remote endpoint.
141      */
142     public void onHttpServerUpgrade(Http2Settings settings) throws Http2Exception {
143         if (!connection().isServer()) {
144             throw connectionError(PROTOCOL_ERROR, "Server-side HTTP upgrade requested for a client");
145         }
146         if (prefaceSent() || decoder.prefaceReceived()) {
147             throw connectionError(PROTOCOL_ERROR, "HTTP upgrade must occur before HTTP/2 preface is sent or received");
148         }
149 
150         // Apply the settings but no ACK is necessary.
151         encoder.remoteSettings(settings);
152 
153         // Create a stream in the half-closed state.
154         connection().remote().createStream(HTTP_UPGRADE_STREAM_ID, true);
155     }

157     @Override
158     public void flush(ChannelHandlerContext ctx) throws Http2Exception {
159         // Trigger pending writes in the remote flow controller.
160         encoder.flowController().writePendingBytes();
161         try {
162             ctx.flush();
163         } catch (Throwable t) {
164             throw new Http2Exception(INTERNAL_ERROR, "Error flushing" , t);
165         }
166     }

168     private abstract class BaseDecoder {
169         public abstract void decode(ChannelHandlerContext ctx, ByteBuf in, List<Object> out) throws Exception;
170         public void handlerRemoved(ChannelHandlerContext ctx) throws Exception { }
171         public void channelActive(ChannelHandlerContext ctx) throws Exception { }
172 
173         public void channelInactive(ChannelHandlerContext ctx) throws Exception {
174             // Connection has terminated, close the encoder and decoder.
175             encoder().close();
176             decoder().close();
177 
178             final Http2Connection connection = connection();
179             // Check if there are streams to avoid the overhead of creating the ChannelFuture.
180             if (connection.numActiveStreams() > 0) {
181                 final ChannelFuture future = ctx.newSucceededFuture();
182                 connection.forEachActiveStream(new Http2StreamVisitor() {
183                     @Override
184                     public boolean visit(Http2Stream stream) throws Http2Exception {
185                         closeStream(stream, future);
186                         return true;
187                     }
188                 });
189             }
190         }

193          * Determine if the HTTP/2 connection preface been sent.
194          */
195         public boolean prefaceSent() {
196             return true;
197         }

200     private final class PrefaceDecoder extends BaseDecoder {
201         private ByteBuf clientPrefaceString;
202         private boolean prefaceSent;
203 
204         public PrefaceDecoder(ChannelHandlerContext ctx) {
205             clientPrefaceString = clientPrefaceString(encoder.connection());
206             // This handler was just added to the context. In case it was handled after
207             // the connection became active, send the connection preface now.
208             sendPreface(ctx);
209         }
210 
211         @Override
212         public boolean prefaceSent() {
213             return prefaceSent;
214         }
215 
216         @Override
217         public void decode(ChannelHandlerContext ctx, ByteBuf in, List<Object> out) throws Exception {
218             try {
219                 if (readClientPrefaceString(in) && verifyFirstFrameIsSettings(in)) {
220                     // After the preface is read, it is time to hand over control to the post initialized decoder.
221                     byteDecoder = new FrameDecoder();
222                     byteDecoder.decode(ctx, in, out);
223                 }
224             } catch (Throwable e) {
225                 onError(ctx, e);
226             }
227         }
228 
229         @Override
230         public void channelActive(ChannelHandlerContext ctx) throws Exception {
231             // The channel just became active - send the connection preface to the remote endpoint.
232             sendPreface(ctx);
233         }
234 
235         @Override
236         public void channelInactive(ChannelHandlerContext ctx) throws Exception {
237             cleanup();
238             super.channelInactive(ctx);
239         }

243          */
244         @Override
245         public void handlerRemoved(ChannelHandlerContext ctx) throws Exception {
246             cleanup();
247         }

249         /**
250          * Releases the {@code clientPrefaceString}. Any active streams will be left in the open.
251          */
252         private void cleanup() {
253             if (clientPrefaceString != null) {
254                 clientPrefaceString.release();
255                 clientPrefaceString = null;
256             }
257         }
258 
259         /**
260          * Decodes the client connection preface string from the input buffer.
261          *
262          * @return {@code true} if processing of the client preface string is complete. Since client preface strings can
263          *         only be received by servers, returns true immediately for client endpoints.
264          */
265         private boolean readClientPrefaceString(ByteBuf in) throws Http2Exception {
266             if (clientPrefaceString == null) {
267                 return true;
268             }
269 
270             int prefaceRemaining = clientPrefaceString.readableBytes();
271             int bytesRead = min(in.readableBytes(), prefaceRemaining);
272 
273             // If the input so far doesn't match the preface, break the connection.
274             if (bytesRead == 0 || !ByteBufUtil.equals(in, in.readerIndex(),
275                     clientPrefaceString, clientPrefaceString.readerIndex(), bytesRead)) {
276                 String receivedBytes = hexDump(in, in.readerIndex(),
277                         min(in.readableBytes(), clientPrefaceString.readableBytes()));
278                 throw connectionError(PROTOCOL_ERROR, "HTTP/2 client preface string missing or corrupt. " +
279                         "Hex dump for received bytes: %s", receivedBytes);
280             }
281             in.skipBytes(bytesRead);
282             clientPrefaceString.skipBytes(bytesRead);
283 
284             if (!clientPrefaceString.isReadable()) {
285                 // Entire preface has been read.
286                 clientPrefaceString.release();
287                 clientPrefaceString = null;
288                 return true;
289             }
290             return false;
291         }
292 
293         /**
294          * Peeks at that the next frame in the buffer and verifies that it is a non-ack {@code SETTINGS} frame.
295          *
296          * @param in the inbound buffer.
297          * @return {@code} true if the next frame is a non-ack {@code SETTINGS} frame, {@code false} if more
298          * data is required before we can determine the next frame type.
299          * @throws Http2Exception thrown if the next frame is NOT a non-ack {@code SETTINGS} frame.
300          */
301         private boolean verifyFirstFrameIsSettings(ByteBuf in) throws Http2Exception {
302             if (in.readableBytes() < 5) {
303                 // Need more data before we can see the frame type for the first frame.
304                 return false;
305             }
306 
307             short frameType = in.getUnsignedByte(in.readerIndex() + 3);
308             short flags = in.getUnsignedByte(in.readerIndex() + 4);
309             if (frameType != SETTINGS || (flags & Http2Flags.ACK) != 0) {
310                 throw connectionError(PROTOCOL_ERROR, "First received frame was not SETTINGS. " +
311                         "Hex dump for first 5 bytes: %s", hexDump(in, in.readerIndex(), 5));
312             }
313             return true;
314         }
315 
316         /**
317          * Sends the HTTP/2 connection preface upon establishment of the connection, if not already sent.
318          */
319         private void sendPreface(ChannelHandlerContext ctx) {
320             if (prefaceSent || !ctx.channel().isActive()) {
321                 return;
322             }
323 
324             prefaceSent = true;
325 
326             if (!connection().isServer()) {
327                 // Clients must send the preface string as the first bytes on the connection.
328                 ctx.write(connectionPrefaceBuf()).addListener(ChannelFutureListener.CLOSE_ON_FAILURE);
329             }
330 
331             // Both client and server must send their initial settings.
332             encoder.writeSettings(ctx, initialSettings, ctx.newPromise()).addListener(
333                     ChannelFutureListener.CLOSE_ON_FAILURE);
334         }
335     }

385     @Override
386     public void channelWritabilityChanged(ChannelHandlerContext ctx) throws Exception {
387         // Writability is expected to change while we are writing. We cannot allow this event to trigger reentering
388         // the allocation and write loop. Reentering the event loop will lead to over or illegal allocation.
389         try {
390             if (ctx.channel().isWritable()) {

392             }
393             encoder.flowController().channelWritabilityChanged();
394         } finally {
395             super.channelWritabilityChanged(ctx);
396         }

420     @Override
421     public void close(ChannelHandlerContext ctx, ChannelPromise promise) throws Exception {
422         // Avoid NotYetConnectedException
423         if (!ctx.channel().isActive()) {
424             ctx.close(promise);
425             return;
426         }
427 
428         ChannelFuture future = goAway(ctx, null);
429         ctx.flush();
430         doGracefulShutdown(ctx, future, promise);
431     }

433     private void doGracefulShutdown(ChannelHandlerContext ctx, ChannelFuture future, ChannelPromise promise) {
434         // If there are no active streams, close immediately after the send is complete.
435         // Otherwise wait until all streams are inactive.
436         if (isGracefulShutdownComplete()) {
437             future.addListener(new ClosingChannelFutureListener(ctx, promise));
438         } else {
439             closeListener = new ClosingChannelFutureListener(ctx, promise,
440                                         gracefulShutdownTimeoutMillis, MILLISECONDS);
441         }
442     }

539     /**
540      * Central handler for all exceptions caught during HTTP/2 processing.
541      */
542     @Override
543     public void onError(ChannelHandlerContext ctx, Throwable cause) {
544         Http2Exception embedded = getEmbeddedHttp2Exception(cause);
545         if (isStreamError(embedded)) {
546             onStreamError(ctx, cause, (StreamException) embedded);
547         } else if (embedded instanceof CompositeStreamException) {
548             CompositeStreamException compositException = (CompositeStreamException) embedded;
549             for (StreamException streamException : compositException) {
550                 onStreamError(ctx, cause, streamException);
551             }
552         } else {
553             onConnectionError(ctx, cause, embedded);
554         }
555         ctx.flush();
556     }

558     /**
559      * Called by the graceful shutdown logic to determine when it is safe to close the connection. Returns {@code true}
560      * if the graceful shutdown has completed and the connection can be safely closed. This implementation just
561      * guarantees that there are no active streams. Subclasses may override to provide additional checks.
562      */
563     protected boolean isGracefulShutdownComplete() {
564         return connection().numActiveStreams() == 0;
565     }

567     /**
568      * Handler for a connection error. Sends a GO_AWAY frame to the remote endpoint. Once all
569      * streams are closed, the connection is shut down.
570      *
571      * @param ctx the channel context
572      * @param cause the exception that was caught
573      * @param http2Ex the {@link Http2Exception} that is embedded in the causality chain. This may
574      *            be {@code null} if it's an unknown exception.
575      */
576     protected void onConnectionError(ChannelHandlerContext ctx, Throwable cause, Http2Exception http2Ex) {
577         if (http2Ex == null) {
578             http2Ex = new Http2Exception(INTERNAL_ERROR, cause.getMessage(), cause);
579         }
580 
581         ChannelPromise promise = ctx.newPromise();
582         ChannelFuture future = goAway(ctx, http2Ex);
583         switch (http2Ex.shutdownHint()) {
584         case GRACEFUL_SHUTDOWN:
585             doGracefulShutdown(ctx, future, promise);
586             break;
587         default:
588             future.addListener(new ClosingChannelFutureListener(ctx, promise));
589             break;
590         }
591     }

610     @Override
611     public ChannelFuture resetStream(final ChannelHandlerContext ctx, int streamId, long errorCode,
612             final ChannelPromise promise) {
613         final Http2Stream stream = connection().stream(streamId);
614         if (stream == null || stream.isResetSent()) {
615             // Don't write a RST_STREAM frame if we are not aware of the stream, or if we have already written one.
616             return promise.setSuccess();
617         }
618 
619         final ChannelFuture future;
620         if (stream.state() == IDLE || (connection().local().created(stream) && !stream.isHeaderSent())) {
621             // The other endpoint doesn't know about the stream yet, so we can't actually send
622             // the RST_STREAM frame. The HTTP/2 spec also disallows sending RST_STREAM for IDLE streams.
623             future = promise.setSuccess();
624         } else {
625             future = frameWriter().writeRstStream(ctx, streamId, errorCode, promise);
626         }
627 
628         // Synchronously set the resetSent flag to prevent any subsequent calls
629         // from resulting in multiple reset frames being sent.
630         stream.resetSent();
631 
632         future.addListener(new ChannelFutureListener() {
633             @Override
634             public void operationComplete(ChannelFuture future) throws Exception {
635                 if (future.isSuccess()) {
636                     closeStream(stream, promise);
637                 } else {
638                     // The connection will be closed and so no need to change the resetSent flag to false.
639                     onConnectionError(ctx, future.cause(), null);
640                 }
641             }
642         });
643 
644         return future;
645     }

647     @Override
648     public ChannelFuture goAway(final ChannelHandlerContext ctx, final int lastStreamId, final long errorCode,
649                                 final ByteBuf debugData, ChannelPromise promise) {
650         try {
651             final Http2Connection connection = connection();
652             if (connection.goAwaySent() && lastStreamId > connection.remote().lastStreamKnownByPeer()) {
653                 throw connectionError(PROTOCOL_ERROR, "Last stream identifier must not increase between " +
654                                                       "sending multiple GOAWAY frames (was '%d', is '%d').",
655                                                       connection.remote().lastStreamKnownByPeer(),
656                                                       lastStreamId);
657             }
658             connection.goAwaySent(lastStreamId, errorCode, debugData);
659 
660             // Need to retain before we write the buffer because if we do it after the refCnt could already be 0 and
661             // result in an IllegalRefCountException.
662             debugData.retain();
663             ChannelFuture future = frameWriter().writeGoAway(ctx, lastStreamId, errorCode, debugData, promise);
664 
665             if (future.isDone()) {
666                 processGoAwayWriteResult(ctx, lastStreamId, errorCode, debugData, future);
667             } else {
668                 future.addListener(new ChannelFutureListener() {
669                     @Override
670                     public void operationComplete(ChannelFuture future) throws Exception {
671                         processGoAwayWriteResult(ctx, lastStreamId, errorCode, debugData, future);
672                     }
673                 });
674             }
675 
676             return future;
677         } catch (Throwable cause) { // Make sure to catch Throwable because we are doing a retain() in this method.
678             debugData.release();
679             return promise.setFailure(cause);
680         }
681     }

684      * Closes the connection if the graceful shutdown process has completed.
685      * @param future Represents the status that will be passed to the {@link #closeListener}.
686      */
687     private void checkCloseConnection(ChannelFuture future) {
688         // If this connection is closing and the graceful shutdown has completed, close the connection
689         // once this operation completes.
690         if (closeListener != null && isGracefulShutdownComplete()) {
691             ChannelFutureListener closeListener = Http2ConnectionHandler.this.closeListener;
692             // This method could be called multiple times
693             // and we don't want to notify the closeListener multiple times.
694             Http2ConnectionHandler.this.closeListener = null;
695             try {
696                 closeListener.operationComplete(future);
697             } catch (Exception e) {
698                 throw new IllegalStateException("Close listener threw an unexpected exception", e);
699             }
700         }
701     }

704      * Close the remote endpoint with with a {@code GO_AWAY} frame. Does <strong>not</strong> flush
705      * immediately, this is the responsibility of the caller.

707     private ChannelFuture goAway(ChannelHandlerContext ctx, Http2Exception cause) {
708         long errorCode = cause != null ? cause.error().code() : NO_ERROR.code();
709         ByteBuf debugData = Http2CodecUtil.toByteBuf(ctx, cause);

721     private static void processGoAwayWriteResult(final ChannelHandlerContext ctx, final int lastStreamId,
722             final long errorCode, final ByteBuf debugData, ChannelFuture future) {
723         try {
724             if (future.isSuccess()) {
725                 if (errorCode != NO_ERROR.code()) {
726                     if (logger.isDebugEnabled()) {
727                         logger.debug(

729                                         "debugData '%s'. Forcing shutdown of the connection.",
730                                         lastStreamId, errorCode, debugData.toString(UTF_8)),
731                                         future.cause());
732                     }
733                     ctx.close();
734                 }
735             } else {
736                 if (logger.isErrorEnabled()) {
737                     logger.error(
738                             format("Sending GOAWAY failed: lastStreamId '%d', errorCode '%d', " +
739                                     "debugData '%s'. Forcing shutdown of the connection.",
740                                     lastStreamId, errorCode, debugData.toString(UTF_8)), future.cause());
741                 }
742                 ctx.close();
743             }
744         } finally {
745             // We're done with the debug data now.
746             debugData.release();
747         }

751      * Closes the channel when the future completes.

753     private static final class ClosingChannelFutureListener implements ChannelFutureListener {
754         private final ChannelHandlerContext ctx;
755         private final ChannelPromise promise;
756         private final ScheduledFuture<?> timeoutTask;

759             this.ctx = ctx;
760             this.promise = promise;
761             timeoutTask = null;
762         }
763 
764         ClosingChannelFutureListener(final ChannelHandlerContext ctx, final ChannelPromise promise,
765                                      long timeout, TimeUnit unit) {
766             this.ctx = ctx;
767             this.promise = promise;
768             timeoutTask = ctx.executor().schedule(new OneTimeTask() {
769                 @Override

771                     ctx.close(promise);
772                 }
773             }, timeout, unit);
774         }
775 
776         @Override
777         public void operationComplete(ChannelFuture sentGoAwayFuture) throws Exception {
778             if (timeoutTask != null) {
