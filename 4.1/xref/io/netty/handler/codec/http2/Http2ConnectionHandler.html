139     /**
140      * Handles the client-side (cleartext) upgrade from HTTP to HTTP/2.
141      * Reserves local stream 1 for the HTTP/2 response.
142      */
143     public void onHttpClientUpgrade() throws Http2Exception {
144         if (connection().isServer()) {
145             throw connectionError(PROTOCOL_ERROR, "Client-side HTTP upgrade requested for a server");
146         }
147         if (!prefaceSent()) {
148             // If the preface was not sent yet it most likely means the handler was not added to the pipeline before
149             // calling this method.
150             throw connectionError(INTERNAL_ERROR, "HTTP upgrade must occur after preface was sent");
151         }
152         if (decoder.prefaceReceived()) {
153             throw connectionError(PROTOCOL_ERROR, "HTTP upgrade must occur before HTTP/2 preface is received");
154         }
155 
156         // Create a local stream used for the HTTP cleartext upgrade.
157         connection().local().createStream(HTTP_UPGRADE_STREAM_ID, true);
158     }
159 
160     /**
161      * Handles the server-side (cleartext) upgrade from HTTP to HTTP/2.
162      * @param settings the settings for the remote endpoint.
163      */
164     public void onHttpServerUpgrade(Http2Settings settings) throws Http2Exception {
165         if (!connection().isServer()) {
166             throw connectionError(PROTOCOL_ERROR, "Server-side HTTP upgrade requested for a client");
167         }
168         if (!prefaceSent()) {
169             // If the preface was not sent yet it most likely means the handler was not added to the pipeline before
170             // calling this method.
171             throw connectionError(INTERNAL_ERROR, "HTTP upgrade must occur after preface was sent");
172         }
173         if (decoder.prefaceReceived()) {
174             throw connectionError(PROTOCOL_ERROR, "HTTP upgrade must occur before HTTP/2 preface is received");
175         }
176 
177         // Apply the settings but no ACK is necessary.
178         encoder.remoteSettings(settings);
179 
180         // Create a stream in the half-closed state.
181         connection().remote().createStream(HTTP_UPGRADE_STREAM_ID, true);
182     }

184     @Override
185     public void flush(ChannelHandlerContext ctx) {
186         try {
187             // Trigger pending writes in the remote flow controller.
188             encoder.flowController().writePendingBytes();
189             ctx.flush();
190         } catch (Http2Exception e) {
191             onError(ctx, true, e);
192         } catch (Throwable cause) {
193             onError(ctx, true, connectionError(INTERNAL_ERROR, cause, "Error flushing"));
194         }

224         PrefaceDecoder(ChannelHandlerContext ctx) throws Exception {
225             clientPrefaceString = clientPrefaceString(encoder.connection());
226             // This handler was just added to the context. In case it was handled after
227             // the connection became active, send the connection preface now.

236         @Override
237         public void decode(ChannelHandlerContext ctx, ByteBuf in, List<Object> out) throws Exception {
238             try {
239                 if (ctx.channel().isActive() && readClientPrefaceString(in) && verifyFirstFrameIsSettings(in)) {
240                     // After the preface is read, it is time to hand over control to the post initialized decoder.
241                     byteDecoder = new FrameDecoder();
242                     byteDecoder.decode(ctx, in, out);
243                 }
244             } catch (Throwable e) {
245                 onError(ctx, false, e);
246             }
247         }

279         /**
280          * Decodes the client connection preface string from the input buffer.
281          *
282          * @return {@code true} if processing of the client preface string is complete. Since client preface strings can
283          *         only be received by servers, returns true immediately for client endpoints.
284          */
285         private boolean readClientPrefaceString(ByteBuf in) throws Http2Exception {
286             if (clientPrefaceString == null) {
287                 return true;
288             }
289 
290             int prefaceRemaining = clientPrefaceString.readableBytes();
291             int bytesRead = min(in.readableBytes(), prefaceRemaining);
292 
293             // If the input so far doesn't match the preface, break the connection.
294             if (bytesRead == 0 || !ByteBufUtil.equals(in, in.readerIndex(),
295                                                       clientPrefaceString, clientPrefaceString.readerIndex(),
296                                                       bytesRead)) {
297                 int maxSearch = 1024; // picked because 512 is too little, and 2048 too much
298                 int http1Index =
299                     ByteBufUtil.indexOf(HTTP_1_X_BUF, in.slice(in.readerIndex(), min(in.readableBytes(), maxSearch)));
300                 if (http1Index != -1) {
301                     String chunk = in.toString(in.readerIndex(), http1Index - in.readerIndex(), CharsetUtil.US_ASCII);
302                     throw connectionError(PROTOCOL_ERROR, "Unexpected HTTP/1.x request: %s", chunk);
303                 }
304                 String receivedBytes = hexDump(in, in.readerIndex(),
305                                                min(in.readableBytes(), clientPrefaceString.readableBytes()));
306                 throw connectionError(PROTOCOL_ERROR, "HTTP/2 client preface string missing or corrupt. " +
307                                                       "Hex dump for received bytes: %s", receivedBytes);
308             }
309             in.skipBytes(bytesRead);
310             clientPrefaceString.skipBytes(bytesRead);
311 
312             if (!clientPrefaceString.isReadable()) {
313                 // Entire preface has been read.
314                 clientPrefaceString.release();
315                 clientPrefaceString = null;
316                 return true;
317             }
318             return false;
319         }
320 
321         /**
322          * Peeks at that the next frame in the buffer and verifies that it is a non-ack {@code SETTINGS} frame.
323          *
324          * @param in the inbound buffer.
325          * @return {@code true} if the next frame is a non-ack {@code SETTINGS} frame, {@code false} if more
326          * data is required before we can determine the next frame type.
327          * @throws Http2Exception thrown if the next frame is NOT a non-ack {@code SETTINGS} frame.
328          */
329         private boolean verifyFirstFrameIsSettings(ByteBuf in) throws Http2Exception {
330             if (in.readableBytes() < 5) {
331                 // Need more data before we can see the frame type for the first frame.
332                 return false;
333             }
334 
335             short frameType = in.getUnsignedByte(in.readerIndex() + 3);
336             short flags = in.getUnsignedByte(in.readerIndex() + 4);
337             if (frameType != SETTINGS || (flags & Http2Flags.ACK) != 0) {
338                 throw connectionError(PROTOCOL_ERROR, "First received frame was not SETTINGS. " +
339                                                       "Hex dump for first 5 bytes: %s",
340                                       hexDump(in, in.readerIndex(), 5));
341             }
342             return true;
343         }

345         /**
346          * Sends the HTTP/2 connection preface upon establishment of the connection, if not already sent.
347          */
348         private void sendPreface(ChannelHandlerContext ctx) throws Exception {
349             if (prefaceSent || !ctx.channel().isActive()) {
350                 return;
351             }
352 
353             prefaceSent = true;
354 
355             final boolean isClient = !connection().isServer();
356             if (isClient) {
357                 // Clients must send the preface string as the first bytes on the connection.
358                 ctx.write(connectionPrefaceBuf()).addListener(ChannelFutureListener.CLOSE_ON_FAILURE);
359             }
360 
361             // Both client and server must send their initial settings.
362             encoder.writeSettings(ctx, initialSettings, ctx.newPromise()).addListener(
363                     ChannelFutureListener.CLOSE_ON_FAILURE);
364 
365             if (isClient) {
366                 // If this handler is extended by the user and we directly fire the userEvent from this context then
367                 // the user will not see the event. We should fire the event starting with this handler so this class
368                 // (and extending classes) have a chance to process the event.
369                 userEventTriggered(ctx, Http2ConnectionPrefaceAndSettingsFrameWrittenEvent.INSTANCE);
370             }
371         }
372     }

424         // Writability is expected to change while we are writing. We cannot allow this event to trigger reentering
425         // the allocation and write loop. Reentering the event loop will lead to over or illegal allocation.
426         try {
427             if (ctx.channel().isWritable()) {

457     @Override
458     public void close(ChannelHandlerContext ctx, ChannelPromise promise) throws Exception {
459         if (decoupleCloseAndGoAway) {
460             ctx.close(promise);
461             return;
462         }
463         promise = promise.unvoid();
464         // Avoid NotYetConnectedException and avoid sending before connection preface
465         if (!ctx.channel().isActive() || !prefaceSent()) {
466             ctx.close(promise);
467             return;
468         }
469 
470         // If the user has already sent a GO_AWAY frame they may be attempting to do a graceful shutdown which requires
471         // sending multiple GO_AWAY frames. We should only send a GO_AWAY here if one has not already been sent. If
472         // a GO_AWAY has been sent we send a empty buffer just so we can wait to close until all other data has been
473         // flushed to the OS.
474         // https://github.com/netty/netty/issues/5307
475         ChannelFuture f = connection().goAwaySent() ? ctx.write(EMPTY_BUFFER) : goAway(ctx, null, ctx.newPromise());
476         ctx.flush();
477         doGracefulShutdown(ctx, f, promise);
478     }

480     private ChannelFutureListener newClosingChannelFutureListener(
481             ChannelHandlerContext ctx, ChannelPromise promise) {
482         long gracefulShutdownTimeoutMillis = this.gracefulShutdownTimeoutMillis;
483         return gracefulShutdownTimeoutMillis < 0 ?
484                 new ClosingChannelFutureListener(ctx, promise) :
485                 new ClosingChannelFutureListener(ctx, promise, gracefulShutdownTimeoutMillis, MILLISECONDS);
486     }

488     private void doGracefulShutdown(ChannelHandlerContext ctx, ChannelFuture future, final ChannelPromise promise) {
489         final ChannelFutureListener listener = newClosingChannelFutureListener(ctx, promise);
490         if (isGracefulShutdownComplete()) {
491             // If there are no active streams, close immediately after the GO_AWAY write completes or the timeout
492             // elapsed.
493             future.addListener(listener);
494         } else {
495             // If there are active streams we should wait until they are all closed before closing the connection.
496 
497             // The ClosingChannelFutureListener will cascade promise completion. We need to always notify the
498             // new ClosingChannelFutureListener when the graceful close completes if the promise is not null.
499             if (closeListener == null) {
500                 closeListener = listener;
501             } else if (promise != null) {
502                 final ChannelFutureListener oldCloseListener = closeListener;
503                 closeListener = new ChannelFutureListener() {
504                     @Override
505                     public void operationComplete(ChannelFuture future) throws Exception {
506                         try {
507                             oldCloseListener.operationComplete(future);
508                         } finally {
509                             listener.operationComplete(future);
510                         }
511                     }
512                 };
513             }
514         }
515     }

532     @Override
533     public void channelReadComplete(ChannelHandlerContext ctx) throws Exception {
534         // Trigger flush after read on the assumption that flush is cheap if there is nothing to write and that
535         // for flow-control the read may release window that causes data to be written that can now be flushed.
536         try {
537             // First call channelReadComplete0(...) as this may produce more data that we want to flush
538             channelReadComplete0(ctx);
539         } finally {
540             flush(ctx);
541         }
542     }
543 
544     final void channelReadComplete0(ChannelHandlerContext ctx) {
545         // Discard bytes of the cumulation buffer if needed.
546         discardSomeReadBytes();
547 
548         // Ensure we never stale the HTTP/2 Channel. Flow-control is enforced by HTTP/2.
549         //
550         // See https://tools.ietf.org/html/rfc7540#section-5.2.2
551         if (!ctx.channel().config().isAutoRead()) {
552             ctx.read();
553         }
554 
555         ctx.fireChannelReadComplete();
556     }

627     /**
628      * Central handler for all exceptions caught during HTTP/2 processing.
629      */
630     @Override
631     public void onError(ChannelHandlerContext ctx, boolean outbound, Throwable cause) {
632         Http2Exception embedded = getEmbeddedHttp2Exception(cause);
633         if (isStreamError(embedded)) {
634             onStreamError(ctx, outbound, cause, (StreamException) embedded);
635         } else if (embedded instanceof CompositeStreamException) {
636             CompositeStreamException compositException = (CompositeStreamException) embedded;
637             for (StreamException streamException : compositException) {
638                 onStreamError(ctx, outbound, cause, streamException);
639             }
640         } else {
641             onConnectionError(ctx, outbound, cause, embedded);
642         }
643         ctx.flush();
644     }

661      * @param cause the exception that was caught
662      * @param http2Ex the {@link Http2Exception} that is embedded in the causality chain. This may
663      *            be {@code null} if it's an unknown exception.
664      */
665     protected void onConnectionError(ChannelHandlerContext ctx, boolean outbound,
666                                      Throwable cause, Http2Exception http2Ex) {
667         if (http2Ex == null) {
668             http2Ex = new Http2Exception(INTERNAL_ERROR, cause.getMessage(), cause);
669         }
670 
671         ChannelPromise promise = ctx.newPromise();
672         ChannelFuture future = goAway(ctx, http2Ex, ctx.newPromise());
673         if (http2Ex.shutdownHint() == Http2Exception.ShutdownHint.GRACEFUL_SHUTDOWN) {
674             doGracefulShutdown(ctx, future, promise);
675         } else {
676             future.addListener(newClosingChannelFutureListener(ctx, promise));
677         }

680     /**
681      * Handler for a stream error. Sends a {@code RST_STREAM} frame to the remote endpoint and closes the
682      * stream.
683      *
684      * @param ctx the channel context
685      * @param outbound {@code true} if the error was caused by an outbound operation.
686      * @param cause the exception that was caught
687      * @param http2Ex the {@link StreamException} that is embedded in the causality chain.
688      */
689     protected void onStreamError(ChannelHandlerContext ctx, boolean outbound,
690                                  @SuppressWarnings("unused") Throwable cause, StreamException http2Ex) {
691         final int streamId = http2Ex.streamId();
692         Http2Stream stream = connection().stream(streamId);
693 
694         //if this is caused by reading headers that are too large, send a header with status 431
695         if (http2Ex instanceof Http2Exception.HeaderListSizeException &&
696             ((Http2Exception.HeaderListSizeException) http2Ex).duringDecode() &&
697             connection().isServer()) {
698 
699             // NOTE We have to check to make sure that a stream exists before we send our reply.
700             // We likely always create the stream below as the stream isn't created until the
701             // header block is completely processed.
702 
703             // The case of a streamId referring to a stream which was already closed is handled
704             // by createStream and will land us in the catch block below
705             if (stream == null) {
706                 try {
707                     stream = encoder.connection().remote().createStream(streamId, true);
708                 } catch (Http2Exception e) {
709                     resetUnknownStream(ctx, streamId, http2Ex.error().code(), ctx.newPromise());
710                     return;
711                 }
712             }
713 
714             // ensure that we have not already sent headers on this stream
715             if (stream != null && !stream.isHeadersSent()) {
716                 try {
717                     handleServerHeaderDecodeSizeError(ctx, stream);
718                 } catch (Throwable cause2) {
719                     onError(ctx, outbound, connectionError(INTERNAL_ERROR, cause2, "Error DecodeSizeError"));
720                 }
721             }
722         }
723 
724         if (stream == null) {
725             if (!outbound || connection().local().mayHaveCreatedStream(streamId)) {
726                 resetUnknownStream(ctx, streamId, http2Ex.error().code(), ctx.newPromise());
727             }
728         } else {
729             resetStream(ctx, stream, http2Ex.error().code(), ctx.newPromise());
730         }
731     }

771                                      ChannelPromise promise) {

780     private ChannelFuture resetStream(final ChannelHandlerContext ctx, final Http2Stream stream,
781                                       long errorCode, ChannelPromise promise) {
782         promise = promise.unvoid();
783         if (stream.isResetSent()) {
784             // Don't write a RST_STREAM frame if we have already written one.
785             return promise.setSuccess();
786         }
787         // Synchronously set the resetSent flag to prevent any subsequent calls
788         // from resulting in multiple reset frames being sent.
789         //
790         // This needs to be done before we notify the promise as the promise may have a listener attached that
791         // call resetStream(...) again.
792         stream.resetSent();
793 
794         final ChannelFuture future;
795         // If the remote peer is not aware of the steam, then we are not allowed to send a RST_STREAM
796         // https://tools.ietf.org/html/rfc7540#section-6.4.
797         if (stream.state() == IDLE ||
798             connection().local().created(stream) && !stream.isHeadersSent() && !stream.isPushPromiseSent()) {
799             future = promise.setSuccess();
800         } else {
801             future = frameWriter().writeRstStream(ctx, stream.id(), errorCode, promise);
802         }
803         if (future.isDone()) {
804             processRstStreamWriteResult(ctx, stream, future);
805         } else {
806             future.addListener(new ChannelFutureListener() {
807                 @Override
808                 public void operationComplete(ChannelFuture future) throws Exception {
809                     processRstStreamWriteResult(ctx, stream, future);
810                 }
811             });
812         }
813 
814         return future;
815     }

817     @Override
818     public ChannelFuture goAway(final ChannelHandlerContext ctx, final int lastStreamId, final long errorCode,
819                                 final ByteBuf debugData, ChannelPromise promise) {
820         promise = promise.unvoid();
821         final Http2Connection connection = connection();
822         try {
823             if (!connection.goAwaySent(lastStreamId, errorCode, debugData)) {
824                 debugData.release();
825                 promise.trySuccess();
826                 return promise;
827             }
828         } catch (Throwable cause) {
829             debugData.release();
830             promise.tryFailure(cause);
831             return promise;
832         }
833 
834         // Need to retain before we write the buffer because if we do it after the refCnt could already be 0 and
835         // result in an IllegalRefCountException.
836         debugData.retain();
837         ChannelFuture future = frameWriter().writeGoAway(ctx, lastStreamId, errorCode, debugData, promise);
838 
839         if (future.isDone()) {
840             processGoAwayWriteResult(ctx, lastStreamId, errorCode, debugData, future);
841         } else {
842             future.addListener(new ChannelFutureListener() {
843                 @Override
844                 public void operationComplete(ChannelFuture future) throws Exception {
845                     processGoAwayWriteResult(ctx, lastStreamId, errorCode, debugData, future);
846                 }
847             });
848         }
849 
850         return future;
851     }

853     /**
854      * Closes the connection if the graceful shutdown process has completed.
855      * @param future Represents the status that will be passed to the {@link #closeListener}.
856      */
857     private void checkCloseConnection(ChannelFuture future) {
858         // If this connection is closing and the graceful shutdown has completed, close the connection
859         // once this operation completes.
860         if (closeListener != null && isGracefulShutdownComplete()) {
861             ChannelFutureListener closeListener = this.closeListener;
862             // This method could be called multiple times
863             // and we don't want to notify the closeListener multiple times.
864             this.closeListener = null;
865             try {
866                 closeListener.operationComplete(future);
867             } catch (Exception e) {
868                 throw new IllegalStateException("Close listener threw an unexpected exception", e);
869             }
870         }
871     }

873     /**
874      * Close the remote endpoint with with a {@code GO_AWAY} frame. Does <strong>not</strong> flush
875      * immediately, this is the responsibility of the caller.
876      */
877     private ChannelFuture goAway(ChannelHandlerContext ctx, Http2Exception cause, ChannelPromise promise) {
878         long errorCode = cause != null ? cause.error().code() : NO_ERROR.code();
879         int lastKnownStream;
880         if (cause != null && cause.shutdownHint() == Http2Exception.ShutdownHint.HARD_SHUTDOWN) {
881             // The hard shutdown could have been triggered during header processing, before updating
882             // lastStreamCreated(). Specifically, any connection errors encountered by Http2FrameReader or HPACK
883             // decoding will fail to update the last known stream. So we must be pessimistic.
884             // https://github.com/netty/netty/issues/10670
885             lastKnownStream = Integer.MAX_VALUE;
886         } else {
887             lastKnownStream = connection().remote().lastStreamCreated();
888         }
889         return goAway(ctx, lastKnownStream, errorCode, Http2CodecUtil.toByteBuf(ctx, cause), promise);
890     }

892     private void processRstStreamWriteResult(ChannelHandlerContext ctx, Http2Stream stream, ChannelFuture future) {
893         if (future.isSuccess()) {
894             closeStream(stream, future);
895         } else {
896             // The connection will be closed and so no need to change the resetSent flag to false.
897             onConnectionError(ctx, true, future.cause(), null);
898         }
899     }

901     private void closeConnectionOnError(ChannelHandlerContext ctx, ChannelFuture future) {
902         if (!future.isSuccess()) {

914     private static void processGoAwayWriteResult(final ChannelHandlerContext ctx, final int lastStreamId,
915                                                  final long errorCode, final ByteBuf debugData, ChannelFuture future) {
916         try {
917             if (future.isSuccess()) {
918                 if (errorCode != NO_ERROR.code()) {
919                     if (logger.isDebugEnabled()) {
920                         logger.debug("{} Sent GOAWAY: lastStreamId '{}', errorCode '{}', " +
921                                      "debugData '{}'. Forcing shutdown of the connection.",
922                                      ctx.channel(), lastStreamId, errorCode, debugData.toString(UTF_8), future.cause());
923                     }
924                     ctx.close();
925                 }
926             } else {
927                 if (logger.isDebugEnabled()) {
928                     logger.debug("{} Sending GOAWAY failed: lastStreamId '{}', errorCode '{}', " +
929                                  "debugData '{}'. Forcing shutdown of the connection.",
930                                  ctx.channel(), lastStreamId, errorCode, debugData.toString(UTF_8), future.cause());
931                 }
932                 ctx.close();
933             }
934         } finally {
935             // We're done with the debug data now.
936             debugData.release();
937         }

940     /**
941      * Closes the channel when the future completes.
942      */
943     private static final class ClosingChannelFutureListener implements ChannelFutureListener {
944         private final ChannelHandlerContext ctx;
945         private final ChannelPromise promise;
946         private final ScheduledFuture<?> timeoutTask;
947         private boolean closed;
948 
949         ClosingChannelFutureListener(ChannelHandlerContext ctx, ChannelPromise promise) {
950             this.ctx = ctx;
951             this.promise = promise;
952             timeoutTask = null;
953         }
954 
955         ClosingChannelFutureListener(final ChannelHandlerContext ctx, final ChannelPromise promise,
956                                      long timeout, TimeUnit unit) {
957             this.ctx = ctx;
958             this.promise = promise;
959             timeoutTask = ctx.executor().schedule(new Runnable() {
960                 @Override
961                 public void run() {
962                     doClose();
963                 }
964             }, timeout, unit);
965         }
966 
967         @Override
968         public void operationComplete(ChannelFuture sentGoAwayFuture) {
969             if (timeoutTask != null) {
970                 timeoutTask.cancel(false);
971             }
972             doClose();
973         }
974 
975         private void doClose() {
976             // We need to guard against multiple calls as the timeout may trigger close() first and then it will be
977             // triggered again because of operationComplete(...) is called.
978             if (closed) {
979                 // This only happens if we also scheduled a timeout task.
980                 assert timeoutTask != null;
981                 return;
982             }
983             closed = true;
984             if (promise == null) {
985                 ctx.close();
986             } else {
987                 ctx.close(promise);
988             }
989         }
990     }
