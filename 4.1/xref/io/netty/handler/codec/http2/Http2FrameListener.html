16  package io.netty.handler.codec.http2;
17  
18  import io.netty.buffer.ByteBuf;
19  import io.netty.channel.ChannelHandlerContext;
20  
21  /**
22   * An listener of HTTP/2 frames.
23   */
24  public interface Http2FrameListener {
25      /**
26       * Handles an inbound {@code DATA} frame.
27       *
28       * @param ctx the context from the handler where the frame was read.
29       * @param streamId the subject stream for the frame.
30       * @param data payload buffer for the frame. This buffer will be released by the codec.
31       * @param padding the number of padding bytes found at the end of the frame.
32       * @param endOfStream Indicates whether this is the last frame to be sent from the remote endpoint for this stream.
33       * @return the number of bytes that have been processed by the application. The returned bytes are used by the
34       * inbound flow controller to determine the appropriate time to expand the inbound flow control window (i.e. send
35       * {@code WINDOW_UPDATE}). Returning a value equal to the length of {@code data} + {@code padding} will effectively
36       * opt-out of application-level flow control for this frame. Returning a value less than the length of {@code data}
37       * + {@code padding} will defer the returning of the processed bytes, which the application must later return via
38       * {@link Http2LocalFlowController#consumeBytes(Http2Stream, int)}. The returned value must
39       * be >= {@code 0} and <= {@code data.readableBytes()} + {@code padding}.
40       */
41      int onDataRead(ChannelHandlerContext ctx, int streamId, ByteBuf data, int padding,
42                     boolean endOfStream) throws Http2Exception;
43  
44      /**
45       * Handles an inbound {@code HEADERS} frame.
46       * <p>
47       * Only one of the following methods will be called for each {@code HEADERS} frame sequence.
48       * One will be called when the {@code END_HEADERS} flag has been received.
49       * <ul>
50       * <li>{@link #onHeadersRead(ChannelHandlerContext, int, Http2Headers, int, boolean)}</li>
51       * <li>{@link #onHeadersRead(ChannelHandlerContext, int, Http2Headers, int, short, boolean, int, boolean)}</li>
52       * <li>{@link #onPushPromiseRead(ChannelHandlerContext, int, int, Http2Headers, int)}</li>
53       * </ul>
54       * <p>
55       * To say it another way; the {@link Http2Headers} will contain all of the headers
56       * for the current message exchange step (additional queuing is not necessary).
57       *
58       * @param ctx the context from the handler where the frame was read.
59       * @param streamId the subject stream for the frame.
60       * @param headers the received headers.
61       * @param padding the number of padding bytes found at the end of the frame.
62       * @param endOfStream Indicates whether this is the last frame to be sent from the remote endpoint
63       *            for this stream.
64       */
65      void onHeadersRead(ChannelHandlerContext ctx, int streamId, Http2Headers headers, int padding,
66              boolean endOfStream) throws Http2Exception;
67  
68      /**
69       * Handles an inbound {@code HEADERS} frame with priority information specified.
70       * Only called if {@code END_HEADERS} encountered.
71       * <p>
72       * Only one of the following methods will be called for each {@code HEADERS} frame sequence.
73       * One will be called when the {@code END_HEADERS} flag has been received.
74       * <ul>
75       * <li>{@link #onHeadersRead(ChannelHandlerContext, int, Http2Headers, int, boolean)}</li>
76       * <li>{@link #onHeadersRead(ChannelHandlerContext, int, Http2Headers, int, short, boolean, int, boolean)}</li>
77       * <li>{@link #onPushPromiseRead(ChannelHandlerContext, int, int, Http2Headers, int)}</li>
78       * </ul>
79       * <p>
80       * To say it another way; the {@link Http2Headers} will contain all of the headers
81       * for the current message exchange step (additional queuing is not necessary).
82       *
83       * @param ctx the context from the handler where the frame was read.
84       * @param streamId the subject stream for the frame.
85       * @param headers the received headers.
86       * @param streamDependency the stream on which this stream depends, or 0 if dependent on the
87       *            connection.
88       * @param weight the new weight for the stream.
89       * @param exclusive whether or not the stream should be the exclusive dependent of its parent.
90       * @param padding the number of padding bytes found at the end of the frame.
91       * @param endOfStream Indicates whether this is the last frame to be sent from the remote endpoint
92       *            for this stream.
93       */
94      void onHeadersRead(ChannelHandlerContext ctx, int streamId, Http2Headers headers,
95              int streamDependency, short weight, boolean exclusive, int padding, boolean endOfStream)
96              throws Http2Exception;
97  
98      /**
99       * Handles an inbound {@code PRIORITY} frame.
100      * <p>
101      * Note that is it possible to have this method called and no stream object exist for either
102      * {@code streamId}, {@code streamDependency}, or both. This is because the {@code PRIORITY} frame can be
103      * sent/received when streams are in the {@code CLOSED} state.
104      *
105      * @param ctx the context from the handler where the frame was read.
106      * @param streamId the subject stream for the frame.
107      * @param streamDependency the stream on which this stream depends, or 0 if dependent on the
108      *            connection.
109      * @param weight the new weight for the stream.
110      * @param exclusive whether or not the stream should be the exclusive dependent of its parent.
111      */
112     void onPriorityRead(ChannelHandlerContext ctx, int streamId, int streamDependency,
113             short weight, boolean exclusive) throws Http2Exception;
114 
115     /**
116      * Handles an inbound {@code RST_STREAM} frame.
117      *
118      * @param ctx the context from the handler where the frame was read.
119      * @param streamId the stream that is terminating.
120      * @param errorCode the error code identifying the type of failure.
121      */
122     void onRstStreamRead(ChannelHandlerContext ctx, int streamId, long errorCode) throws Http2Exception;
123 
124     /**
125      * Handles an inbound {@code SETTINGS} acknowledgment frame.
126      * @param ctx the context from the handler where the frame was read.
127      */
128     void onSettingsAckRead(ChannelHandlerContext ctx) throws Http2Exception;
129 
130     /**
131      * Handles an inbound {@code SETTINGS} frame.
132      *
133      * @param ctx the context from the handler where the frame was read.
134      * @param settings the settings received from the remote endpoint.
135      */
136     void onSettingsRead(ChannelHandlerContext ctx, Http2Settings settings) throws Http2Exception;
137 
138     /**
139      * Handles an inbound {@code PING} frame.
140      *
141      * @param ctx the context from the handler where the frame was read.
142      * @param data the payload of the frame. If this buffer needs to be retained by the listener
143      *            they must make a copy.
144      */
145     void onPingRead(ChannelHandlerContext ctx, ByteBuf data) throws Http2Exception;
146 
147     /**
148      * Handles an inbound {@code PING} acknowledgment.
149      *
150      * @param ctx the context from the handler where the frame was read.
151      * @param data the payload of the frame. If this buffer needs to be retained by the listener
152      *            they must make a copy.
153      */
154     void onPingAckRead(ChannelHandlerContext ctx, ByteBuf data) throws Http2Exception;
155 
156     /**
157      * Handles an inbound {@code PUSH_PROMISE} frame. Only called if {@code END_HEADERS} encountered.
158      * <p>
159      * Promised requests MUST be authoritative, cacheable, and safe.
160      * See <a href="https://tools.ietf.org/html/draft-ietf-httpbis-http2-17#section-8.2">[RFC http2], Seciton 8.2</a>.
161      * <p>
162      * Only one of the following methods will be called for each {@code HEADERS} frame sequence.
163      * One will be called when the {@code END_HEADERS} flag has been received.
164      * <ul>
165      * <li>{@link #onHeadersRead(ChannelHandlerContext, int, Http2Headers, int, boolean)}</li>
166      * <li>{@link #onHeadersRead(ChannelHandlerContext, int, Http2Headers, int, short, boolean, int, boolean)}</li>
167      * <li>{@link #onPushPromiseRead(ChannelHandlerContext, int, int, Http2Headers, int)}</li>
168      * </ul>
169      * <p>
170      * To say it another way; the {@link Http2Headers} will contain all of the headers
171      * for the current message exchange step (additional queuing is not necessary).
172      *
173      * @param ctx the context from the handler where the frame was read.
174      * @param streamId the stream the frame was sent on.
175      * @param promisedStreamId the ID of the promised stream.
176      * @param headers the received headers.
177      * @param padding the number of padding bytes found at the end of the frame.
178      */
179     void onPushPromiseRead(ChannelHandlerContext ctx, int streamId, int promisedStreamId,
180             Http2Headers headers, int padding) throws Http2Exception;
181 
182     /**
183      * Handles an inbound {@code GO_AWAY} frame.
184      *
185      * @param ctx the context from the handler where the frame was read.
186      * @param lastStreamId the last known stream of the remote endpoint.
187      * @param errorCode the error code, if abnormal closure.
188      * @param debugData application-defined debug data. If this buffer needs to be retained by the
189      *            listener they must make a copy.
190      */
191     void onGoAwayRead(ChannelHandlerContext ctx, int lastStreamId, long errorCode, ByteBuf debugData)
192             throws Http2Exception;
193 
194     /**
195      * Handles an inbound {@code WINDOW_UPDATE} frame.
196      *
197      * @param ctx the context from the handler where the frame was read.
198      * @param streamId the stream the frame was sent on.
199      * @param windowSizeIncrement the increased number of bytes of the remote endpoint's flow
200      *            control window.
201      */
202     void onWindowUpdateRead(ChannelHandlerContext ctx, int streamId, int windowSizeIncrement)
203             throws Http2Exception;
204 
205     /**
206      * Handler for a frame not defined by the HTTP/2 spec.
207      *
208      * @param ctx the context from the handler where the frame was read.
209      * @param frameType the frame type from the HTTP/2 header.
210      * @param streamId the stream the frame was sent on.
211      * @param flags the flags in the frame header.
212      * @param payload the payload of the frame.
213      */
214     void onUnknownFrame(ChannelHandlerContext ctx, byte frameType, int streamId, Http2Flags flags, ByteBuf payload)
215             throws Http2Exception;
