29  /**
30   * Logs HTTP2 frames for debugging purposes.
31   */

40      private final InternalLogger logger;

60      public void logData(Direction direction, ChannelHandlerContext ctx, int streamId, ByteBuf data, int padding,
61              boolean endStream) {
62          if (enabled()) {
63              log(direction,
64                      "%s DATA: streamId=%d, padding=%d, endStream=%b, length=%d, bytes=%s",
65                      ctx.channel(), streamId, padding, endStream, data.readableBytes(), toString(data));
66          }
67      }

159     private boolean enabled() {
160         return logger.isEnabled(level);
161     }
162 
163     private String toString(ByteBuf buf) {
164         if (level == InternalLogLevel.TRACE || buf.readableBytes() <= BUFFER_LENGTH_THRESHOLD) {
165             // Log the entire buffer.
166             return ByteBufUtil.hexDump(buf);
167         }
168 
169         // Otherwise just log the first 64 bytes.
170         int length = Math.min(buf.readableBytes(), BUFFER_LENGTH_THRESHOLD);
171         return ByteBufUtil.hexDump(buf, buf.readerIndex(), length) + "...";
172     }
173 
174     private void log(Direction direction, String format, Object... args) {
175         StringBuilder b = new StringBuilder(200);
176         b.append("\n----------------")
177                 .append(direction.name())
178                 .append("--------------------\n")
179                 .append(String.format(format, args))
180                 .append("\n------------------------------------");
181         logger.log(level, b.toString());
182     }
