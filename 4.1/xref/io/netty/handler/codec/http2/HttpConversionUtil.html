162     /**
163      * Apply HTTP/2 rules while translating status code to {@link HttpResponseStatus}
164      *
165      * @param status The status from an HTTP/2 frame
166      * @return The HTTP/1.x status
167      * @throws Http2Exception If there is a problem translating from HTTP/2 to HTTP/1.x
168      */
169     public static HttpResponseStatus parseStatus(CharSequence status) throws Http2Exception {
170         HttpResponseStatus result;
171         try {
172             result = HttpResponseStatus.parseLine(status);
173             if (result == HttpResponseStatus.SWITCHING_PROTOCOLS) {
174                 throw connectionError(PROTOCOL_ERROR, "Invalid HTTP/2 status code '%d'", result.code());
175             }
176         } catch (Http2Exception e) {
177             throw e;
178         } catch (Throwable t) {
179             throw connectionError(PROTOCOL_ERROR, t,
180                             "Unrecognized HTTP status code '%s' encountered in translation to HTTP/1.x", status);
181         }
182         return result;
183     }

185     /**
186      * Create a new object to contain the response data
187      *
188      * @param streamId The stream associated with the response
189      * @param http2Headers The initial set of HTTP/2 headers to create the response with
190      * @param alloc The {@link ByteBufAllocator} to use to generate the content of the message
191      * @param validateHttpHeaders <ul>
192      *        <li>{@code true} to validate HTTP headers in the http-codec</li>
193      *        <li>{@code false} not to validate HTTP headers in the http-codec</li>
194      *        </ul>
195      * @return A new response object which represents headers/data
196      * @throws Http2Exception see {@link #addHttp2ToHttpHeaders(int, Http2Headers, FullHttpMessage, boolean)}
197      */
198     public static FullHttpResponse toHttpResponse(int streamId, Http2Headers http2Headers, ByteBufAllocator alloc,
199                                                   boolean validateHttpHeaders)
200                     throws Http2Exception {
201         HttpResponseStatus status = parseStatus(http2Headers.status());
202         // HTTP/2 does not define a way to carry the version or reason phrase that is included in an
203         // HTTP/1.1 status line.
204         FullHttpResponse msg = new DefaultFullHttpResponse(HttpVersion.HTTP_1_1, status, alloc.buffer(),
205                                                            validateHttpHeaders);
206         try {
207             addHttp2ToHttpHeaders(streamId, http2Headers, msg, false);
208         } catch (Http2Exception e) {
209             msg.release();
210             throw e;
211         } catch (Throwable t) {
212             msg.release();
213             throw streamError(streamId, PROTOCOL_ERROR, t, "HTTP/2 to HTTP/1.x headers conversion error");
214         }
215         return msg;
216     }

218     /**
219      * Create a new object to contain the request data
220      *
221      * @param streamId The stream associated with the request
222      * @param http2Headers The initial set of HTTP/2 headers to create the request with
223      * @param alloc The {@link ByteBufAllocator} to use to generate the content of the message
224      * @param validateHttpHeaders <ul>
225      *        <li>{@code true} to validate HTTP headers in the http-codec</li>
226      *        <li>{@code false} not to validate HTTP headers in the http-codec</li>
227      *        </ul>
228      * @return A new request object which represents headers/data
229      * @throws Http2Exception see {@link #addHttp2ToHttpHeaders(int, Http2Headers, FullHttpMessage, boolean)}
230      */
231     public static FullHttpRequest toHttpRequest(int streamId, Http2Headers http2Headers, ByteBufAllocator alloc,
232                                                 boolean validateHttpHeaders)
233                     throws Http2Exception {
234         // HTTP/2 does not define a way to carry the version identifier that is included in the HTTP/1.1 request line.
235         final CharSequence method = checkNotNull(http2Headers.method(),
236                 "method header cannot be null in conversion to HTTP/1.x");
237         final CharSequence path = checkNotNull(http2Headers.path(),
238                 "path header cannot be null in conversion to HTTP/1.x");
239         FullHttpRequest msg = new DefaultFullHttpRequest(HttpVersion.HTTP_1_1, HttpMethod.valueOf(method
240                         .toString()), path.toString(), alloc.buffer(), validateHttpHeaders);
241         try {
242             addHttp2ToHttpHeaders(streamId, http2Headers, msg, false);
243         } catch (Http2Exception e) {
244             msg.release();
245             throw e;
246         } catch (Throwable t) {
247             msg.release();
248             throw streamError(streamId, PROTOCOL_ERROR, t, "HTTP/2 to HTTP/1.x headers conversion error");
249         }
250         return msg;
251     }

270     /**
271      * Translate and add HTTP/2 headers to HTTP/1.x headers.
272      *
273      * @param streamId The stream associated with {@code sourceHeaders}.
274      * @param inputHeaders The HTTP/2 headers to convert.
275      * @param outputHeaders The object which will contain the resulting HTTP/1.x headers..
276      * @param httpVersion What HTTP/1.x version {@code outputHeaders} should be treated as when doing the conversion.
277      * @param isTrailer {@code true} if {@code outputHeaders} should be treated as trailing headers.
278      * {@code false} otherwise.
279      * @param isRequest {@code true} if the {@code outputHeaders} will be used in a request message.
280      * {@code false} for response message.
281      * @throws Http2Exception If not all HTTP/2 headers can be translated to HTTP/1.x.
282      */
283     public static void addHttp2ToHttpHeaders(int streamId, Http2Headers inputHeaders, HttpHeaders outputHeaders,
284             HttpVersion httpVersion, boolean isTrailer, boolean isRequest) throws Http2Exception {
285         Http2ToHttpHeaderTranslator translator = new Http2ToHttpHeaderTranslator(streamId, outputHeaders, isRequest);
286         try {
287             for (Entry<CharSequence, CharSequence> entry : inputHeaders) {
288                 translator.translate(entry);
289             }
290         } catch (Http2Exception ex) {
291             throw ex;
292         } catch (Throwable t) {
293             throw streamError(streamId, PROTOCOL_ERROR, t, "HTTP/2 to HTTP/1.x headers conversion error");
294         }
295 
296         outputHeaders.remove(HttpHeaderNames.TRANSFER_ENCODING);
297         outputHeaders.remove(HttpHeaderNames.TRAILER);
298         if (!isTrailer) {
299             outputHeaders.setInt(ExtensionHeaderNames.STREAM_ID.text(), streamId);
300             HttpUtil.setKeepAlive(outputHeaders, httpVersion, true);
301         }
302     }

304     /**
305      * Converts the given HTTP/1.x headers into HTTP/2 headers.
306      * The following headers are only used if they can not be found in from the {@code HOST} header or the
307      * {@code Request-Line} as defined by <a href="https://tools.ietf.org/html/rfc7230">rfc7230</a>
308      * <ul>
309      * <li>{@link ExtensionHeaderNames#AUTHORITY}</li>
310      * <li>{@link ExtensionHeaderNames#SCHEME}</li>
311      * </ul>
312      * {@link ExtensionHeaderNames#PATH} is ignored and instead extracted from the {@code Request-Line}.
313      */
314     public static Http2Headers toHttp2Headers(HttpMessage in, boolean validateHeaders) {
315         HttpHeaders inHeaders = in.headers();
316         final Http2Headers out = new DefaultHttp2Headers(validateHeaders, inHeaders.size());
317         if (in instanceof HttpRequest) {
318             HttpRequest request = (HttpRequest) in;
319             URI requestTargetUri = URI.create(request.uri());
320             out.path(toHttp2Path(requestTargetUri));
321             out.method(request.method().asciiName());
322             setHttp2Scheme(inHeaders, requestTargetUri, out);
323 
324             if (!isOriginForm(requestTargetUri) && !isAsteriskForm(requestTargetUri)) {
325                 // Attempt to take from HOST header before taking from the request-line
326                 String host = inHeaders.getAsString(HttpHeaderNames.HOST);
327                 setHttp2Authority((host == null || host.isEmpty()) ? requestTargetUri.getAuthority() : host, out);
328             }
329         } else if (in instanceof HttpResponse) {
330             HttpResponse response = (HttpResponse) in;
331             out.status(new AsciiString(Integer.toString(response.status().code())));
332         }
333 
334         // Add the HTTP headers which have not been consumed above
335         toHttp2Headers(inHeaders, out);
336         return out;
337     }

340         if (inHeaders.isEmpty()) {

349     public static void toHttp2Headers(HttpHeaders inHeaders, Http2Headers out) {
350         Iterator<Entry<CharSequence, CharSequence>> iter = inHeaders.iteratorCharSequence();
351         while (iter.hasNext()) {
352             Entry<CharSequence, CharSequence> entry = iter.next();
353             final AsciiString aName = AsciiString.of(entry.getKey()).toLowerCase();
354             if (!HTTP_TO_HTTP2_HEADER_BLACKLIST.contains(aName)) {
355                 // https://tools.ietf.org/html/rfc7540#section-8.1.2.2 makes a special exception for TE
356                 if (aName.contentEqualsIgnoreCase(HttpHeaderNames.TE) &&
357                     !AsciiString.contentEqualsIgnoreCase(entry.getValue(), HttpHeaderValues.TRAILERS)) {
358                     throw new IllegalArgumentException("Invalid value for " + HttpHeaderNames.TE + ": " +
359                                                         entry.getValue());
360                 }
361                 if (aName.contentEqualsIgnoreCase(HttpHeaderNames.COOKIE)) {
362                     AsciiString value = AsciiString.of(entry.getValue());
363                     // split up cookies to allow for better compression
364                     // https://tools.ietf.org/html/rfc7540#section-8.1.2.5
365                     try {
366                         int index = value.forEachByte(FIND_SEMI_COLON);
367                         if (index != -1) {
368                             int start = 0;
369                             do {
370                                 out.add(HttpHeaderNames.COOKIE, value.subSequence(start, index, false));
371                                 // skip 2 characters "; " (see https://tools.ietf.org/html/rfc6265#section-4.2.1)
372                                 start = index + 2;
373                             } while (start < value.length() &&
374                                     (index = value.forEachByte(start, value.length() - start, FIND_SEMI_COLON)) != -1);
375                             if (start >= value.length()) {
376                                 throw new IllegalArgumentException("cookie value is of unexpected format: " + value);
377                             }
378                             out.add(HttpHeaderNames.COOKIE, value.subSequence(start, value.length(), false));
379                         } else {
380                             out.add(HttpHeaderNames.COOKIE, value);
381                         }
382                     } catch (Exception e) {
383                         // This is not expect to happen because FIND_SEMI_COLON never throws but must be caught
384                         // because of the ByteProcessor interface.
385                         throw new IllegalStateException(e);
386                     }
387                 } else {
388                     out.add(aName, entry.getValue());
389                 }
390             }
391         }
392     }

394     /**
395      * Generate a HTTP/2 {code :path} from a URI in accordance with
396      * <a href="https://tools.ietf.org/html/rfc7230#section-5.3">rfc7230, 5.3</a>.
397      */
398     private static AsciiString toHttp2Path(URI uri) {
399         StringBuilder pathBuilder = new StringBuilder(length(uri.getRawPath()) +
400                 length(uri.getRawQuery()) + length(uri.getRawFragment()) + 2);
401         if (!isNullOrEmpty(uri.getRawPath())) {
402             pathBuilder.append(uri.getRawPath());
403         }
404         if (!isNullOrEmpty(uri.getRawQuery())) {
405             pathBuilder.append('?');
406             pathBuilder.append(uri.getRawQuery());
407         }
408         if (!isNullOrEmpty(uri.getRawFragment())) {
409             pathBuilder.append('#');
410             pathBuilder.append(uri.getRawFragment());
411         }
412         String path = pathBuilder.toString();
413         return path.isEmpty() ? EMPTY_REQUEST_PATH : new AsciiString(path);
414     }

416     private static void setHttp2Authority(String autority, Http2Headers out) {
417         // The authority MUST NOT include the deprecated "userinfo" subcomponent
418         if (autority != null) {
419             int endOfUserInfo = autority.indexOf('@');
420             if (endOfUserInfo < 0) {
421                 out.authority(new AsciiString(autority));
422             } else if (endOfUserInfo + 1 < autority.length()) {
423                 out.authority(new AsciiString(autority.substring(endOfUserInfo + 1)));
424             } else {
425                 throw new IllegalArgumentException("autority: " + autority);
426             }
427         }
428     }

430     private static void setHttp2Scheme(HttpHeaders in, URI uri, Http2Headers out) {
431         String value = uri.getScheme();
432         if (value != null) {
433             out.scheme(new AsciiString(value));
434             return;
435         }
436 
437         // Consume the Scheme extension header if present
438         CharSequence cValue = in.get(ExtensionHeaderNames.SCHEME.text());
439         if (cValue != null) {
440             out.scheme(AsciiString.of(cValue));
441             return;
442         }
443 
444         if (uri.getPort() == HTTPS.port()) {
445             out.scheme(HTTPS.name());
446         } else if (uri.getPort() == HTTP.port()) {
447             out.scheme(HTTP.name());
448         } else {
449             throw new IllegalArgumentException(":scheme must be specified. " +
450                     "see https://tools.ietf.org/html/rfc7540#section-8.1.2.3");
451         }

454     /**
455      * Utility which translates HTTP/2 headers to HTTP/1 headers.
456      */
457     private static final class Http2ToHttpHeaderTranslator {
458         /**
459          * Translations from HTTP/2 header name to the HTTP/1.x equivalent.
460          */
461         private static final CharSequenceMap<AsciiString>
462             REQUEST_HEADER_TRANSLATIONS = new CharSequenceMap<AsciiString>();
463         private static final CharSequenceMap<AsciiString>
464             RESPONSE_HEADER_TRANSLATIONS = new CharSequenceMap<AsciiString>();
465         static {
466             RESPONSE_HEADER_TRANSLATIONS.add(Http2Headers.PseudoHeaderName.AUTHORITY.value(),
467                             HttpHeaderNames.HOST);
468             RESPONSE_HEADER_TRANSLATIONS.add(Http2Headers.PseudoHeaderName.SCHEME.value(),
469                             ExtensionHeaderNames.SCHEME.text());
470             REQUEST_HEADER_TRANSLATIONS.add(RESPONSE_HEADER_TRANSLATIONS);
471             RESPONSE_HEADER_TRANSLATIONS.add(Http2Headers.PseudoHeaderName.PATH.value(),
472                             ExtensionHeaderNames.PATH.text());
473         }
474 
475         private final int streamId;
476         private final HttpHeaders output;
477         private final CharSequenceMap<AsciiString> translations;
478 
479         /**
480          * Create a new instance
481          *
482          * @param output The HTTP/1.x headers object to store the results of the translation
483          * @param request if {@code true}, translates headers using the request translation map. Otherwise uses the
484          *        response translation map.
485          */
486         Http2ToHttpHeaderTranslator(int streamId, HttpHeaders output, boolean request) {
487             this.streamId = streamId;
488             this.output = output;
489             translations = request ? REQUEST_HEADER_TRANSLATIONS : RESPONSE_HEADER_TRANSLATIONS;
490         }
491 
492         public void translate(Entry<CharSequence, CharSequence> entry) throws Http2Exception {
493             final CharSequence name = entry.getKey();
494             final CharSequence value = entry.getValue();
495             AsciiString translatedName = translations.get(name);
496             if (translatedName != null) {
497                 output.add(translatedName, AsciiString.of(value));
498             } else if (!Http2Headers.PseudoHeaderName.isPseudoHeader(name)) {
499                 // https://tools.ietf.org/html/rfc7540#section-8.1.2.3
500                 // All headers that start with ':' are only valid in HTTP/2 context
501                 if (name.length() == 0 || name.charAt(0) == ':') {
502                     throw streamError(streamId, PROTOCOL_ERROR,
503                             "Invalid HTTP/2 header '%s' encountered in translation to HTTP/1.x", name);
504                 }
505                 if (HttpHeaderNames.COOKIE.equals(name)) {
506                     // combine the cookie values into 1 header entry.
507                     // https://tools.ietf.org/html/rfc7540#section-8.1.2.5
508                     String existingCookie = output.get(HttpHeaderNames.COOKIE);
509                     output.set(HttpHeaderNames.COOKIE,
510                                (existingCookie != null) ? (existingCookie + "; " + value) : value);
511                 } else {
512                     output.add(name, value);
513                 }
514             }
515         }
516     }
